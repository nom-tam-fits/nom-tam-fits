61  /**
62   * This class provides access to routines to allow users to read and write FITS
63   * files. <br>
64   * <b> Description of the Package </b>
65   * <p>
66   * This FITS package attempts to make using FITS files easy, but does not do
67   * exhaustive error checking. Users should not assume that just because a FITS
68   * file can be read and written that it is necessarily legal FITS. These classes
69   * try to make it easy to transform between arrays of Java primitives and their
70   * FITS encodings.
71   * <ul>
72   * <li>The Fits class provides capabilities to read and write data at the HDU
73   * level, and to add and delete HDU's from the current Fits object. A large
74   * number of constructors are provided which allow users to associate the Fits
75   * object with some form of external data. This external data may be in a
76   * compressed format.
77   * <p>
78   * Note that this association is limited, it only specifies where the various
79   * read methods should read data from. It does not automatically read the data
80   * content and store the results. To ensure that the external content has been
81   * read and parsed the user may wish to invoke the read() method after creating
82   * the Fits object associated with external data. E.g.,
83   * 
84   * <pre>
85   *     File fl = ...  ; 
86   *     Fits f = new Fits(fl); // Or we could have used one of the other constructors.
87   *     // At this point the Fits object is empty.
88   *     f.read();    // Read the external data into the Fits object
89   *     // At this point the Fits object should have one or more HDUs depending
90   *     // upon the external content.
91   * </pre>
92   * 
93   * Users can choose to read only some of the HDUs in a given input, and may add
94   * HDU's that were either read from other files or generated by the program. See
95   * the various read and addHDU methods.
96   * <li>The FitsFactory class is a factory class which is used to create HDUs.
97   * HDU's can be of a number of types derived from the abstract class BasicHDU.
98   * The hierarchy of HDUs is:
99   * <ul>
100  * <li>BasicHDU
101  * <ul>
102  * <li>ImageHDU
103  * <li>RandomGroupsHDU
104  * <li>TableHDU
105  * <ul>
106  * <li>BinaryTableHDU
107  * <li>AsciiTableHDU
108  * </ul>
109  * <li>UndefinedHDU
110  * </ul>
111  * </ul>
112  * <li>The Header class provides many functions to add, delete and read header
113  * keywords in a variety of formats.
114  * <li>The HeaderCard class provides access to the structure of a FITS header
115  * card.
116  * <li>The header package defines sets of enumerations that allow users to
117  * create and access header keywords in a controlled way.
118  * <li>The Data class is an abstract class which provides the basic methods for
119  * reading and writing FITS data. It provides methods to get the the actual
120  * underlying arrays and detailed methods for manipulation specific to the
121  * different data types.
122  * <li>The TableHDU class provides a large number of methods to access and
123  * modify information in tables.
124  * <li>The utilities package includes simple tools to copy and list FITS files.
125  * </ul>
126  * 
127  * @version 1.16
128  */

248     /**
249      * Associate the FITS object with a file or URL. The string is assumed to be
250      * a URL if it begins one of the protocol strings. If the string ends in .gz
251      * it is assumed that the data is in a compressed format. All string
252      * comparisons are case insensitive.
253      * 
254      * @param filename
255      *            The name of the file or URL to be processed. The content of
256      *            this file will not be read into the Fits object until the user
257      *            makes some explicit request.
258      * @param compressed
259      *            is the file compressed?
260      * @throws FitsException
261      *             Thrown if unable to find or open a file or URL from the
262      *             string given.
263      **/
264     @SuppressWarnings("resource")
265     public Fits(String filename, boolean compressed) throws FitsException {
266         if (filename == null) {
267             throw new FitsException("Null FITS Identifier String");
268         }
269         try {
270             File fil = new File(filename);
271             if (fil.exists()) {
272                 fileInit(fil, compressed);
273                 return;
274             }
275         } catch (Exception e) {
276             LOG.log(Level.FINE, "not a file " + filename, e);
277             throw new FitsException("could not detect type of " + filename, e);
278         }
279         try {
280             InputStream str = Thread.currentThread().getContextClassLoader().getResourceAsStream(filename);
281             if (str != null) {
282                 streamInit(str);
283                 return;
284             }
285         } catch (Exception e) {
286             LOG.log(Level.FINE, "not a resource " + filename, e);
287             throw new FitsException("could not detect type of " + filename, e);
288         }
289         try {
290             InputStream is = FitsUtil.getURLStream(new URL(filename), 0);
291             streamInit(is);
292             return;
293         } catch (Exception e) {
294             LOG.log(Level.FINE, "not a url " + filename, e);
295             throw new FitsException("could not detect type of " + filename, e);
296         }
297 
298     }

426     /**
427      * Delete an HDU from the HDU list.
428      * 
429      * @param n
430      *            The index of the HDU to be deleted. If n is 0 and there is
431      *            more than one HDU present, then the next HDU will be converted
432      *            from an image to primary HDU if possible. If not a dummy
433      *            header HDU will then be inserted.
434      * @throws FitsException
435      *             if the HDU could not be deleted.
436      */
437     public void deleteHDU(int n) throws FitsException {
438         int size = getNumberOfHDUs();
439         if (n < 0 || n >= size) {
440             throw new FitsException("Attempt to delete non-existent HDU:" + n);
441         }
442         this.hduList.remove(n);
443         if (n == 0 && size > 1) {
444             BasicHDU<?> newFirst = this.hduList.get(0);
445             if (newFirst.canBePrimary()) {
446                 newFirst.setPrimaryHDU(true);
447             } else {
448                 insertHDU(BasicHDU.getDummyHDU(), 0);
449             }
450         }
451     }

477     /**
478      * Returns the n'th HDU. If the HDU is already read simply return a pointer
479      *         to the cached data. Otherwise read the associated stream until
480      *         the n'th HDU is read.
481      * @param n
482      *            The index of the HDU to be read. The primary HDU is index 0.
483      * @return The n'th HDU or null if it could not be found.
484      * @throws FitsException
485      *             if the header could not be read
486      * @throws IOException
487      *             if the underlying buffer threw an error
488      */
489     public BasicHDU<?> getHDU(int n) throws FitsException, IOException {
490         int size = getNumberOfHDUs();
491         for (int i = size; i <= n; i += 1) {
492             BasicHDU<?> hdu = readHDU();
493             if (hdu == null) {
494                 return null;
495             }
496         }
497         return this.hduList.get(n);
498     }

520     /**
521      * Insert a FITS object into the list of HDUs.
522      * 
523      * @param myHDU
524      *            The HDU to be inserted into the list of HDUs.
525      * @param position
526      *            The location at which the HDU is to be inserted.
527      * @throws FitsException
528      *             if the HDU could not be inserted.
529      */
530     public void insertHDU(BasicHDU<?> myHDU, int position) throws FitsException {
531         if (myHDU == null) {
532             return;
533         }
534         if (position < 0 || position > getNumberOfHDUs()) {
535             throw new FitsException("Attempt to insert HDU at invalid location: " + position);
536         }
537         try {
538             if (position == 0) {
539                 // Note that the previous initial HDU is no longer the first.
540                 // If we were to insert tables backwards from last to first,
541                 // we could get a lot of extraneous DummyHDUs but we currently
542                 // do not worry about that.
543                 if (getNumberOfHDUs() > 0) {
544                     this.hduList.get(0).setPrimaryHDU(false);
545                 }
546                 if (myHDU.canBePrimary()) {
547                     myHDU.setPrimaryHDU(true);
548                     this.hduList.add(0, myHDU);
549                 } else {
550                     insertHDU(BasicHDU.getDummyHDU(), 0);
551                     myHDU.setPrimaryHDU(false);
552                     this.hduList.add(1, myHDU);
553                 }
554             } else {
555                 myHDU.setPrimaryHDU(false);
556                 this.hduList.add(position, myHDU);
557             }
558         } catch (NoSuchElementException e) {
559             throw new FitsException("hduList inconsistency in insertHDU", e);
560         }
561     }

563     /**
564      * Initialize using buffered random access. This implies that the data is
565      * uncompressed.
566      * 
567      * @param file
568      *            the file to open
569      * @throws FitsException
570      *             if the file could not be read
571      */
572     protected void randomInit(File file) throws FitsException {
573 
574         String permissions = "r";
575         if (!file.exists() || !file.canRead()) {
576             throw new FitsException("Non-existent or unreadable file");
577         }
578         if (file.canWrite()) {
579             permissions += "w";
580         }
581         try {
582             this.dataStr = new FitsFile(file, permissions);
583             ((FitsFile) this.dataStr).seek(0);
584         } catch (IOException e) {
585             throw new FitsException("Unable to open file " + file.getPath(), e);
586         }
587     }

589     /**
590      * Return all HDUs for the Fits object. If the FITS file is associated with
591      * an external stream make sure that we have exhausted the stream.
592      * 
593      * @return an array of all HDUs in the Fits object. Returns null if there
594      *         are no HDUs associated with this object.
595      * @throws FitsException
596      *             if the reading failed.
597      */
598     public BasicHDU<?>[] read() throws FitsException {
599         readToEnd();
600         int size = getNumberOfHDUs();
601         if (size == 0) {
602             return new BasicHDU<?>[0];
603         }
604         return this.hduList.toArray(new BasicHDU<?>[size]);
605     }

624     /**
625      * Read the next HDU on the default input stream.
626      * 
627      * @return The HDU read, or null if an EOF was detected. Note that null is
628      *         only returned when the EOF is detected immediately at the
629      *         beginning of reading the HDU.
630      * @throws FitsException
631      *             if the header could not be read
632      * @throws IOException
633      *             if the underlying buffer threw an error
634      */
635     public BasicHDU<?> readHDU() throws FitsException, IOException {
636         if (this.dataStr == null || this.atEOF) {
637             if (this.dataStr == null) {
638                 LOG.warning("trying to read a hdu, without an input source!");
639             }
640             return null;
641         }
642         if (this.dataStr instanceof RandomAccess && this.lastFileOffset > 0) {
643             FitsUtil.reposition(this.dataStr, this.lastFileOffset);
644         }
645 
646         Header hdr = Header.readHeader(this.dataStr);
647         if (hdr == null) {
648             this.atEOF = true;
649             return null;
650         }
651 
652         Data data = hdr.makeData();
653         try {
654             data.read(this.dataStr);
655         } catch (PaddingException e) {
656             // Stream end before required padding after data...
657             e.updateHeader(hdr);
658             LOG.warning("Missing padding after data segment.");
659         }
660 
661 
662         // Check for truncation even if we successfully skipped to the expected end
663         // since skip may allow going beyond the EOF.
664         if (this.dataStr.checkTruncated()) {
665             // File ends before required padding after data...
666             // (files allow skipping beyond the end, which is why we don't
667             // catch it above)
668             LOG.warning("Missing padding after data segment.");
669         }
670 
671         this.lastFileOffset = FitsUtil.findOffset(this.dataStr);
672         BasicHDU<Data> nextHDU = FitsFactory.hduFactory(hdr, data);
673         this.hduList.add(nextHDU);
674 
675         return nextHDU;
676     }

678     /**
679      * Read to the end of the associated input stream
680      * 
681      * @throws FitsException
682      *             if the operation failed
683      */
684     private void readToEnd() throws FitsException {
685         try {
686             while (this.dataStr != null && !this.atEOF) {
687                 if (readHDU() == null) {
688                     if (getNumberOfHDUs() == 0) {
689                         throw new FitsException("Not FITS file.");
690                     }
691                     return;
692                 }
693             }

761     /**
762      * Skip the next HDU on the default input stream.
763      * 
764      * @throws FitsException
765      *             if the HDU could not be skipped
766      * @throws IOException
767      *             if the underlying stream failed
768      */
769     public void skipHDU() throws FitsException, IOException {
770         if (this.atEOF) {
771             return;
772         } 
773 
774         Header hdr = new Header(this.dataStr);
775         int dataSize = (int) hdr.getDataSize();
776         this.dataStr.skipAllBytes(dataSize);
777         if (this.dataStr instanceof RandomAccess) {
778             this.lastFileOffset = ((RandomAccess) this.dataStr).getFilePointer();
779         }
780     }
