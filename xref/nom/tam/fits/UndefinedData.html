34  import static nom.tam.fits.header.Standard.BITPIX;
35  import static nom.tam.fits.header.Standard.EXTEND;
36  import static nom.tam.fits.header.Standard.GCOUNT;
37  import static nom.tam.fits.header.Standard.NAXIS;
38  import static nom.tam.fits.header.Standard.NAXISn;
39  import static nom.tam.fits.header.Standard.PCOUNT;
40  import static nom.tam.util.LoggerHelper.getLogger;
41  
42  import java.io.EOFException;
43  import java.io.IOException;
44  import java.util.logging.Level;
45  import java.util.logging.Logger;
46  
47  import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
48  import nom.tam.fits.header.Standard;
49  import nom.tam.util.ArrayDataInput;
50  import nom.tam.util.ArrayDataOutput;
51  import nom.tam.util.ArrayFuncs;
52  
53  /**
54   * This class provides a simple holder for data which is not handled by other
55   * classes.
56   */
57  public class UndefinedData extends Data {
58  
59      private static final Logger LOG = getLogger(UndefinedData.class);
60  
61      private static final int BITS_PER_BYTE = 8;
62  
63      private byte[] data;
64  
65      public UndefinedData(Header h) throws FitsException {
66  
67          /**
68           * Just get a byte buffer to hold the data.
69           */
70          // Bug fix by Vincenzo Forzi.
71          int naxis = h.getIntValue(NAXIS);
72  
73          int size = naxis > 0 ? 1 : 0;
74          for (int i = 0; i < naxis; i += 1) {
75              size *= h.getIntValue(NAXISn.n(i + 1));
76          }
77          size += h.getIntValue(PCOUNT);
78          if (h.getIntValue(GCOUNT) > 1) {
79              size *= h.getIntValue(GCOUNT);
80          }
81          size *= Math.abs(h.getIntValue(BITPIX) / BITS_PER_BYTE);
82  
83          this.data = new byte[size];
84      }
85  
86      /**
87       * Create an UndefinedData object using the specified object.
88       * 
89       * @param x
90       *            object to create the hdu from
91       */
92      public UndefinedData(Object x) {
93          this.data = new byte[(int) ArrayFuncs.computeLSize(x)];
94          ArrayFuncs.copyInto(x, this.data);
95      }
96  
97      /**
98       * Fill header with keywords that describe data.
99       * 
100      * @param head
101      *            The FITS header
102      */
103     @Override
104     protected void fillHeader(Header head) {
105         try {
106             Standard.context(UndefinedData.class);
107             head.setXtension("UNKNOWN");
108             head.setBitpix(BasicHDU.BITPIX_BYTE);
109             head.setNaxes(1);
110             head.addValue(NAXISn.n(1), this.data.length);
111             head.addValue(PCOUNT, 0);
112             head.addValue(GCOUNT, 1);
113             // Just in case!
114             head.addValue(EXTEND, true);
115         } catch (HeaderCardException e) {
116             LOG.log(Level.SEVERE, "Unable to create unknown header", e);
117         } finally {
118             Standard.context(null);
119         }
120     }
121 
122     @Override
123     @SuppressFBWarnings(value = "EI_EXPOSE_REP", justification = "intended exposure of mutable data")
124     public Object getData() {
125         return this.data;
126     }
127 
128     /** Get the size in bytes of the data */
129     @Override
130     protected long getTrueSize() {
131         return this.data.length;
132     }
133 
134     @Override
135     public void read(ArrayDataInput i) throws FitsException {
136         setFileOffset(i);
137         try {
138             i.readFully(this.data);
139         } catch (IOException e) {
140             throw new FitsException("Unable to read unknown data:" + e);
141         }
142 
143         int pad = FitsUtil.padding(getTrueSize());
144         try {
145             i.skipAllBytes(pad);
146         } catch (EOFException e) {
147             throw new PaddingException("EOF skipping padding in undefined data", this, e);
148         } catch (IOException e) {
149             throw new FitsException("Error skipping padding in undefined data", e);
150         }
151     }
152 
153     @Override
154     public void write(ArrayDataOutput o) throws FitsException {
155         try {
156             o.write(this.data);
157         } catch (IOException e) {
158             throw new FitsException("IO Error on unknown data write" + e);
159         }
160         FitsUtil.pad(o, getTrueSize());
161     }
162 }
