34  /**
35   * This data dictionary contains FITS keywords that have been widely used within
36   * the astronomical community. It is recommended that these keywords only be
37   * used as defined here. These are the Keywords that describe the observation.
38   * 
39   * <pre>
40   * @see <a href="http://heasarc.gsfc.nasa.gov/docs/fcg/common_dict.html">http://heasarc.gsfc.nasa.gov/docs/fcg/common_dict.html</a>
41   * </pre>
42   * 
43   * @author Richard van Nieuwenhoven
44   */
45  public enum ObservationDurationDescription implements IFitsHeader {
46      /**
47       * The value field shall contain a character string that gives the date on
48       * which the observation ended. This keyword has the same format, and is
49       * used in conjunction with, the standard DATA-OBS keyword that gives the
50       * starting date of the observation. These 2 keywords may give either the
51       * calendar date using the 'yyyy-mm-dd' format, or may give the full date
52       * and time using the 'yyyy-mm-ddThh:mm:ss.sss' format.
53       */
54      DATE_END("DATE-END", SOURCE.HEASARC, HDU.ANY, VALUE.STRING, "date of the end of observation"),
55      /**
56       * The value field shall contain a floating point number giving the
57       * difference between the stop and start times of the observation in units
58       * of seconds. This keyword is synonymous with the TELAPSE keyword.
59       */
60      ELAPTIME(SOURCE.UCOLICK, HDU.ANY, VALUE.REAL, "elapsed time of the observation"),
61      /**
62       * The value field shall contain a floating point number giving the exposure
63       * time of the observation in units of seconds. The exact definition of
64       * 'exposure time' is mission dependent and may, for example, include
65       * corrections for shutter open and close duration, detector dead time,
66       * vignetting, or other effects. This keyword is synonymous with the EXPTIME
67       * keyword.
68       */
69      EXPOSURE(SOURCE.HEASARC, HDU.ANY, VALUE.REAL, "exposure time"),
70      /**
71       * The value field shall contain a floating point number giving the exposure
72       * time of the observation in units of seconds. The exact definition of
73       * 'exposure time' is mission dependent and may, for example, include
74       * corrections for shutter open and close duration, detector dead time,
75       * vignetting, or other effects. This keyword is synonymous with the
76       * EXPOSURE keyword.
77       */
78      EXPTIME(SOURCE.NOAO, HDU.ANY, VALUE.REAL, "exposure time"),
79      /**
80       * The value field shall contain a floating point number giving the total
81       * integrated exposure time in units of seconds corrected for detector 'dead
82       * time' effects which reduce the net efficiency of the detector. The ratio
83       * of LIVETIME/ONTIME gives the mean dead time correction during the
84       * observation, which lies in the range 0.0 to 1.0.
85       */
86      LIVETIME(SOURCE.HEASARC, HDU.ANY, VALUE.REAL, "exposure time after deadtime correction"),
87      /**
88       * The value field shall contain a floating point number giving the total
89       * integrated exposure time of the observation in units of seconds. ONTIME
90       * may be less than TELAPSE if there were intevals during the observation in
91       * which the target was not observed (e.g., the shutter was closed, or the
92       * detector power was turned off).
93       */
94      ONTIME(SOURCE.HEASARC, HDU.ANY, VALUE.REAL, "integration time during the observation"),
95      /**
96       * The value field shall contain a floating point number giving the
97       * difference between the stop and start times of the observation in units
98       * of seconds. This keyword is synonymous with the ELAPTIME keyword.
99       */
100     TELAPSE(SOURCE.HEASARC, HDU.ANY, VALUE.REAL, "elapsed time of the observation"),
101     /**
102      * The value field shall contain a character string that gives the time at
103      * which the observation ended. This keyword is used in conjunction with the
104      * DATE-END keyword to give the ending time of the observation; the DATE-END
105      * keyword gives the ending calendar date, with format 'yyyy-mm-dd', and
106      * TIME-END gives the time within that day using the format
107      * 'hh:mm:ss.sss...'. This keyword should not be used if the time is
108      * included directly as part of the DATE-END keyword value with the format
109      * 'yyyy-mm-ddThh:mm:ss.sss'.
110      */
111     TIME_END("TIME-END", SOURCE.HEASARC, HDU.ANY, VALUE.STRING, "time at the end of the observation"),
112     /**
113      * The value field shall contain a character string that gives the time at
114      * which the observation started. This keyword is used in conjunction with
115      * the standard DATE-OBS keyword to give the starting time of the
116      * observation; the DATE-OBS keyword gives the starting calendar date, with
117      * format 'yyyy-mm-dd', and TIME-OBS gives the time within that day using
118      * the format 'hh:mm:ss.sss...'. This keyword should not be used if the time
119      * is included directly as part of the DATE-OBS keyword value with the
120      * format 'yyyy-mm-ddThh:mm:ss.sss'.
121      */
122     TIME_OBS("TIME-OBS", SOURCE.HEASARC, HDU.ANY, VALUE.STRING, "time at the start of the observation");
123 
124     @SuppressWarnings("CPD-START")
125     private final IFitsHeader key;
126 
127     ObservationDurationDescription(SOURCE status, HDU hdu, VALUE valueType, String comment) {
128         this.key = new FitsHeaderImpl(name(), status, hdu, valueType, comment);
129     }
130 
131     ObservationDurationDescription(String key, SOURCE status, HDU hdu, VALUE valueType, String comment) {
132         this.key = new FitsHeaderImpl(key == null ? name() : key, status, hdu, valueType, comment);
133     }
134 
135     @Override
136     public String comment() {
137         return this.key.comment();
138     }
139 
140     @Override
141     public HDU hdu() {
142         return this.key.hdu();
143     }
144 
145     @Override
146     public String key() {
147         return this.key.key();
148     }
149 
150     @Override
151     public IFitsHeader n(int... number) {
152         return this.key.n(number);
153     }
154 
155     @Override
156     public SOURCE status() {
157         return this.key.status();
158     }
159 
160     @Override
161     @SuppressWarnings("CPD-END")
162     public VALUE valueType() {
163         return this.key.valueType();
164     }
