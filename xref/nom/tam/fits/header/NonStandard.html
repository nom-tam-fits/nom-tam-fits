34  /**
35   * This data dictionary contains FITS keywords that have been widely used within
36   * the astronomical community. It is recommended that these keywords only be
37   * used as defined here. These are the Keywords that denote non-standard FITS
38   * keyword format conventions.
39   * 
40   * <pre>
41   * @see <a href="http://heasarc.gsfc.nasa.gov/docs/fcg/common_dict.html">http://heasarc.gsfc.nasa.gov/docs/fcg/common_dict.html</a>
42   * </pre>
43   * 
44   * @author Richard van Nieuwenhoven
45   */
46  public enum NonStandard implements IFitsHeader {
47      /**
48       * @deprecated The CONTINUE key is now part of the FITS 4.0 standard, so use
49       * {@link Standard#CONTINUE} instead.
50       */
51      @Deprecated
52      CONTINUE(SOURCE.HEASARC, HDU.ANY, VALUE.NONE, "denotes the CONTINUE long string keyword convention"),
53      /**
54       * The HIERARCH keyword, when followed by spaces in columns 9 and 10 of the
55       * FITS card image, indicates that the ESO HIERARCH keyword convention
56       * should be used to interpret the name and value of the keyword. The
57       * HIERARCH keyword formally has no value because it is not followed by an
58       * equals sign in column 9. Under the HIERARCH convention the actual name of
59       * the keyword begins in column 11 of the card image and is terminated by
60       * the equal sign ('=') character. The name can contain any number of
61       * characters as long as it fits within columns 11 and 80 of the card image
62       * and also leaves enough space for the equal sign separator and the value
63       * field. The name can contain any printable ASCII text character, including
64       * spaces and lower-case characters, except for the equal sign character
65       * which serves as the terminator of the name field. Leading and trailing
66       * spaces in the name field are not significant, but embedded spaces within
67       * the name are significant. The value field follows the equals sign and
68       * must conform to the syntax for a free-format value field as defined in
69       * the FITS Standard. The value field may be null, in which case it contains
70       * only space characters, otherwise it may contain either a character string
71       * enclosed in single quotes, the logical constant T or F, an integer
72       * number, a floating point number, a complex integer number, or a complex
73       * floating point number. The value field may be followed by an optional
74       * comment string. The comment field must be separated from the value field
75       * by a slash character ('/'). It is recommended that the slash character be
76       * preceeded and followed by a space character. Example: "HIERARCH Filter
77       * Wheel = 12 / filter position". In this example the logical name of the
78       * keyword is 'Filter Wheel' and the value is 12.
79       */
80      HIERARCH(SOURCE.ESO, HDU.ANY, VALUE.NONE, "denotes the HIERARCH keyword convention"),
81      /**
82       * The presence of this keyword with a value = T in an extension key
83       * indicates that the keywords contained in the primary key (except the FITS
84       * Mandatory keywords, and any COMMENT, HISTORY or 'blank' keywords) are to
85       * be inherited, or logically included in that extension key.
86       */
87      INHERIT(SOURCE.STScI, HDU.EXTENSION, VALUE.LOGICAL, "denotes the INHERIT keyword convention");
88  
89      @SuppressWarnings("CPD-START")
90      private final IFitsHeader key;
91  
92      NonStandard(IFitsHeader.SOURCE status, HDU hdu, VALUE valueType, String comment) {
93          this.key = new FitsHeaderImpl(name(), status, hdu, valueType, comment);
94      }
95  
96      @Override
97      public String comment() {
98          return this.key.comment();
99      }
100 
101     @Override
102     public HDU hdu() {
103         return this.key.hdu();
104     }
105 
106     @Override
107     public String key() {
108         return this.key.key();
109     }
110 
111     @Override
112     public IFitsHeader n(int... number) {
113         return this.key.n(number);
114     }
115 
116     @Override
117     public SOURCE status() {
118         return this.key.status();
119     }
120 
121     @Override
122     @SuppressWarnings("CPD-END")
123     public VALUE valueType() {
124         return this.key.valueType();
125     }
126 }
