34  /**
35   * This data dictionary contains FITS keywords that have been widely used within
36   * the astronomical community. It is recommended that these keywords only be
37   * used as defined here. These are the Keywords that denote non-standard FITS
38   * keyword format conventions.
39   * 
40   * <pre>
41   * @see <a href="http://heasarc.gsfc.nasa.gov/docs/fcg/common_dict.html">http://heasarc.gsfc.nasa.gov/docs/fcg/common_dict.html</a>
42   * </pre>
43   * 
44   * @author Richard van Nieuwenhoven
45   */
46  public enum NonStandard implements IFitsHeader {
47      /**
48       * The CONTINUE keyword, when followed by spaces in columns 9 and 10 of the
49       * card image and a character string enclosed in single quotes starting in
50       * column 11 or higher, indicates that the quoted string should be treated
51       * as a continuation of the character string value in the previous key
52       * keyword. To conform to this convention, the character string value on the
53       * previous keyword must end with the ampersand character ('&amp;'), but the
54       * ampersand is not part of the value string and should be deleted before
55       * concatenating the strings together. The character string value may be
56       * continued on any number of consecutive CONTINUE keywords, thus
57       * effectively allowing arbitrarily long strings to be written as keyword
58       * values.
59       */
60      CONTINUE(SOURCE.HEASARC, HDU.ANY, VALUE.NONE, "denotes the CONTINUE long string keyword convention"),
61      /**
62       * The HIERARCH keyword, when followed by spaces in columns 9 and 10 of the
63       * FITS card image, indicates that the ESO HIERARCH keyword convention
64       * should be used to interpret the name and value of the keyword. The
65       * HIERARCH keyword formally has no value because it is not followed by an
66       * equals sign in column 9. Under the HIERARCH convention the actual name of
67       * the keyword begins in column 11 of the card image and is terminated by
68       * the equal sign ('=') character. The name can contain any number of
69       * characters as long as it fits within columns 11 and 80 of the card image
70       * and also leaves enough space for the equal sign separator and the value
71       * field. The name can contain any printable ASCII text character, including
72       * spaces and lower-case characters, except for the equal sign character
73       * which serves as the terminator of the name field. Leading and trailing
74       * spaces in the name field are not significant, but embedded spaces within
75       * the name are significant. The value field follows the equals sign and
76       * must conform to the syntax for a free-format value field as defined in
77       * the FITS Standard. The value field may be null, in which case it contains
78       * only space characters, otherwise it may contain either a character string
79       * enclosed in single quotes, the logical constant T or F, an integer
80       * number, a floating point number, a complex integer number, or a complex
81       * floating point number. The value field may be followed by an optional
82       * comment string. The comment field must be separated from the value field
83       * by a slash character ('/'). It is recommended that the slash character be
84       * preceeded and followed by a space character. Example: "HIERARCH Filter
85       * Wheel = 12 / filter position". In this example the logical name of the
86       * keyword is 'Filter Wheel' and the value is 12.
87       */
88      HIERARCH(SOURCE.ESO, HDU.ANY, VALUE.NONE, "denotes the HIERARCH keyword convention"),
89      /**
90       * The presence of this keyword with a value = T in an extension key
91       * indicates that the keywords contained in the primary key (except the FITS
92       * Mandatory keywords, and any COMMENT, HISTORY or 'blank' keywords) are to
93       * be inherited, or logically included in that extension key.
94       */
95      INHERIT(SOURCE.STScI, HDU.EXTENSION, VALUE.LOGICAL, "denotes the INHERIT keyword convention");
96  
97      private IFitsHeader key;
98  
99      private NonStandard(IFitsHeader.SOURCE status, HDU hdu, VALUE valueType, String comment) {
100         this.key = new FitsHeaderImpl(name(), status, hdu, valueType, comment);
101     }
102 
103     @Override
104     public String comment() {
105         return this.key.comment();
106     }
107 
108     @Override
109     public HDU hdu() {
110         return this.key.hdu();
111     }
112 
113     @Override
114     public String key() {
115         return this.key.key();
116     }
117 
118     @Override
119     public IFitsHeader n(int... number) {
120         return this.key.n(number);
121     }
122 
123     @Override
124     public SOURCE status() {
125         return this.key.status();
126     }
127 
128     @Override
129     public VALUE valueType() {
130         return this.key.valueType();
131     }
132 }
