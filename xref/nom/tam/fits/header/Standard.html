647     Standard(SOURCE status, HDU hdu, VALUE valueType, String comment, StandardCommentReplacement... replacements) {

652     Standard(String headerName, SOURCE status, HDU hdu, VALUE valueType, String comment, StandardCommentReplacement... replacements) {
653         this.key = new FitsHeaderImpl(headerName == null ? name() : headerName, status, hdu, valueType, comment);
654         this.commentReplacements = replacements;
655     }

657     @Override
658     public String comment() {
659         Class<?> contextClass = COMMENT_CONTEXT.get();
660         if (contextClass == null) {
661             contextClass = Object.class;
662         }
663         for (StandardCommentReplacement stdCommentReplacement : commentReplacements) {
664             if (stdCommentReplacement.getContext().isAssignableFrom(contextClass)) {
665                 if (stdCommentReplacement.getComment() != null) {
666                     return stdCommentReplacement.getComment();
667                 }
668             }
669         }
670         return this.key.comment();
671     }

675         return this.key.hdu();
676     }

699     public static void context(Class<?> clazz) {
700         COMMENT_CONTEXT.set(clazz);
701     }

703     /**
704      * scan for a comment with the specified reference key.
705      * 
706      * @param commentKey
707      *            the reference key
708      * @return the comment for the reference key
709      */
710     public String getCommentByKey(String commentKey) {
711         for (StandardCommentReplacement commentReplacement : commentReplacements) {
712             if (commentReplacement.getRef().equals(commentKey)) {
713                 String foundcommentReplacement = commentReplacement.getComment();
714                 if (foundcommentReplacement == null) {
715                     return comment();
716                 }
717                 return foundcommentReplacement;
718             }
719         }
720         return null;
721     }

723     /**
724      * set the comment for the specified reference key.
725      * 
726      * @param commentKey
727      *            the reference key
728      * @param value
729      *            the comment to set when the fits key is used.
730      */
731     public void setCommentByKey(String commentKey, String value) {
732         for (StandardCommentReplacement commentReplacement : commentReplacements) {
733             if (commentReplacement.getRef().equals(commentKey)) {
734                 commentReplacement.setComment(value);
735                 return;
736             }
737         }
738     }
