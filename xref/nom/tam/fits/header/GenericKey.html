34  import java.util.Collections;
35  import java.util.HashMap;
36  import java.util.Map;
37  
38  import nom.tam.fits.header.IFitsHeader.HDU;
39  import nom.tam.fits.header.IFitsHeader.SOURCE;
40  import nom.tam.fits.header.IFitsHeader.VALUE;
41  
42  /**
43   * generic key interface, create an IFitsHeader from a key.
44   *
45   * @author ritchie
46   */
47  public final class GenericKey {
48  
49      private static final int NUMBER_BASE = 10;
50  
51      /**
52       * cache of all standard keys, for reusing the standards.
53       */
54      private static final Map<String, IFitsHeader> STANDARD_KEYS;
55  
56      static {
57          Map<String, IFitsHeader> headers = new HashMap<>();
58          for (IFitsHeader key : Standard.values()) {
59              headers.put(key.key(), key);
60          }
61          for (IFitsHeader key : Checksum.values()) {
62              headers.put(key.key(), key);
63          }
64          for (IFitsHeader key : Compression.values()) {
65              headers.put(key.key(), key);
66          }
67          for (IFitsHeader key : DataDescription.values()) {
68              headers.put(key.key(), key);
69          }
70          for (IFitsHeader key : InstrumentDescription.values()) {
71              headers.put(key.key(), key);
72          }
73          for (IFitsHeader key : NonStandard.values()) {
74              headers.put(key.key(), key);
75          }
76          for (IFitsHeader key : ObservationDescription.values()) {
77              headers.put(key.key(), key);
78          }
79          STANDARD_KEYS = Collections.unmodifiableMap(headers);
80      }
81  
82      /**
83       * create a fits header key from a free string
84       *
85       * @param key
86       *            the string to create the key for
87       * @return the IFitsHeader implementation for the key.
88       */
89      public static IFitsHeader create(String key) {
90          IFitsHeader result = STANDARD_KEYS.get(key);
91          if (result == null) {
92              result = new FitsHeaderImpl(key, SOURCE.UNKNOWN, HDU.ANY, VALUE.ANY, "");
93          }
94          return result;
95      }
96  
97      /**
98       * create a array of generic fits header keys from a array of string keys.
99       *
100      * @param keys
101      *            the array of string keys
102      * @return the array of IFitsHeaderKeys.
103      */
104     public static IFitsHeader[] create(String[] keys) {
105         IFitsHeader[] result = new IFitsHeader[keys.length];
106         for (int index = 0; index < result.length; index++) {
107             result[index] = create(keys[index]);
108         }
109         return result;
110     }
111 
112     public static int getN(String card) {
113         int index = card.length() - 1;
114         int n = 0;
115         while (index >= 0 && Character.isDigit(card.charAt(index))) {
116             n = n * NUMBER_BASE + card.charAt(index) - '0';
117             index--;
118         }
119         return n;
120     }
121 
122     /**
123      * lookup a string key in the standard key sets.
124      *
125      * @param key
126      *            the fits key to search.
127      * @return the found fits key or null
128      */
129     public static IFitsHeader lookup(String key) {
130         int keyLength = key.length();
131         if (keyLength > 0 && Character.isDigit(key.charAt(keyLength - 1))) {
132             StringBuilder builder = new StringBuilder();
133             for (int index = 0; index < keyLength; index++) {
134                 char character = key.charAt(index);
135                 if (Character.isDigit(character)) {
136                     if (builder.charAt(builder.length() - 1) != 'n') {
137                         builder.append('n');
138                     }
139                 } else {
140                     builder.append(character);
141                 }
142             }
143             return STANDARD_KEYS.get(builder.toString());
144         }
145         return STANDARD_KEYS.get(key);
146     }
147 
148     /**
149      * utility class do not instanciate it.
150      */
151     private GenericKey() {
152     }
153 }
