32  import nom.tam.fits.header.IFitsHeader.HDU;
33  import nom.tam.fits.header.IFitsHeader.STATUS;
34  import nom.tam.fits.header.IFitsHeader.VALUE;
35  
36  /**
37   * File checksum keywords. This data dictionary contains FITS keywords that have
38   * been widely used within the astronomical community. It is recommended that
39   * these keywords only be used as defined here. These are the keywords that
40   * describe the data or the FITS file itself {@link http
41   * ://heasarc.gsfc.nasa.gov/docs/fcg/common_dict.html}
42   * 
43   * @author Richard van Nieuwenhoven
44   */
45  public enum DataDescription implements IFitsHeader {
46      /**
47       * The value field shall contain a character string giving a title that is
48       * suitable for display purposes, e.g., for annotation on images or plots of
49       * the data contained in the HDU.
50       */
51      TITLE(STATUS.ROSAT, HDU.ANY, VALUE.STRING, "title for the observation or data"),
52      /**
53       * The value field shall contain a character string giving the the host file
54       * name used to record the original data.
55       */
56      FILENAME(STATUS.NOAO, HDU.ANY, VALUE.STRING, "name of the file "),
57  
58      /**
59       * The value field shall contain a character string giving the file type
60       * suffix of the host file name. The full file name typically consists of
61       * the root name (see ROOTNAME) followed by a file type suffix, separated by
62       * the period ('.') character.
63       */
64      FILETYPE(STATUS.UNKNOWN, HDU.ANY, VALUE.STRING, "type of file"),
65  
66      /**
67       * The value field shall contain a character string giving the root of the
68       * host file name. The full file name typically consists of the root name
69       * followed by a file type suffix (see FILETYPE), separated by the period
70       * ('.') character.
71       */
72      ROOTNAME(STATUS.UNKNOWN, HDU.ANY, VALUE.STRING, "rootname of the file"),
73  
74      /**
75       * The value field shall contain a character string that uniquely defines
76       * the configuration state, or version, of the the software processing
77       * system that generated the data contained in the HDU. This keyword differs
78       * from the CREATOR keyword in that it give the name and version of the
79       * overall processing system and not just the name and version of a single
80       * program.
81       */
82      CONFIGUR(STATUS.INTEGRAL, HDU.ANY, VALUE.STRING, "software configuration used to process the data"),
83  
84      /**
85       * The value field shall contain an integer giving the number of standard
86       * extensions contained in the FITS file. This keyword may only be used in
87       * the primary array key.
88       */
89      NEXTEND(STATUS.STScI, HDU.PRIMARY, VALUE.INTEGER, "Number of standard extensions"),
90  
91      /**
92       * This keyword is synonymous to the standard EXTNAME keyword except that it
93       * may also be used in the primary key. It is recommended that the HDUNAME
94       * and EXTNAME keywords should not both be given in the same HDU key, but if
95       * they are, then the HDUNAME keyword will have precedence.
96       */
97      HDUNAME(STATUS.UNKNOWN, HDU.ANY, VALUE.STRING, "descriptive name of the HDU"),
98  
99      /**
100      * This keyword is synonymous to the standard EXTVER keyword except that it
101      * may also be used in the primary key. It is recommended that the HDUVER
102      * and EXTVER keywords should not both be given in the same HDU key, but if
103      * they are, then the HDUVER keyword will have precedence.
104      */
105     HDUVER(STATUS.UNKNOWN, HDU.ANY, VALUE.INTEGER, "version number of the HDU"),
106     /**
107      * This keyword is synonymous to the standard EXTLEVEL keyword except that
108      * it may also be used in the primary key. It is recommended that the
109      * HDULEVEL and EXTLEVEL keywords should not both be given in the same HDU
110      * key, but if they are, then the HDULEVEL keyword will have precedence.
111      */
112     HDULEVEL(STATUS.UNKNOWN, HDU.ANY, VALUE.INTEGER, "hierarchical level of the HDU"),
113 
114     /**
115      * The value field of this indexed keyword shall contain a floating point
116      * number specifying the lower bound of the legal range of physical values
117      * that may be represented in column n of the table. The column may contain
118      * values that are less than this legal minimum value but the interpretation
119      * of such values is not defined here. The value of this keyword is
120      * typically used as the mininum value when constructing a histogram of the
121      * values in the column. This keyword may only be used in 'TABLE' or
122      * 'BINTABLE' extensions.
123      */
124     TLMINn(STATUS.HEASARC, HDU.TABLE, VALUE.REAL, "minimum legal value in the column"),
125     /**
126      * The value field of this indexed keyword shall contain a floating point
127      * number specifying the upper bound of the legal range of physical values
128      * that may be represented in column n of the table. The column may contain
129      * values that are greater than this legal maximum value but the
130      * interpretation of such values is not defined here. The value of this
131      * keyword is typically used as the maxinum value when constructing a
132      * histogram of the values in the column. This keyword may only be used in
133      * 'TABLE' or 'BINTABLE' extensions.
134      */
135     TLMAXn(STATUS.HEASARC, HDU.TABLE, VALUE.REAL, "maximum legal value in the column"),
136     /**
137      * The value field of this indexed keyword shall contain a floating point
138      * number specifying the minimum valid physical value represented in column
139      * n of the table, exclusive of any special values. This keyword may only be
140      * used in 'TABLE' or 'BINTABLE' extensions and is analogous to the DATAMIN
141      * keyword used for FITS images.
142      */
143     TDMINn(STATUS.HEASARC, HDU.TABLE, VALUE.REAL, "minimum physical value in the column"),
144     /**
145      * The value field of this indexed keyword shall contain a floating point
146      * number specifying the maximum valid physical value represented in column
147      * n of the table, exclusive of any special values. This keyword may only be
148      * used in 'TABLE' or 'BINTABLE' extensions and is analogous to the DATAMAX
149      * keyword used for FITS images.
150      */
151     TDMAXn(STATUS.HEASARC, HDU.TABLE, VALUE.REAL, "maximum physical value in the column"),
152     /**
153      * The value field of this indexed keyword shall contain a floating point
154      * number specifying the suggested bin size when producing a histogram of
155      * the values in column n. This keyword is typically used in conjunction the
156      * TLMINn and TLMAXn keywords when constructing a histogram of the values in
157      * column n, such that the histogram ranges from TLMINn to TLMAXn with the
158      * histogram bin size given by TDBINn. This keyword may only be used in
159      * 'TABLE' or 'BINTABLE' extensions.
160      */
161     TDBINn(STATUS.CXC, HDU.TABLE, VALUE.REAL, "default histogram bin size for the column"),
162     /**
163      * The value field shall contain a character string that defines the order
164      * in which the rows in the current FITS ASCII or binary table extension
165      * have been sorted. The character string lists the name (as given by the
166      * TTYPEn keyword) of the primary sort column, optionally followed by the
167      * names of any secondary sort column(s). The presence of this keyword
168      * indicates that the rows in the table have been sorted first by the values
169      * in the primary sort column; any rows that have the same value in the
170      * primary column have been further sorted by the values in the secondary
171      * sort column and so on for all the specified columns. If more than one
172      * column is specified by TSORTKEY then the names must be separated by a
173      * comma. One or more spaces are also allowed between the comma and the
174      * following column name. By default, columns are sorted in ascending order,
175      * but a minus sign may precede the column name to indicate that the rows
176      * are sorted in descending order. This keyword may only be used in 'TABLE'
177      * or 'BINTABLE' extensions. Example: TSORTKEY = 'TIME, RA, DEC'.
178      */
179     TSORTKEY(STATUS.HEASARC, HDU.TABLE, VALUE.STRING, "defines the sort order of a table"),
180     /**
181      * The value field shall contain a character string giving the name, and
182      * optionally, the version of the program that originally created the
183      * current FITS HDU. This keyword is synonymous with the CREATOR keyword.
184      * Example: 'TASKNAME V1.2.3'
185      */
186     PROGRAM(STATUS.UCOLICK, HDU.ANY, VALUE.STRING, "the name of the software task that created the file"),
187     /**
188      * The value field shall contain a character string giving the name, and
189      * optionally, the version of the program that originally created the
190      * current FITS HDU. This keyword is synonymous with the PROGRAM keyword.
191      * Example: 'TASKNAME V1.2.3'
192      */
193     CREATOR(STATUS.HEASARC, HDU.ANY, VALUE.STRING, "the name of the software task that created the file"),
194     /**
195      * The value field shall contain a character string that identifies the
196      * domain to which the associated HDUCLASn keywords apply. This keyword
197      * typically identifies the institution or project that has defined the
198      * allowed set of values for the associated hierarchical HDUCLASn keywords.
199      */
200     HDUCLASS(STATUS.HEASARC, HDU.ANY, VALUE.STRING, "general identifier for the classification of the data"),
201     /**
202      * The value field shall contain a character string that gives a reference
203      * to a document that describes the allowed values that may be assigned to
204      * the HDUCLASn data classification keywords.
205      */
206     HDUDOC(STATUS.HEASARC, HDU.ANY, VALUE.STRING, "reference to document describing the data format"),
207     /**
208      * The value field shall contain a character string that gives the specific
209      * version of the document referenced by HDUDOC.
210      */
211     HDUVERS(STATUS.HEASARC, HDU.ANY, VALUE.STRING, "specific version of the document referenced by HDUDOC"),
212     /**
213      * The value fields of this hierarchical set of indexed keywords shall
214      * contain character strings that classify the type of data contained in the
215      * HDU. The HDUCLAS1 keyword gives the highest, most general data
216      * classification, and the HDUCLAS2 and higher keywords provide
217      * progressively more detailed subclassifications of the data.
218      */
219     HDUCLASn(STATUS.HEASARC, HDU.ANY, VALUE.STRING, "hierarchical classification of the data");
220 
221     private IFitsHeader key;
222 
223     private DataDescription(IFitsHeader.STATUS status, HDU hdu, VALUE valueType, String comment) {
224         this.key = new FitsHeaderImpl(name(), status, hdu, valueType, comment);
225     }
226 
227     @Override
228     public String key() {
229         return key.key();
230     }
231 
232     @Override
233     public STATUS status() {
234         return key.status();
235     }
236 
237     @Override
238     public HDU hdu() {
239         return key.hdu();
240     }
241 
242     @Override
243     public VALUE valueType() {
244         return key.valueType();
245     }
246 
247     @Override
248     public String comment() {
249         return key.comment();
250     }
251 
252     @Override
253     public IFitsHeader n(int number) {
254         return key.n(number);
255     }
256 }
