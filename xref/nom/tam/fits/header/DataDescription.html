34  /**
35   * This data dictionary contains FITS keywords that have been widely used within
36   * the astronomical community. It is recommended that these keywords only be
37   * used as defined here. These are the keywords that describe the data or the
38   * FITS file itself
39   * 
40   * <pre>
41   * @see <a href="http://heasarc.gsfc.nasa.gov/docs/fcg/common_dict.html">http://heasarc.gsfc.nasa.gov/docs/fcg/common_dict.html</a>
42   * </pre>
43   * 
44   * @author Richard van Nieuwenhoven
45   */
46  public enum DataDescription implements IFitsHeader {
47      /**
48       * The value field shall contain a character string that uniquely defines
49       * the configuration state, or version, of the the software processing
50       * system that generated the data contained in the HDU. This keyword differs
51       * from the CREATOR keyword in that it give the name and version of the
52       * overall processing system and not just the name and version of a single
53       * program.
54       */
55      CONFIGUR(SOURCE.INTEGRAL, HDU.ANY, VALUE.STRING, "software configuration used to process the data"),
56      /**
57       * The value field shall contain a character string giving the name, and
58       * optionally, the version of the program that originally created the
59       * current FITS HDU. This keyword is synonymous with the PROGRAM keyword.
60       * Example: 'TASKNAME V1.2.3'
61       */
62      CREATOR(SOURCE.HEASARC, HDU.ANY, VALUE.STRING, "the name of the software task that created the file"),
63  
64      /**
65       * The value field shall contain a character string giving the the host file
66       * name used to record the original data.
67       */
68      FILENAME(SOURCE.NOAO, HDU.ANY, VALUE.STRING, "name of the file "),
69  
70      /**
71       * The value field shall contain a character string giving the file type
72       * suffix of the host file name. The full file name typically consists of
73       * the root name (see ROOTNAME) followed by a file type suffix, separated by
74       * the period ('.') character.
75       */
76      FILETYPE(SOURCE.UNKNOWN, HDU.ANY, VALUE.STRING, "type of file"),
77  
78      /**
79       * The value fields of this hierarchical set of indexed keywords shall
80       * contain character strings that classify the type of data contained in the
81       * HDU. The HDUCLAS1 keyword gives the highest, most general data
82       * classification, and the HDUCLAS2 and higher keywords provide
83       * progressively more detailed subclassifications of the data.
84       */
85      HDUCLASn(SOURCE.HEASARC, HDU.ANY, VALUE.STRING, "hierarchical classification of the data"),
86  
87      /**
88       * The value field shall contain a character string that identifies the
89       * domain to which the associated HDUCLASn keywords apply. This keyword
90       * typically identifies the institution or project that has defined the
91       * allowed set of values for the associated hierarchical HDUCLASn keywords.
92       */
93      HDUCLASS(SOURCE.HEASARC, HDU.ANY, VALUE.STRING, "general identifier for the classification of the data"),
94  
95      /**
96       * The value field shall contain a character string that gives a reference
97       * to a document that describes the allowed values that may be assigned to
98       * the HDUCLASn data classification keywords.
99       */
100     HDUDOC(SOURCE.HEASARC, HDU.ANY, VALUE.STRING, "reference to document describing the data format"),
101 
102     /**
103      * This keyword is synonymous to the standard EXTLEVEL keyword except that
104      * it may also be used in the primary key. It is recommended that the
105      * HDULEVEL and EXTLEVEL keywords should not both be given in the same HDU
106      * key, but if they are, then the HDULEVEL keyword will have precedence.
107      */
108     HDULEVEL(SOURCE.UNKNOWN, HDU.ANY, VALUE.INTEGER, "hierarchical level of the HDU"),
109     /**
110      * This keyword is synonymous to the standard EXTNAME keyword except that it
111      * may also be used in the primary key. It is recommended that the HDUNAME
112      * and EXTNAME keywords should not both be given in the same HDU key, but if
113      * they are, then the HDUNAME keyword will have precedence.
114      */
115     HDUNAME(SOURCE.UNKNOWN, HDU.ANY, VALUE.STRING, "descriptive name of the HDU"),
116 
117     /**
118      * This keyword is synonymous to the standard EXTVER keyword except that it
119      * may also be used in the primary key. It is recommended that the HDUVER
120      * and EXTVER keywords should not both be given in the same HDU key, but if
121      * they are, then the HDUVER keyword will have precedence.
122      */
123     HDUVER(SOURCE.UNKNOWN, HDU.ANY, VALUE.INTEGER, "version number of the HDU"),
124     /**
125      * The value field shall contain a character string that gives the specific
126      * version of the document referenced by HDUDOC.
127      */
128     HDUVERS(SOURCE.HEASARC, HDU.ANY, VALUE.STRING, "specific version of the document referenced by HDUDOC"),
129     /**
130      * The value field shall contain an integer giving the number of standard
131      * extensions contained in the FITS file. This keyword may only be used in
132      * the primary array key.
133      */
134     NEXTEND(SOURCE.STScI, HDU.PRIMARY, VALUE.INTEGER, "Number of standard extensions"),
135     /**
136      * The value field shall contain a character string giving the name, and
137      * optionally, the version of the program that originally created the
138      * current FITS HDU. This keyword is synonymous with the CREATOR keyword.
139      * Example: 'TASKNAME V1.2.3'
140      */
141     PROGRAM(SOURCE.UCOLICK, HDU.ANY, VALUE.STRING, "the name of the software task that created the file"),
142     /**
143      * The value field shall contain a character string giving the root of the
144      * host file name. The full file name typically consists of the root name
145      * followed by a file type suffix (see FILETYPE), separated by the period
146      * ('.') character.
147      */
148     ROOTNAME(SOURCE.UNKNOWN, HDU.ANY, VALUE.STRING, "rootname of the file"),
149     /**
150      * The value field of this indexed keyword shall contain a floating point
151      * number specifying the suggested bin size when producing a histogram of
152      * the values in column n. This keyword is typically used in conjunction the
153      * TLMINn and TLMAXn keywords when constructing a histogram of the values in
154      * column n, such that the histogram ranges from TLMINn to TLMAXn with the
155      * histogram bin size given by TDBINn. This keyword may only be used in
156      * 'TABLE' or 'BINTABLE' extensions.
157      */
158     TDBINn(SOURCE.CXC, HDU.TABLE, VALUE.REAL, "default histogram bin size for the column"),
159     /**
160      * The value field of this indexed keyword shall contain a floating point
161      * number specifying the maximum valid physical value represented in column
162      * n of the table, exclusive of any special values. This keyword may only be
163      * used in 'TABLE' or 'BINTABLE' extensions and is analogous to the DATAMAX
164      * keyword used for FITS images.
165      */
166     TDMAXn(SOURCE.HEASARC, HDU.TABLE, VALUE.REAL, "maximum value in the column"),
167     /**
168      * The value field of this indexed keyword shall contain a floating point
169      * number specifying the minimum valid physical value represented in column
170      * n of the table, exclusive of any special values. This keyword may only be
171      * used in 'TABLE' or 'BINTABLE' extensions and is analogous to the DATAMIN
172      * keyword used for FITS images.
173      */
174     TDMINn(SOURCE.HEASARC, HDU.TABLE, VALUE.REAL, "minimum value in the column"),
175     /**
176      * The value field shall contain a character string giving a title that is
177      * suitable for display purposes, e.g., for annotation on images or plots of
178      * the data contained in the HDU.
179      */
180     TITLE(SOURCE.ROSAT, HDU.ANY, VALUE.STRING, "title for the observation or data"),
181     /**
182      * The value field of this indexed keyword shall contain a floating point
183      * number specifying the upper bound of the legal range of physical values
184      * that may be represented in column n of the table. The column may contain
185      * values that are greater than this legal maximum value but the
186      * interpretation of such values is not defined here. The value of this
187      * keyword is typically used as the maxinum value when constructing a
188      * histogram of the values in the column. This keyword may only be used in
189      * 'TABLE' or 'BINTABLE' extensions.
190      */
191     TLMAXn(SOURCE.HEASARC, HDU.TABLE, VALUE.REAL, "maximum legal value in the column"),
192     /**
193      * The value field of this indexed keyword shall contain a floating point
194      * number specifying the lower bound of the legal range of physical values
195      * that may be represented in column n of the table. The column may contain
196      * values that are less than this legal minimum value but the interpretation
197      * of such values is not defined here. The value of this keyword is
198      * typically used as the mininum value when constructing a histogram of the
199      * values in the column. This keyword may only be used in 'TABLE' or
200      * 'BINTABLE' extensions.
201      */
202     TLMINn(SOURCE.HEASARC, HDU.TABLE, VALUE.REAL, "minimum legal value in the column"),
203     /**
204      * The value field shall contain a character string that defines the order
205      * in which the rows in the current FITS ASCII or binary table extension
206      * have been sorted. The character string lists the name (as given by the
207      * TTYPEn keyword) of the primary sort column, optionally followed by the
208      * names of any secondary sort column(s). The presence of this keyword
209      * indicates that the rows in the table have been sorted first by the values
210      * in the primary sort column; any rows that have the same value in the
211      * primary column have been further sorted by the values in the secondary
212      * sort column and so on for all the specified columns. If more than one
213      * column is specified by TSORTKEY then the names must be separated by a
214      * comma. One or more spaces are also allowed between the comma and the
215      * following column name. By default, columns are sorted in ascending order,
216      * but a minus sign may precede the column name to indicate that the rows
217      * are sorted in descending order. This keyword may only be used in 'TABLE'
218      * or 'BINTABLE' extensions. Example: TSORTKEY = 'TIME, RA, DEC'.
219      */
220     TSORTKEY(SOURCE.HEASARC, HDU.TABLE, VALUE.STRING, "defines the sort order of a table");
221 
222     @SuppressWarnings("CPD-START")
223     private final IFitsHeader key;
224 
225     DataDescription(IFitsHeader.SOURCE status, HDU hdu, VALUE valueType, String comment) {
226         this.key = new FitsHeaderImpl(name(), status, hdu, valueType, comment);
227     }
228 
229     @Override
230     public String comment() {
231         return this.key.comment();
232     }
233 
234     @Override
235     public HDU hdu() {
236         return this.key.hdu();
237     }
238 
239     @Override
240     public String key() {
241         return this.key.key();
242     }
243 
244     @Override
245     public IFitsHeader n(int... number) {
246         return this.key.n(number);
247     }
248 
249     @Override
250     public SOURCE status() {
251         return this.key.status();
252     }
253 
254     @Override
255     @SuppressWarnings("CPD-END")
256     public VALUE valueType() {
257         return this.key.valueType();
258     }
259 }
