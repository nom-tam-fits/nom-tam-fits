34  /**
35   * <p>
36   * This data dictionary contains FITS keywords that have been widely used within the astronomical community. It is
37   * recommended that these keywords only be used as defined here. These are the keywords that describe the data or the
38   * FITS file itself
39   * </p>
40   * <p>
41   * See <a href=
42   * "http://heasarc.gsfc.nasa.gov/docs/fcg/common_dict.html">http://heasarc.gsfc.nasa.gov/docs/fcg/common_dict.html</a>
43   * </p>
44   * 
45   * @author Richard van Nieuwenhoven
46   */
47  public enum DataDescription implements IFitsHeader {
48      /**
49       * The value field shall contain a character string that uniquely defines the configuration state, or version, of
50       * the the software processing system that generated the data contained in the HDU. This keyword differs from the
51       * CREATOR keyword in that it give the name and version of the overall processing system and not just the name and
52       * version of a single program.
53       */
54      CONFIGUR(SOURCE.INTEGRAL, HDU.ANY, VALUE.STRING, "software configuration used to process the data"),
55      /**
56       * The value field shall contain a character string giving the name, and optionally, the version of the program that
57       * originally created the current FITS HDU. This keyword is synonymous with the PROGRAM keyword. Example: 'TASKNAME
58       * V1.2.3'
59       */
60      CREATOR(SOURCE.HEASARC, HDU.ANY, VALUE.STRING, "the name of the software task that created the file"),
61  
62      /**
63       * The value field shall contain a character string giving the the host file name used to record the original data.
64       */
65      FILENAME(SOURCE.NOAO, HDU.ANY, VALUE.STRING, "name of the file "),
66  
67      /**
68       * The value field shall contain a character string giving the file type suffix of the host file name. The full file
69       * name typically consists of the root name (see ROOTNAME) followed by a file type suffix, separated by the period
70       * ('.') character.
71       */
72      FILETYPE(SOURCE.UNKNOWN, HDU.ANY, VALUE.STRING, "type of file"),
73  
74      /**
75       * The value fields of this hierarchical set of indexed keywords shall contain character strings that classify the
76       * type of data contained in the HDU. The HDUCLAS1 keyword gives the highest, most general data classification, and
77       * the HDUCLAS2 and higher keywords provide progressively more detailed subclassifications of the data.
78       */
79      HDUCLASn(SOURCE.HEASARC, HDU.ANY, VALUE.STRING, "hierarchical classification of the data"),
80  
81      /**
82       * The value field shall contain a character string that identifies the domain to which the associated HDUCLASn
83       * keywords apply. This keyword typically identifies the institution or project that has defined the allowed set of
84       * values for the associated hierarchical HDUCLASn keywords.
85       */
86      HDUCLASS(SOURCE.HEASARC, HDU.ANY, VALUE.STRING, "general identifier for the classification of the data"),
87  
88      /**
89       * The value field shall contain a character string that gives a reference to a document that describes the allowed
90       * values that may be assigned to the HDUCLASn data classification keywords.
91       */
92      HDUDOC(SOURCE.HEASARC, HDU.ANY, VALUE.STRING, "reference to document describing the data format"),
93  
94      /**
95       * This keyword is synonymous to the standard EXTLEVEL keyword except that it may also be used in the primary key.
96       * It is recommended that the HDULEVEL and EXTLEVEL keywords should not both be given in the same HDU key, but if
97       * they are, then the HDULEVEL keyword will have precedence.
98       */
99      HDULEVEL(SOURCE.UNKNOWN, HDU.ANY, VALUE.INTEGER, "hierarchical level of the HDU"),
100     /**
101      * This keyword is synonymous to the standard EXTNAME keyword except that it may also be used in the primary key. It
102      * is recommended that the HDUNAME and EXTNAME keywords should not both be given in the same HDU key, but if they
103      * are, then the HDUNAME keyword will have precedence.
104      */
105     HDUNAME(SOURCE.UNKNOWN, HDU.ANY, VALUE.STRING, "descriptive name of the HDU"),
106 
107     /**
108      * This keyword is synonymous to the standard EXTVER keyword except that it may also be used in the primary key. It
109      * is recommended that the HDUVER and EXTVER keywords should not both be given in the same HDU key, but if they are,
110      * then the HDUVER keyword will have precedence.
111      */
112     HDUVER(SOURCE.UNKNOWN, HDU.ANY, VALUE.INTEGER, "version number of the HDU"),
113     /**
114      * The value field shall contain a character string that gives the specific version of the document referenced by
115      * HDUDOC.
116      */
117     HDUVERS(SOURCE.HEASARC, HDU.ANY, VALUE.STRING, "specific version of the document referenced by HDUDOC"),
118     /**
119      * The value field shall contain an integer giving the number of standard extensions contained in the FITS file.
120      * This keyword may only be used in the primary array key.
121      */
122     NEXTEND(SOURCE.STScI, HDU.PRIMARY, VALUE.INTEGER, "Number of standard extensions"),
123     /**
124      * The value field shall contain a character string giving the name, and optionally, the version of the program that
125      * originally created the current FITS HDU. This keyword is synonymous with the CREATOR keyword. Example: 'TASKNAME
126      * V1.2.3'
127      */
128     PROGRAM(SOURCE.UCOLICK, HDU.ANY, VALUE.STRING, "the name of the software task that created the file"),
129     /**
130      * The value field shall contain a character string giving the root of the host file name. The full file name
131      * typically consists of the root name followed by a file type suffix (see FILETYPE), separated by the period ('.')
132      * character.
133      */
134     ROOTNAME(SOURCE.UNKNOWN, HDU.ANY, VALUE.STRING, "rootname of the file"),
135     /**
136      * The value field of this indexed keyword shall contain a floating point number specifying the suggested bin size
137      * when producing a histogram of the values in column n. This keyword is typically used in conjunction the TLMINn
138      * and TLMAXn keywords when constructing a histogram of the values in column n, such that the histogram ranges from
139      * TLMINn to TLMAXn with the histogram bin size given by TDBINn. This keyword may only be used in 'TABLE' or
140      * 'BINTABLE' extensions.
141      */
142     TDBINn(SOURCE.CXC, HDU.TABLE, VALUE.REAL, "default histogram bin size for the column"),
143     /**
144      * The value field of this indexed keyword shall contain a floating point number specifying the maximum valid
145      * physical value represented in column n of the table, exclusive of any special values. This keyword may only be
146      * used in 'TABLE' or 'BINTABLE' extensions and is analogous to the DATAMAX keyword used for FITS images.
147      */
148     TDMAXn(SOURCE.HEASARC, HDU.TABLE, VALUE.REAL, "maximum value in the column"),
149     /**
150      * The value field of this indexed keyword shall contain a floating point number specifying the minimum valid
151      * physical value represented in column n of the table, exclusive of any special values. This keyword may only be
152      * used in 'TABLE' or 'BINTABLE' extensions and is analogous to the DATAMIN keyword used for FITS images.
153      */
154     TDMINn(SOURCE.HEASARC, HDU.TABLE, VALUE.REAL, "minimum value in the column"),
155     /**
156      * The value field shall contain a character string giving a title that is suitable for display purposes, e.g., for
157      * annotation on images or plots of the data contained in the HDU.
158      */
159     TITLE(SOURCE.ROSAT, HDU.ANY, VALUE.STRING, "title for the observation or data"),
160     /**
161      * The value field of this indexed keyword shall contain a floating point number specifying the upper bound of the
162      * legal range of physical values that may be represented in column n of the table. The column may contain values
163      * that are greater than this legal maximum value but the interpretation of such values is not defined here. The
164      * value of this keyword is typically used as the maxinum value when constructing a histogram of the values in the
165      * column. This keyword may only be used in 'TABLE' or 'BINTABLE' extensions.
166      */
167     TLMAXn(SOURCE.HEASARC, HDU.TABLE, VALUE.REAL, "maximum legal value in the column"),
168     /**
169      * The value field of this indexed keyword shall contain a floating point number specifying the lower bound of the
170      * legal range of physical values that may be represented in column n of the table. The column may contain values
171      * that are less than this legal minimum value but the interpretation of such values is not defined here. The value
172      * of this keyword is typically used as the mininum value when constructing a histogram of the values in the column.
173      * This keyword may only be used in 'TABLE' or 'BINTABLE' extensions.
174      */
175     TLMINn(SOURCE.HEASARC, HDU.TABLE, VALUE.REAL, "minimum legal value in the column"),
176     /**
177      * The value field shall contain a character string that defines the order in which the rows in the current FITS
178      * ASCII or binary table extension have been sorted. The character string lists the name (as given by the TTYPEn
179      * keyword) of the primary sort column, optionally followed by the names of any secondary sort column(s). The
180      * presence of this keyword indicates that the rows in the table have been sorted first by the values in the primary
181      * sort column; any rows that have the same value in the primary column have been further sorted by the values in
182      * the secondary sort column and so on for all the specified columns. If more than one column is specified by
183      * TSORTKEY then the names must be separated by a comma. One or more spaces are also allowed between the comma and
184      * the following column name. By default, columns are sorted in ascending order, but a minus sign may precede the
185      * column name to indicate that the rows are sorted in descending order. This keyword may only be used in 'TABLE' or
186      * 'BINTABLE' extensions. Example: TSORTKEY = 'TIME, RA, DEC'.
187      */
188     TSORTKEY(SOURCE.HEASARC, HDU.TABLE, VALUE.STRING, "defines the sort order of a table");
189 
190     @SuppressWarnings("CPD-START")
191     private final IFitsHeader key;
192 
193     DataDescription(IFitsHeader.SOURCE status, HDU hdu, VALUE valueType, String comment) {
194         this.key = new FitsHeaderImpl(name(), status, hdu, valueType, comment);
195     }
196 
197     @Override
198     public String comment() {
199         return this.key.comment();
200     }
201 
202     @Override
203     public HDU hdu() {
204         return this.key.hdu();
205     }
206 
207     @Override
208     public String key() {
209         return this.key.key();
210     }
211 
212     @Override
213     public IFitsHeader n(int... number) {
214         return this.key.n(number);
215     }
216 
217     @Override
218     public SOURCE status() {
219         return this.key.status();
220     }
221 
222     @Override
223     @SuppressWarnings("CPD-END")
224     public VALUE valueType() {
225         return this.key.valueType();
226     }
227 }
