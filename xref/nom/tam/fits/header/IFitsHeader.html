28  public interface IFitsHeader {
29  
30      public enum HDU {
31          ANY,
32          IMAGE,
33          PRIMARY,
34          ASCII_TABLE,
35          BINTABLE,
36          TABLE,
37          GROUPS,
38          EXTENSION
39      }
40  
41      public enum STATUS {
42          /**
43           * defined mandatory by the fits standard.
44           */
45          MANDATORY,
46  
47          /**
48           * defined reserved by the fits standard.
49           */
50          RESERVED,
51          /**
52           * <pre>
53           * {@link http://heasarc.gsfc.nasa.gov/docs/heasarc/ofwg/docs/general/checksum/checksum.html}
54           * </pre>
55           */
56          CHECKSUM,
57          INTEGRAL,
58          /**
59           * rosat no link availabe.
60           */
61          ROSAT,
62          /**
63           * <pre>
64           * {@link http://heasarc.gsfc.nasa.gov/docs/heasarc/ofwg/docs/ofwg_recomm/r13.html}
65           * </pre>
66           */
67          HEASARC,
68          /**
69           * <pre>
70           * {@link http://iraf.noao.edu/iraf/web/projects/ccdmosaic/imagedef/fitsdic.html}
71           * </pre>
72           */
73          NOAO,
74          /**
75           * <pre>
76           * {@link http://www.ucolick.org/cgi-bin/Tcl/demos.cgi}
77           * </pre>
78           */
79          UCOLICK,
80          /**
81           * <pre>
82           * {@link http://arcdev.hq.eso.org/dicb/dicd/dic-1-1.4.html}
83           * </pre>
84           */
85          ESO,
86          /**
87           * <pre>
88           * {@link http://tucana.noao.edu/ADASS/adass_proc/adass_95/zaraten/zaraten.html}
89           * </pre>
90           */
91          STScI,
92          /**
93           * <pre>
94           * {@link http://archive.sbig.com/pdffiles/SBFITSEXT_1r0.pdf}
95           * </pre>
96           */
97          SBIG,
98          /**
99           * <pre>
100          * {@link http://www.cyanogen.com/help/maximdl/FITS_File_Header_Definitions.htm}
101          * </pre>
102          */
103         MaxImDL,
104         /**
105          * http://cxc.harvard.edu/contrib/arots/fits/content.txt.
106          */
107         CXC,
108         /**
109          * developed over time.
110          */
111         UNKNOWN;
112     }

115         NONE,
116         STRING,
117         INTEGER,
118         LOGICAL,
119         REAL
120     }

123 
124     STATUS status();
125 
126     HDU hdu();
127 
128     VALUE valueType();
129 
130     String comment();
131 
132     IFitsHeader n(int number);
133 }
