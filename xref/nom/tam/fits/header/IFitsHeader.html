34  public interface IFitsHeader {
35  
36      enum HDU {
37          ANY, ASCII_TABLE, BINTABLE, EXTENSION, GROUPS, IMAGE, PRIMARY, PRIMARY_EXTENSION, TABLE
38      }
39  
40      enum SOURCE {
41          /**
42           * Checksum keywords. See
43           * <a href="http://heasarc.gsfc.nasa.gov/docs/heasarc/ofwg/docs/general/checksum/checksum.html">checksum doc</a>
44           */
45          CHECKSUM("http://heasarc.gsfc.nasa.gov/docs/heasarc/ofwg/docs/general/checksum/checksum.html"),
46  
47          /**
48           * CXC keywords. See <a href=
49           * "http://cxc.harvard.edu/contrib/arots/fits/content.txt">http://cxc.harvard.edu/contrib/arots/fits/content.txt</a>
50           */
51          CXC("http://cxc.harvard.edu/contrib/arots/fits/content.txt"),
52          /**
53           * ESO keywords. See <a href=
54           * "http://arcdev.hq.eso.org/dicb/dicd/dic-1-1.4.html">http://arcdev.hq.eso.org/dicb/dicd/dic-1-1.4.html</a>
55           */
56          ESO("http://arcdev.hq.eso.org/dicb/dicd/dic-1-1.4.html"),
57          /**
58           * HEASARC keywords. See <a href=
59           * "http://heasarc.gsfc.nasa.gov/docs/heasarc/ofwg/docs/ofwg_recomm/r13.html">http://heasarc.gsfc.nasa.gov/docs/heasarc/ofwg/docs/ofwg_recomm/r13.html</a>
60           */
61          HEASARC("http://heasarc.gsfc.nasa.gov/docs/heasarc/ofwg/docs/ofwg_recomm/r13.html"),
62          /**
63           * integral.
64           */
65          INTEGRAL(null),
66          /**
67           * Mandatory keywords defined by the fits standard.
68           */
69          MANDATORY("http://heasarc.gsfc.nasa.gov/docs/fcg/standard_dict.html"),
70          /**
71           * MaxImDL keywords. See <a href=
72           * "http://www.cyanogen.com/help/maximdl/FITS_File_Header_Definitions.htm">http://www.cyanogen.com/help/maximdl/FITS_File_Header_Definitions.htm</a>
73           */
74          MaxImDL("http://www.cyanogen.com/help/maximdl/FITS_File_Header_Definitions.htm"),
75          /**
76           * NOAO keywords. See <a href=
77           * "http://iraf.noao.edu/iraf/web/projects/ccdmosaic/imagedef/fitsdic.html">http://iraf.noao.edu/iraf/web/projects/ccdmosaic/imagedef/fitsdic.html</a>
78           */
79          NOAO("http://iraf.noao.edu/iraf/web/projects/ccdmosaic/imagedef/fitsdic.html"),
80          /**
81           * Reserved keywords specified by the FITS standard.
82           */
83          RESERVED("http://heasarc.gsfc.nasa.gov/docs/fcg/standard_dict.html"),
84          /**
85           * ROSAT keywords. (No link available.)
86           */
87          ROSAT(null),
88          /**
89           * SBIG keywords. See <a href=
90           * "http://archive.sbig.com/pdffiles/SBFITSEXT_1r0.pdf">http://archive.sbig.com/pdffiles/SBFITSEXT_1r0.pdf</a>
91           */
92          SBIG("http://archive.sbig.com/pdffiles/SBFITSEXT_1r0.pdf"),
93          /**
94           * STScI keywords. See <a href=
95           * "http://tucana.noao.edu/ADASS/adass_proc/adass_95/zaraten/zaraten.html">http://tucana.noao.edu/ADASS/adass_proc/adass_95/zaraten/zaraten.html</a>
96           */
97          STScI("http://tucana.noao.edu/ADASS/adass_proc/adass_95/zaraten/zaraten.html"),
98          /**
99           * UCOLICK keywords. See <a href="http://www.ucolick.org">http://www.ucolick.org</a>
100          */
101         UCOLICK("http://www.ucolick.org"),
102         /**
103          * developed over time, source long forgotten.
104          */
105         UNKNOWN(null);
106 
107         private final String url;
108 
109         SOURCE(String url) {
110             this.url = url;
111         }
112 
113         public String url() {
114             return this.url;
115         }
116     }
117 
118     enum VALUE {
119         INTEGER, LOGICAL, NONE, REAL, COMPLEX, STRING, ANY
120     }
121 
122     String comment();
123 
124     HDU hdu();
125 
126     String key();
127 
128     IFitsHeader n(int... number);
129 
130     SOURCE status();
131 
132     VALUE valueType();
133 }
