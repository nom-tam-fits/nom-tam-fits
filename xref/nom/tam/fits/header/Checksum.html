34  /**
35   * File checksum keywords. This data dictionary contains FITS keywords that have
36   * been widely used within the astronomical community. It is recommended that
37   * these keywords only be used as defined here.
38   * 
39   * <pre>
40   * @see <a href="http://heasarc.gsfc.nasa.gov/docs/fcg/common_dict.html">http://heasarc.gsfc.nasa.gov/docs/fcg/common_dict.html</a>
41   * </pre>
42   * 
43   * @author Richard van Nieuwenhoven
44   */
45  public enum Checksum implements IFitsHeader {
46      /**
47       * The value field of the CHECKSUM keyword shall contain a 16 character
48       * string, left justified starting in column 12, containing the ASCII
49       * encoded complement of the checksum of the FITS HDU (Header and Data
50       * Unit). The algorithm shall be the 32-bit 1's complement checksum and the
51       * ASCII encoding that are described in the checksum proposal. The checksum
52       * is accumulated in FITS datastream order on the same HDU, identical in all
53       * respects, except that the value of the CHECKSUM keyword shall be set to
54       * the string '0000000000000000' (ASCII 0's, hex 30) before the checksum is
55       * computed.
56       */
57      CHECKSUM(HDU.ANY, VALUE.STRING, "checksum for the current HDU"),
58      /**
59       * The value field of the CHECKVER keyword shall contain a string, unique in
60       * the first 8 characters, which distinguishes between any future
61       * alternative checksum algorithms which may be defined. The default value
62       * for a missing keyword shall be 'COMPLEMENT' which will represent the
63       * algorithm defined in the current proposal. It is recommended that this
64       * keyword be omitted from headers which implement the default ASCII encoded
65       * 32-bit 1's complement algorithm.
66       */
67      CHECKVER(HDU.ANY, VALUE.STRING, "version of checksum algorithm"),
68  
69      /**
70       * The value field of the DATASUM keyword shall be a character string
71       * containing the unsigned integer value of the checksum of the data records
72       * of the HDU. For dataless HDU's, this keyword may either be omitted, or
73       * the value field shall contain the string value '0', which is preferred. A
74       * missing DATASUM keyword asserts no knowledge of the checksum of the data
75       * records.
76       */
77      DATASUM(HDU.ANY, VALUE.STRING, "checksum of the data records");
78  
79      private IFitsHeader key;
80  
81      private Checksum(HDU hdu, VALUE valueType, String comment) {
82          this.key = new FitsHeaderImpl(name(), IFitsHeader.SOURCE.CHECKSUM, hdu, valueType, comment);
83      }
84  
85      @Override
86      public String comment() {
87          return key.comment();
88      }
89  
90      @Override
91      public HDU hdu() {
92          return key.hdu();
93      }
94  
95      @Override
96      public String key() {
97          return key.key();
98      }
99  
100     @Override
101     public IFitsHeader n(int... number) {
102         return key.n(number);
103     }
104 
105     @Override
106     public SOURCE status() {
107         return key.status();
108     }
109 
110     @Override
111     public VALUE valueType() {
112         return key.valueType();
113     }
114 
115 }
