34  /**
35   * This data dictionary contains FITS keywords that have been widely used within
36   * the astronomical community. It is recommended that these keywords only be
37   * used as defined here. These are the Keywords that describe the observation.
38   * 
39   * <pre>
40   * @see <a href="http://heasarc.gsfc.nasa.gov/docs/fcg/common_dict.html">http://heasarc.gsfc.nasa.gov/docs/fcg/common_dict.html</a>
41   * </pre>
42   * 
43   * @author Richard van Nieuwenhoven
44   */
45  public enum ObservationDescription implements IFitsHeader {
46      /**
47       * The value field shall contain a floating point number giving the air mass
48       * during the observation by a ground based telescope. The value of the
49       * airmass is often approximated by the secant of the elevation angle and
50       * has a value of 1.0 at the zenith and increases towards the horizon. This
51       * value is assumed to correspond to the start of the observation unless
52       * another interpretation is clearly explained in the comment field.
53       */
54      AIRMASS(SOURCE.NOAO, HDU.ANY, VALUE.REAL, "air mass"),
55      /**
56       * The value field gives the declination of the observation. It may be
57       * expressed either as a floating point number in units of decimal degrees,
58       * or as a character string in 'dd:mm:ss.sss' format where the decimal point
59       * and number of fractional digits are optional. The coordinate reference
60       * frame is given by the RADECSYS keyword, and the coordinate epoch is given
61       * by the EQUINOX keyword. Example: -47.25944 or '-47:15:34.00'.
62       */
63      DEC(SOURCE.NOAO, HDU.ANY, VALUE.STRING, "declination of the observed object"),
64      /**
65       * The value field shall contain a floating point number giving the nominal
66       * declination of the pointing direction in units of decimal degrees. The
67       * coordinate reference frame is given by the RADECSYS keyword, and the
68       * coordinate epoch is given by the EQUINOX keyword. The precise definition
69       * of this keyword is instrument-specific, but typically the nominal
70       * direction corresponds to the direction to which the instrument was
71       * requested to point. The DEC_PNT keyword should be used to give the actual
72       * pointed direction.
73       */
74      DEC_NOM(SOURCE.HEASARC, HDU.ANY, VALUE.REAL, "nominal declination of the observation"),
75      /**
76       * The value field shall contain a floating point number giving the
77       * declination of the observed object in units of decimal degrees. The
78       * coordinate reference frame is given by the RADECSYS keyword, and the
79       * coordinate epoch is given by the EQUINOX keyword.
80       */
81      DEC_OBJ(SOURCE.HEASARC, HDU.ANY, VALUE.REAL, "declination of the observed object"),
82      /**
83       * The value field shall contain a floating point number giving the
84       * declination of the pointing direction in units of decimal degrees. The
85       * coordinate reference frame is given by the RADECSYS keyword, and the
86       * coordinate epoch is given by the EQUINOX keyword. The precise definition
87       * of this keyword is instrument-specific, but typically the pointed
88       * direction corresponds to the optical axis of the instrument. This keyword
89       * gives a mean value in cases where the pointing axis was not fixed during
90       * the entire observation.
91       */
92      DEC_PNT(SOURCE.HEASARC, HDU.ANY, VALUE.REAL, "declination of the pointed direction of the instrument"),
93      /**
94       * The value field shall contain a floating point number giving the
95       * declination of the space craft (or telescope platform) X axis during the
96       * observation in decimal degrees. The coordinate reference frame is given
97       * by the RADECSYS keyword, and the coordinate epoch is given by the EQUINOX
98       * keyword. This keyword gives a mean value in cases where the axis was not
99       * fixed during the entire observation.
100      */
101     DEC_SCX(SOURCE.HEASARC, HDU.ANY, VALUE.REAL, "declination of the X spacecraft axis"),
102     /**
103      * The value field shall contain a floating point number giving the
104      * declination of the space craft (or telescope platform) Z axis during the
105      * observation in decimal degrees. The coordinate reference frame is given
106      * by the RADECSYS keyword, and the coordinate epoch is given by the EQUINOX
107      * keyword. This keyword gives a mean value in cases where the axis was not
108      * fixed during the entire observation.
109      */
110     DEC_SCZ(SOURCE.HEASARC, HDU.ANY, VALUE.REAL, "declination of the Z spacecraft axis"),
111     /**
112      * The value field shall contain a floating point number giving the
113      * geographic latitude from which the observation was made in units of
114      * degrees.
115      */
116     LATITUDE(SOURCE.UCOLICK, HDU.ANY, VALUE.REAL, "geographic latitude of the observation"),
117     /**
118      * The value field shall contain a floating point number giving the angle
119      * between the direction of the observation (e.g., the optical axis of the
120      * telescope or the position of the target) and the moon, measured in
121      * degrees.
122      */
123     MOONANGL(SOURCE.STScI, HDU.ANY, VALUE.REAL, "angle between the observation and the moon"),
124     /**
125      * The value field shall contain a character string giving a name for the
126      * observed object that conforms to the IAU astronomical object naming
127      * conventions. The value of this keyword is more strictly constrained than
128      * for the standard OBJECT keyword which in practice has often been used to
129      * record other ancillary information about the observation (e.g. filter,
130      * exposure time, weather conditions, etc.).
131      */
132     OBJNAME(SOURCE.NOAO, HDU.ANY, VALUE.STRING, "AU name of observed object"),
133     /**
134      * The value field shall contain a character string which uniquely
135      * identifies the dataset contained in the FITS file. This is typically a
136      * sequence number that can contain a mixture of numerical and character
137      * values. Example: '10315-01-01-30A'
138      */
139     OBS_ID(SOURCE.HEASARC, HDU.ANY, VALUE.STRING, "unique observation ID"),
140     /**
141      * The value field shall contain a floating point number giving the position
142      * angle of the y axis of the detector projected on the sky, in degrees east
143      * of north. This keyword is synonymous with the CROTA2 WCS keyword.
144      */
145     ORIENTAT(SOURCE.STScI, HDU.IMAGE, VALUE.REAL, "position angle of image y axis (deg. E of N)"),
146     /**
147      * The value field shall contain a floating point number giving the position
148      * angle of the relevant aspect of telescope pointing axis and/or instrument
149      * on the sky in units of degrees east of north. It commonly applies to the
150      * orientation of a slit mask.
151      */
152     PA_PNT(SOURCE.UCOLICK, HDU.ANY, VALUE.REAL, "position angle of the pointing"),
153     /**
154      * The value field gives the Right Ascension of the observation. It may be
155      * expressed either as a floating point number in units of decimal degrees,
156      * or as a character string in 'HH:MM:SS.sss' format where the decimal point
157      * and number of fractional digits are optional. The coordinate reference
158      * frame is given by the RADECSYS keyword, and the coordinate epoch is given
159      * by the EQUINOX keyword. Example: 180.6904 or '12:02:45.7'.
160      */
161     RA(SOURCE.NOAO, HDU.ANY, VALUE.STRING, "R.A. of the observation"),
162     /**
163      * The value field shall contain a floating point number giving the nominal
164      * Right Ascension of the pointing direction in units of decimal degrees.
165      * The coordinate reference frame is given by the RADECSYS keyword, and the
166      * coordinate epoch is given by the EQUINOX keyword. The precise definition
167      * of this keyword is instrument-specific, but typically the nominal
168      * direction corresponds to the direction to which the instrument was
169      * requested to point. The RA_PNT keyword should be used to give the actual
170      * pointed direction.
171      */
172     RA_NOM(SOURCE.HEASARC, HDU.ANY, VALUE.REAL, "nominal R.A. of the observation"),
173     /**
174      * The value field shall contain a floating point number giving the Right
175      * Ascension of the observed object in units of decimal degrees. The
176      * coordinate reference frame is given by the RADECSYS keyword, and the
177      * coordinate epoch is given by the EQUINOX keyword.
178      */
179     RA_OBJ(SOURCE.HEASARC, HDU.ANY, VALUE.REAL, "R.A. of the observed object"),
180     /**
181      * The value field shall contain a floating point number giving the Right
182      * Ascension of the pointing direction in units of decimal degrees. The
183      * coordinate reference frame is given by the RADECSYS keyword, and the
184      * coordinate epoch is given by the EQUINOX keyword. The precise definition
185      * of this keyword is instrument-specific, but typically the pointed
186      * direction corresponds to the optical axis of the instrument. This keyword
187      * gives a mean value in cases where the pointing axis was not fixed during
188      * the entire observation.
189      */
190     RA_PNT(SOURCE.HEASARC, HDU.ANY, VALUE.REAL, "R.A. of the pointed direction of the instrument"),
191     /**
192      * The value field shall contain a floating point number giving the Right
193      * Ascension of the space craft (or telescope platform) X axis during the
194      * observation in decimal degrees. The coordinate reference frame is given
195      * by the RADECSYS keyword, and the coordinate epoch is given by the EQUINOX
196      * keyword. This keyword gives a mean value in cases where the axis was not
197      * fixed during the entire observation.
198      */
199     RA_SCX(SOURCE.HEASARC, HDU.ANY, VALUE.REAL, "R.A. of the X spacecraft axis"),
200     /**
201      * The value field shall contain a floating point number giving the Right
202      * Ascension of the space craft (or telescope platform) Y axis during the
203      * observation in decimal degrees. The coordinate reference frame is given
204      * by the RADECSYS keyword, and the coordinate epoch is given by the EQUINOX
205      * keyword. This keyword gives a mean value in cases where the axis was not
206      * fixed during the entire observation.
207      */
208     RA_SCY(SOURCE.HEASARC, HDU.ANY, VALUE.REAL, "R.A. of the Y spacecraft axis"),
209     /**
210      * The value field shall contain a floating point number giving the Right
211      * Ascension of the space craft (or telescope platform) Z axis during the
212      * observation in decimal degrees. The coordinate reference frame is given
213      * by the RADECSYS keyword, and the coordinate epoch is given by the EQUINOX
214      * keyword. This keyword gives a mean value in cases where the axis was not
215      * fixed during the entire observation.
216      */
217     RA_SCZ(SOURCE.HEASARC, HDU.ANY, VALUE.REAL, "R.A. of the Z spacecraft axis"),
218     /**
219      * The value field shall contain a floating point number giving the angle
220      * between the direction of the observation (e.g., the optical axis of the
221      * telescope or the position of the target) and the sun, measured in
222      * degrees.
223      */
224     SUNANGLE(SOURCE.STScI, HDU.ANY, VALUE.REAL, "angle between the observation and the sun");
225 
226     private IFitsHeader key;
227 
228     private ObservationDescription(IFitsHeader.SOURCE status, HDU hdu, VALUE valueType, String comment) {
229         key = new FitsHeaderImpl(name(), status, hdu, valueType, comment);
230     }
231 
232     @Override
233     public String comment() {
234         return key.comment();
235     }
236 
237     @Override
238     public HDU hdu() {
239         return key.hdu();
240     }
241 
242     @Override
243     public String key() {
244         return key.key();
245     }
246 
247     @Override
248     public IFitsHeader n(int... number) {
249         return key.n(number);
250     }
251 
252     @Override
253     public SOURCE status() {
254         return key.status();
255     }
256 
257     @Override
258     public VALUE valueType() {
259         return key.valueType();
260     }
261 }
