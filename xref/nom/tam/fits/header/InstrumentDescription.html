34  /**
35   * This data dictionary contains FITS keywords that have been widely used within
36   * the astronomical community. It is recommended that these keywords only be
37   * used as defined here. These are the Keywords that describe the instrument
38   * that took the data.
39   * 
40   * <pre>
41   * @see <a href="http://heasarc.gsfc.nasa.gov/docs/fcg/common_dict.html">http://heasarc.gsfc.nasa.gov/docs/fcg/common_dict.html</a>
42   * </pre>
43   * 
44   * @author Richard van Nieuwenhoven
45   */
46  public enum InstrumentDescription implements IFitsHeader {
47      /**
48       * The value field shall contain a character string which gives the name of
49       * the instrumental aperture though which the observation was made. This
50       * keyword is typically used in instruments which have a selection of
51       * apertures which restrict the field of view of the detector.
52       */
53      APERTURE(SOURCE.UNKNOWN, HDU.ANY, VALUE.STRING, "name of field of view aperture"),
54      /**
55       * The value field shall contain a character string which identifies the
56       * configuration or mode of the pre-processing software that operated on the
57       * raw instrumental data to generate the data that is recorded in the FITS
58       * file. Example: some X-ray satellite data may be recorded in 'BRIGHT',
59       * 'FAINT', or 'FAST' data mode.
60       */
61      DATAMODE(SOURCE.HEASARC, HDU.ANY, VALUE.STRING, "pre-processor data mode"),
62      /**
63       * The value field shall contain a character string giving the name of the
64       * detector within the instrument that was used to make the observation.
65       * Example: 'CCD1'
66       */
67      DETNAM(SOURCE.HEASARC, HDU.ANY, VALUE.STRING, "name of the detector used to make the observation"),
68      /**
69       * The value field shall contain a character string which gives the name of
70       * the filter that was used during the observation to select or modify the
71       * radiation that was transmitted to the detector. More than 1 filter may be
72       * listed by using the FILTERn indexed keyword. The value 'none' or 'NONE'
73       * indicates that no filter was used.
74       */
75      FILTER(SOURCE.HEASARC, HDU.ANY, VALUE.STRING, "name of filter used during the observation"),
76      /**
77       * The value field of this indexed keyword shall contain a character string
78       * which gives the name of one of multiple filters that were used during the
79       * observation to select or modify the radiation that was transmitted to the
80       * detector. The value 'none' or 'NONE' indicates that no filter was used.
81       */
82      FILTERn(SOURCE.HEASARC, HDU.ANY, VALUE.STRING, "name of filters used during the observation"),
83      /**
84       * The value field shall contain a character string which gives the name of
85       * the defraction grating that was used during the observation. More than 1
86       * grating may be listed by using the GRATINGn indexed keyword. The value
87       * 'none' or 'NONE' indicates that no grating was used.
88       */
89      GRATING(SOURCE.HEASARC, HDU.ANY, VALUE.STRING, "name of the grating used during the observation."),
90      /**
91       * The value field of this indexed keyword shall contain a character string
92       * which gives the name of one of multiple defraction gratings that were
93       * used during the observation. The value 'none' or 'NONE' indicates that no
94       * grating was used.
95       */
96      GRATINGn(SOURCE.HEASARC, HDU.ANY, VALUE.STRING, "name of gratings used during the observation."),
97      /**
98       * The value field shall contain a character string which gives the
99       * observing mode of the observation. This is used in cases where the
100      * instrument or detector can be configured to operate in different modes
101      * which significantly affect the resulting data. Examples: 'SLEW',
102      * 'RASTER', or 'POINTING'
103      */
104     OBS_MODE(SOURCE.HEASARC, HDU.ANY, VALUE.STRING, "instrumental mode of the observation"),
105     /**
106      * The value field shall contain an integer giving the data value at which
107      * the detector becomes saturated. This keyword value may differ from the
108      * maximum value implied by the BITPIX in that more bits may be allocated in
109      * the FITS pixel values than the detector can accommodate.
110      */
111     SATURATE(SOURCE.STScI, HDU.ANY, VALUE.INTEGER, "Data value at which saturation occurs");
112 
113     private IFitsHeader key;
114 
115     private InstrumentDescription(IFitsHeader.SOURCE status, HDU hdu, VALUE valueType, String comment) {
116         key = new FitsHeaderImpl(name(), status, hdu, valueType, comment);
117     }
118 
119     @Override
120     public String comment() {
121         return key.comment();
122     }
123 
124     @Override
125     public HDU hdu() {
126         return key.hdu();
127     }
128 
129     @Override
130     public String key() {
131         return key.key();
132     }
133 
134     @Override
135     public IFitsHeader n(int... number) {
136         return key.n(number);
137     }
138 
139     @Override
140     public SOURCE status() {
141         return key.status();
142     }
143 
144     @Override
145     public VALUE valueType() {
146         return key.valueType();
147     }
148 }
