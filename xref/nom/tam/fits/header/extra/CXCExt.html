178     @SuppressWarnings("CPD-START")
179     private final IFitsHeader key;
180 
181     CXCExt(String comment) {
182         this.key = new FitsHeaderImpl(name(), IFitsHeader.SOURCE.CXC, HDU.ANY, VALUE.STRING, comment);
183     }
184 
185     @Override
186     public String comment() {
187         return this.key.comment();
188     }
189 
190     @Override
191     public HDU hdu() {
192         return this.key.hdu();
193     }
194 
195     @Override
196     public String key() {
197         return this.key.key();
198     }
199 
200     @Override
201     public IFitsHeader n(int... number) {
202         return this.key.n(number);
203     }
204 
205     @Override
206     public SOURCE status() {
207         return this.key.status();
208     }
209 
210     @Override
211     @SuppressWarnings("CPD-END")
212     public VALUE valueType() {
213         return this.key.valueType();
214     }
