34  import nom.tam.fits.header.FitsHeaderImpl;
35  import nom.tam.fits.header.IFitsHeader;
36  
37  /**
38   * This is the file represents the common keywords between CXC and STSclExt
39   * 
40   * @author Richard van Nieuwenhoven
41   */
42  public enum CXCStclSharedExt implements IFitsHeader {
43      /**
44       * Clock correction applied
45       * <p>
46       * T
47       * </p>
48       */
49      CLOCKAPP("Clock correction applied"),
50      /**
51       * 1998-01-01T00:00:00 (TT) expressed in MJD (TT)
52       */
53      MJDREF("1998-01-01T00:00:00 (TT) expressed in MJD"),
54      /**
55       * Spacecraft clock
56       */
57      TASSIGN("Spacecraft clock"),
58      /**
59       * Time resolution of data (in seconds)
60       */
61      TIMEDEL("Time resolution of data (in seconds)"),
62      /**
63       * No pathlength corrections
64       */
65      TIMEREF("No pathlength corrections"),
66      /**
67       * Units of time e.g. 's'
68       */
69      TIMEUNIT("Units of time "),
70      /**
71       * AXAF FITS design document
72       */
73      TIMVERSN("AXAF FITS design document"),
74      /**
75       * Clock correction (if not zero)
76       */
77      TIMEZERO("Clock correction (if not zero)"),
78      /**
79       * As in the "TIME" column: raw space craft clock;
80       */
81      TSTART("As in the \"TIME\" column: raw space craft clock;"),
82      /**
83       * add TIMEZERO and MJDREF for absolute TT
84       */
85      TSTOP("add TIMEZERO and MJDREF for absolute TT");
86  
87      @SuppressWarnings("CPD-START")
88      private final IFitsHeader key;
89  
90      CXCStclSharedExt(String comment) {
91          this.key = new FitsHeaderImpl(name(), IFitsHeader.SOURCE.CXC, HDU.ANY, VALUE.STRING, comment);
92      }
93  
94      @Override
95      public String comment() {
96          return this.key.comment();
97      }
98  
99      @Override
100     public HDU hdu() {
101         return this.key.hdu();
102     }
103 
104     @Override
105     public String key() {
106         return this.key.key();
107     }
108 
109     @Override
110     public IFitsHeader n(int... number) {
111         return this.key.n(number);
112     }
113 
114     @Override
115     public SOURCE status() {
116         return this.key.status();
117     }
118 
119     @Override
120     @SuppressWarnings("CPD-END")
121     public VALUE valueType() {
122         return this.key.valueType();
123     }
124 }
