37  /**

44   * </p>
45   * 
46   * <pre>
47   * http://www.cyanogen.com/help/maximdl/FITS_File_Header_Definitions.htm

73      BOLTDEW(VALUE.REAL, "Boltwood Cloud Sensor dewpoint in degrees C."),

77      BOLTHUM(VALUE.REAL, "Boltwood Cloud Sensor humidity in percent."),

216     TILEXY(VALUE.REAL, "tile position within a mosaic"),

226     @SuppressWarnings("CPD-START")
227     private final IFitsHeader key;
228 
229     MaxImDLExt(String key, VALUE valueType, String comment) {
230         this.key = new FitsHeaderImpl(key, IFitsHeader.SOURCE.MaxImDL, HDU.IMAGE, valueType, comment);
231     }
232 
233     MaxImDLExt(VALUE valueType, String comment) {
234         this.key = new FitsHeaderImpl(name(), IFitsHeader.SOURCE.MaxImDL, HDU.IMAGE, valueType, comment);
235     }
236 
237     @Override
238     public String comment() {
239         return this.key.comment();
240     }
241 
242     @Override
243     public HDU hdu() {
244         return this.key.hdu();
245     }
246 
247     @Override
248     public String key() {
249         return this.key.key();
250     }
251 
252     @Override
253     public IFitsHeader n(int... number) {
254         return this.key.n(number);
255     }
256 
257     @Override
258     public SOURCE status() {
259         return this.key.status();
260     }
261 
262     @Override
263     @SuppressWarnings("CPD-END")
264     public VALUE valueType() {
265         return this.key.valueType();
266     }
