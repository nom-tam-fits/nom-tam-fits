216     @SuppressWarnings("CPD-START")
217     private final IFitsHeader key;
218 
219     STScIExt(String comment) {
220         this.key = new FitsHeaderImpl(name(), IFitsHeader.SOURCE.CXC, HDU.ANY, VALUE.STRING, comment);
221     }
222 
223     STScIExt(String key, String comment) {
224         this.key = new FitsHeaderImpl(name(), IFitsHeader.SOURCE.CXC, HDU.ANY, VALUE.STRING, comment);
225     }
226 
227     @Override
228     public String comment() {
229         return this.key.comment();
230     }
231 
232     @Override
233     public HDU hdu() {
234         return this.key.hdu();
235     }
236 
237     @Override
238     public String key() {
239         return this.key.key();
240     }
241 
242     @Override
243     public IFitsHeader n(int... number) {
244         return this.key.n(number);
245     }
246 
247     @Override
248     public SOURCE status() {
249         return this.key.status();
250     }

252     @Override
253     @SuppressWarnings("CPD-END")
254     public VALUE valueType() {
255         return this.key.valueType();
