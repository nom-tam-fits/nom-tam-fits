32  import nom.tam.fits.header.IFitsHeader.HDU;
33  import nom.tam.fits.header.IFitsHeader.STATUS;
34  import nom.tam.fits.header.IFitsHeader.VALUE;
35  
36  /**
37   * File checksum keywords. This data dictionary contains FITS keywords that have
38   * been widely used within the astronomical community. It is recommended that
39   * these keywords only be used as defined here. These are the Hierarchical file
40   * grouping keywords. {@link http
41   * ://heasarc.gsfc.nasa.gov/docs/fcg/common_dict.html}
42   * 
43   * @author Richard van Nieuwenhoven
44   */
45  public enum HierarchicalGrouping implements IFitsHeader {
46      /**
47       * the grouping table name. TODO: find description?
48       */
49      GRPNAME(STATUS.HEASARC, HDU.TABLE, VALUE.STRING, "the grouping table name"),
50      /**
51       * TODO: find description?
52       */
53      GRPIDn(STATUS.HEASARC, HDU.TABLE, VALUE.STRING, ""),
54      /**
55       * TODO: find description?
56       */
57      GRPLCn(STATUS.HEASARC, HDU.TABLE, VALUE.STRING, "");
58  
59      private IFitsHeader key;
60  
61      private HierarchicalGrouping(IFitsHeader.STATUS status, HDU hdu, VALUE valueType, String comment) {
62          this.key = new FitsHeaderImpl(name(), status, hdu, valueType, comment);
63      }
64  
65      @Override
66      public String key() {
67          return key.key();
68      }
69  
70      @Override
71      public STATUS status() {
72          return key.status();
73      }
74  
75      @Override
76      public HDU hdu() {
77          return key.hdu();
78      }
79  
80      @Override
81      public VALUE valueType() {
82          return key.valueType();
83      }
84  
85      @Override
86      public String comment() {
87          return key.comment();
88      }
89  
90      @Override
91      public IFitsHeader n(int number) {
92          return key.n(number);
93      }
94  }
