34  /**
35   * This data dictionary contains FITS keywords that have been widely used within
36   * the astronomical community. It is recommended that these keywords only be
37   * used as defined here. These are the Hierarchical file grouping keywords.
38   * 
39   * <pre>
40   * @see <a href="http://heasarc.gsfc.nasa.gov/docs/fcg/common_dict.html">http://heasarc.gsfc.nasa.gov/docs/fcg/common_dict.html</a>
41   * </pre>
42   * 
43   * @author Richard van Nieuwenhoven
44   */
45  public enum HierarchicalGrouping implements IFitsHeader {
46      /**
47       * TODO: find description?
48       */
49      GRPIDn(SOURCE.HEASARC, HDU.TABLE, VALUE.STRING, ""),
50      /**
51       * TODO: find description?
52       */
53      GRPLCn(SOURCE.HEASARC, HDU.TABLE, VALUE.STRING, ""),
54      /**
55       * the grouping table name. TODO: find description?
56       */
57      GRPNAME(SOURCE.HEASARC, HDU.TABLE, VALUE.STRING, "the grouping table name");
58  
59      @SuppressWarnings("CPD-START")
60      private final IFitsHeader key;
61  
62      HierarchicalGrouping(IFitsHeader.SOURCE status, HDU hdu, VALUE valueType, String comment) {
63          this.key = new FitsHeaderImpl(name(), status, hdu, valueType, comment);
64      }
65  
66      @Override
67      public String comment() {
68          return this.key.comment();
69      }
70  
71      @Override
72      public HDU hdu() {
73          return this.key.hdu();
74      }
75  
76      @Override
77      public String key() {
78          return this.key.key();
79      }
80  
81      @Override
82      public IFitsHeader n(int... number) {
83          return this.key.n(number);
84      }
85  
86      @Override
87      public SOURCE status() {
88          return this.key.status();
89      }
90  
91      @Override
92      @SuppressWarnings("CPD-END")
93      public VALUE valueType() {
94          return this.key.valueType();
95      }
96  }
