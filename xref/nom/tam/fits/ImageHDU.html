51  /**
52   * FITS image header/data unit
53   */
54  public class ImageHDU extends BasicHDU<ImageData> {
55  
56      private static final Logger LOG = getLogger(ImageHDU.class);
57  
58      /**
59       * @return Encapsulate an object as an ImageHDU.
60       * @param o
61       *            object to encapsulate
62       * @throws FitsException
63       *             if the operation failed
64       */
65      public static ImageData encapsulate(Object o) throws FitsException {
66          return new ImageData(o);
67      }
68  
69      /**
70       * @return is this object can be described as a FITS image.
71       * @param o
72       *            The Object being tested.
73       */
74      public static boolean isData(Object o) {
75          if (o.getClass().isArray()) {
76              ElementType<?> type = ElementType.forClass(ArrayFuncs.getBaseClass(o));
77              return type != ElementType.BOOLEAN && //
78                      type != ElementType.STRING && //
79                      type != ElementType.UNKNOWN;
80  
81          }
82          return false;
83      }
84  
85      /**
86       * Check that this HDU has a valid header for this type.
87       * 
88       * @param hdr
89       *            header to check
90       * @return <CODE>true</CODE> if this HDU has a valid header.
91       */
92      public static boolean isHeader(Header hdr) {
93          boolean found = hdr.getBooleanValue(SIMPLE);
94          if (!found) {
95              String xtension = hdr.getStringValue(XTENSION);
96              xtension = xtension == null ? "" : xtension.trim();
97              if (XTENSION_IMAGE.equals(xtension) || "IUEIMAGE".equals(xtension)) {
98                  found = true;
99              }
100         }
101         if (!found) {
102             return false;
103         }
104         return !hdr.getBooleanValue(GROUPS);
105     }
106 
107     public static Data manufactureData(Header hdr) throws FitsException {
108         return new ImageData(hdr);
109     }
110 
111     /**
112      * @return Create a header that describes the given image data.
113      * @param d
114      *            The image to be described.
115      * @throws FitsException
116      *             if the object does not contain valid image data.
117      */
118     public static Header manufactureHeader(Data d) throws FitsException {
119         if (d == null) {
120             return null;
121         }
122 
123         Header h = new Header();
124         d.fillHeader(h);
125 
126         return h;
127     }
128 
129     /**
130      * Build an image HDU using the supplied data.
131      * 
132      * @param h
133      *            the header for the image.
134      * @param d
135      *            the data used in the image.
136      * @throws FitsException
137      *             if there was a problem with the data.
138      */
139     public ImageHDU(Header h, ImageData d) throws FitsException {
140         super(h, d);
141     }
142 
143     /** Indicate that Images can appear at the beginning of a FITS dataset */
144     @Override
145     protected boolean canBePrimary() {
146         return true;
147     }
148 
149     public StandardImageTiler getTiler() {
150         return this.myData.getTiler();
151     }
152 
153     /**
154      * Print out some information about this HDU.
155      */
156     @Override
157     public void info(PrintStream stream) {
158         if (isHeader(this.myHeader)) {
159             stream.println("  Image");
160         } else {
161             stream.println("  Image (bad header)");
162         }
163 
164         stream.println("      Header Information:");
165         stream.println("         BITPIX=" + this.myHeader.getIntValue(BITPIX, -1));
166         int naxis = this.myHeader.getIntValue(NAXIS, -1);
167         stream.println("         NAXIS=" + naxis);
168         for (int i = 1; i <= naxis; i += 1) {
169             stream.println("         NAXIS" + i + "=" + this.myHeader.getIntValue(NAXISn.n(i), -1));
170         }
171 
172         stream.println("      Data information:");
173         try {
174             if (this.myData.getData() == null) {
175                 stream.println("        No Data");
176             } else {
177                 stream.println("         " + ArrayFuncs.arrayDescription(this.myData.getData()));
178             }
179         } catch (Exception e) {
180             LOG.log(Level.SEVERE, "Unable to get image data", e);
181             stream.println("      Unable to get data");
182         }
183     }
184 
185     /** Change the Image from/to primary */
186     @Override
187     protected void setPrimaryHDU(boolean status) {
188 
189         try {
190             super.setPrimaryHDU(status);
191         } catch (FitsException e) {
192             LOG.log(Level.SEVERE, "Impossible exception in ImageData", e);
193         }
194 
195         if (status) {
196             this.myHeader.setSimple(true);
197         } else {
198             this.myHeader.setXtension(XTENSION_IMAGE);
199         }
200     }
201 }
