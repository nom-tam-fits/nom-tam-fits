48  /**
49   * <p>
50   * Converts a single 80-character wide FITS header record into a header card. See
51   * {@link HeaderCard#create(String)} for a description of the rules that guide parsing.
52   * </p>
53   * <p>
54   * When parsing header records that violate FITS standards, the violations can be logged
55   * or will throw appropriate excpetions (depending on the severity of the standard 
56   * violation and whether {@link FitsFactory#setAllowHeaderRepairs(boolean)} is
57   * enabled or not. The logging of violations is disabled by default, but may be
58   * controlled via {@link Header#setParserWarningsEnabled(boolean)}.
59   * </p>
60   * 
61   *
62   * @author Attila Kovacs
63   * 
64   * @see FitsFactory#setAllowHeaderRepairs(boolean)
65   * @see Header#setParserWarningsEnabled(boolean)
66   */
67  class HeaderCardParser {

76      /** regexp for IEEE floats */
77      private static final Pattern DECIMAL_REGEX = Pattern.compile("[+-]?\\d+(\\.\\d*)?([dDeE][+-]?\\d+)?");

79      /** regexp for complex numbers */
80      private static final Pattern COMPLEX_REGEX = Pattern.compile("\\(\\s*" + DECIMAL_REGEX + "\\s*,\\s*" + DECIMAL_REGEX + "\\s*\\)");

82      /** regexp for decimal integers. */
83      private static final Pattern INT_REGEX = Pattern.compile("[+-]?\\d+");
84      
85      /** regexp for hexadecimal integers. */
86      private static final Pattern HEX_REGEX = Pattern.compile("[+-]?[\\dA-Fa-f]+");

191     /**
192      * Returns whether the line contained a quoted string value. By default, strings with missing end quotes are no
193      * considered string values, but rather as comments. To allow processing lines with missing quotes as string
194      * values, you must set {@link FitsFactory#setAllowHeaderRepairs(boolean)} to <code>true</code> prior to parsing
195      * a header line with the missing end quote.
196      * 
197      * @return true if the value was quoted.
198      * 
199      * @see FitsFactory#setAllowHeaderRepairs(boolean)
200      */
201     boolean isString() {
202         if (type == null) {
203             return false;
204         }
205         return String.class.isAssignableFrom(type);
206     }

208     /**
209      * <p>
210      * Returns the inferred Java class for the value stored in the header record, such as a
211      * {@link String} class, a {@link Boolean} class, an integer type ({@link Integer}, {@link Long}, or {@link BigInteger})
212      * class, a decimal type ({@link Float}, {@link Double}, or {@link BigDecimal}) class,
213      * a {@link ComplexValue} class, or <code>null</code>. For number types, it returns the
214      * 'smallest' type that can be used to represent the string value. 
215      * </p>
216      * <p>
217      * Its an inferred type as
218      * the true underlying type that was used to create the value is lost. For example, the
219      * value <code>42</code> may have been written from any integer type, including <code>byte</code>
220      * or <code>short<code>, but this routine will guess it to be an <code>int</code> ({@link Integer}
221      * type. As such, it may not be equal to {@link HeaderCard#valueType()} from which the
222      * record was created, and hence should not be used for round-trip testing of type equality.
223      * </p>
224      * 
225      * @return  the inferred type of the stored serialized (string) value, or <code>null</code>
226      *          if the value does not seem to match any of the supported value types.
227      * 
228      * @see HeaderCard#valueType()
229      */
230     Class<?> getInferredType() {
231         return type;
232     }

234     /**
235      * Parses a fits keyword from a card and standardizes it (trim, uppercase, and hierarch with dots).
236      */
237     private void parseKey() {
238         /*
239          * AK: The parsing of headers should never be stricter that the writing, such that any header written by
240          * this library can be parsed back without errors. (And, if anything, the parsing should be more permissive
241          * to allow reading FITS produced by other libraries, which may be less stringent in their rules). The
242          * original implementation strongly enforced the ESO HIERARCH convention when reading, but not at all for
243          * writing. Here is a tolerant hierarch parser that will read back any hierarch key that was written by this
244          * library. The input FITS can use any space or even '.' to separate the hierarchies, and the hierarchical
245          * elements may contain any ASCII characters other than those used for separating. It is more in line with
246          * what we do with standard keys too.
247          */
248 
249         // Find the '=' in the line, if any...
250         int iEq = line.indexOf('=');
251         
252         // The stem is in the first 8 characters or what precedes an '=' character
253         // before that.
254         int endStem = (iEq >= 0 && iEq <= HeaderCard.MAX_KEYWORD_LENGTH) ? iEq : HeaderCard.MAX_KEYWORD_LENGTH;
255         endStem = Math.min(line.length(), endStem);
256 
257         String rawStem = line.substring(0, endStem).trim();
258         
259         // Check for space at the start of the keyword...
260         if (endStem > 0 && !rawStem.isEmpty()) {
261             if (Character.isSpaceChar(line.charAt(0))) {
262                 LOG.warning("[" + sanitize(rawStem) + "] Non-standard starting with a space (trimming).");
263             }
264         }
265         
266         String stem = rawStem.toUpperCase();
267         
268         if (!stem.equals(rawStem)) {
269             LOG.warning("[" + sanitize(rawStem) + "] Non-standard lower-case letter(s) in base keyword.");
270         }   
271         
272         key = stem;
273         parsePos = endStem;
274 
275         // If not using HIERARCH, then be very resilient, and return whatever key the first 8 chars make...
276         if (!FitsFactory.getUseHierarch()) {
277             return;
278         }
279 
280         // If the line does not have an '=', can only be a simple key
281         if (iEq < 0) {
282             return;
283         }
284 
285         // If it's not a HIERARCH keyword, then return the simple key.
286         if (!stem.equals(HIERARCH.key())) {
287             return;
288         }
289 
290         // Compose the hierarchical key...
291         StringTokenizer tokens = new StringTokenizer(line.substring(stem.length(), iEq), " \t\r\n.");
292         StringBuilder builder = new StringBuilder(stem);
293 
294         while (tokens.hasMoreTokens()) {
295             String token = tokens.nextToken();
296 
297             parsePos = line.indexOf(token, parsePos) + token.length();
298 
299             // Add a . to separate hierarchies
300             builder.append('.');
301             builder.append(token);
302         }
303 
304         key = builder.toString();
305         
306         if (HIERARCH.key().equals(key)) {
307             // The key is only HIERARCH, without a hierarchical keyword after it...
308             LOG.warning("HIERARCH base keyword without HIERARCH-style long key after it.");
309             return;
310         }
311 
312         if (!FitsFactory.getHierarchFormater().isCaseSensitive()) {
313             key = key.toUpperCase(Locale.US);
314         }
315         
316         try {
317             HeaderCard.validateKey(key);
318         } catch (IllegalArgumentException e) {
319             LOG.warning(e.getMessage());
320         }
321     }

323     /**
324      * Advances the parse position to skip any spaces at the current parse position, and returns whether there is
325      * anything left in the line after the spaces...
326      * 
327      * @return <code>true</code> if there is more non-space characters in the string, otherwise <code>false</code>
328      */
329     private boolean skipSpaces() {
330         for (; parsePos < line.length(); parsePos++) {
331             if (!Character.isSpaceChar(line.charAt(parsePos))) {
332                 // Line has non-space characters left to parse...
333                 return true;
334             }
335         }
336         // nothing left to parse.
337         return false;
338     }
339 
340     /**
341      * Parses the comment components starting from the current parse position. After this call the parse position is
342      * set to the end of the string. The leading '/' (if found) is not included in the comment.
343      */
344     private void parseComment() {
345         if (!skipSpaces()) {
346             // nothing left to parse.
347             return;
348         }
349         
350         // if no value, then everything is comment from here on...
351         if (value != null) {
352             if (line.charAt(parsePos) == '/') {
353                 // Skip the '/' itself, the comment is whatever is after it.
354                 parsePos++;
355             } else {
356                 // Junk after a string value -- interpret it as the start of the comment...
357                 LOG.warning("[" + sanitize(getKey()) + "] Junk after value (included in the comment).");
358             }
359         }
360 
361         comment = line.substring(parsePos);
362         parsePos = line.length();
363         
364         try {
365             HeaderCard.validateChars(comment);
366         } catch (IllegalArgumentException e) {
367             LOG.warning("[" + sanitize(getKey()) + "]: " + e.getMessage());
368         }
369     }

371     /**
372      * Parses the value component from the current parse position. The parse position is advanced to the first
373      * character after the value specification in the line. If the header line does not contain a value component,
374      * then the value field of this object is set to <code>null</code>.
375      * 
376      * @throws UnclosedQuoteException if there is a missing end-quote and header repairs aren't allowed.
377      * 
378      * @see FitsFactory#setAllowHeaderRepairs(boolean)
379      * 
380      */
381     private void parseValue() throws UnclosedQuoteException {
382         if (key.isEmpty()) {
383             // the entire line is a comment.
384             return;
385         }
386 
387         if (!skipSpaces()) {
388             // nothing left to parse.
389             return;
390         }
391 
392         if (CONTINUE.key().equals(key)) {
393             parseValueBody();
394         } else if (line.charAt(parsePos) == '=') {
395            
396             if (parsePos < HeaderCard.MAX_KEYWORD_LENGTH) {
397                 LOG.warning("[" + sanitize(key) + "] assigmment before byte " + (HeaderCard.MAX_KEYWORD_LENGTH + 1) + " for key '" + sanitize(key) + "'.");
398             }
399             if (parsePos + 1 >= line.length()) {
400                 LOG.warning("[" + sanitize(key) + "] Record ends with '='.");
401             } else if (line.charAt(parsePos + 1) != ' ') {
402                 LOG.warning("[" + sanitize(key) + "] Missing required standard space after '='.");
403             }    
404             
405             if (parsePos > HeaderCard.MAX_KEYWORD_LENGTH) {
406                 // equal sign = after the 9th char -- only supported with hierarch keys...
407                 if (!key.startsWith(HIERARCH.key() + ".")) {
408                     LOG.warning("[" + sanitize(key) + "] Possibly misplaced '=' (after byte 9).");
409                     // It's not a HIERARCH key
410                     return;
411                 }
412             } 
413 
414             parsePos++;
415             parseValueBody();
416         }
417         
418         try {
419             HeaderCard.validateChars(value);
420         } catch (IllegalArgumentException e) {
421             LOG.warning("[" + sanitize(getKey()) + "] " + e.getMessage());
422         }
423     }
424 
425     /**
426      * Parses the value body from the current parse position. The parse position is advanced to the first character
427      * after the value specification in the line. If the header line does not contain a value component, then the
428      * value field of this object is set to <code>null</code>.
429      * 
430      * @throws UnclosedQuoteException if there is a missing end-quote and header repairs aren't allowed.
431      * 
432      * @see FitsFactory#setAllowHeaderRepairs(boolean)
433      * 
434      */
435     private void parseValueBody() throws UnclosedQuoteException {
436         if (!skipSpaces()) {
437             // nothing left to parse.
438             return;
439         }
440 
441         if (isNextQuote()) {
442             // Parse as a string value, or else throw an exception.
443             parseStringValue();
444         } else {
445             int end = line.indexOf('/', parsePos);
446             if (end < 0) {
447                 end = line.length();
448             }
449             value = line.substring(parsePos, end).trim();
450             parsePos = end;
451             this.type = getInferredValueType(key, value);
452         }
453 
454     }
455 
456     /**
457      * Checks if the next character, at the current parse position, is a single quote.
458      * 
459      * @return <code>true</code> if the next character on the line exists and is a single quote, otherwise
460      *             <code>false</code>.
461      */
462     private boolean isNextQuote() {
463         if (parsePos >= line.length()) {
464             // nothing left to parse.
465             return false;
466         }
467         return line.charAt(parsePos) == '\'';
468     }

470     /**
471      * Returns the string fom a parsed string value component, with trailing spaces removed. It preserves leading
472      * spaces.
473      * 
474      * @param buf the parsed string value.
475      * 
476      * @return the string value with trailing spaces removed.
477      */
478     private static String getNoTrailingSpaceString(StringBuilder buf) {
479         int to = buf.length();
480 
481         // Remove trailing spaces only!
482         while (--to >= 0) {
483             if (!Character.isSpaceChar(buf.charAt(to))) {
484                 break;
485             }
486         }
487 
488         return to < 0 ? "" : buf.substring(0, to + 1);
489     }
490 
491     /**
492      * Parses a quoted string value starting at the current parse position. If successful, the parse position is
493      * updated to after the string. Otherwise, the parse position is advanced only to skip leading spaces starting
494      * from the input position.
495      * 
496      * @throws UnclosedQuoteException if there is a missing end-quote and header repairs aren't allowed.
497      * 
498      * @see FitsFactory#setAllowHeaderRepairs(boolean)
499      */
500     private void parseStringValue() throws UnclosedQuoteException {
501         this.type = String.class;
502         StringBuilder buf = new StringBuilder(HeaderCard.MAX_VALUE_LENGTH);
503 
504         // Build the string value, up to the end quote and paying attention to double
505         // quotes inside the string, which are translated to single quotes within
506         // the string value itself.
507         for (++parsePos; parsePos < line.length(); parsePos++) {
508             if (isNextQuote()) {
509                 parsePos++;
510 
511                 if (!isNextQuote()) {
512                     // Closing single quote;
513                     value = getNoTrailingSpaceString(buf);
514                     return;
515                 }
516             } 
517             buf.append(line.charAt(parsePos));
518         }
519         
520         // String with missing end quote
521         if (FitsFactory.isAllowHeaderRepairs()) {
522             LOG.warning("[" + sanitize(key) + "] Ignored missing end quote (value parsed to end of record).");
523             value = getNoTrailingSpaceString(buf);
524         } else {
525             throw new UnclosedQuoteException(line);
526         }
527     }

529     /**
530      * Returns the inferred Java class for the specified value. See {@link #getInferredType()}
531      * for a more detailed description.
532      * 
533      * @param value     the serialized (string) representation of a FITS header value.
534      * @return          the inferred type of the specified serialized (string) value, or <code>null</code>
535      *                  if the value does not seem to match any of the supported value types. <code>null</code>
536      *                  values default to <code>Boolean.class</code>.
537      */
538     private static Class<?> getInferredValueType(String key, String value) {
539         // TODO We never call this with null locally, so the following check is dead code here...
540 //        if (value == null) {
541 //            return Boolean.class;
542 //        }
543         if (value.isEmpty()) {
544             LOG.warning("[" + sanitize(key) + "] Null non-string value (defaulted to Boolean.class).");
545             return Boolean.class;
546         }
547            
548         String trimmedValue = value.trim().toUpperCase();
549         
550         if ("T".equals(trimmedValue) || "F".equals(trimmedValue)) {
551             return Boolean.class;
552         } else if (INT_REGEX.matcher(trimmedValue).matches()) {
553                 return getIntegerType(trimmedValue); 
554         } else if (DECIMAL_REGEX.matcher(trimmedValue).matches()) {
555             return getDecimalType(trimmedValue); 
556         } else if (HEX_REGEX.matcher(trimmedValue).matches()) {
557             return getIntegerType(trimmedValue);
558         } else if (COMPLEX_REGEX.matcher(trimmedValue).matches()) {
559             return ComplexValue.class;
560         }
561         
562         LOG.warning("[" + sanitize(key) + "] Unrecognised non-string value type '" + sanitize(trimmedValue) + "'.");
563         
564         return null;
565     }

567     /**
568      * Returns the guessed decimal type of a string representation of a decimal value.
569      *
570      * @param value     the string representation of a decimal value.
571      * 
572      * @return the      The Java class ({@link Float}, {@link Double}, or {@link BigDecimal}) 
573      *                  that can be used to represent the value with the precision provided.
574      *                  
575      * @see #getInferredValueType()
576      * @see #getIntegerType(String)
577      */
578     private static Class<? extends Number> getDecimalType(String value) {
579         value = value.toUpperCase(Locale.US);
580         boolean hasD = (value.indexOf('D') >= 0);
581         
582         if (hasD) {
583             // Convert the Double Scientific Notation specified by FITS to pure IEEE.
584             value = value.replace('D', 'E');
585         }
586      
587         BigDecimal big = new BigDecimal(value);
588         
589         // Check for zero, and deal with it separately...
590         if (big.stripTrailingZeros().equals(BigDecimal.ZERO)) {
591             int decimals = big.scale();
592             if (decimals <= FlexFormat.FLOAT_DECIMALS) {
593                 return hasD ? Double.class : Float.class;
594             }
595             if (decimals <= FlexFormat.DOUBLE_DECIMALS) {
596                 return Double.class;
597             }
598             return BigDecimal.class;
599         }
600        
601         // Now non-zero values...
602         int decimals = big.precision() - 1;
603         float f = big.floatValue();        
604         if (decimals <= FlexFormat.FLOAT_DECIMALS && (f != 0.0F) && Float.isFinite(f)) {
605             return hasD ? Double.class : Float.class;
606         } 
607         
608         double d = big.doubleValue();
609         if (decimals <= FlexFormat.DOUBLE_DECIMALS && (d != 0.0) && Double.isFinite(d)) {
610             return Double.class;
611         }
612         return BigDecimal.class;
613     }

615     /**
616      * Returns the guessed integer type of a string representation of a integer value.
617      *
618      * @param value     the string representation of an integer value.
619      * 
620      * @return the      The Java class ({@link Integer}, {@link Long}, or {@link BigInteger})
621      *                  that can be used to represent the value with the number of digits provided.
622      *                  
623      * @see #getInferredValueType()
624      * @see #getDecimalType(String)
625      */
626     private static Class<? extends Number> getIntegerType(String value) { 
627         try {
628             int bits = new BigInteger(value).bitLength();      
629             if (bits < Integer.SIZE) {
630                 return Integer.class;
631             }
632             if (bits < Long.SIZE) {
633                 return Long.class;
634             }   
635             return BigInteger.class;
636         } catch (NumberFormatException e) {
637             // Try hexadecimal...
638             long l = Long.decode("0x" + value);
639             return (l == (int) l) ? Integer.class : Long.class; 
640         }  
641     }

644         return HeaderCard.sanitize(text);
