34  import java.io.IOException;
35  
36  import nom.tam.util.ArrayDataInput;
37  import nom.tam.util.ArrayDataOutput;
38  
39  /**
40   * This interface allows to easily perform basic I/O operations on a FITS
41   * element.
42   */
43  public interface FitsElement {
44  
45      /**
46       * @return the byte at which this element begins. This is only available if
47       *         the data is originally read from a random access medium.
48       */
49      long getFileOffset();
50  
51      /**
52       * @return The size of this element in bytes
53       */
54      long getSize();
55  
56      /**
57       * @deprecated This method is poorly conceived as we cannot really read from 
58       *              just any <code>ArrayDataInput</code> but only those that
59       *              utilize {@link nom.tam.util.FitsDecoder} to convert binary data to
60       *              Java types. As such, this method is inherently unsafe as it can 
61       *              be used to properly interpret FITS files.
62       *              It will be removed in a future release of this library,
63       *              and will be replaced with a new <code>read(FitsDecoder)</code>
64       *              method that offers similar functionality in a safe way. 
65       * 
66       * Read a data array into the current object and if needed position to the
67       * beginning of the next FITS block.
68       * 
69       * @param in
70       *            The input data stream
71       * @throws FitsException
72       *             if the read was unsuccessful.
73       * @throws IOException
74       *             if the read was unsuccessful.
75       */
76      @Deprecated
77      void read(ArrayDataInput in) throws FitsException, IOException;
78  
79      /**
80       * Reset the input stream to point to the beginning of this element
81       * 
82       * @return True if the reset succeeded.
83       */
84      boolean reset();
85  
86      /**
87       * Rewrite the contents of the element in place. The data must have been
88       * originally read from a random access device, and the size of the element
89       * may not have changed.
90       * 
91       * @throws FitsException
92       *             if the rewrite was unsuccessful.
93       * @throws IOException
94       *             if the rewrite was unsuccessful.
95       */
96      void rewrite() throws FitsException, IOException;
97  
98      /**
99       * @return <code>true</code> if this element can be rewritten?
100      */
101     boolean rewriteable();    
102     
103     /**
104      * @deprecated This method is poorly conceived as we cannot really write FITS content to 
105      *              just any <code>ArrayDataOutput</code> but only to ones that utilize
106      *              {@link nom.tam.util.FitsEncoder} to convert Java types to FITS binary
107      *              format. As such, this
108      *              method is inherently unsafe as it can be used to create unreadable FITS files.
109      *              It will be removed from the public API in a future release of this library,
110      *              and will be replaced with a new <code>write(FitsEncoder)</code>
111      *              method that offers similar functionality in a safe way. 
112      * 
113      * Write the contents of the element to a data sink.
114      * 
115      * @param out
116      *            The data sink.
117      * @throws FitsException
118      *             if the write was unsuccessful.
119      * @throws IOException
120      *             if the write was unsuccessful.
121      */
122     @Deprecated
123     void write(ArrayDataOutput out) throws FitsException, IOException;
124 }
