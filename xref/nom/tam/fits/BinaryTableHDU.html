37  /** FITS binary table header/data unit */
38  public class BinaryTableHDU extends TableHDU {
39  
40      private BinaryTable table;
41  
42      /** The standard column keywords for a binary table. */
43      private String[] keyStems = {
44          "TTYPE",
45          "TFORM",
46          "TUNIT",
47          "TNULL",
48          "TSCAL",
49          "TZERO",
50          "TDISP",
51          "TDIM"
52      };

54      public BinaryTableHDU(Header hdr, Data datum) {
55  
56          super((TableData) datum);
57          myHeader = hdr;
58          myData = datum;
59          table = (BinaryTable) datum;
60  
61      }

63      /**
64       * Create data from a binary table header.
65       * 
66       * @param header
67       *            the template specifying the binary table.
68       * @exception FitsException
69       *                if there was a problem with the header.
70       */
71      public static Data manufactureData(Header header) throws FitsException {
72          return new BinaryTable(header);
73      }
74  
75      public Data manufactureData() throws FitsException {
76          return manufactureData(myHeader);
77      }

79      /**
80       * Build a binary table HDU from the supplied data.
81       * 
82       * @param data
83       *            the data used to build the binary table. This is typically
84       *            some kind of array of objects.
85       * @exception FitsException
86       *                if there was a problem with the data.
87       */
88      public static Header manufactureHeader(Data data) throws FitsException {
89          Header hdr = new Header();
90          data.fillHeader(hdr);
91          return hdr;
92      }

94      /** Encapsulate data in a BinaryTable data type */
95      public static Data encapsulate(Object o) throws FitsException {
96  
97          if (o instanceof nom.tam.util.ColumnTable) {
98              return new BinaryTable((nom.tam.util.ColumnTable) o);
99          } else if (o instanceof Object[][]) {
100             return new BinaryTable((Object[][]) o);
101         } else if (o instanceof Object[]) {
102             return new BinaryTable((Object[]) o);
103         } else {
104             throw new FitsException("Unable to encapsulate object of type:" + o.getClass().getName() + " as BinaryTable");
105         }
106     }

108     /**
109      * Check that this is a valid binary table header.
110      * 
111      * @param header
112      *            to validate.
113      * @return <CODE>true</CODE> if this is a binary table header.
114      */
115     public static boolean isHeader(Header header) {
116         String xten = header.getStringValue("XTENSION");
117         if (xten == null) {
118             return false;
119         }
120         xten = xten.trim();
121         if (xten.equals("BINTABLE") || xten.equals("A3DTABLE")) {
122             return true;
123         } else {
124             return false;
125         }
126     }
127 
128     /**
129      * Check that this HDU has a valid header.
130      * 
131      * @return <CODE>true</CODE> if this HDU has a valid header.
132      */
133     public boolean isHeader() {
134         return isHeader(myHeader);
135     }

137     /*
138      * Check if this data object is consistent with a binary table. There are
139      * three options: a column table object, an Object[][], or an Object[]. This
140      * routine doesn't check that the dimensions of arrays are properly
141      * consistent.
142      */
143     public static boolean isData(Object o) {
144 
145         if (o instanceof nom.tam.util.ColumnTable || o instanceof Object[][] || o instanceof Object[]) {
146             return true;
147         } else {
148             return false;
149         }
150     }

152     /**
153      * Add a column without any associated header information.
154      * 
155      * @param data
156      *            The column data to be added. Data should be an Object[] where
157      *            type of all of the constituents is identical. The length of
158      *            data should match the other columns. <b> Note:</b> It is valid
159      *            for data to be a 2 or higher dimensionality primitive array.
160      *            In this case the column index is the first (in Java speak)
161      *            index of the array. E.g., if called with int[30][20][10], the
162      *            number of rows in the table should be 30 and this column will
163      *            have elements which are 2-d integer arrays with TDIM =
164      *            (10,20).
165      * @exception FitsException
166      *                the column could not be added.
167      */
168     public int addColumn(Object data) throws FitsException {
169 
170         int col = table.addColumn(data);
171         table.pointToColumn(getNCols() - 1, myHeader);
172         return col;
173     }

175     // Need to tell header about the Heap before writing.
176     public void write(ArrayDataOutput ado) throws FitsException {
177 
178         int oldSize = myHeader.getIntValue("PCOUNT");
179         if (oldSize != table.getHeapSize()) {
180             myHeader.addValue("PCOUNT", table.getHeapSize(), "ntf::binarytablehdu:pcount:1");
181         }
182 
183         if (myHeader.getIntValue("PCOUNT") == 0) {
184             myHeader.deleteKey("THEAP");
185         } else {
186             myHeader.getIntValue("TFIELDS");
187             int offset = myHeader.getIntValue("NAXIS1") * myHeader.getIntValue("NAXIS2") + table.getHeapOffset();
188             myHeader.addValue("THEAP", offset, "ntf::binarytablehdu:theap:1");
189         }
190 
191         super.write(ado);
192     }

194     /**
195      * Convert a column in the table to complex. Only tables with appropriate
196      * types and dimensionalities can be converted. It is legal to call this on
197      * a column that is already complex.
198      * 
199      * @param index
200      *            The 0-based index of the column to be converted.
201      * @return Whether the column can be converted
202      * @throws FitsException
203      */
204     public boolean setComplexColumn(int index) throws FitsException {
205         boolean status = false;
206         if (table.setComplexColumn(index)) {
207 
208             // No problem with the data. Make sure the header
209             // is right.
210 
211             int[] dimens = table.getDimens()[index];
212             Class base = table.getBases()[index];
213 
214             int dim = 1;
215             String tdim = "";
216             String sep = "";
217             // Don't loop over all values.
218             // The last is the [2] for the complex data.
219             for (int i = 0; i < dimens.length - 1; i += 1) {
220                 dim *= dimens[i];
221                 tdim = dimens[i] + sep + tdim;
222                 sep = ",";
223             }
224             String suffix = "C"; // For complex
225             // Update the TFORMn keyword.
226             if (base == double.class) {
227                 suffix = "M";
228             }
229 
230             // Worry about variable length columns.
231             String prefix = "";
232             if (table.isVarCol(index)) {
233                 prefix = "P";
234                 dim = 1;
235                 if (table.isLongVary(index)) {
236                     prefix = "Q";
237                 }
238             }
239 
240             // Now update the header.
241             myHeader.findCard("TFORM" + (index + 1));
242             HeaderCard hc = myHeader.nextCard();
243             String oldComment = hc.getComment();
244             if (oldComment == null) {
245                 oldComment = "Column converted to complex";
246             }
247             myHeader.addValue("TFORM" + (index + 1), dim + prefix + suffix, oldComment);
248             if (tdim.length() > 0) {
249                 myHeader.addValue("TDIM" + (index + 1), "(" + tdim + ")", "ntf::binarytablehdu:tdimN:1");
250             } else {
251                 // Just in case there used to be a TDIM card that's no longer
252                 // needed.
253                 myHeader.removeCard("TDIM" + (index + 1));
254             }
255             status = true;
256         }
257         return status;
258     }

260     private void prtField(String type, String field) {
261         String val = myHeader.getStringValue(field);
262         if (val != null) {
263             System.out.print(type + '=' + val + "; ");
264         }
265     }

267     /**
268      * Print out some information about this HDU.
269      */
270     public void info() {
271 
272         BinaryTable myData = (BinaryTable) this.myData;
273 
274         System.out.println("  Binary Table");
275         System.out.println("      Header Information:");
276 
277         int nhcol = myHeader.getIntValue("TFIELDS", -1);
278         int nrow = myHeader.getIntValue("NAXIS2", -1);
279         int rowsize = myHeader.getIntValue("NAXIS1", -1);
280 
281         System.out.print("          " + nhcol + " fields");
282         System.out.println(", " + nrow + " rows of length " + rowsize);
283 
284         for (int i = 1; i <= nhcol; i += 1) {
285             System.out.print("           " + i + ":");
286             prtField("Name", "TTYPE" + i);
287             prtField("Format", "TFORM" + i);
288             prtField("Dimens", "TDIM" + i);
289             System.out.println("");
290         }
291 
292         System.out.println("      Data Information:");
293         if (myData == null || table.getNRows() == 0 || table.getNCols() == 0) {
294             System.out.println("         No data present");
295             if (table.getHeapSize() > 0) {
296                 System.out.println("         Heap size is: " + table.getHeapSize() + " bytes");
297             }
298         } else {
299 
300             System.out.println("          Number of rows=" + table.getNRows());
301             System.out.println("          Number of columns=" + table.getNCols());
302             if (table.getHeapSize() > 0) {
303                 System.out.println("          Heap size is: " + table.getHeapSize() + " bytes");
304             }
305             Object[] cols = table.getFlatColumns();
306             for (int i = 0; i < cols.length; i += 1) {
307                 System.out.println("           " + i + ":" + ArrayFuncs.arrayDescription(cols[i]));
308             }
309         }
310     }
