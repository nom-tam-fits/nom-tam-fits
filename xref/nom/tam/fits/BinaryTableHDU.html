76      /**
77       * @return Encapsulate data in a BinaryTable data type
78       * @param o
79       *            data to encapsulate
80       * @throws FitsException
81       *             if the type of the data is not usable as data
82       */
83      public static BinaryTable encapsulate(Object o) throws FitsException {
84          if (o instanceof nom.tam.util.ColumnTable) {
85              return new BinaryTable((nom.tam.util.ColumnTable<?>) o);
86          } else if (o instanceof Object[][]) {
87              return new BinaryTable((Object[][]) o);
88          } else if (o instanceof Object[]) {
89              return new BinaryTable((Object[]) o);
90          } else {
91              throw new FitsException("Unable to encapsulate object of type:" + o.getClass().getName() + " as BinaryTable");
92          }
93      }

95      /*
96       * Check if this data object is consistent with a binary table. There are
97       * three options: a column table object, an Object[][], or an Object[]. This
98       * routine doesn't check that the dimensions of arrays are properly
99       * consistent.
100      */
101     public static boolean isData(Object o) {
102         return o instanceof nom.tam.util.ColumnTable || o instanceof Object[][] || o instanceof Object[];
103     }

105     /**
106      * Check that this is a valid binary table header.
107      *
108      * @param header
109      *            to validate.
110      * @return <CODE>true</CODE> if this is a binary table header.
111      */
112     public static boolean isHeader(Header header) {
113         String xten = header.getStringValue(XTENSION);
114         if (xten == null) {
115             return false;
116         }
117         xten = xten.trim();
118         return xten.equals(XTENSION_BINTABLE) || xten.equals("A3DTABLE");
119     }

165     /**
166      * Print out some information about this HDU.
167      */
168     @Override
169     public void info(PrintStream stream) {
170 
171         BinaryTable myData = this.myData;
172 
173         stream.println("  Binary Table");
174         stream.println("      Header Information:");
175 
176         int nhcol = this.myHeader.getIntValue(TFIELDS, -1);
177         int nrow = this.myHeader.getIntValue(NAXIS2, -1);
178         int rowsize = this.myHeader.getIntValue(NAXIS1, -1);
179 
180         stream.print("          " + nhcol + " fields");
181         stream.println(", " + nrow + " rows of length " + rowsize);
182 
183         for (int i = 1; i <= nhcol; i += 1) {
184             stream.print("           " + i + ":");
185             prtField(stream, "Name", TTYPEn.n(i).key());
186             prtField(stream, "Format", TFORMn.n(i).key());
187             prtField(stream, "Dimens", TDIMn.n(i).key());
188             stream.println("");
189         }
190 
191         stream.println("      Data Information:");
192         if (myData == null || this.myData.getNRows() == 0 || this.myData.getNCols() == 0) {
193             stream.println("         No data present");
194             if (this.myData.getHeapSize() > 0) {
195                 stream.println("         Heap size is: " + this.myData.getHeapSize() + " bytes");
196             }
197         } else {
198 
199             stream.println("          Number of rows=" + this.myData.getNRows());
200             stream.println("          Number of columns=" + this.myData.getNCols());
201             if (this.myData.getHeapSize() > 0) {
202                 stream.println("          Heap size is: " + this.myData.getHeapSize() + " bytes");
203             }
204             Object[] cols = this.myData.getFlatColumns();
205             for (int i = 0; i < cols.length; i += 1) {
206                 stream.println("           " + i + ":" + ArrayFuncs.arrayDescription(cols[i]));
207             }
208         }
209     }

227     /**
228      * Convert a column in the table to complex. Only tables with appropriate
229      * types and dimensionalities can be converted. It is legal to call this on
230      * a column that is already complex.
231      *
232      * @param index
233      *            The 0-based index of the column to be converted.
234      * @return Whether the column can be converted
235      * @throws FitsException
236      *             if the header could not be adapted
237      */
238     public boolean setComplexColumn(int index) throws FitsException {
239         Standard.context(BinaryTable.class);
240         boolean status = false;
241         if (this.myData.setComplexColumn(index)) {
242             // No problem with the data. Make sure the header
243             // is right.
244             BinaryTable.ColumnDesc colDesc = this.myData.getDescriptor(index);
245             int dim = 1;
246             String tdim = "";
247             String sep = "";
248             // Don't loop over all values.
249             // The last is the [2] for the complex data.
250             int[] dimens = colDesc.getDimens();
251             for (int i = 0; i < dimens.length - 1; i += 1) {
252                 dim *= dimens[i];
253                 tdim = dimens[i] + sep + tdim;
254                 sep = ",";
255             }
256             String suffix = "C"; // For complex
257             // Update the TFORMn keyword.
258 
259             if (colDesc.getBase() == double.class) {
260                 suffix = "M";
261             }
262             // Worry about variable length columns.
263             String prefix = "";
264             if (this.myData.getDescriptor(index).isVarying()) {
265                 prefix = "P";
266                 dim = 1;
267                 if (this.myData.getDescriptor(index).isLongVary()) {
268                     prefix = "Q";
269                 }
270             }
271             // Now update the header.
272             this.myHeader.findCard(TFORMn.n(index + 1));
273             HeaderCard hc = this.myHeader.nextCard();
274             String oldComment = hc.getComment();
275             if (oldComment == null) {
276                 oldComment = "Column converted to complex";
277             }
278             this.myHeader.card(TFORMn.n(index + 1)).value(dim + prefix + suffix).comment(oldComment);
279             if (tdim.length() > 0) {
280                 this.myHeader.addValue(TDIMn.n(index + 1), "(" + tdim + ")");
281             } else {
282                 // Just in case there used to be a TDIM card that's no longer
283                 // needed.
284                 this.myHeader.deleteKey(TDIMn.n(index + 1));
285             }
286             status = true;
287         }
288         Standard.context(null);
289         return status;
290     }

292     // Need to tell header about the Heap before writing.
293     @Override
294     public void write(ArrayDataOutput ado) throws FitsException {
295 
296         int oldSize = this.myHeader.getIntValue(PCOUNT);
297         if (oldSize != this.myData.getHeapSize()) {
298             this.myHeader.addValue(PCOUNT, this.myData.getHeapSize());
299         }
300 
301         if (this.myHeader.getIntValue(PCOUNT) == 0) {
302             this.myHeader.deleteKey(THEAP);
303         } else {
304             this.myHeader.getIntValue(TFIELDS);
305             int offset = this.myHeader.getIntValue(NAXIS1) * this.myHeader.getIntValue(NAXIS2) + this.myData.getHeapOffset();
306             this.myHeader.addValue(THEAP, offset);
307         }
308 
309         super.write(ado);
310     }
