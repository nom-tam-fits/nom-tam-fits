39  /**
40   * Converts {@link HeaderCard}s into one or more 80-character wide FITS header records. It is a
41   * replacement for {@link nom.tam.fits.utilities.FitsLineAppender}, which is still available
42   * for external use for backward compatibility, but is no longer used internally in this 
43   * library itself.
44   * 
45   * @author Attila Kovacs
46   * 
47   * @since 1.16
48   */
49  class HeaderCardFormatter {
50      
51      /**
52       * The FITS settings to use, such as support for long strings, support for HIERARCH-style cards, or the
53       * use of 'D' for high-precision exponential values. These settings control when and how header cards
54       * are represented exactly in the FITS header.
55       * 
56       */
57      private FitsSettings settings;
58      
59      /** The length of two single quotes. */
60      private static final int QUOTES_LENGTH = 2;
61      
62      /** 
63       * Character sequence that comes after a value field, and before the comment string in the
64       * header record. While only a '/' character is really required, we like to add spaces around
65       * it for a more pleasing visual of the resulting header record. The space before the
66       * '/' is strongly recommended by the FITS standard.
67       */
68      private static final String COMMENT_PREFIX = " / ";
69      
70      /**
71       * Long string comments should not add a space after the '/', because we want to preserve spaces in continued
72       * long string comments, hece we start the comment immediately after the '/' to ensure that
73       * internal spaces in wrapped comments remain intact and properly accounted for. The space before the
74       * '/' is strongly recommended by the FITS standard.
75       */
76      private static final String LONG_COMMENT_PREFIX = " /";
77      
78      /** 
79       * The alignment position of card comments for a more pleasing visual experience. Comments will be
80       * aligned to this position, provided the lengths of all fields allow for it.
81       */
82      private static final int COMMENT_ALIGN = 30;
83  
84      /**
85       * In older FITS standards there was a requirement that a closing quote for string values may not
86       * come before byte 20 (counted from 1) in the header record. To ensure that, strings need to be
87       * padded with blank spaces to push the closing quote out to that position, if necessary. While
88       * it is no longer required by the current FITS standard, it is possible (or even likely) that
89       * some existing tools rely on the earlier requirement. Therefore, we will abide by the requirements
90       * of the older standard. (In the future, we may make this requirement optional, and controllable
91       * through the API). 
92       */
93      private static final int MIN_STRING_END = 19;
94      
95      /** whatever fits after "CONTINUE  '' /" */
96      private static final int MAX_LONG_END_COMMENT = 68 - LONG_COMMENT_PREFIX.length(); 
97

101      * @param settings      the local FITS settings to use by this card formatter.
102      * 
103      * @see #HeaderCardFormatter()
104      */
105     HeaderCardFormatter(FitsSettings settings) {
106         this.settings = settings;
107     }

109     /**
110      * Converts a {@link HeaderCard} to one or more 80-character wide FITS header records
111      * following the FITS rules, and the various conventions that are allowed by the 
112      * FITS settings with which this card formatter instance was created. 
113      * 
114      * @param card          the header card object
115      * @return              the correspoinding FITS header snipplet, as one or more
116      *                      80-character wide header 'records'.
117      * @throws HierarchNotEnabledException
118      *                      if the cards is a HIERARCH-style card, but support
119      *                      for HIERARCH keywords is not enabled in the FITS settings 
120      *                      used by this formatter.
121      * @throws LongValueException       
122      *                      if the (non-string) value stored in the card cannot fit
123      *                      in the header record.
124      * @throws LongStringsNotEnabledException
125      *                      if the card contains a string value that cannot fit into
126      *                      a single header record, and the use of long string is
127      *                      not enabled in the FITS settings used by this formatter.
128      *                      
129      * @see FitsFactory#setLongStringsEnabled(boolean)
130      */
131     String toString(HeaderCard card) 
132             throws HierarchNotEnabledException, LongValueException, LongStringsNotEnabledException {
133         StringBuffer buf = new StringBuffer(HeaderCard.FITS_HEADER_CARD_SIZE);
134         
135         appendKey(buf, card);
136         
137         int valueStart = appendValue(buf, card);
138         int valueEnd = buf.length();
139         
140         appendComment(buf, card);
141         
142         if (!card.isCommentStyleCard()) {
143             realign(buf, card.isStringValue() ? valueEnd : valueStart, valueEnd);
144         }
145         
146         pad(buf);
147         
148         return HeaderCard.sanitize(new String(buf));
149     }

151     /**
152      * Adds the FITS keyword to the header record (normally at the beginning).
153      * 
154      * @param buf           The string buffer in which we are building the header record.
155      * @param card          The header card to be formatted.
156      * @throws HierarchNotEnabledException      
157      *                      if the card contains a HIERARCH-style long keyword, but support
158      *                      for these has not been enabled in the settings used by this
159      *                      formatter.
160      * @throws LongValueException
161      *                      if the HIERARCH keyword is itself too long to fit on the
162      *                      record without leaving a minimum amount of space for a value.
163      *                      
164      * @see FitsFactory#setUseHierarch(boolean)
165      */
166     private void appendKey(StringBuffer buf, HeaderCard card) throws HierarchNotEnabledException, LongValueException {
167         String key = card.getKey();
168 
169         if (card.hasHierarchKey()) {
170             if (!settings.isUseHierarch()) {
171                 throw new HierarchNotEnabledException(key);
172             }
173             key = settings.getHierarchKeyFormatter().toHeaderString(key);
174             if (key.length() > HeaderCard.MAX_HIERARCH_KEYWORD_LENGTH) {
175                 // Truncate HIERARCH keywords as necessary to fit.
176                 // This is really just a second parachute here. Normally, HeaderCards
177                 // won't allow creation or setting longer keywords...
178                 throw new LongValueException(key, HeaderCard.MAX_HIERARCH_KEYWORD_LENGTH);
179             }            
180         } else {
181             // Just to be certain, we'll make sure base keywords are upper-case, if they
182             // were not already.
183             key = key.toUpperCase();
184         }
185         
186         buf.append(key);
187         
188         padTo(buf, HeaderCard.MAX_KEYWORD_LENGTH);
189     }

192     /**
193      * Adds the FITS value to the header record (normally after the keyword), including the standard "= " assigment marker
194      * in front of it, or the non-standard "=" (without space after) if {@link FitsFactory#setSkipBlankAfterAssign(boolean)}
195      * is set <code>true</code>.
196      * 
197      * @param buf       The string buffer in which we are building the header record.
198      * @param card      The header card to be formatted.
199      * @return          the buffer position at which the appended value starts, or the 
200      *                  last posirtion if a value was not added at all. (This is used
201      *                  for realigning later...)
202      * @throws LongValueException
203      *                  if the card contained a non-string value that is too long to fit
204      *                  in the space available in the current record.
205      * @throws LongStringsNotEnabledException
206      *                  if the card contains a string value that cannot fit into
207      *                  a single header record, and the use of long string is
208      *                  not enabled in the FITS settings used by this formatter.
209      */
210     private int appendValue(StringBuffer buf, HeaderCard card) throws LongValueException, LongStringsNotEnabledException {
211         String value = card.getValue();
212         
213         if (card.isCommentStyleCard()) {
214             // omment-style card. Nothing to do here...
215             return buf.length();
216         }
217 
218         // Add assignment sequence "= "
219         buf.append(getAssignString());
220        
221         if (value == null) {
222             // 'null' value, nothing more to append.
223             return buf.length();
224         }
225         
226         int valueStart = buf.length();        
227 
228         if (card.isStringValue()) {
229             int from = appendQuotedValue(buf, card, 0);
230             while (from < value.length()) {
231                 pad(buf);
232                 buf.append(CONTINUE.key() + "  ");
233                 from += appendQuotedValue(buf, card, from);
234             }
235         // TODO We prevent the creation of cards with longer values, so the following check is dead code here.    
236 //        } else if (value.length() > available) {
237 //            throw new LongValueException(available, card.getKey(), card.getValue());
238         } else {
239             append(buf, value, 0);
240         }
241         
242         return valueStart;
243     }

245     /**
246      * Returns the minimum size of a truncated header comment. When truncating header comments
247      * we should preserve at least the first word of the comment string wholly...
248      * 
249      * @param card      The header card to be formatted.
250      * @return          the length of the first word in the comment string
251      */
252     private int getMinTruncatedCommentSize(HeaderCard card) {
253         String comment = card.getComment();
254         
255         // TODO We check for null before calling, so this is dead code here...
256 //        if (comment == null) {
257 //            return 0;
258 //        }
259         
260         int firstWordLength = comment.indexOf(' ');
261         if (firstWordLength < 0) {
262             firstWordLength = comment.length();
263         }
264         
265         return COMMENT_PREFIX.length() + firstWordLength;
266     }

268     /**
269      * Appends the comment to the header record, or as much of it as possible, but never
270      * less than the first word (at minimum).
271      * 
272      * @param buf       The string buffer in which we are building the header record.
273      * @param card      The header card to be formatted.
274      * @return          <code>true</code> if the comment was fully represented in the
275      *                  record, or <code>false</code> if it was truncated or fully
276      *                  ommitted.
277      */
278     private boolean appendComment(StringBuffer buf, HeaderCard card) {
279         String comment = card.getComment();
280         if (comment == null) {
281             return true;
282         }
283         
284         if (comment.isEmpty()) {
285             return true;
286         }
287         
288         int available = getAvailable(buf);
289         boolean longCommentOK = FitsFactory.isLongStringsEnabled() && card.isStringValue();
290         
291         if (!card.isCommentStyleCard() && longCommentOK) {
292             if (COMMENT_PREFIX.length() + card.getComment().length() > available) {
293                 // No room for a complete regular comment, but we can do a long string comment...
294                 appendLongStringComment(buf, card);
295                 return true;
296             }
297         }
298 
299         if (card.isCommentStyleCard()) {
300             // '  ' instead of '= '
301             available--;
302         } else {    
303             // ' / '
304             available -= COMMENT_PREFIX.length();
305             if (getMinTruncatedCommentSize(card) > available) {
306                 if (!longCommentOK) {
307                     return false;
308                 }
309             }
310         }
311         
312         if (card.isCommentStyleCard()) {
313             buf.append(' ');
314         } else {
315             buf.append(COMMENT_PREFIX);
316         }
317         
318         if (available >= comment.length()) {
319             buf.append(comment);
320             return true;
321         }
322         
323         buf.append(comment.substring(0, available));
324         return false;
325     }

327     /**
328      * Realigns the header record (single records only!) for more pleasing visual appearance
329      * by adding padding after a string value, or before a non-string value, as necessary
330      * to push the comment field to the alignment position, if it's possible without truncating
331      * the existing record.
332      * 
333      * @param buf       The string buffer in which we are building the header record.
334      * @param at        The position at which to insert padding
335      * @param from      The position in the record that is to be pushed to the alignment
336      *                  position.
337      *                  
338      * @return          <code>true</code> if the card was successfully realigned. Otherwise
339      *                  <code>false</code>.
340      */
341     private boolean realign(StringBuffer buf, int at, int from) {
342         if (buf.length() >= HeaderCard.FITS_HEADER_CARD_SIZE) {
343             // No space on first card, or it's a long string card, which stays unaligned.
344             return false;
345         }
346         
347         if (from > COMMENT_ALIGN) {
348             // We are beyond the alignment point already...
349             return false;
350         }
351         
352         return realign(buf, at, from, COMMENT_ALIGN);     
353     }
354     
355     /**
356      * Realigns the header record (single records only!) for more pleasing visual appearance
357      * by adding padding after a string value, or before a non-string value, as necessary
358      * to push the comment field to the specified alignment position, if it's possible without 
359      * truncating the existing record
360      * 
361      * @param buf       The string buffer in which we are building the header record.
362      * @param at        The position at which to insert padding
363      * @param from      The position in the record that is to be pushed to the alignment
364      *                  position.
365      * @param to        The new alignment position.
366      * @return          <code>true</code> if the card was successfully realigned. Otherwise
367      *                  <code>false</code>.
368      */
369     private boolean realign(StringBuffer buf, int at, int from, int to) {
370         int spaces = to - from;
371         
372         if (spaces < 1 || spaces > getAvailable(buf)) {
373             // No space left in card to align the the specified position.
374             return false;
375         }
376         
377         StringBuffer sBuf = new StringBuffer(spaces);
378         while (--spaces >= 0) {
379             sBuf.append(' ');
380         }
381         
382         buf.insert(at, sBuf.toString());
383         
384         return true;
385     }

387     /**
388      * Adds a long string comment. When long strings are enabled, it is possible to fully
389      * preserve a comment of any length after a string value, by wrapping into multiple
390      * records with CONTINUE keywords. Crucially, we will want to do this in a way as to 
391      * preserve internal spaces within the comment, when wrapped into multiple records.
392      * 
393      * @param buf       The string buffer in which we are building the header record.
394      * @param card      The header card to be formatted.
395      */
396     private void appendLongStringComment(StringBuffer buf, HeaderCard card) {     
397         // We can wrap the comment to our delight, with CONTINUE!
398         int iLast = buf.length() - 1;
399         String comment = card.getComment();
400         
401         // We need to amend the last string to end with '&'
402         if (getAvailable(buf) >= LONG_COMMENT_PREFIX.length() + comment.length()) {
403             // We can append the entire comment, easy...
404             buf.append(LONG_COMMENT_PREFIX);
405             append(buf, comment, 0);
406             return;
407         }   
408         
409         // Add '&' to the end of the string value.
410         // appendQuotedValue() must always leave space for it!
411         buf.setCharAt(iLast, '&');
412         buf.append("'");
413         
414         int from = 0;
415         
416         int available = getAvailable(buf);
417         
418         // If there is room for a standard inline comment, then go for it
419         if (available >= COMMENT_PREFIX.length()) {
420             buf.append(COMMENT_PREFIX);
421         } else {
422             // Add a CONTINUE card with an empty string and try again...
423             pad(buf);
424             buf.append(CONTINUE.key() + "  ''");
425             appendComment(buf, card);
426             return;
427         }
428             
429         from = append(buf, comment, 0);
430         
431         // Now add records as needed to write the comment fully...
432         while (from < comment.length()) {
433             pad(buf);
434             buf.append(CONTINUE.key() + "  ");
435             buf.append((comment.length() >= from + MAX_LONG_END_COMMENT) ? "'&'" : "''");
436             buf.append(LONG_COMMENT_PREFIX);
437             from += append(buf, comment, from);
438         }
439     }

441     /**
442      * Appends as many characters as possible from a string, starting at the
443      * specified string position, into the header record. 
444      * 
445      * @param buf       The string buffer in which we are building the header record.
446      * @param text      The string from which to append characters up to the end
447      *                  of the record.
448      * @param from      The starting position in the string
449      * @return          the number of characters deposited into the header
450      *                  record from the string after the starting position.
451      */
452     private int append(StringBuffer buf, String text, int from) {
453         int available = getAvailable(buf);
454 
455         int n = Math.min(available, text.length() - from);
456         if (n < 1) {
457             return 0;
458         }
459     
460         for (int i = 0; i < n; i++) {
461             buf.append(text.charAt(from + i));
462         }
463      
464         return n;        
465     }

467     /**
468      * Appends quoted text from the specified string position, until the end of the string is reached, or 
469      * until the 80-character header record is full. It replaces quotes in the string with doubled quotes,
470      * while making sure that not unclosed quotes are left and there is space for an '&' character
471      * for
472      * 
473      * @param buf       The string buffer in which we are building the header record.
474      * @param card      The header card whose value to quote in the header record.
475      * @param from      The starting position in the string.
476      * @return          the number of characters consumed from the string, which may be different
477      *                  from the number of characters deposited as each single quote in
478      *                  the input string is represented as 2 single quotes in the record.
479      */
480     private int appendQuotedValue(StringBuffer buf, HeaderCard card, int from) {
481         // Always leave room for an extra & character at the end...
482         int available = getAvailable(buf) - QUOTES_LENGTH;
483         
484         // If long strings are enabled leave space for '&' at the end.
485         if (FitsFactory.isLongStringsEnabled() && card.getComment() != null) {
486             if (card.getComment().length() > 0) {
487                 available--;
488             }
489         }
490         
491         String text = card.getValue();
492 
493         // TODO We check for null before calling, so this is dead code here...
494 //        if (text == null) {
495 //            return 0;
496 //        }
497 
498         // The the remaining part of the string fits in the space with the
499         // quoted quotes, then it's easy...
500         if (available >= text.length() - from) {
501             String escaped = text.substring(from).replace("'", "''");
502             if (escaped.length() <= available) {
503                 buf.append('\'');
504                 buf.append(escaped);
505                 
506                 // Earlier versions of the FITS standard required that the closing quote
507                 // does not come before byte 20. It's no longer required but older tools
508                 // may still expect it, so let's conform. This only affects single
509                 // record card, but not continued long strings...
510                 if (buf.length() < MIN_STRING_END) {
511                     padTo(buf, MIN_STRING_END);
512                 }
513                 
514                 buf.append('\'');
515                 
516                 
517                 return text.length() - from;
518             }
519         }
520         
521         if (!FitsFactory.isLongStringsEnabled()) {
522             throw new LongStringsNotEnabledException(card.getKey() + "= " + card.getValue());
523         }
524         
525         // Now, we definitely need space for '&' at the end...
526         available = getAvailable(buf) - QUOTES_LENGTH - 1;
527             
528         // We need room for an '&' character at the end also... 
529         // TODO Again we prevent this ever occuring before we reach this point, so it is dead code...
530 //        if (available < 1) {
531 //            return 0;
532 //        }
533         
534         // Opening quote
535         buf.append("'");
536 
537         // For counting the characters consumed from the input
538         int consumed = 0;
539         
540         for (int i = 0; i < available; i++, consumed++) {
541             // TODO We already know we cannot show the whole string on one line, so this is dead code...
542 //            if (from + i >= text.length()) {
543 //                // Reached end of string;
544 //                break;
545 //            }
546             
547             char c = text.charAt(from + consumed);
548             
549             if (c == '\'') {
550                 // Quoted quotes take up 2 spaces...
551                 i++;
552                 if (i + 1 < available) {
553                     // Only append the quoted quote if there is room for both.
554                     buf.append("''");
555                 } else {
556                     // Otherwise leave the value quote unconsumed.
557                     break;
558                 }
559             } else {
560                 // Append a non-quote character.
561                 buf.append(c);
562             }
563         }
564         
565         // & and Closing quote
566         buf.append("&'");
567         
568         return consumed;        
569     }

571     /**
572      * Adds a specific amount of padding (empty spaces) in the header record. 
573      * 
574      * @param buf   The string buffer in which we are building the header record.
575      * @param n     the number of empty spaces to add.
576      */
577     private void pad(StringBuffer buf, int n) {
578         for (int i = n; --i >= 0;) {
579             buf.append(' ');
580         }
581     }

583     /**
584      * Pads the current header record with empty spaces to up to the end of the 
585      * 80-character record.
586      * 
587      * @param buf   The string buffer in which we are building the header record.
588      */
589     private void pad(StringBuffer buf) {
590         pad(buf, getAvailable(buf));
591     }
592     
593     /**
594      * Adds padding (empty spaces) in the header record, up to the specified
595      * position within the record.
596      * 
597      * @param buf   The string buffer in which we are building the header record.
598      * @param to    The position in the record to which to pad with spaces.
599      */
600     private void padTo(StringBuffer buf, int to) {
601         for (int pos = buf.length() % HeaderCard.FITS_HEADER_CARD_SIZE; pos < to; pos++) {
602             buf.append(' ');
603         }
604     }

606     /**
607      * Returns the number of characters available for remaining fields in the
608      * current record. Empty records will return 0.
609      * 
610      * @param buf   The string buffer in which we are building the header record.
611      * @return      the number of characters still available in the currently started
612      *              80-character header record. Empty records will return 0.
613      */
614     private int getAvailable(StringBuffer buf) {
615         return (HeaderCard.FITS_HEADER_CARD_SIZE - buf.length() % HeaderCard.FITS_HEADER_CARD_SIZE) % HeaderCard.FITS_HEADER_CARD_SIZE;
616     }

618     /**
619      * Returns the assignment string to use between the keyword and the value. The
620      * FITS standard requires the 2-character sequence "= ", but for some reason we
621      * allow to skip the required space after the '=' if {@link FitsFactory#setSkipBlankAfterAssign(boolean)}
622      * is set to <code>true</code>...
623      * 
624      * @return      The character sequence to insert between the keyword and the
625      *              value.
626      *              
627      * @see #getAssignLength()
628      */
629     @SuppressWarnings("deprecation")
630     static String getAssignString() {
631         return FitsFactory.isSkipBlankAfterAssign() ? "=" : "= ";
632     }
633     
634     /**
635      * Returns the number of characters we use for assignment. Normally, it
636      * should be 2 as per FITS standard, but if {@link FitsFactory#setSkipBlankAfterAssign(boolean)}
637      * is set to <code>true</code>, it may be only 1.
638      * 
639      * @return      The number of characters that should be between the keyword and the value
640      *              indicating assignment.
641      *              
642      * @see #getAssignString()
643      */
644     @SuppressWarnings("deprecation")
645     static int getAssignLength() {
646         int n = 1;
647         if (!FitsFactory.isSkipBlankAfterAssign()) {
648             n++;
649         }
650         return n;
651     }
