132     /**
133      * Create an ASCII table given a header
134      *
135      * @param hdr
136      *            The header describing the table
137      * @throws FitsException
138      *             if the operation failed
139      */
140     public AsciiTable(Header hdr) throws FitsException {
141 
142         this.nRows = hdr.getIntValue(NAXIS2);
143         this.nFields = hdr.getIntValue(TFIELDS);
144         this.rowLen = hdr.getIntValue(NAXIS1);
145 
146         this.types = new Class[this.nFields];
147         this.offsets = new int[this.nFields];
148         this.lengths = new int[this.nFields];
149         this.nulls = new String[this.nFields];
150 
151         for (int i = 0; i < this.nFields; i += 1) {
152             this.offsets[i] = hdr.getIntValue(TBCOLn.n(i + 1)) - 1;
153             String s = hdr.getStringValue(TFORMn.n(i + 1));
154             if (this.offsets[i] < 0 || s == null) {
155                 throw new FitsException("Invalid Specification for column:" + (i + 1));
156             }
157             s = s.trim();
158             char c = s.charAt(0);
159             s = s.substring(1);
160             if (s.indexOf('.') > 0) {
161                 s = s.substring(0, s.indexOf('.'));
162             }
163             this.lengths[i] = Integer.parseInt(s);
164 
165             switch (c) {
166                 case 'A':
167                     this.types[i] = String.class;
168                     break;
169                 case 'I':
170                     if (this.lengths[i] > MAX_INTEGER_LENGTH) {
171                         this.types[i] = long.class;
172                     } else {
173                         this.types[i] = int.class;
174                     }
175                     break;
176                 case 'F':
177                 case 'E':
178                     this.types[i] = float.class;
179                     break;
180                 case 'D':
181                     this.types[i] = double.class;
182                     break;
183                 default:
184                     throw new FitsException("could not parse column type of ascii table");
185             }
186 
187             this.nulls[i] = hdr.getStringValue(TNULLn.n(i + 1));
188             if (this.nulls[i] != null) {
189                 this.nulls[i] = this.nulls[i].trim();
190             }
191         }
192     }

194     int addColInfo(int col, Cursor<String, HeaderCard> iter) throws HeaderCardException {
195 
196         String tform = null;
197         if (this.types[col] == String.class) {
198             tform = "A" + this.lengths[col];
199         } else if (this.types[col] == int.class || this.types[col] == long.class) {
200             tform = "I" + this.lengths[col];
201         } else if (this.types[col] == float.class) {
202             tform = "E" + this.lengths[col] + ".0";
203         } else if (this.types[col] == double.class) {
204             tform = "D" + this.lengths[col] + ".0";
205         }
206         Standard.context(AsciiTable.class);
207         IFitsHeader key = TFORMn.n(col + 1);
208         iter.add(new HeaderCard(key.key(), tform, key.comment()));
209         key = TBCOLn.n(col + 1);
210         iter.add(new HeaderCard(key.key(), this.offsets[col] + 1, key.comment()));
211         Standard.context(null);
212         return this.lengths[col];
213     }

215     @Override
216     public int addColumn(Object newCol) throws FitsException {
217         int maxLen = 1;
218         if (newCol instanceof String[]) {
219 
220             String[] sa = (String[]) newCol;
221             for (String element : sa) {
222                 if (element != null && element.length() > maxLen) {
223                     maxLen = element.length();
224                 }
225             }
226         } else if (newCol instanceof double[]) {
227             maxLen = DOUBLE_MAX_LENGTH;
228         } else if (newCol instanceof int[]) {
229             maxLen = INT_MAX_LENGTH;
230         } else if (newCol instanceof long[]) {
231             maxLen = LONG_MAX_LENGTH;
232         } else if (newCol instanceof float[]) {
233             maxLen = FLOAT_MAX_LENGTH;
234         } else {
235             throw new FitsException("Adding invalid type to ASCII table");
236         }
237         addColumn(newCol, maxLen);
238 
239         // Invalidate the buffer
240         this.buffer = null;
241 
242         return this.nFields;
243     }

245     /**
246      * This version of addColumn allows the user to override the default length
247      * associated with each column type.
248      *
249      * @param newCol
250      *            The new column data
251      * @param length
252      *            the requested length for the column
253      * @return the number of columns after this one is added.
254      * @throws FitsException
255      *             if the operation failed
256      */
257     public int addColumn(Object newCol, int length) throws FitsException {
258 
259         if (this.nFields > 0 && Array.getLength(newCol) != this.nRows) {
260             throw new FitsException("New column has different number of rows");
261         }
262 
263         if (this.nFields == 0) {
264             this.nRows = Array.getLength(newCol);
265         }
266 
267         Object[] newData = new Object[this.nFields + 1];
268         int[] newOffsets = new int[this.nFields + 1];
269         int[] newLengths = new int[this.nFields + 1];
270         Class<?>[] newTypes = new Class[this.nFields + 1];
271         String[] newNulls = new String[this.nFields + 1];
272 
273         System.arraycopy(this.data, 0, newData, 0, this.nFields);
274         System.arraycopy(this.offsets, 0, newOffsets, 0, this.nFields);
275         System.arraycopy(this.lengths, 0, newLengths, 0, this.nFields);
276         System.arraycopy(this.types, 0, newTypes, 0, this.nFields);
277         System.arraycopy(this.nulls, 0, newNulls, 0, this.nFields);
278 
279         this.data = newData;
280         this.offsets = newOffsets;
281         this.lengths = newLengths;
282         this.types = newTypes;
283         this.nulls = newNulls;
284 
285         newData[this.nFields] = newCol;
286         this.offsets[this.nFields] = this.rowLen + 1;
287         this.lengths[this.nFields] = length;
288         this.types[this.nFields] = ArrayFuncs.getBaseClass(newCol);
289 
290         this.rowLen += length + 1;
291         if (this.isNull != null) {
292             boolean[] newIsNull = new boolean[this.nRows * (this.nFields + 1)];
293             // Fix the null pointers.
294             int add = 0;
295             for (int i = 0; i < this.isNull.length; i += 1) {
296                 if (i % this.nFields == 0) {
297                     add += 1;
298                 }
299                 if (this.isNull[i]) {
300                     newIsNull[i + add] = true;
301                 }
302             }
303             this.isNull = newIsNull;
304         }
305         this.nFields += 1;
306 
307         // Invalidate the buffer
308         this.buffer = null;
309 
310         return this.nFields;
311     }

313     @Override
314     public int addRow(Object[] newRow) throws FitsException {
315         try {
316             // If there are no fields, then this is the
317             // first row. We need to add in each of the columns
318             // to get the descriptors set up.
319             if (this.nFields == 0) {
320                 for (Object element : newRow) {
321                     addColumn(element);
322                 }
323             } else {
324                 for (int i = 0; i < this.nFields; i += 1) {
325                     Object o = ArrayFuncs.newInstance(this.types[i], this.nRows + 1);
326                     System.arraycopy(this.data[i], 0, o, 0, this.nRows);
327                     System.arraycopy(newRow[i], 0, o, this.nRows, 1);
328                     this.data[i] = o;
329                 }
330                 this.nRows += 1;
331             }
332             // Invalidate the buffer
333             this.buffer = null;
334             return this.nRows;
335         } catch (FitsException e) {
336             throw e;
337         } catch (Exception e) {
338             throw new FitsException("Error deleting row:" + e.getMessage(), e);
339         }
340     }

342     /**
343      * Delete columns from the table.
344      *
345      * @param start
346      *            The first, 0-indexed, column to be deleted.
347      * @param len
348      *            The number of columns to be deleted.
349      * @throws FitsException
350      *             if the operation failed
351      */
352 
353     @Override
354     public void deleteColumns(int start, int len) throws FitsException {
355         ensureData();
356 
357         Object[] newData = new Object[this.nFields - len];
358         int[] newOffsets = new int[this.nFields - len];
359         int[] newLengths = new int[this.nFields - len];
360         Class<?>[] newTypes = new Class[this.nFields - len];
361         String[] newNulls = new String[this.nFields - len];
362 
363         // Copy in the initial stuff...
364         System.arraycopy(this.data, 0, newData, 0, start);
365         // Don't do the offsets here.
366         System.arraycopy(this.lengths, 0, newLengths, 0, start);
367         System.arraycopy(this.types, 0, newTypes, 0, start);
368         System.arraycopy(this.nulls, 0, newNulls, 0, start);
369 
370         // Copy in the final
371         System.arraycopy(this.data, start + len, newData, start, this.nFields - start - len);
372         // Don't do the offsets here.
373         System.arraycopy(this.lengths, start + len, newLengths, start, this.nFields - start - len);
374         System.arraycopy(this.types, start + len, newTypes, start, this.nFields - start - len);
375         System.arraycopy(this.nulls, start + len, newNulls, start, this.nFields - start - len);
376 
377         for (int i = start; i < start + len; i += 1) {
378             this.rowLen -= this.lengths[i] + 1;
379         }
380 
381         this.data = newData;
382         this.offsets = newOffsets;
383         this.lengths = newLengths;
384         this.types = newTypes;
385         this.nulls = newNulls;
386 
387         if (this.isNull != null) {
388             boolean found = false;
389 
390             boolean[] newIsNull = new boolean[this.nRows * (this.nFields - len)];
391             for (int i = 0; i < this.nRows; i += 1) {
392                 int oldOff = this.nFields * i;
393                 int newOff = (this.nFields - len) * i;
394                 for (int col = 0; col < start; col += 1) {
395                     newIsNull[newOff + col] = this.isNull[oldOff + col];
396                     found = found || this.isNull[oldOff + col];
397                 }
398                 for (int col = start + len; col < this.nFields; col += 1) {
399                     newIsNull[newOff + col - len] = this.isNull[oldOff + col];
400                     found = found || this.isNull[oldOff + col];
401                 }
402             }
403             if (found) {
404                 this.isNull = newIsNull;
405             } else {
406                 this.isNull = null;
407             }
408         }
409 
410         // Invalidate the buffer
411         this.buffer = null;
412 
413         this.nFields -= len;
414     }

416     /**
417      * Delete rows from a FITS table
418      *
419      * @param start
420      *            The first (0-indexed) row to be deleted.
421      * @param len
422      *            The number of rows to be deleted.
423      * @throws FitsException
424      *             if the operation failed
425      */
426 
427     @Override
428     public void deleteRows(int start, int len) throws FitsException {
429         try {
430             if (this.nRows == 0 || start < 0 || start >= this.nRows || len <= 0) {
431                 return;
432             }
433             if (start + len > this.nRows) {
434                 len = this.nRows - start;
435             }
436             ensureData();
437             for (int i = 0; i < this.nFields; i += 1) {
438                 Object o = ArrayFuncs.newInstance(this.types[i], this.nRows - len);
439                 System.arraycopy(this.data[i], 0, o, 0, start);
440                 System.arraycopy(this.data[i], start + len, o, start, this.nRows - len - start);
441                 this.data[i] = o;
442             }
443             this.nRows -= len;
444         } catch (FitsException e) {
445             throw e;
446         } catch (Exception e) {
447             throw new FitsException("Error deleting row:" + e.getMessage(), e);
448         }
449     }

462     /**
463      * Move an element from the buffer into a data array.
464      *
465      * @param offset
466      *            The offset within buffer at which the element starts.
467      * @param length
468      *            The number of bytes in the buffer for the element.
469      * @param array
470      *            An array of objects, each of which is a simple array.
471      * @param col
472      *            Which element of array is to be modified?
473      * @param row
474      *            Which index into that element is to be modified?
475      * @param nullFld
476      *            What string signifies a null element?
477      * @throws FitsException
478      *             if the operation failed
479      */
480     private boolean extractElement(int offset, int length, Object[] array, int col, int row, String nullFld) throws FitsException {
481 
482         this.bp.setOffset(offset);
483 
484         if (nullFld != null) {
485             String s = this.bp.getString(length);
486             if (s.trim().equals(nullFld)) {
487                 return false;
488             }
489             this.bp.skip(-length);
490         }
491         try {
492             if (array[col] instanceof String[]) {
493                 ((String[]) array[col])[row] = this.bp.getString(length);
494             } else if (array[col] instanceof int[]) {
495                 ((int[]) array[col])[row] = this.bp.getInt(length);
496             } else if (array[col] instanceof float[]) {
497                 ((float[]) array[col])[row] = this.bp.getFloat(length);
498             } else if (array[col] instanceof double[]) {
499                 ((double[]) array[col])[row] = this.bp.getDouble(length);
500             } else if (array[col] instanceof long[]) {
501                 ((long[]) array[col])[row] = this.bp.getLong(length);
502             } else {
503                 throw new FitsException("Invalid type for ASCII table conversion:" + array[col]);
504             }
505         } catch (FormatException e) {
506             throw new FitsException("Error parsing data at row,col:" + row + "," + col + "  " + e);
507         }
508         return true;
509     }

511     /**
512      * Fill in a header with information that points to this data.
513      *
514      * @param hdr
515      *            The header to be updated with information appropriate to the
516      *            current table data.
517      */
518 
519     @Override
520     public void fillHeader(Header hdr) {
521         try {
522             Standard.context(AsciiTable.class);
523             hdr.setXtension("TABLE");
524             hdr.setBitpix(BasicHDU.BITPIX_BYTE);
525             hdr.setNaxes(2);
526             hdr.setNaxis(1, this.rowLen);
527             hdr.setNaxis(2, this.nRows);
528             Cursor<String, HeaderCard> iter = hdr.iterator();
529             iter.setKey(NAXIS2.key());
530             iter.next();
531             iter.add(new HeaderCard(PCOUNT.key(), 0, PCOUNT.comment()));
532             iter.add(new HeaderCard(GCOUNT.key(), 1, GCOUNT.comment()));
533             iter.add(new HeaderCard(TFIELDS.key(), this.nFields, TFIELDS.comment()));
534 
535             for (int i = 0; i < this.nFields; i += 1) {
536                 addColInfo(i, iter);
537             }
538         } catch (HeaderCardException e) {
539             LOG.log(Level.SEVERE, "ImpossibleException in fillHeader:" + e.getMessage(), e);
540         } finally {
541             Standard.context(null);
542         }
543     }

545     /**
546      * Read some data into the buffer.
547      */
548     private void getBuffer(int size, long offset) throws IOException, FitsException {
549 
550         if (this.currInput == null) {
551             throw new IOException("No stream open to read");
552         }
553 
554         this.buffer = new byte[size];
555         if (offset != 0) {
556             FitsUtil.reposition(this.currInput, offset);
557         }
558         this.currInput.readFully(this.buffer);
559         this.bp = new ByteParser(this.buffer);
560     }

578     /**
579      * Get the ASCII table information. This will actually do the read if it had
580      * previously been deferred
581      *
582      * @return The table data as an Object[] array.
583      * @throws FitsException
584      *             if the operation failed
585      */
586     @Override
587     @SuppressFBWarnings(value = "EI_EXPOSE_REP", justification = "intended exposure of mutable data")
588     public Object getData() throws FitsException {
589 
590         if (this.data == null) {
591 
592             this.data = new Object[this.nFields];
593 
594             for (int i = 0; i < this.nFields; i += 1) {
595                 this.data[i] = ArrayFuncs.newInstance(this.types[i], this.nRows);
596             }
597 
598             if (this.buffer == null) {
599                 long newOffset = FitsUtil.findOffset(this.currInput);
600                 try {
601                     getBuffer(this.nRows * this.rowLen, this.fileOffset);
602 
603                 } catch (IOException e) {
604                     throw new FitsException("Error in deferred read -- file closed prematurely?:" + e.getMessage(), e);
605                 }
606                 FitsUtil.reposition(this.currInput, newOffset);
607             }
608 
609             this.bp.setOffset(0);
610 
611             int rowOffset;
612             for (int i = 0; i < this.nRows; i += 1) {
613                 rowOffset = this.rowLen * i;
614                 for (int j = 0; j < this.nFields; j += 1) {
615                     if (!extractElement(rowOffset + this.offsets[j], this.lengths[j], this.data, j, i, this.nulls[j])) {
616                         if (this.isNull == null) {
617                             this.isNull = new boolean[this.nRows * this.nFields];
618                         }
619 
620                         this.isNull[j + i * this.nFields] = true;
621                     }
622                 }
623             }
624         }
625         return this.data;
626     }

730     /**
731      * Read a single element from the table. This returns an array of dimension
732      * 1.
733      *
734      * @throws FitsException
735      *             if the operation failed
736      */
737     private Object parseSingleElement(int row, int col) throws FitsException {
738 
739         Object[] res = new Object[1];
740         try {
741             getBuffer(this.lengths[col], this.fileOffset + (long) row * (long) this.rowLen + this.offsets[col]);
742         } catch (IOException e) {
743             this.buffer = null;
744             throw new FitsException("Unable to read element");
745         }
746         res[0] = ArrayFuncs.newInstance(this.types[col], 1);
747 
748         if (extractElement(0, this.lengths[col], res, 0, 0, this.nulls[col])) {
749             this.buffer = null;
750             return res[0];
751 
752         } else {
753 
754             this.buffer = null;
755             return null;
756         }
757     }

759     /**
760      * Read a single row from the table. This returns a set of arrays of
761      * dimension 1.
762      *
763      * @throws FitsException
764      *             if the operation failed
765      */
766     private Object[] parseSingleRow(int row) throws FitsException {
767 
768         Object[] res = new Object[this.nFields];
769 
770         try {
771             getBuffer(this.rowLen, this.fileOffset + (long) row * (long) this.rowLen);
772         } catch (IOException e) {
773             throw new FitsException("Unable to read row");
774         }
775 
776         for (int i = 0; i < this.nFields; i += 1) {
777             res[i] = ArrayFuncs.newInstance(this.types[i], 1);
778             if (!extractElement(this.offsets[i], this.lengths[i], res, i, 0, this.nulls[i])) {
779                 res[i] = null;
780             }
781         }
782 
783         // Invalidate buffer for future use.
784         this.buffer = null;
785         return res;
786     }

788     /**
789      * Read in an ASCII table. Reading is deferred if we are reading from a
790      * random access device
791      *
792      * @param str
793      *            the stream to read from
794      * @throws FitsException
795      *             if the operation failed
796      */
797 
798     @Override
799     public void read(ArrayDataInput str) throws FitsException {
800         try {
801             setFileOffset(str);
802             this.currInput = str;
803             if (str instanceof RandomAccess) {
804                 str.skipAllBytes((long) this.nRows * this.rowLen);
805             } else {
806                 if ((long) this.rowLen * this.nRows > Integer.MAX_VALUE) {
807                     throw new FitsException("Cannot read ASCII table > 2 GB");
808                 }
809                 getBuffer(this.rowLen * this.nRows, 0);
810             }
811             str.skipAllBytes(FitsUtil.padding(this.nRows * this.rowLen));
812         } catch (EOFException e) {
813             throw new PaddingException("EOF skipping padding after ASCII Table", this, e);
814         } catch (IOException e) {
815             throw new FitsException("Error skipping padding after ASCII Table", e);
816         }
817     }

871     /**
872      * Mark (or unmark) an element as null. Note that if this FITS file is
873      * latter written out, a TNULL keyword needs to be defined in the
874      * corresponding header. This routine does not add an element for String
875      * columns.
876      *
877      * @param row
878      *            The 0-based row.
879      * @param col
880      *            The 0-based column.
881      * @param flag
882      *            True if the element is to be set to null.
883      */
884     public void setNull(int row, int col, boolean flag) {
885         if (flag) {
886             if (this.isNull == null) {
887                 this.isNull = new boolean[this.nRows * this.nFields];
888             }
889             this.isNull[col + row * this.nFields] = true;
890         } else if (this.isNull != null) {
891             this.isNull[col + row * this.nFields] = false;
892         }
893 
894         // Invalidate the buffer
895         this.buffer = null;
896     }

970     /**
971      * This is called after we delete columns. The HDU doesn't know how to
972      * update the TBCOL entries.
973      *
974      * @param oldNCol
975      *            The number of columns we had before deletion.
976      * @param hdr
977      *            The associated header. @throws FitsException if the operation
978      *            failed
979      */
980 
981     @Override
982     public void updateAfterDelete(int oldNCol, Header hdr) throws FitsException {
983 
984         int offset = 0;
985         for (int i = 0; i < this.nFields; i += 1) {
986             this.offsets[i] = offset;
987             hdr.addValue(TBCOLn.n(i + 1), offset + 1);
988             offset += this.lengths[i] + 1;
989         }
990         for (int i = this.nFields; i < oldNCol; i += 1) {
991             hdr.deleteKey(TBCOLn.n(i + 1));
992         }
993 
994         hdr.addValue(NAXIS1, this.rowLen);
995     }

997     /**
998      * Write the data to an output stream.
999      *
1000      * @param str
1001      *            The output stream to be written to
1002      * @throws FitsException
1003      *             if any IO exception is found or some inconsistency the FITS
1004      *             file arises.
1005      */
1006 
1007     @Override
1008     public void write(ArrayDataOutput str) throws FitsException {
1009         // Make sure we have the data in hand.
1010         ensureData();
1011         // If buffer is still around we can just reuse it,
1012         // since nothing we've done has invalidated it.
1013 
1014         if (this.buffer == null) {
1015 
1016             if (this.data == null) {
1017                 throw new FitsException("Attempt to write undefined ASCII Table");
1018             }
1019 
1020             if ((long) this.nRows * this.rowLen > Integer.MAX_VALUE) {
1021                 throw new FitsException("Cannot write ASCII table > 2 GB");
1022             }
1023 
1024             this.buffer = new byte[this.nRows * this.rowLen];
1025 
1026             this.bp = new ByteParser(this.buffer);
1027             for (int i = 0; i < this.buffer.length; i += 1) {
1028                 this.buffer[i] = (byte) ' ';
1029             }
1030 
1031             ByteFormatter bf = new ByteFormatter();
1032 
1033             for (int i = 0; i < this.nRows; i += 1) {
1034 
1035                 for (int j = 0; j < this.nFields; j += 1) {
1036                     int offset = i * this.rowLen + this.offsets[j];
1037                     int len = this.lengths[j];
1038                     if (this.isNull != null && this.isNull[i * this.nFields + j]) {
1039                         if (this.nulls[j] == null) {
1040                             throw new FitsException("No null value set when needed");
1041                         }
1042                         bf.format(this.nulls[j], this.buffer, offset, len);
1043                     } else {
1044                         if (this.types[j] == String.class) {
1045                             String[] s = (String[]) this.data[j];
1046                             bf.format(s[i], this.buffer, offset, len);
1047                         } else if (this.types[j] == int.class) {
1048                             int[] ia = (int[]) this.data[j];
1049                             bf.format(ia[i], this.buffer, offset, len);
1050                         } else if (this.types[j] == float.class) {
1051                             float[] fa = (float[]) this.data[j];
1052                             bf.format(fa[i], this.buffer, offset, len);
1053                         } else if (this.types[j] == double.class) {
1054                             double[] da = (double[]) this.data[j];
1055                             bf.format(da[i], this.buffer, offset, len);
1056                         } else if (this.types[j] == long.class) {
1057                             long[] la = (long[]) this.data[j];
1058                             bf.format(la[i], this.buffer, offset, len);
1059                         }
1060                     }
1061                 }
1062             }
1063         }
1064 
1065         // Now write the buffer.
1066         try {
1067             str.write(this.buffer);
1068             FitsUtil.pad(str, this.buffer.length, (byte) ' ');
1069         } catch (IOException e) {
1070             throw new FitsException("Error writing ASCII Table data");
1071         }
1072     }
