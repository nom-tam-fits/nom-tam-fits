61  /**
62   * ASCII table data. ASCII tables are meant for human readability without any special tools. However, they are far less
63   * flexible or compact than {@link BinaryTable}. As such, users are generally discouraged from using this type of table
64   * to represent FITS table data. This class only supports scalar entries of type <code>int</code>, <code>long</code>,
65   * <code>float</code>, <code>double</code>, or else <code>String</code> types.

67   * @see AsciiTableHDU
68   * @see BinaryTable

156     /**
157      * <p>
158      * Create an ASCII table given a header, with custom integer handling support.
159      * </p>
160      * <p>
161      * The <code>preferInt</code> parameter controls how columns with format "<code>I10</code>" are handled; this is
162      * tricky because some, but not all, integers that can be represented in 10 characters can be represented as 32-bit
163      * integers. Setting it <code>true</code> may make it more likely to avoid unexpected type changes during
164      * round-tripping, but it also means that some (large number) data in I10 columns may be impossible to read.
165      * </p>
166      * 
167      * @param      hdr           The header describing the table
168      * @param      preferInt     if <code>true</code>, format "I10" columns will be assumed <code>int.class</code>,
169      *                               provided TLMINn/TLMAXn or TDMINn/TDMAXn limits (if defined) allow it. if
170      *                               <code>false</code>, I10 columns that have no clear indication of data range will be
171      *                               assumed <code>long.class</code>.
172      *
173      * @throws     FitsException if the operation failed
174      * 
175      * @deprecated               Use {@link #setI10PreferInt(boolean)} instead prior to reading ASCII tables.
176      */
177     public AsciiTable(Header hdr, boolean preferInt) throws FitsException {
178         String ext = hdr.getStringValue(Standard.XTENSION, Standard.XTENSION_IMAGE);
179 
180         if (!ext.equalsIgnoreCase(Standard.XTENSION_ASCIITABLE)) {
181             throw new FitsException("Not an ASCII table header (XTENSION = " + hdr.getStringValue(Standard.XTENSION) + ")");
182         }
183 
184         nRows = hdr.getIntValue(NAXIS2);
185         nFields = hdr.getIntValue(TFIELDS);
186         rowLen = hdr.getIntValue(NAXIS1);
187 
188         types = new Class[nFields];
189         offsets = new int[nFields];
190         lengths = new int[nFields];
191         nulls = new String[nFields];
192 
193         for (int i = 0; i < nFields; i++) {
194             offsets[i] = hdr.getIntValue(TBCOLn.n(i + 1)) - 1;
195             String s = hdr.getStringValue(TFORMn.n(i + 1));
196             if (offsets[i] < 0 || s == null) {
197                 throw new FitsException("Invalid Specification for column:" + (i + 1));
198             }
199             s = s.trim();
200             char c = s.charAt(0);
201             s = s.substring(1);
202             if (s.indexOf('.') > 0) {
203                 s = s.substring(0, s.indexOf('.'));
204             }
205             lengths[i] = Integer.parseInt(s);
206 
207             switch (c) {
208             case 'A':
209                 types[i] = String.class;
210                 break;
211             case 'I':
212                 if (lengths[i] == MAX_INTEGER_LENGTH) {
213                     types[i] = guessI10Type(i, hdr, preferInt);
214                 } else {
215                     types[i] = lengths[i] > MAX_INTEGER_LENGTH ? long.class : int.class;
216                 }
217                 break;
218             case 'F':
219             case 'E':
220                 types[i] = float.class;
221                 break;
222             case 'D':
223                 types[i] = double.class;
224                 break;
225             default:
226                 throw new FitsException("could not parse column type of ascii table");
227             }
228 
229             nulls[i] = hdr.getStringValue(TNULLn.n(i + 1));
230             if (nulls[i] != null) {
231                 nulls[i] = nulls[i].trim();
232             }
233         }
234     }

236     /**
237      * Creates an ASCII table from existing data in column-major format order.
238      *
239      * @param  columns       The data for scalar-valued columns. Each column must be an array of <code>int[]</code>,
240      *                           <code>long[]</code>, <code>float[]</code>, <code>double[]</code>, or else
241      *                           <code>String[]</code>, containing the same number of elements in each column (the
242      *                           number of rows).
243      * 
244      * @return               a new ASCII table with the data. The tables data may be partially independent from the
245      *                           argument. Modifications to the table data, or that to the argument have undefined
246      *                           effect on the other object. If it is important to decouple them, you can use a
247      *                           {@link ArrayFuncs#deepClone(Object)} of your original data as an argument.
248      * 
249      * @throws FitsException if the argument is not a suitable representation of FITS data in columns
250      * 
251      * @see                  BinaryTable#fromColumnMajor(Object[])
252      * 
253      * @since                1.19
254      */
255     public static AsciiTable fromColumnMajor(Object[] columns) throws FitsException {
256         AsciiTable t = new AsciiTable();
257         for (int i = 0; i < columns.length; i++) {
258             try {
259                 t.addColumn(columns[i]);
260             } catch (Exception e) {
261                 throw new FitsException("col[" + i + "]: " + e.getMessage(), e);
262             }
263         }
264         return t;
265     }

267     /**
268      * Checks if the integer value of a specific key requires <code>long</code> value type to store.
269      *
270      * @param  h   the header
271      * @param  key the keyword to check
272      *
273      * @return     <code>true</code> if the keyword exists and has an integer value that is outside the range of
274      *                 <code>int</code>. Otherwise <code>false</code>
275      *
276      * @see        #guessI10Type(int, Header, boolean)
277      */
278     private boolean requiresLong(Header h, IFitsHeader key, Long dft) {
279         long l = h.getLongValue(key, dft);
280         if (l == dft) {
281             return false;
282         }
283 
284         return (l < Integer.MIN_VALUE || l > Integer.MAX_VALUE);
285     }

287     /**
288      * Guesses what type of values to use to return I10 type table values. Depending on the range of represented values
289      * I10 may fit into <code>int</code> types, or else require <code>long</code> type arrays. Therefore, the method
290      * checks for the presence of standard column limit keywords TLMINn/TLMAXn and TDMINn/TDMAXn and if these exist and
291      * are outside of the range of an <code>int</code> then the call will return <code>long.class</code>. If the header
292      * does not define the data limits (fully), it will return the class the caller prefers. Otherwise (data limits were
293      * defined and fit into the <code>int</code> range) <code>int.class</code> will be returned.
294      *
295      * @param  col       the 0-based table column index
296      * @param  h         the header
297      * @param  preferInt whether we prefer <code>int.class</code> over <code>long.class</code> in case the header does
298      *                       not provide us with a clue.
299      *
300      * @return           <code>long.class</code> if the data requires long or we prefer it. Othwerwise
301      *                       <code>int.class</code>
302      *
303      * @see              #AsciiTable(Header, boolean)
304      */
305     private Class<?> guessI10Type(int col, Header h, boolean preferInt) {
306         col++;
307 
308         if (requiresLong(h, TLMINn.n(col), Long.MAX_VALUE) || requiresLong(h, TLMAXn.n(col), Long.MIN_VALUE)
309                 || requiresLong(h, TDMINn.n(col), Long.MAX_VALUE) || requiresLong(h, TDMAXn.n(col), Long.MIN_VALUE)) {
310             return long.class;
311         }
312 
313         if ((h.containsKey(TLMINn.n(col)) || h.containsKey(TDMINn.n(col))) //
314                 && (h.containsKey(TLMAXn.n(col)) || h.containsKey(TDMAXn.n(col)))) {
315             // There are keywords defining both min/max values, and none of them require long types...
316             return int.class;
317         }
318 
319         return preferInt ? int.class : long.class;
320     }

335     int addColInfo(int col, Cursor<String, HeaderCard> iter) {
336         String tform = null;
337         if (types[col] == String.class) {
338             tform = "A" + lengths[col];
339         } else if (types[col] == int.class || types[col] == long.class) {
340             tform = "I" + lengths[col];
341         } else if (types[col] == float.class) {
342             tform = "E" + lengths[col] + ".0";
343         } else if (types[col] == double.class) {
344             tform = "D" + lengths[col] + ".0";
345         }
346 
347         Standard.context(AsciiTable.class);
348         iter.add(HeaderCard.create(Standard.TFORMn.n(col + 1), tform));
349         iter.add(HeaderCard.create(Standard.TBCOLn.n(col + 1), offsets[col] + 1));
350         Standard.context(null);
351         return lengths[col];
352     }

354     @Override
355     public int addColumn(Object newCol) throws FitsException, IllegalArgumentException {
356         if (newCol == null) {
357             throw new FitsException("data is null");
358         }
359 
360         if (!newCol.getClass().isArray()) {
361             throw new IllegalArgumentException("Not an array: " + newCol.getClass().getName());
362         }
363 
364         int maxLen = 1;
365         if (newCol instanceof String[]) {
366             String[] sa = (String[]) newCol;
367             for (String element : sa) {
368                 if (element != null && element.length() > maxLen) {
369                     maxLen = element.length();
370                 }
371             }
372         } else if (newCol instanceof double[]) {
373             maxLen = DOUBLE_MAX_LENGTH;
374         } else if (newCol instanceof int[]) {
375             maxLen = INT_MAX_LENGTH;
376         } else if (newCol instanceof long[]) {
377             maxLen = LONG_MAX_LENGTH;
378         } else if (newCol instanceof float[]) {
379             maxLen = FLOAT_MAX_LENGTH;
380         } else {
381             throw new FitsException(
382                     "No AsciiTable support for elements of " + newCol.getClass().getComponentType().getName());
383         }
384         addColumn(newCol, maxLen);
385 
386         // Invalidate the buffer
387         buffer = null;
388 
389         return nFields;
390     }

392     /**
393      * Adds an ASCII table column with the specified ASCII text width for storing its elements.
394      *
395      * @param  newCol                   The new column data, which must be one of: <code>int[]</code>,
396      *                                      <code>long[]</code>, <code>float[]</code>, <code>double[]</code>, or else
397      *                                      <code>String[]</code>. If the table already contains data, the length of the
398      *                                      array must match the number of rows already contained in the table.
399      * @param  width                    the ASCII text width of the for the column entries (without the string
400      *                                      termination).
401      *
402      * @return                          the number of columns after this one is added.
403      *
404      * @throws IllegalArgumentException if the column data is not an array or the specified text <code>width</code> is
405      *                                      &le;1.
406      * @throws FitsException            if the column us of an unsupported data type or if the number of entries does
407      *                                      not match the number of rows already contained in the table.
408      * 
409      * @see                             #addColumn(Object)
410      */
411     public int addColumn(Object newCol, int width) throws FitsException, IllegalArgumentException {
412         if (width < 1) {
413             throw new IllegalArgumentException("Illegal ASCII column width: " + width);
414         }
415 
416         if (!newCol.getClass().isArray()) {
417             throw new IllegalArgumentException("Not an array: " + newCol.getClass().getName());
418         }
419 
420         if (nFields > 0 && Array.getLength(newCol) != nRows) {
421             throw new FitsException(
422                     "Mismatched number of rows: expected " + nRows + ", got " + Array.getLength(newCol) + "rows.");
423         }
424 
425         if (nFields == 0) {
426             nRows = Array.getLength(newCol);
427         }
428 
429         Class<?> type = ArrayFuncs.getBaseClass(newCol);
430         if (type != int.class && type != long.class && type != float.class && type != double.class
431                 && type != String.class) {
432             throw new FitsException("No AsciiTable support for elements of " + type.getName());
433         }
434 
435         Object[] newData = new Object[nFields + 1];
436         int[] newOffsets = new int[nFields + 1];
437         int[] newLengths = new int[nFields + 1];
438         Class<?>[] newTypes = new Class[nFields + 1];
439         String[] newNulls = new String[nFields + 1];
440 
441         System.arraycopy(data, 0, newData, 0, nFields);
442         System.arraycopy(offsets, 0, newOffsets, 0, nFields);
443         System.arraycopy(lengths, 0, newLengths, 0, nFields);
444         System.arraycopy(types, 0, newTypes, 0, nFields);
445         System.arraycopy(nulls, 0, newNulls, 0, nFields);
446 
447         data = newData;
448         offsets = newOffsets;
449         lengths = newLengths;
450         types = newTypes;
451         nulls = newNulls;
452 
453         newData[nFields] = newCol;
454         offsets[nFields] = rowLen + 1;
455         lengths[nFields] = width;
456         types[nFields] = ArrayFuncs.getBaseClass(newCol);
457 
458         rowLen += width + 1;
459         if (isNull != null) {
460             boolean[] newIsNull = new boolean[nRows * (nFields + 1)];
461             // Fix the null pointers.
462             int add = 0;
463             for (int i = 0; i < isNull.length; i++) {
464                 if (i % nFields == 0) {
465                     add++;
466                 }
467                 if (isNull[i]) {
468                     newIsNull[i + add] = true;
469                 }
470             }
471             isNull = newIsNull;
472         }
473         nFields++;
474 
475         // Invalidate the buffer
476         buffer = null;
477 
478         return nFields;
479     }

481     /**
482      * Beware that adding rows to ASCII tables may be very inefficient. Avoid addding more than a few rows if you can.
483      */
484     @Override
485     public int addRow(Object[] newRow) throws FitsException {
486         try {
487             // If there are no fields, then this is the
488             // first row. We need to add in each of the columns
489             // to get the descriptors set up.
490             if (nFields == 0) {
491                 for (Object element : newRow) {
492                     addColumn(element);
493                 }
494             } else {
495                 for (int i = 0; i < nFields; i++) {
496                     Object o = ArrayFuncs.newInstance(types[i], nRows + 1);
497                     System.arraycopy(data[i], 0, o, 0, nRows);
498                     System.arraycopy(newRow[i], 0, o, nRows, 1);
499                     data[i] = o;
500                 }
501                 nRows++;
502             }
503             // Invalidate the buffer
504             buffer = null;
505             return nRows;
506         } catch (Exception e) {
507             throw new FitsException("Error adding row:" + e.getMessage(), e);
508         }
509     }

511     @Override
512     public void deleteColumns(int start, int len) throws FitsException {
513         ensureData();
514 
515         Object[] newData = new Object[nFields - len];
516         int[] newOffsets = new int[nFields - len];
517         int[] newLengths = new int[nFields - len];
518         Class<?>[] newTypes = new Class[nFields - len];
519         String[] newNulls = new String[nFields - len];
520 
521         // Copy in the initial stuff...
522         System.arraycopy(data, 0, newData, 0, start);
523         // Don't do the offsets here.
524         System.arraycopy(lengths, 0, newLengths, 0, start);
525         System.arraycopy(types, 0, newTypes, 0, start);
526         System.arraycopy(nulls, 0, newNulls, 0, start);
527 
528         // Copy in the final
529         System.arraycopy(data, start + len, newData, start, nFields - start - len);
530         // Don't do the offsets here.
531         System.arraycopy(lengths, start + len, newLengths, start, nFields - start - len);
532         System.arraycopy(types, start + len, newTypes, start, nFields - start - len);
533         System.arraycopy(nulls, start + len, newNulls, start, nFields - start - len);
534 
535         for (int i = start; i < start + len; i++) {
536             rowLen -= lengths[i] + 1;
537         }
538 
539         data = newData;
540         offsets = newOffsets;
541         lengths = newLengths;
542         types = newTypes;
543         nulls = newNulls;
544 
545         if (isNull != null) {
546             boolean found = false;
547 
548             boolean[] newIsNull = new boolean[nRows * (nFields - len)];
549             for (int i = 0; i < nRows; i++) {
550                 int oldOff = nFields * i;
551                 int newOff = (nFields - len) * i;
552                 for (int col = 0; col < start; col++) {
553                     newIsNull[newOff + col] = isNull[oldOff + col];
554                     found = found || isNull[oldOff + col];
555                 }
556                 for (int col = start + len; col < nFields; col++) {
557                     newIsNull[newOff + col - len] = isNull[oldOff + col];
558                     found = found || isNull[oldOff + col];
559                 }
560             }
561             if (found) {
562                 isNull = newIsNull;
563             } else {
564                 isNull = null;
565             }
566         }
567 
568         // Invalidate the buffer
569         buffer = null;
570 
571         nFields -= len;
572     }

574     /**
575      * Beware that repeatedly deleting rows from ASCII tables may be very inefficient. Avoid calling this more than once
576      * (or a few times) if you can.
577      */
578     @Override
579     public void deleteRows(int start, int len) throws FitsException {
580         if (nRows == 0 || start < 0 || start >= nRows || len <= 0) {
581             return;
582         }
583         if (start + len > nRows) {
584             len = nRows - start;
585         }
586 
587         ensureData();
588 
589         for (int i = 0; i < nFields; i++) {
590             try {
591                 Object o = ArrayFuncs.newInstance(types[i], nRows - len);
592                 System.arraycopy(data[i], 0, o, 0, start);
593                 System.arraycopy(data[i], start + len, o, start, nRows - len - start);
594                 data[i] = o;
595             } catch (Exception e) {
596                 throw new FitsException("Error deleting row: " + e.getMessage(), e);
597             }
598 
599         }
600         nRows -= len;
601     }

603     @Override
604     protected void loadData(ArrayDataInput in) throws IOException, FitsException {
605         currInput = in;
606 
607         if (buffer == null) {
608             getBuffer((long) nRows * rowLen, 0);
609         }
610 
611         data = new Object[nFields];
612         for (int i = 0; i < nFields; i++) {
613             data[i] = ArrayFuncs.newInstance(types[i], nRows);
614         }
615 
616         bp.setOffset(0);
617 
618         int rowOffset;
619         for (int i = 0; i < nRows; i++) {
620             rowOffset = rowLen * i;
621             for (int j = 0; j < nFields; j++) {
622                 try {
623                     if (!extractElement(rowOffset + offsets[j], lengths[j], data, j, i, nulls[j])) {
624                         if (isNull == null) {
625                             isNull = new boolean[nRows * nFields];
626                         }
627 
628                         isNull[j + i * nFields] = true;
629                     }
630                 } catch (ArrayIndexOutOfBoundsException e) {

643     /**
644      * Move an element from the buffer into a data array.
645      *
646      * @param  offset        The offset within buffer at which the element starts.
647      * @param  length        The number of bytes in the buffer for the element.
648      * @param  array         An array of objects, each of which is a simple array.
649      * @param  col           Which element of array is to be modified?
650      * @param  row           Which index into that element is to be modified?
651      * @param  nullFld       What string signifies a null element?
652      *
653      * @throws FitsException if the operation failed
654      */
655     private boolean extractElement(int offset, int length, Object[] array, int col, int row, String nullFld)
656             throws FitsException {
657 
658         bp.setOffset(offset);
659 
660         if (nullFld != null) {
661             String s = bp.getString(length);
662             if (s.trim().equals(nullFld)) {
663                 return false;
664             }
665             bp.skip(-length);
666         }
667         try {
668             if (array[col] instanceof String[]) {
669                 ((String[]) array[col])[row] = bp.getString(length);
670             } else if (array[col] instanceof int[]) {
671                 ((int[]) array[col])[row] = bp.getInt(length);
672             } else if (array[col] instanceof float[]) {
673                 ((float[]) array[col])[row] = bp.getFloat(length);
674             } else if (array[col] instanceof double[]) {
675                 ((double[]) array[col])[row] = bp.getDouble(length);
676             } else if (array[col] instanceof long[]) {
677                 ((long[]) array[col])[row] = bp.getLong(length);
678             } else {
679                 throw new FitsException("Invalid type for ASCII table conversion:" + array[col]);
680             }
681         } catch (FormatException e) {
682             throw new FitsException("Error parsing data at row,col:" + row + "," + col + "  ", e);
683         }
684         return true;
685     }

687     @Override
688     protected void fillHeader(Header h) {
689         h.deleteKey(Standard.SIMPLE);
690         h.deleteKey(Standard.EXTEND);
691 
692         Standard.context(AsciiTable.class);
693 
694         Cursor<String, HeaderCard> c = h.iterator();
695         c.add(HeaderCard.create(Standard.XTENSION, Standard.XTENSION_ASCIITABLE));
696         c.add(HeaderCard.create(Standard.BITPIX, Bitpix.BYTE.getHeaderValue()));
697         c.add(HeaderCard.create(Standard.NAXIS, 2));
698         c.add(HeaderCard.create(Standard.NAXIS1, rowLen));
699         c.add(HeaderCard.create(Standard.NAXIS2, nRows));
700         c.add(HeaderCard.create(Standard.PCOUNT, 0));
701         c.add(HeaderCard.create(Standard.GCOUNT, 1));
702         c.add(HeaderCard.create(Standard.TFIELDS, nFields));
703 
704         for (int i = 0; i < nFields; i++) {
705             addColInfo(i, c);
706         }
707 
708         Standard.context(null);
709     }

711     /**
712      * Read some data into the buffer.
713      */
714     private void getBuffer(long size, long offset) throws IOException, FitsException {
715 
716         if (currInput == null) {
717             throw new IOException("No stream open to read");
718         }
719 
720         if (size > Integer.MAX_VALUE) {
721             throw new FitsException("Cannot read ASCII table > 2 GB");
722         }
723 
724         buffer = new byte[(int) size];
725         if (offset != 0) {
726             FitsUtil.reposition(currInput, offset);
727         }
728         currInput.readFully(buffer);
729         bp = new ByteParser(buffer);
730     }

821     /**
822      * Read a single element from the table. This returns an array of dimension 1.
823      *
824      * @throws FitsException if the operation failed
825      */
826     private Object parseSingleElement(int row, int col) throws FitsException {
827 
828         Object[] res = new Object[1];
829         try {
830             getBuffer(lengths[col], getFileOffset() + (long) row * (long) rowLen + offsets[col]);
831         } catch (IOException e) {
832             buffer = null;
833             throw new FitsException("Unable to read element", e);
834         }
835         res[0] = ArrayFuncs.newInstance(types[col], 1);
836 
837         boolean success = extractElement(0, lengths[col], res, 0, 0, nulls[col]);
838         buffer = null;
839 
840         return success ? res[0] : null;
841     }

843     /**
844      * Read a single row from the table. This returns a set of arrays of dimension 1.
845      *
846      * @throws FitsException if the operation failed
847      */
848     private Object[] parseSingleRow(int row) throws FitsException {
849 
850         Object[] res = new Object[nFields];
851 
852         try {
853             getBuffer(rowLen, getFileOffset() + (long) row * (long) rowLen);
854         } catch (IOException e) {
855             throw new FitsException("Unable to read row", e);
856         }
857 
858         for (int i = 0; i < nFields; i++) {
859             res[i] = ArrayFuncs.newInstance(types[i], 1);
860             if (!extractElement(offsets[i], lengths[i], res, i, 0, nulls[i])) {
861                 res[i] = null;
862             }
863         }
864 
865         // Invalidate buffer for future use.
866         buffer = null;
867         return res;
868     }

979      * @deprecated It is not entirely foolproof for keeping the header in sync -- it is better to (re)wrap tables in a
980      *                 new HDU and editing the header as necessary to incorporate custom entries. May be removed from
981      *                 the API in the future.
982      */
983     @Override
984     public void updateAfterDelete(int oldNCol, Header hdr) throws FitsException {
985 
986         int offset = 0;
987         for (int i = 0; i < nFields; i++) {
988             offsets[i] = offset;
989             hdr.addValue(TBCOLn.n(i + 1), offset + 1);
990             offset += lengths[i] + 1;
991         }
992         for (int i = nFields; i < oldNCol; i++) {
993             hdr.deleteKey(TBCOLn.n(i + 1));
994         }
995 
996         hdr.addValue(NAXIS1, rowLen);
997     }

999     @Override
1000     public void write(ArrayDataOutput str) throws FitsException {
1001         // Make sure we have the data in hand.
1002         if (str != currInput) {
1003             ensureData();
1004         }
1005 
1006         // If buffer is still around we can just reuse it,
1007         // since nothing we've done has invalidated it.
1008         if (data == null) {
1009             throw new FitsException("Attempt to write undefined ASCII Table");
1010         }
1011 
1012         if ((long) nRows * rowLen > Integer.MAX_VALUE) {
1013             throw new FitsException("Cannot write ASCII table > 2 GB");
1014         }
1015 
1016         buffer = new byte[nRows * rowLen];
1017 
1018         bp = new ByteParser(buffer);
1019         for (int i = 0; i < buffer.length; i++) {
1020             buffer[i] = (byte) ' ';
1021         }
1022 
1023         ByteFormatter bf = new ByteFormatter();
1024 
1025         for (int i = 0; i < nRows; i++) {
1026 
1027             for (int j = 0; j < nFields; j++) {
1028                 int offset = i * rowLen + offsets[j];
1029                 int len = lengths[j];
1030                 if (isNull != null && isNull[i * nFields + j]) {
1031                     if (nulls[j] == null) {
1032                         throw new FitsException("No null value set when needed");
1033                     }
1034                     bf.format(nulls[j], buffer, offset, len);
1035                 } else if (types[j] == String.class) {
1036                     String[] s = (String[]) data[j];
1037                     bf.format(s[i], buffer, offset, len);
1038                 } else if (types[j] == int.class) {
1039                     int[] ia = (int[]) data[j];
1040                     bf.format(ia[i], buffer, offset, len);
1041                 } else if (types[j] == float.class) {
1042                     float[] fa = (float[]) data[j];
1043                     bf.format(fa[i], buffer, offset, len);
1044                 } else if (types[j] == double.class) {
1045                     double[] da = (double[]) data[j];
1046                     bf.format(da[i], buffer, offset, len);
1047                 } else if (types[j] == long.class) {
1048                     long[] la = (long[]) data[j];
1049                     bf.format(la[i], buffer, offset, len);
1050                 }
1051             }
1052         }
1053 
1054         // Now write the buffer.
1055         try {
1056             str.write(buffer);
1057             FitsUtil.pad(str, buffer.length, (byte) ' ');
1058         } catch (IOException e) {
1059             throw new FitsException("Error writing ASCII Table data", e);
1060         }
1061     }
