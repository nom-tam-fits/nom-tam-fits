152     /**
153      * <p>
154      * Create an ASCII table given a header, with custom integer handling
155      * support.
156      * </p>
157      *
158      * <p>The <code>preferInt</code> parameter controls how columns
159      * with format "<code>I10</code>" are handled; this is tricky because some,
160      * but not all, integers that can be represented in 10 characters can
161      * be represented as 32-bit integers.  Setting it <code>true</code> may make it
162      * more likely to avoid unexpected type changes during round-tripping,
163      * but it also means that some (large number) data in I10 columns may
164      * be impossible to read.
165      * </p>
166      * 
167      * @param hdr
168      *            The header describing the table
169      * @param preferInt
170      *            if <code>true</code>, format "I10" columns will be assumed
171      *            <code>int.class</code>, provided TLMINn/TLMAXn or TDMINn/TDMAXn 
172      *            limits (if defined) allow it. 
173      *            if <code>false</code>, I10 columns that have no clear indication 
174      *            of data range will be assumed <code>long.class</code>.
175      * @throws FitsException
176      *             if the operation failed
177      */
178     public AsciiTable(Header hdr, boolean preferInt) throws FitsException {
179 
180         this.nRows = hdr.getIntValue(NAXIS2);
181         this.nFields = hdr.getIntValue(TFIELDS);
182         this.rowLen = hdr.getIntValue(NAXIS1);
183 
184         this.types = new Class[this.nFields];
185         this.offsets = new int[this.nFields];
186         this.lengths = new int[this.nFields];
187         this.nulls = new String[this.nFields];
188         
189 
190         for (int i = 0; i < this.nFields; i += 1) {
191             this.offsets[i] = hdr.getIntValue(TBCOLn.n(i + 1)) - 1;
192             String s = hdr.getStringValue(TFORMn.n(i + 1));
193             if (this.offsets[i] < 0 || s == null) {
194                 throw new FitsException("Invalid Specification for column:" + (i + 1));
195             }
196             s = s.trim();
197             char c = s.charAt(0);
198             s = s.substring(1);
199             if (s.indexOf('.') > 0) {
200                 s = s.substring(0, s.indexOf('.'));
201             }
202             this.lengths[i] = Integer.parseInt(s);
203 
204             switch (c) {
205                 case 'A':
206                     this.types[i] = String.class;
207                     break;
208                 case 'I':
209                     if (this.lengths[i] == MAX_INTEGER_LENGTH) {
210                         this.types[i] = guessI10Type(i, hdr, preferInt);
211                     } else {
212                         this.types[i] = this.lengths[i] > MAX_INTEGER_LENGTH ? long.class : int.class;
213                     }
214                     break;
215                 case 'F':
216                 case 'E':
217                     this.types[i] = float.class;
218                     break;
219                 case 'D':
220                     this.types[i] = double.class;
221                     break;
222                 default:
223                     throw new FitsException("could not parse column type of ascii table");
224             }
225 
226             this.nulls[i] = hdr.getStringValue(TNULLn.n(i + 1));
227             if (this.nulls[i] != null) {
228                 this.nulls[i] = this.nulls[i].trim();
229             }
230         }
231     }

233     /**
234      * Checks if the integer value of a specific key requires <code>long</code>
235      * value type to store.
236      * 
237      * @param h     the header
238      * @param key   the keyword to check
239      * @return      <code>true</code> if the keyword exists and has an integer value that is
240      *              outside the range of <code>int</code>. Otherwise <code>false</code>
241      *              
242      * @see #guessI10Type(int, Header, boolean)
243      */
244     private boolean requiresLong(Header h, IFitsHeader key, Long dft) {
245         long l = h.getLongValue(key, dft);
246         if (l == dft) {
247             return false;
248         }
249 
250         return (l < Integer.MIN_VALUE || l > Integer.MAX_VALUE);
251     }

253     /**
254      * Guesses what type of values to use to return I10 type table values. Depending on the
255      * range of represented values I10 may fit into <code>int</code> types, or else
256      * require <code>long</code> type arrays. Therefore, the method checks for the
257      * presence of standard column limit keywords TLMINn/TLMAXn and TDMINn/TDMAXn
258      * and if these exist and are outside of the range of an <code>int</code> then
259      * the call will return <code>long.class</code>. If the header does not define the
260      * data limits (fully), it will return the class the caller prefers. Otherwise (data limits
261      * were defined and fit into the <code>int</code> range) <code>int.class</code> will
262      * be returned.
263      * 
264      * @param col           the 0-based table column index
265      * @param h             the header
266      * @param preferInt     whether we prefer <code>int.class</code> over <code>long.class</code>
267      *                      in case the header does not provide us with a clue. 
268      * @return              <code>long.class</code> if the data requires long or
269      *                      we prefer it. Othwerwise <code>int.class</code>
270      *                      
271      * @see #AsciiTable(Header, boolean)
272      */
273     private Class<?> guessI10Type(int col, Header h, boolean preferInt) {
274         col++;
275         
276         if (requiresLong(h, TLMINn.n(col), Long.MAX_VALUE)) {
277             return long.class;
278         }
279         if (requiresLong(h, TLMAXn.n(col), Long.MIN_VALUE)) {
280             return long.class;
281         }
282         if (requiresLong(h, TDMINn.n(col), Long.MAX_VALUE)) {
283             return long.class;
284         }
285         if (requiresLong(h, TDMAXn.n(col), Long.MIN_VALUE)) {
286             return long.class;
287         }
288         
289         if ((h.containsKey(TLMINn.n(col)) || h.containsKey(TDMINn.n(col))) //
290                 && (h.containsKey(TLMAXn.n(col)) || h.containsKey(TDMAXn.n(col)))) {
291             // There are keywords defining both min/max values, and none of them require long types...
292             return int.class;
293         }
294         
295         return preferInt ? int.class : long.class;
296     }

311     int addColInfo(int col, Cursor<String, HeaderCard> iter) throws HeaderCardException {
312 
313         String tform = null;
314         if (this.types[col] == String.class) {
315             tform = "A" + this.lengths[col];
316         } else if (this.types[col] == int.class || this.types[col] == long.class) {
317             tform = "I" + this.lengths[col];
318         } else if (this.types[col] == float.class) {
319             tform = "E" + this.lengths[col] + ".0";
320         } else if (this.types[col] == double.class) {
321             tform = "D" + this.lengths[col] + ".0";
322         }
323         Standard.context(AsciiTable.class);
324         IFitsHeader key = TFORMn.n(col + 1);
325         iter.add(new HeaderCard(key.key(), tform, key.comment()));
326         key = TBCOLn.n(col + 1);
327         iter.add(new HeaderCard(key.key(), this.offsets[col] + 1, key.comment()));
328         Standard.context(null);
329         return this.lengths[col];
330     }

332     @Override
333     public int addColumn(Object newCol) throws FitsException {
334         int maxLen = 1;
335         if (newCol instanceof String[]) {
336 
337             String[] sa = (String[]) newCol;
338             for (String element : sa) {
339                 if (element != null && element.length() > maxLen) {
340                     maxLen = element.length();
341                 }
342             }
343         } else if (newCol instanceof double[]) {
344             maxLen = DOUBLE_MAX_LENGTH;
345         } else if (newCol instanceof int[]) {
346             maxLen = INT_MAX_LENGTH;
347         } else if (newCol instanceof long[]) {
348             maxLen = LONG_MAX_LENGTH;
349         } else if (newCol instanceof float[]) {
350             maxLen = FLOAT_MAX_LENGTH;
351         } else {
352             throw new FitsException("Adding invalid type to ASCII table");
353         }
354         addColumn(newCol, maxLen);
355 
356         // Invalidate the buffer
357         this.buffer = null;
358 
359         return this.nFields;
360     }

362     /**
363      * This version of addColumn allows the user to override the default length
364      * associated with each column type.
365      *
366      * @param newCol
367      *            The new column data
368      * @param length
369      *            the requested length for the column
370      * @return the number of columns after this one is added.
371      * @throws FitsException
372      *             if the operation failed
373      */
374     public int addColumn(Object newCol, int length) throws FitsException {
375 
376         if (this.nFields > 0 && Array.getLength(newCol) != this.nRows) {
377             throw new FitsException("New column has different number of rows");
378         }
379 
380         if (this.nFields == 0) {
381             this.nRows = Array.getLength(newCol);
382         }
383 
384         Object[] newData = new Object[this.nFields + 1];
385         int[] newOffsets = new int[this.nFields + 1];
386         int[] newLengths = new int[this.nFields + 1];
387         Class<?>[] newTypes = new Class[this.nFields + 1];
388         String[] newNulls = new String[this.nFields + 1];
389 
390         System.arraycopy(this.data, 0, newData, 0, this.nFields);
391         System.arraycopy(this.offsets, 0, newOffsets, 0, this.nFields);
392         System.arraycopy(this.lengths, 0, newLengths, 0, this.nFields);
393         System.arraycopy(this.types, 0, newTypes, 0, this.nFields);
394         System.arraycopy(this.nulls, 0, newNulls, 0, this.nFields);
395 
396         this.data = newData;
397         this.offsets = newOffsets;
398         this.lengths = newLengths;
399         this.types = newTypes;
400         this.nulls = newNulls;
401 
402         newData[this.nFields] = newCol;
403         this.offsets[this.nFields] = this.rowLen + 1;
404         this.lengths[this.nFields] = length;
405         this.types[this.nFields] = ArrayFuncs.getBaseClass(newCol);
406 
407         this.rowLen += length + 1;
408         if (this.isNull != null) {
409             boolean[] newIsNull = new boolean[this.nRows * (this.nFields + 1)];
410             // Fix the null pointers.
411             int add = 0;
412             for (int i = 0; i < this.isNull.length; i += 1) {
413                 if (i % this.nFields == 0) {
414                     add += 1;
415                 }
416                 if (this.isNull[i]) {
417                     newIsNull[i + add] = true;
418                 }
419             }
420             this.isNull = newIsNull;
421         }
422         this.nFields += 1;
423 
424         // Invalidate the buffer
425         this.buffer = null;
426 
427         return this.nFields;
428     }

430     @Override
431     public int addRow(Object[] newRow) throws FitsException {
432         try {
433             // If there are no fields, then this is the
434             // first row. We need to add in each of the columns
435             // to get the descriptors set up.
436             if (this.nFields == 0) {
437                 for (Object element : newRow) {
438                     addColumn(element);
439                 }
440             } else {
441                 for (int i = 0; i < this.nFields; i += 1) {
442                     Object o = ArrayFuncs.newInstance(this.types[i], this.nRows + 1);
443                     System.arraycopy(this.data[i], 0, o, 0, this.nRows);
444                     System.arraycopy(newRow[i], 0, o, this.nRows, 1);
445                     this.data[i] = o;
446                 }
447                 this.nRows += 1;
448             }
449             // Invalidate the buffer
450             this.buffer = null;
451             return this.nRows;
452         } catch (Exception e) {
453             throw new FitsException("Error addnig row:" + e.getMessage(), e);
454         }
455     }

457     /**
458      * Delete columns from the table.
459      *
460      * @param start
461      *            The first, 0-indexed, column to be deleted.
462      * @param len
463      *            The number of columns to be deleted.
464      * @throws FitsException
465      *             if the operation failed
466      */
467 
468     @Override
469     public void deleteColumns(int start, int len) throws FitsException {
470         ensureData();
471 
472         Object[] newData = new Object[this.nFields - len];
473         int[] newOffsets = new int[this.nFields - len];
474         int[] newLengths = new int[this.nFields - len];
475         Class<?>[] newTypes = new Class[this.nFields - len];
476         String[] newNulls = new String[this.nFields - len];
477 
478         // Copy in the initial stuff...
479         System.arraycopy(this.data, 0, newData, 0, start);
480         // Don't do the offsets here.
481         System.arraycopy(this.lengths, 0, newLengths, 0, start);
482         System.arraycopy(this.types, 0, newTypes, 0, start);
483         System.arraycopy(this.nulls, 0, newNulls, 0, start);
484 
485         // Copy in the final
486         System.arraycopy(this.data, start + len, newData, start, this.nFields - start - len);
487         // Don't do the offsets here.
488         System.arraycopy(this.lengths, start + len, newLengths, start, this.nFields - start - len);
489         System.arraycopy(this.types, start + len, newTypes, start, this.nFields - start - len);
490         System.arraycopy(this.nulls, start + len, newNulls, start, this.nFields - start - len);
491 
492         for (int i = start; i < start + len; i += 1) {
493             this.rowLen -= this.lengths[i] + 1;
494         }
495 
496         this.data = newData;
497         this.offsets = newOffsets;
498         this.lengths = newLengths;
499         this.types = newTypes;
500         this.nulls = newNulls;
501 
502         if (this.isNull != null) {
503             boolean found = false;
504 
505             boolean[] newIsNull = new boolean[this.nRows * (this.nFields - len)];
506             for (int i = 0; i < this.nRows; i += 1) {
507                 int oldOff = this.nFields * i;
508                 int newOff = (this.nFields - len) * i;
509                 for (int col = 0; col < start; col += 1) {
510                     newIsNull[newOff + col] = this.isNull[oldOff + col];
511                     found = found || this.isNull[oldOff + col];
512                 }
513                 for (int col = start + len; col < this.nFields; col += 1) {
514                     newIsNull[newOff + col - len] = this.isNull[oldOff + col];
515                     found = found || this.isNull[oldOff + col];
516                 }
517             }
518             if (found) {
519                 this.isNull = newIsNull;
520             } else {
521                 this.isNull = null;
522             }
523         }
524 
525         // Invalidate the buffer
526         this.buffer = null;
527 
528         this.nFields -= len;
529     }

531     /**
532      * Delete rows from a FITS table
533      *
534      * @param start
535      *            The first (0-indexed) row to be deleted.
536      * @param len
537      *            The number of rows to be deleted.
538      * @throws FitsException
539      *             if the operation failed
540      */
541 
542     @Override
543     public void deleteRows(int start, int len) throws FitsException {
544         try {
545             if (this.nRows == 0 || start < 0 || start >= this.nRows || len <= 0) {
546                 return;
547             }
548             if (start + len > this.nRows) {
549                 len = this.nRows - start;
550             }
551             ensureData();
552             for (int i = 0; i < this.nFields; i += 1) {
553                 Object o = ArrayFuncs.newInstance(this.types[i], this.nRows - len);
554                 System.arraycopy(this.data[i], 0, o, 0, start);
555                 System.arraycopy(this.data[i], start + len, o, start, this.nRows - len - start);
556                 this.data[i] = o;
557             }
558             this.nRows -= len;
559         } catch (FitsException e) {
560             throw e;
561         } catch (Exception e) {
562             throw new FitsException("Error deleting row:" + e.getMessage(), e);
563         }
564     }

577     /**
578      * Move an element from the buffer into a data array.
579      *
580      * @param offset
581      *            The offset within buffer at which the element starts.
582      * @param length
583      *            The number of bytes in the buffer for the element.
584      * @param array
585      *            An array of objects, each of which is a simple array.
586      * @param col
587      *            Which element of array is to be modified?
588      * @param row
589      *            Which index into that element is to be modified?
590      * @param nullFld
591      *            What string signifies a null element?
592      * @throws FitsException
593      *             if the operation failed
594      */
595     private boolean extractElement(int offset, int length, Object[] array, int col, int row, String nullFld) throws FitsException {
596 
597         this.bp.setOffset(offset);
598 
599         if (nullFld != null) {
600             String s = this.bp.getString(length);
601             if (s.trim().equals(nullFld)) {
602                 return false;
603             }
604             this.bp.skip(-length);
605         }
606         try {
607             if (array[col] instanceof String[]) {
608                 ((String[]) array[col])[row] = this.bp.getString(length);
609             } else if (array[col] instanceof int[]) {
610                 ((int[]) array[col])[row] = this.bp.getInt(length);
611             } else if (array[col] instanceof float[]) {
612                 ((float[]) array[col])[row] = this.bp.getFloat(length);
613             } else if (array[col] instanceof double[]) {
614                 ((double[]) array[col])[row] = this.bp.getDouble(length);
615             } else if (array[col] instanceof long[]) {
616                 ((long[]) array[col])[row] = this.bp.getLong(length);
617             } else {
618                 throw new FitsException("Invalid type for ASCII table conversion:" + array[col]);
619             }
620         } catch (FormatException e) {
621             throw new FitsException("Error parsing data at row,col:" + row + "," + col + "  ", e);
622         }
623         return true;
624     }

634     @Override
635     public void fillHeader(Header hdr) {
636         try {
637             Standard.context(AsciiTable.class);
638             hdr.setXtension("TABLE");
639             hdr.setBitpix(Bitpix.BYTE);
640             hdr.setNaxes(2);
641             hdr.setNaxis(1, this.rowLen);
642             hdr.setNaxis(2, this.nRows);
643             Cursor<String, HeaderCard> iter = hdr.iterator();
644             iter.setKey(NAXIS2.key());
645             iter.next();
646             iter.add(new HeaderCard(PCOUNT.key(), 0, PCOUNT.comment()));
647             iter.add(new HeaderCard(GCOUNT.key(), 1, GCOUNT.comment()));
648             iter.add(new HeaderCard(TFIELDS.key(), this.nFields, TFIELDS.comment()));
649 
650             for (int i = 0; i < this.nFields; i += 1) {
651                 addColInfo(i, iter);
652             }
653         } catch (HeaderCardException e) {
654             LOG.log(Level.SEVERE, "ImpossibleException in fillHeader:" + e.getMessage(), e);
655         } finally {
656             Standard.context(null);
657         }
658     }

660     /**
661      * Read some data into the buffer.
662      */
663     private void getBuffer(int size, long offset) throws IOException, FitsException {
664 
665         if (this.currInput == null) {
666             throw new IOException("No stream open to read");
667         }
668 
669         this.buffer = new byte[size];
670         if (offset != 0) {
671             FitsUtil.reposition(this.currInput, offset);
672         }
673         this.currInput.readFully(this.buffer);
674         this.bp = new ByteParser(this.buffer);
675     }

693     /**
694      * Get the ASCII table information. This will actually do the read if it had
695      * previously been deferred
696      *
697      * @return The table data as an Object[] array.
698      * @throws FitsException
699      *             if the operation failed
700      */
701     @Override
702     @SuppressFBWarnings(value = "EI_EXPOSE_REP", justification = "intended exposure of mutable data")
703     public Object getData() throws FitsException {
704 
705         if (this.data == null) {
706 
707             this.data = new Object[this.nFields];
708 
709             for (int i = 0; i < this.nFields; i += 1) {
710                 this.data[i] = ArrayFuncs.newInstance(this.types[i], this.nRows);
711             }
712 
713             if (this.buffer == null) {
714                 long newOffset = FitsUtil.findOffset(this.currInput);
715                 try {
716                     getBuffer(this.nRows * this.rowLen, this.fileOffset);
717 
718                 } catch (IOException e) {
719                     throw new FitsException("Error in deferred read -- file closed prematurely?:" + e.getMessage(), e);
720                 }
721                 FitsUtil.reposition(this.currInput, newOffset);
722             }
723 
724             this.bp.setOffset(0);
725 
726             int rowOffset;
727             for (int i = 0; i < this.nRows; i += 1) {
728                 rowOffset = this.rowLen * i;
729                 for (int j = 0; j < this.nFields; j += 1) {
730                     if (!extractElement(rowOffset + this.offsets[j], this.lengths[j], this.data, j, i, this.nulls[j])) {
731                         if (this.isNull == null) {
732                             this.isNull = new boolean[this.nRows * this.nFields];
733                         }
734 
735                         this.isNull[j + i * this.nFields] = true;
736                     }
737                 }
738             }
739         }
740         return this.data;
741     }

842     /**
843      * Read a single element from the table. This returns an array of dimension
844      * 1.
845      *
846      * @throws FitsException
847      *             if the operation failed
848      */
849     private Object parseSingleElement(int row, int col) throws FitsException {
850 
851         Object[] res = new Object[1];
852         try {
853             getBuffer(this.lengths[col], this.fileOffset + (long) row * (long) this.rowLen + this.offsets[col]);
854         } catch (IOException e) {
855             this.buffer = null;
856             throw new FitsException("Unable to read element", e);
857         }
858         res[0] = ArrayFuncs.newInstance(this.types[col], 1);
859 
860         if (extractElement(0, this.lengths[col], res, 0, 0, this.nulls[col])) {
861             this.buffer = null;
862             return res[0];
863         } 
864         
865         this.buffer = null;
866         return null;
867     }

869     /**
870      * Read a single row from the table. This returns a set of arrays of
871      * dimension 1.
872      *
873      * @throws FitsException
874      *             if the operation failed
875      */
876     private Object[] parseSingleRow(int row) throws FitsException {
877 
878         Object[] res = new Object[this.nFields];
879 
880         try {
881             getBuffer(this.rowLen, this.fileOffset + (long) row * (long) this.rowLen);
882         } catch (IOException e) {
883             throw new FitsException("Unable to read row", e);
884         }
885 
886         for (int i = 0; i < this.nFields; i += 1) {
887             res[i] = ArrayFuncs.newInstance(this.types[i], 1);
888             if (!extractElement(this.offsets[i], this.lengths[i], res, i, 0, this.nulls[i])) {
889                 res[i] = null;
890             }
891         }
892 
893         // Invalidate buffer for future use.
894         this.buffer = null;
895         return res;
896     }

898     /**
899      * Read in an ASCII table. Reading is deferred if we are reading from a
900      * random access device
901      *
902      * @param str
903      *            the stream to read from
904      * @throws FitsException
905      *             if the operation failed
906      */
907 
908     @Override
909     public void read(ArrayDataInput str) throws FitsException {
910         try {
911             setFileOffset(str);
912             this.currInput = str;
913             if (str instanceof RandomAccess) {
914                 str.skipAllBytes((long) this.nRows * this.rowLen);
915             } else {
916                 if ((long) this.rowLen * this.nRows > Integer.MAX_VALUE) {
917                     throw new FitsException("Cannot read ASCII table > 2 GB");
918                 }
919                 getBuffer(this.rowLen * this.nRows, 0);
920             }
921             str.skipAllBytes(FitsUtil.padding(this.nRows * this.rowLen));
922         } catch (EOFException e) {
923             throw new PaddingException("EOF skipping padding after ASCII Table", this, e);
924         } catch (IOException e) {
925             throw new FitsException("Error skipping padding after ASCII Table", e);
926         }
927     }

1080     /**
1081      * This is called after we delete columns. The HDU doesn't know how to
1082      * update the TBCOL entries.
1083      *
1084      * @param oldNCol
1085      *            The number of columns we had before deletion.
1086      * @param hdr
1087      *            The associated header. @throws FitsException if the operation
1088      *            failed
1089      */
1090 
1091     @Override
1092     public void updateAfterDelete(int oldNCol, Header hdr) throws FitsException {
1093 
1094         int offset = 0;
1095         for (int i = 0; i < this.nFields; i += 1) {
1096             this.offsets[i] = offset;
1097             hdr.addValue(TBCOLn.n(i + 1), offset + 1);
1098             offset += this.lengths[i] + 1;
1099         }
1100         for (int i = this.nFields; i < oldNCol; i += 1) {
1101             hdr.deleteKey(TBCOLn.n(i + 1));
1102         }
1103 
1104         hdr.addValue(NAXIS1, this.rowLen);
1105     }

1107     /**
1108      * Write the data to an output stream.
1109      *
1110      * @param str
1111      *            The output stream to be written to
1112      * @throws FitsException
1113      *             if any IO exception is found or some inconsistency the FITS
1114      *             file arises.
1115      */
1116 
1117     @Override
1118     public void write(ArrayDataOutput str) throws FitsException {
1119         // Make sure we have the data in hand.
1120         ensureData();
1121         // If buffer is still around we can just reuse it,
1122         // since nothing we've done has invalidated it.
1123 
1124         if (this.buffer == null) {
1125 
1126             if (this.data == null) {
1127                 throw new FitsException("Attempt to write undefined ASCII Table");
1128             }
1129 
1130             if ((long) this.nRows * this.rowLen > Integer.MAX_VALUE) {
1131                 throw new FitsException("Cannot write ASCII table > 2 GB");
1132             }
1133 
1134             this.buffer = new byte[this.nRows * this.rowLen];
1135 
1136             this.bp = new ByteParser(this.buffer);
1137             for (int i = 0; i < this.buffer.length; i += 1) {
1138                 this.buffer[i] = (byte) ' ';
1139             }
1140 
1141             ByteFormatter bf = new ByteFormatter();
1142 
1143             for (int i = 0; i < this.nRows; i += 1) {
1144 
1145                 for (int j = 0; j < this.nFields; j += 1) {
1146                     int offset = i * this.rowLen + this.offsets[j];
1147                     int len = this.lengths[j];
1148                     if (this.isNull != null && this.isNull[i * this.nFields + j]) {
1149                         if (this.nulls[j] == null) {
1150                             throw new FitsException("No null value set when needed");
1151                         }
1152                         bf.format(this.nulls[j], this.buffer, offset, len);
1153                     } else {
1154                         if (this.types[j] == String.class) {
1155                             String[] s = (String[]) this.data[j];
1156                             bf.format(s[i], this.buffer, offset, len);
1157                         } else if (this.types[j] == int.class) {
1158                             int[] ia = (int[]) this.data[j];
1159                             bf.format(ia[i], this.buffer, offset, len);
1160                         } else if (this.types[j] == float.class) {
1161                             float[] fa = (float[]) this.data[j];
1162                             bf.format(fa[i], this.buffer, offset, len);
1163                         } else if (this.types[j] == double.class) {
1164                             double[] da = (double[]) this.data[j];
1165                             bf.format(da[i], this.buffer, offset, len);
1166                         } else if (this.types[j] == long.class) {
1167                             long[] la = (long[]) this.data[j];
1168                             bf.format(la[i], this.buffer, offset, len);
1169                         }
1170                     }
1171                 }
1172             }
1173         }
1174 
1175         // Now write the buffer.
1176         try {
1177             str.write(this.buffer);
1178             FitsUtil.pad(str, this.buffer.length, (byte) ' ');
1179         } catch (IOException e) {
1180             throw new FitsException("Error writing ASCII Table data", e);
1181         }
1182     }
