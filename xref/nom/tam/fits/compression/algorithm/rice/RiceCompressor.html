46  /**
47   * The original compression was designed by Rice, Yeh, and Miller the code was
48   * written by Richard White at STSc at the STScI and included (ported to c and
49   * adapted) in cfitsio by William Pence, NASA/GSFC. That code was then ported to
50   * java by R. van Nieuwenhoven. Later it was massively refactored to harmonize
51   * the different compression algorithms and reduce the duplicate code pieces
52   * without obscuring the algorithm itself as far as possible.
53   *
54   * @author Richard White
55   * @author William Pence
56   * @author Richard van Nieuwenhoven
57   * 
58   * @param <T> the genetic type of NIO buffer on which this compressor operates.
59   */
60  public abstract class RiceCompressor<T extends Buffer> implements ICompressor<T> {

62      public static class ByteRiceCompressor extends RiceCompressor<ByteBuffer> {
63  
64          private ByteBuffer pixelBuffer;
65  
66          public ByteRiceCompressor(RiceCompressOption option) {
67              super(option.setDefaultBytePix(ElementType.BYTE.size()));
68          }
69  
70          @Override
71          public boolean compress(ByteBuffer buffer, ByteBuffer writeBuffer) {
72              this.pixelBuffer = buffer;
73              super.compress(buffer.limit(), this.pixelBuffer.get(this.pixelBuffer.position()), new BitBuffer(writeBuffer));
74              return true;
75          }
76  
77          @Override
78          public void decompress(ByteBuffer readBuffer, ByteBuffer buffer) {
79              this.pixelBuffer = buffer;
80              super.decompressBuffer(readBuffer, buffer.limit());
81          }
82  
83          @Override
84          protected int nextPixel() {
85              return this.pixelBuffer.get();
86          }
87  
88          @Override
89          protected void nextPixel(int pixel) {
90              this.pixelBuffer.put((byte) pixel);
91          }
92      }

246     private RiceCompressor(RiceCompressOption option) {
247         this.blockSize = option.getBlockSize();
248         if (option.getBytePix() == ElementType.BYTE.size()) {
249             this.fsBits = FS_BITS_FOR_BYTE;
250             this.fsMax = FS_MAX_FOR_BYTE;
251             this.bitsPerPixel = FitsIO.BITS_OF_1_BYTE;
252         } else if (option.getBytePix() == ElementType.SHORT.size()) {
253             this.fsBits = FS_BITS_FOR_SHORT;
254             this.fsMax = FS_MAX_FOR_SHORT;
255             this.bitsPerPixel = FitsIO.BITS_OF_2_BYTES;
256         } else if (option.getBytePix() == ElementType.INT.size()) {
257             this.fsBits = FS_BITS_FOR_INT;
258             this.fsMax = FS_MAX_FOR_INT;
259             this.bitsPerPixel = FitsIO.BITS_OF_4_BYTES;
260         } else {
261             throw new UnsupportedOperationException("Rice only supports 1/2/4 type per pixel");
262         }
263         /*
264          * From bsize derive: FSBITS = # bits required to store FS FSMAX =
265          * maximum value for FS BBITS = bits/pixel for direct coding
266          */
267         this.bBits = 1 << this.fsBits;
268     }

270     /**
271      * <p>
272      * undo mapping and differencing Note that some of these operations will
273      * overflow the unsigned int arithmetic -- that's OK, it all works out to
274      * give the right answers in the output file.
275      * </p>
276      * <p>
277      * In java this is more complicated because of the missing unsigned
278      * integers. trying to simulate the behavior
279      * </p>
280      *
281      * @param lastpix
282      *            the current last pix value
283      * @param diff
284      *            the difference to "add"
285      * @return return the new lastpiy value
286      */
287     private long undoMappingAndDifferencing(long lastpix, long diff) {
288         diff &= UNSIGNED_INTEGER_MASK;
289         if ((diff & 1) == 0) {
290             diff = diff >>> 1;
291         } else {
292             diff = diff >>> 1 ^ UNSIGNED_INTEGER_MASK;
293         }
294         lastpix = diff + lastpix & UNSIGNED_INTEGER_MASK;
295         nextPixel((int) lastpix);
296         return lastpix;
297     }

299     /**
300      * compress the integer tiledImageOperation on a rise compressed byte
301      * buffer.
302      *
303      * @param dataLength
304      *            length of the data to compress
305      * @param firstPixel
306      *            the value of the first pixel
307      * @param buffer
308      *            the buffer to write to
309      */
310     protected void compress(final int dataLength, int firstPixel, BitBuffer buffer) {
311         /* the first difference will always be zero */
312         int lastpix = firstPixel;
313         /* write out first int value to the first 4 bytes of the buffer */
314         buffer.putInt(firstPixel, this.bitsPerPixel);
315         int thisblock = this.blockSize;
316         for (int i = 0; i < dataLength; i += this.blockSize) {
317             /* last block may be shorter */
318             if (dataLength - i < this.blockSize) {
319                 thisblock = dataLength - i;
320             }
321             /*
322              * Compute differences of adjacent pixels and map them to unsigned
323              * values. Note that this may overflow the integer variables --
324              * that's OK, because we can recover when decompressing. If we were
325              * compressing shorts or bytes, would want to do this arithmetic
326              * with short/byte working variables (though diff will still be
327              * passed as an int.) compute sum of mapped pixel values at same
328              * time use double precision for sum to allow 32-bit integer inputs
329              */
330             long[] diff = new long[this.blockSize];
331             double pixelsum = 0.0;
332             int nextpix;
333             /*
334              * tiledImageOperation for differences mapped to non-negative values
335              */
336             for (int j = 0; j < thisblock; j++) {
337                 nextpix = nextPixel();
338                 long pdiff = (nextpix - lastpix);
339                 diff[j] = (pdiff < 0 ? (pdiff << 1) ^ UNSIGNED_INTEGER_MASK : pdiff << 1) & UNSIGNED_INTEGER_MASK;
340                 pixelsum += diff[j];
341                 lastpix = nextpix;
342             }
343 
344             /*
345              * compute number of bits to split from sum
346              */
347             double dpsum = (pixelsum - thisblock / 2d - 1d) / thisblock;
348             if (dpsum < 0) {
349                 dpsum = 0.0;
350             }
351             long psum = (long) dpsum >> 1;
352             int fs;
353             for (fs = 0; psum > 0; fs++) { // NOSONAR
354                 psum >>= 1;
355             }
356 
357             /*
358              * write the codes fsbits ID bits used to indicate split level
359              */
360             if (fs >= this.fsMax) {
361                 /*
362                  * Special high entropy case when FS >= fsmax Just write pixel
363                  * difference values directly, no Rice coding at all.
364                  */
365                 buffer.putInt(this.fsMax + 1, this.fsBits);
366                 for (int j = 0; j < thisblock; j++) {
367                     buffer.putLong(diff[j], this.bBits);
368                 }
369             } else if (fs == 0 && pixelsum == 0) { // NOSONAR
370                 /*
371                  * special low entropy case when FS = 0 and pixelsum=0 (all
372                  * pixels in block are zero.) Output a 0 and return
373                  */
374                 buffer.putInt(0, this.fsBits);
375             } else {
376                 /* normal case: not either very high or very low entropy */
377                 buffer.putInt(fs + 1, this.fsBits);
378                 int fsmask = (1 << fs) - 1;
379                 /*
380                  * local copies of bit buffer to improve optimization
381                  */
382                 int bitsToGo = buffer.missingBitsInCurrentByte();
383                 int bitBuffer = buffer.bitbuffer() >> bitsToGo;
384                 buffer.movePosition(bitsToGo - BITS_OF_1_BYTE);
385                 for (int j = 0; j < thisblock; j++) {
386                     int v = (int) diff[j];
387                     int top = v >> fs;
388                     /*
389                      * top is coded by top zeros + 1
390                      */
391                     if (bitsToGo >= top + 1) {
392                         bitBuffer <<= top + 1;
393                         bitBuffer |= 1;
394                         bitsToGo -= top + 1;
395                     } else {
396                         bitBuffer <<= bitsToGo;
397                         buffer.putByte((byte) (bitBuffer & BYTE_MASK));
398                         for (top -= bitsToGo; top >= BITS_OF_1_BYTE; top -= BITS_OF_1_BYTE) {
399                             buffer.putByte((byte) 0);
400                         }
401                         bitBuffer = 1;
402                         bitsToGo = BITS_OF_1_BYTE - 1 - top;
403                     }
404                     /*
405                      * bottom FS bits are written without coding code is
406                      * output_nbits, moved into this routine to reduce overheads
407                      * This code potentially breaks if FS>24, so I am limiting
408                      * FS to 24 by choice of FSMAX above.
409                      */
410                     if (fs > 0) {
411                         bitBuffer <<= fs;
412                         bitBuffer |= v & fsmask;
413                         bitsToGo -= fs;
414                         while (bitsToGo <= 0) {
415                             buffer.putByte((byte) (bitBuffer >> -bitsToGo & BYTE_MASK));
416                             bitsToGo += BITS_OF_1_BYTE;
417                         }
418                     }
419                 }
420                 buffer.putByte((byte) (bitBuffer & BYTE_MASK), BITS_OF_1_BYTE - bitsToGo);
421             }
422         }
423         buffer.close();
424     }

426     /**
427      * decompress the readbuffer and fill the pixelarray.
428      *
429      * @param readBuffer
430      *            input buffer
431      * @param nx
432      *            the number of pixel to uncompress
433      */
434     protected void decompressBuffer(final ByteBuffer readBuffer, final int nx) {
435         /* first x bytes of input buffer contain the value of the first */
436         /* x byte integer value, without any encoding */
437         long lastpix = 0L;
438         if (this.bitsPerPixel == ElementType.BYTE.bitPix()) {
439             lastpix = readBuffer.get() & UNSIGNED_BYTE_MASK;
440         } else if (this.bitsPerPixel == ElementType.SHORT.bitPix()) {
441             lastpix = readBuffer.getShort() & UNSIGNED_SHORT_MASK;
442         } else if (this.bitsPerPixel == ElementType.INT.bitPix()) {
443             lastpix = readBuffer.getInt() & UNSIGNED_INTEGER_MASK;
444         }
445         long b = readBuffer.get() & BYTE_MASK; /* bit buffer */
446         int nbits = BITS_PER_BYTE; /* number of bits remaining in b */
447         for (int i = 0; i < nx;) {
448             /* get the FS value from first fsbits */
449             nbits -= this.fsBits;
450             while (nbits < 0) {
451                 b = b << BITS_PER_BYTE | readBuffer.get() & BYTE_MASK;
452                 nbits += BITS_PER_BYTE;
453             }
454             long fs = (b >>> nbits) - 1L;
455 
456             b &= (1 << nbits) - 1;
457             /* loop over the next block */
458             int imax = i + this.blockSize;
459             if (imax > nx) {
460                 imax = nx;
461             }
462             if (fs < 0) {
463                 /* low-entropy case, all zero differences */
464                 for (; i < imax; i++) {
465                     nextPixel((int) lastpix);
466                 }
467             } else if (fs == this.fsMax) {
468                 /* high-entropy case, directly coded pixel values */
469                 for (; i < imax; i++) {
470                     int k = this.bBits - nbits;
471                     long diff = b << k;
472                     for (k -= BITS_PER_BYTE; k >= 0; k -= BITS_PER_BYTE) {
473                         b = readBuffer.get() & BYTE_MASK;
474                         diff |= b << k;
475                     }
476                     if (nbits > 0) {
477                         b = readBuffer.get() & BYTE_MASK;
478                         diff |= b >>> -k;
479                         b &= (1 << nbits) - 1L;
480                     } else {
481                         b = 0;
482                     }
483                     lastpix = undoMappingAndDifferencing(lastpix, diff);
484                 }
485             } else {
486                 /* normal case, Rice coding */
487                 for (; i < imax; i++) {
488                     /* count number of leading zeros */
489                     while (b == 0) {
490                         nbits += BITS_PER_BYTE;
491                         b = readBuffer.get() & BYTE_MASK;
492                     }
493                     long nzero = nbits - NONZERO_COUNT[(int) (b & BYTE_MASK)];
494                     nbits -= nzero + 1;
495                     /* flip the leading one-bit */
496                     b ^= 1 << nbits;
497                     /* get the FS trailing bits */
498                     nbits -= fs;
499                     while (nbits < 0) {
500                         b = b << BITS_PER_BYTE | readBuffer.get() & BYTE_MASK;
501                         nbits += BITS_PER_BYTE;
502                     }
503                     long diff = nzero << fs | b >> nbits;
504                     b &= (1 << nbits) - 1L;
505 
506                     lastpix = undoMappingAndDifferencing(lastpix, diff);
507                 }
508             }
509         }
510         if (readBuffer.limit() > readBuffer.position()) {
511             LOG.warning("decompressing left over some extra bytes got: " + readBuffer.limit() + " but needed only " + readBuffer.position());
512         }
513 
514     }
