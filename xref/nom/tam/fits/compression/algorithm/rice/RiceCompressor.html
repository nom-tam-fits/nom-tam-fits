46  /**
47   * (<i>for internal use</i>) The Rice compression algorithm. The original compression was designed by Rice, Yeh, and
48   * Miller the code was written by Richard White at STScI at the STScI and included (ported to c and adapted) in cfitsio
49   * by William Pence, NASA/GSFC. That code was then ported to java by R. van Nieuwenhoven. Later it was massively
50   * refactored to harmonize the different compression algorithms and reduce the duplicate code pieces without obscuring
51   * the algorithm itself as far as possible.
52   *
53   * @author     Richard White
54   * @author     William Pence
55   * @author     Richard van Nieuwenhoven
56   *
57   * @param  <T> the genetic type of NIO buffer on which this compressor operates.
58   */
59  @SuppressWarnings({"deprecation", "javadoc"})
60  public abstract class RiceCompressor<T extends Buffer> implements ICompressor<T> {

234     private RiceCompressor(RiceCompressOption option) throws UnsupportedOperationException {
235         blockSize = option.getBlockSize();
236         if (option.getBytePix() == ElementType.BYTE.size()) {
237             fsBits = FS_BITS_FOR_BYTE;
238             fsMax = FS_MAX_FOR_BYTE;
239             bitsPerPixel = FitsIO.BITS_OF_1_BYTE;
240         } else if (option.getBytePix() == ElementType.SHORT.size()) {
241             fsBits = FS_BITS_FOR_SHORT;
242             fsMax = FS_MAX_FOR_SHORT;
243             bitsPerPixel = FitsIO.BITS_OF_2_BYTES;
244         } else if (option.getBytePix() == ElementType.INT.size()) {
245             fsBits = FS_BITS_FOR_INT;
246             fsMax = FS_MAX_FOR_INT;
247             bitsPerPixel = FitsIO.BITS_OF_4_BYTES;
248         } else {
249             throw new UnsupportedOperationException("Implemented for 1/2/4 bytes only");
250         }
251         /*
252          * From bsize derive: FSBITS = # bits required to store FS FSMAX = maximum value for FS BBITS = bits/pixel for
253          * direct coding
254          */
255         bBits = 1 << fsBits;
256     }

258     /**
259      * <p>
260      * undo mapping and differencing Note that some of these operations will overflow the unsigned int arithmetic --
261      * that's OK, it all works out to give the right answers in the output file.
262      * </p>
263      * <p>
264      * In java this is more complicated because of the missing unsigned integers. trying to simulate the behavior
265      * </p>
266      *
267      * @param  lastpix the current last pix value
268      * @param  diff    the difference to "add"
269      *
270      * @return         return the new lastpiy value
271      */
272     private long undoMappingAndDifferencing(long lastpix, long diff) {
273         diff &= UNSIGNED_INTEGER_MASK;
274         if ((diff & 1) == 0) {
275             diff = diff >>> 1;
276         } else {
277             diff = diff >>> 1 ^ UNSIGNED_INTEGER_MASK;
278         }
279         lastpix = diff + lastpix & UNSIGNED_INTEGER_MASK;
280         nextPixel((int) lastpix);
281         return lastpix;
282     }

284     /**
285      * compress the integer tiledImageOperation on a rise compressed byte buffer.
286      *
287      * @param dataLength length of the data to compress
288      * @param firstPixel the value of the first pixel
289      * @param buffer     the buffer to write to
290      */
291     protected void compress(final int dataLength, int firstPixel, BitBuffer buffer) {
292         /* the first difference will always be zero */
293         int lastpix = firstPixel;
294         /* write out first int value to the first 4 bytes of the buffer */
295         buffer.putInt(firstPixel, bitsPerPixel);
296         int thisblock = blockSize;
297         for (int i = 0; i < dataLength; i += blockSize) {
298             /* last block may be shorter */
299             if (dataLength - i < blockSize) {
300                 thisblock = dataLength - i;
301             }
302             /*
303              * Compute differences of adjacent pixels and map them to unsigned values. Note that this may overflow the
304              * integer variables -- that's OK, because we can recover when decompressing. If we were compressing shorts
305              * or bytes, would want to do this arithmetic with short/byte working variables (though diff will still be
306              * passed as an int.) compute sum of mapped pixel values at same time use double precision for sum to allow
307              * 32-bit integer inputs
308              */
309             long[] diff = new long[blockSize];
310             double pixelsum = 0.0;
311             int nextpix;
312             /*
313              * tiledImageOperation for differences mapped to non-negative values
314              */
315             for (int j = 0; j < thisblock; j++) {
316                 nextpix = nextPixel();
317                 long pdiff = (nextpix - lastpix);
318                 diff[j] = (pdiff < 0 ? (pdiff << 1) ^ UNSIGNED_INTEGER_MASK : pdiff << 1) & UNSIGNED_INTEGER_MASK;
319                 pixelsum += diff[j];
320                 lastpix = nextpix;
321             }
322 
323             /*
324              * compute number of bits to split from sum
325              */
326             double dpsum = (pixelsum - thisblock / 2d - 1d) / thisblock;
327             if (dpsum < 0) {
328                 dpsum = 0.0;
329             }
330             long psum = (long) dpsum >> 1;
331             int fs;
332             for (fs = 0; psum > 0; fs++) { // NOSONAR
333                 psum >>= 1;
334             }
335 
336             /*
337              * write the codes fsbits ID bits used to indicate split level
338              */
339             if (fs >= fsMax) {
340                 /*
341                  * Special high entropy case when FS >= fsmax Just write pixel difference values directly, no Rice
342                  * coding at all.
343                  */
344                 buffer.putInt(fsMax + 1, fsBits);
345                 for (int j = 0; j < thisblock; j++) {
346                     buffer.putLong(diff[j], bBits);
347                 }
348             } else if (fs == 0 && pixelsum == 0) { // NOSONAR
349                 /*
350                  * special low entropy case when FS = 0 and pixelsum=0 (all pixels in block are zero.) Output a 0 and
351                  * return
352                  */
353                 buffer.putInt(0, fsBits);
354             } else {
355                 /* normal case: not either very high or very low entropy */
356                 buffer.putInt(fs + 1, fsBits);
357                 int fsmask = (1 << fs) - 1;
358                 /*
359                  * local copies of bit buffer to improve optimization
360                  */
361                 int bitsToGo = buffer.missingBitsInCurrentByte();
362                 int bitBuffer = buffer.bitbuffer() >> bitsToGo;
363                 buffer.movePosition(bitsToGo - BITS_OF_1_BYTE);
364                 for (int j = 0; j < thisblock; j++) {
365                     int v = (int) diff[j];
366                     int top = v >> fs;
367                     /*
368                      * top is coded by top zeros + 1
369                      */
370                     if (bitsToGo >= top + 1) {
371                         bitBuffer <<= top + 1;
372                         bitBuffer |= 1;
373                         bitsToGo -= top + 1;
374                     } else {
375                         bitBuffer <<= bitsToGo;
376                         buffer.putByte((byte) (bitBuffer & BYTE_MASK));
377                         for (top -= bitsToGo; top >= BITS_OF_1_BYTE; top -= BITS_OF_1_BYTE) {
378                             buffer.putByte((byte) 0);
379                         }
380                         bitBuffer = 1;
381                         bitsToGo = BITS_OF_1_BYTE - 1 - top;
382                     }
383                     /*
384                      * bottom FS bits are written without coding code is output_nbits, moved into this routine to reduce
385                      * overheads This code potentially breaks if FS>24, so I am limiting FS to 24 by choice of FSMAX
386                      * above.
387                      */
388                     if (fs > 0) {
389                         bitBuffer <<= fs;
390                         bitBuffer |= v & fsmask;
391                         bitsToGo -= fs;
392                         while (bitsToGo <= 0) {
393                             buffer.putByte((byte) (bitBuffer >> -bitsToGo & BYTE_MASK));
394                             bitsToGo += BITS_OF_1_BYTE;
395                         }
396                     }
397                 }
398                 buffer.putByte((byte) (bitBuffer & BYTE_MASK), BITS_OF_1_BYTE - bitsToGo);
399             }
400         }
401         buffer.close();
402     }

404     /**
405      * decompress the readbuffer and fill the pixelarray.
406      *
407      * @param readBuffer input buffer
408      * @param nx         the number of pixel to uncompress
409      */
410     protected void decompressBuffer(final ByteBuffer readBuffer, final int nx) {
411         /* first x bytes of input buffer contain the value of the first */
412         /* x byte integer value, without any encoding */
413         long lastpix = 0L;
414         if (bitsPerPixel == ElementType.BYTE.bitPix()) {
415             lastpix = readBuffer.get() & UNSIGNED_BYTE_MASK;
416         } else if (bitsPerPixel == ElementType.SHORT.bitPix()) {
417             lastpix = readBuffer.getShort() & UNSIGNED_SHORT_MASK;
418         } else {
419             // Must be (this.bitsPerPixel == ElementType.INT.bitPix())
420             lastpix = readBuffer.getInt() & UNSIGNED_INTEGER_MASK;
421         }
422         long b = readBuffer.get() & BYTE_MASK; /* bit buffer */
423         int nbits = BITS_PER_BYTE; /* number of bits remaining in b */
424         for (int i = 0; i < nx;) {
425             /* get the FS value from first fsbits */
426             nbits -= fsBits;
427             while (nbits < 0) {
428                 b = b << BITS_PER_BYTE | readBuffer.get() & BYTE_MASK;
429                 nbits += BITS_PER_BYTE;
430             }
431             long fs = (b >>> nbits) - 1L;
432 
433             b &= (1 << nbits) - 1;
434             /* loop over the next block */
435             int imax = i + blockSize;
436             if (imax > nx) {
437                 imax = nx;
438             }
439             if (fs < 0) {
440                 /* low-entropy case, all zero differences */
441                 for (; i < imax; i++) {
442                     nextPixel((int) lastpix);
443                 }
444             } else if (fs == fsMax) {
445                 /* high-entropy case, directly coded pixel values */
446                 for (; i < imax; i++) {
447                     int k = bBits - nbits;
448                     long diff = b << k;
449                     for (k -= BITS_PER_BYTE; k >= 0; k -= BITS_PER_BYTE) {
450                         b = readBuffer.get() & BYTE_MASK;
451                         diff |= b << k;
452                     }
453                     if (nbits > 0) {
454                         b = readBuffer.get() & BYTE_MASK;
455                         diff |= b >>> -k;
456                         b &= (1 << nbits) - 1L;
457                     } else {
458                         b = 0;
459                     }
460                     lastpix = undoMappingAndDifferencing(lastpix, diff);
461                 }
462             } else {
463                 /* normal case, Rice coding */
464                 for (; i < imax; i++) {
465                     /* count number of leading zeros */
466                     while (b == 0) {
467                         nbits += BITS_PER_BYTE;
468                         b = readBuffer.get() & BYTE_MASK;
469                     }
470                     long nzero = nbits - NONZERO_COUNT[(int) (b & BYTE_MASK)];
471                     nbits -= nzero + 1;
472                     /* flip the leading one-bit */
473                     b ^= 1 << nbits;
474                     /* get the FS trailing bits */
475                     nbits -= fs;
476                     while (nbits < 0) {
477                         b = b << BITS_PER_BYTE | readBuffer.get() & BYTE_MASK;
478                         nbits += BITS_PER_BYTE;
479                     }
480                     long diff = nzero << fs | b >> nbits;
481                     b &= (1 << nbits) - 1L;
482 
483                     lastpix = undoMappingAndDifferencing(lastpix, diff);
484                 }
485             }
486         }
487         if (readBuffer.limit() > readBuffer.position()) {
488             LOG.warning("decompressing left over some extra bytes got: " + readBuffer.limit() + " but needed only "
489                     + readBuffer.position());
490         }
491 
492     }
