40  /**
41   * The original decompression code was written by Doug Tody, NRAO and included
42   * (ported to c and adapted) in cfitsio by William Pence, NASA/GSFC. That code
43   * was then ported to Java by R. van Nieuwenhoven. Later it was massively
44   * refactored to harmonize the different compression algorithms and reduce the
45   * duplicate code pieces without obscuring the algorithm itself as good as
46   * possible.
47   *
48   * @author Doug Tody
49   * @author William Pence
50   * @author Richard van Nieuwenhoven
51   */
52  public abstract class PLIOCompress {

54      public static class BytePLIOCompressor extends PLIOCompress implements ICompressor<ByteBuffer> {
55  
56          private ByteBuffer pixelData;
57  
58          @Override
59          public boolean compress(ByteBuffer buffer, ByteBuffer compressed) {
60              this.pixelData = buffer;
61              compress(compressed.asShortBuffer(), this.pixelData.limit());
62              return true;
63          }
64  
65          @Override
66          public void decompress(ByteBuffer compressed, ByteBuffer buffer) {
67              this.pixelData = buffer;
68              decompress(compressed.asShortBuffer(), this.pixelData.limit());
69          }
70  
71          @Override
72          protected int nextPixel() {
73              return this.pixelData.get();
74          }
75  
76          @Override
77          protected void put(int index, int pixel) {
78              this.pixelData.put(index, (byte) pixel);
79          }
80      }

141     private static final int FIRST_VALUE_WITH_13_BIT = 4096;
142 
143     private static final int FIRST_VALUE_WITH_14_BIT = 8192;
144 
145     private static final int FIRST_VALUE_WITH_15_BIT = 16384;
146 
147     private static final int FIRST_VALUE_WITH_16_BIT = 32768;
148 
149     private static final int HEADER_SIZE_FIELD1 = 3;
150 
151     private static final int HEADER_SIZE_FIELD2 = 4;
152 
153     private static final int LAST_VALUE_FITTING_IN_12_BIT = FIRST_VALUE_WITH_13_BIT - 1;
154 
155     private static final int MINI_HEADER_SIZE = 3;
156 
157     private static final int MINI_HEADER_SIZE_FIELD = 2;
158 
159     /**
160      * The exact meaning of this var is not clear at the moment of porting the
161      * algorithm to Java.
162      */
163     private static final int N20481 = 20481;
164 
165     private static final int OPCODE_1 = 1;
166 
167     private static final int OPCODE_2 = 2;
168 
169     private static final int OPCODE_3 = 3;
170 
171     private static final int OPCODE_4 = 4;
172 
173     private static final int OPCODE_5 = 5;
174 
175     private static final int OPCODE_6 = 6;
176 
177     private static final int OPCODE_7 = 7;
178 
179     private static final int OPCODE_8 = 8;

181     private static final short[] PLIO_HEADER = {
182         (short) 0,
183         (short) 7,
184         (short) -100,
185         (short) 0,
186         (short) 0,
187         (short) 0,
188         (short) 0
189     };
190 
191     private static final int SHIFT_12_BITS = 12;
192 
193     private static final int SHIFT_15_BITS = 15;
194 
195     private static final int VALUE_OF_BIT_13_AND14_ON = 12288;

197     /**
198      * PL_P2L -- Convert a pixel tiledImageOperation to a line list. The length
199      * of the list is returned as the function value.
200      *
201      * @param compressedData
202      *            encoded line list
203      * @param npix
204      *            number of pixels to convert
205      */
206     protected void compress(ShortBuffer compressedData, int npix) {
207         compressedData.put(PLIO_HEADER);
208         final int xe = npix - 1;
209         int op = PLIO_HEADER.length;
210         /* Computing MAX */
211         int pv = Math.max(0, nextPixel());
212         int x1 = 0;
213         int iz = 0;
214         int hi = 1;
215         int nv = 0;
216         for (int ip = 0; ip <= xe; ++ip) {
217             if (ip < xe) {
218                 /* Computing MAX */
219                 nv = Math.max(0, nextPixel());
220                 if (nv == pv) {
221                     continue;
222                 }
223                 if (pv == 0) {
224                     pv = nv;
225                     x1 = ip + 1;
226                     continue;
227                 }
228             } else {
229                 if (pv == 0) {
230                     x1 = xe + 1;
231                 }
232             }
233 
234             int np = ip - x1 + 1;
235             int nz = x1 - iz;
236             boolean skip = false;
237             if (pv > 0) {
238                 int dv = pv - hi;
239                 if (dv != 0) {
240                     hi = pv;
241                     if (Math.abs(dv) > LAST_VALUE_FITTING_IN_12_BIT) {
242                         compressedData.put(op, (short) ((pv & LAST_VALUE_FITTING_IN_12_BIT) + FIRST_VALUE_WITH_13_BIT));
243                         ++op;
244                         compressedData.put(op, (short) (pv / FIRST_VALUE_WITH_13_BIT));
245                         ++op;
246                     } else {
247                         if (dv < 0) {
248                             compressedData.put(op, (short) (-dv + VALUE_OF_BIT_13_AND14_ON));
249                         } else {
250                             compressedData.put(op, (short) (dv + FIRST_VALUE_WITH_14_BIT));
251                         }
252                         ++op;
253                         if (np == 1 && nz == 0) {
254                             int v = compressedData.get(op - 1);
255                             compressedData.put(op - 1, (short) (v | FIRST_VALUE_WITH_15_BIT));
256                             skip = true;
257                         }
258                     }
259                 }
260             }
261             if (!skip) {
262                 if (nz > 0) {
263                     while (nz > 0) {
264                         compressedData.put(op, (short) Math.min(LAST_VALUE_FITTING_IN_12_BIT, nz));
265                         ++op;
266                         nz += -LAST_VALUE_FITTING_IN_12_BIT;
267                     }
268                     if (np == 1 && pv > 0) {
269                         compressedData.put(op - 1, (short) (compressedData.get(op - 1) + N20481));
270                         skip = true;
271                     }
272                 }
273             }
274             if (!skip) {
275                 while (np > 0) {
276                     compressedData.put(op, (short) (Math.min(LAST_VALUE_FITTING_IN_12_BIT, np) + FIRST_VALUE_WITH_15_BIT));
277                     ++op;
278                     np += -LAST_VALUE_FITTING_IN_12_BIT;
279                 }
280             }
281             x1 = ip + 1;
282             iz = x1;
283             pv = nv;
284         }
285         compressedData.put(HEADER_SIZE_FIELD1, (short) (op % FIRST_VALUE_WITH_16_BIT));
286         compressedData.put(HEADER_SIZE_FIELD2, (short) (op / FIRST_VALUE_WITH_16_BIT));
287         compressedData.position(op);
288     }

290     /**
291      * PL_L2PI -- Translate a PLIO line list into an integer pixel
292      * tiledImageOperation. The number of pixels output (always npix) is
293      * returned as the function value.
294      *
295      * @param compressedData
296      *            encoded line list
297      * @param npix
298      *            number of pixels to convert
299      * @return number of pixels converted
300      */
301     protected int decompress(ShortBuffer compressedData, int npix) {
302         int llfirt;
303         int lllen;
304         if (!(compressedData.get(2) > 0)) {
305             lllen = (compressedData.get(HEADER_SIZE_FIELD2) << SHIFT_15_BITS) + compressedData.get(HEADER_SIZE_FIELD1);
306             llfirt = compressedData.get(1);
307         } else {
308             lllen = compressedData.get(MINI_HEADER_SIZE_FIELD);
309             llfirt = MINI_HEADER_SIZE;
310         }
311         final int xe = npix;
312         int op = 0;
313         int x1 = 1;
314         int pv = 1;
315         for (int ip = llfirt; ip <= lllen; ++ip) {
316             final int opcode = compressedData.get(ip) / FIRST_VALUE_WITH_13_BIT;
317             final int data = compressedData.get(ip) & LAST_VALUE_FITTING_IN_12_BIT;
318             final int sw0001 = opcode + 1;
319             if (sw0001 == OPCODE_1 || sw0001 == OPCODE_5 || sw0001 == OPCODE_6) {
320                 final int x2 = x1 + data - 1;
321                 final int i2 = Math.min(x2, xe);
322                 final int np = i2 - Math.max(x1, 0) + 1;
323                 if (np > 0) {
324                     final int otop = op + np - 1;
325                     if (!(opcode == OPCODE_4)) {
326                         for (int index = op; index <= otop; ++index) {
327                             put(index, 0);
328                         }
329                         if (opcode == OPCODE_5 && i2 == x2) {
330                             put(otop, pv);
331                         }
332                     } else {
333                         for (int index = op; index <= otop; ++index) {
334                             put(index, pv);
335                         }
336                     }
337                     op = otop + 1;
338                 }
339                 x1 = x2 + 1;
340             } else if (sw0001 == OPCODE_2) {
341                 pv = (compressedData.get(ip + 1) << SHIFT_12_BITS) + data;
342                 ++ip;
343             } else if (sw0001 == OPCODE_3) {
344                 pv += data;
345             } else if (sw0001 == OPCODE_4) {
346                 pv -= data;
347             } else if (sw0001 == OPCODE_7) {
348                 pv += data;
349                 if (x1 >= 0 && x1 <= xe) {
350                     put(op, pv);
351                     ++op;
352                 }
353                 ++x1;
354             } else if (sw0001 == OPCODE_8) {
355                 pv -= data;
356                 if (x1 >= 0 && x1 <= xe) {
357                     put(op, pv);
358                     ++op;
359                 }
360                 ++x1;
361             }
362             if (x1 > xe) {
363                 break;
364             }
365         }
366         for (int index = op; index < npix; ++index) {
367             put(index, 0);
368         }
369         return npix;
370     }
