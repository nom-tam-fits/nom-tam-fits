94      /**
95       * Creates a new set of quantization options, to be used together with the specified compression options.
96       * 
97       * @param compressOption Compression-specific options to pair with these quantization options, or <code>null</code>.
98       * 
99       * @since 1.18
100      */
101     public QuantizeOption(ICompressOption compressOption) {
102         parameters = new QuantizeParameters(this);
103         this.compressOption = compressOption;
104     }
105 
106     @Override
107     public QuantizeOption copy() {
108         try {
109             QuantizeOption copy = (QuantizeOption) clone();
110             if (compressOption != null) {
111                 copy.compressOption = compressOption.copy();
112             }
113             copy.parameters = this.parameters.copy(copy);
114             return copy;
115         } catch (CloneNotSupportedException e) {
116             throw new IllegalStateException("option could not be cloned", e);
117         }
118     }

288     @Override
289     public void setParameters(ICompressParameters parameters) {
290         if (parameters instanceof QuantizeParameters) {
291             this.parameters = (QuantizeParameters) parameters.copy(this);
292         } else if (parameters instanceof BundledParameters) {
293             BundledParameters bundle = (BundledParameters) parameters;
294             for (int i = 0; i < bundle.size(); i++) {
295                 setParameters(bundle.get(i));
296             }
297         } else if (compressOption != null) {
298             compressOption.setParameters(parameters);
299         }
300     }

354     @Override
355     public <T> T unwrap(Class<T> clazz) {
356         if (clazz.isAssignableFrom(this.getClass())) {
357             return clazz.cast(this);
358         }
359         if (compressOption != null) {
360             if (clazz.isAssignableFrom(compressOption.getClass())) {
361                 return clazz.cast(compressOption);
362             }
363         }
364         return null;
365     }
