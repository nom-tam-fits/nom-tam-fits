40  /**
41   * Quantization options when they are part of the compression scheme. When compressing tables and images includes
42   * quantization (integer representation of floating point data), users can control how exactly the quantization should
43   * be performed. When reading compressed FITS files, these options will be set automatically based on the header values
44   * recorded in the compressed HDU.
45   * 
46   * @see nom.tam.image.compression.hdu.CompressedImageHDU#setQuantAlgorithm(String)
47   * @see nom.tam.image.compression.hdu.CompressedImageHDU#getCompressOption(Class)
48   */
49  public class QuantizeOption implements ICompressOption {

52       * and including NULL_VALUE. These values may not be used to represent the quantized and scaled floating point pixel
53       * values If lossy Hcompression is used, and the tiledImageOperation contains null values, then it is also possible

55       * little more space value used to represent undefined pixels

104     /**
105      * Creates a new set of quantization options, to be used together with the specified compression options.

107      * @param compressOption Compression-specific options to pair with these quantization options, or <code>null</code>.
108      *

111     public QuantizeOption(ICompressOption compressOption) {
112         parameters = new QuantizeParameters(this);
113         this.compressOption = compressOption;

116     @Override
117     public QuantizeOption copy() {
118         try {
119             QuantizeOption copy = (QuantizeOption) clone();
120             if (compressOption != null) {
121                 copy.compressOption = compressOption.copy();
122             }
123             copy.parameters = parameters.copy(copy);
124             return copy;
125         } catch (CloneNotSupportedException e) {
126             throw new IllegalStateException("option could not be cloned", e);
127         }

632     @Override
633     public void setParameters(ICompressParameters parameters) {
634         if (parameters instanceof QuantizeParameters) {
635             this.parameters = (QuantizeParameters) parameters.copy(this);
636         } else if (parameters instanceof BundledParameters) {
637             BundledParameters bundle = (BundledParameters) parameters;
638             for (int i = 0; i < bundle.size(); i++) {
639                 setParameters(bundle.get(i));
640             }
641         } else if (compressOption != null) {
642             compressOption.setParameters(parameters);
643         }
644     }

711     @Override

713         if (clazz.isAssignableFrom(this.getClass())) {
714             return clazz.cast(this);
715         }
716         if (compressOption != null) {
717             if (clazz.isAssignableFrom(compressOption.getClass())) {
