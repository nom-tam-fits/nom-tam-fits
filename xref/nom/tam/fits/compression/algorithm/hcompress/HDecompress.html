174     /**
175      * log2n is log2 of max(nx,ny) rounded up to next power of 2
176      */
177     private int calculateLog2N(int nmax) {
178         int log2n;
179         log2n = (int) (Math.log(nmax) / Math.log(2.0) + ROUNDING_HALF);
180         if (nmax > 1 << log2n) {
181             log2n += 1;
182         }
183         return log2n;
184     }

186     /**
187      * char *infile; input file long *a; address of output tiledImageOperation [nx][ny] int *nx,*ny; size of output
188      * tiledImageOperation int *scale; scale factor for digitization
189      *
190      * @param infile
191      * @param a
192      */
193     private void decode64(ByteBuffer infile, LongArrayPointer a) {
194         byte[] nbitplanes = new byte[N03];
195         byte[] tmagic = new byte[2];
196 
197         /*
198          * File starts either with special 2-byte magic code or with FITS keyword "SIMPLE  ="
199          */
200         infile.get(tmagic);
201         /*
202          * check for correct magic code value
203          */
204         if (tmagic[0] != CODE_MAGIC[0] || tmagic[1] != CODE_MAGIC[1]) {
205             throw new RuntimeException("Compression error");
206         }
207         this.nx = infile.getInt(); /* x size of image */
208         this.ny = infile.getInt(); /* y size of image */
209         this.scale = infile.getInt(); /* scale factor for digitization */
210 
211         /* sum of all pixels */
212         long sumall = infile.getLong();
213         /* # bits in quadrants */
214 
215         infile.get(nbitplanes);
216 
217         dodecode64(infile, a, nbitplanes);
218         /*
219          * put sum of all pixels back into pixel 0
220          */
221         a.set(0, sumall);
222     }

224     /**
225      * decompress the input byte stream using the H-compress algorithm input - input tiledImageOperation of compressed
226      * bytes a - pre-allocated tiledImageOperation to hold the output uncompressed image nx - returned X axis size ny -
227      * returned Y axis size NOTE: the nx and ny dimensions as defined within this code are reversed from the usual FITS
228      * notation. ny is the fastest varying dimension, which is usually considered the X axis in the FITS image display
229      *
230      * @param input the input buffer to decompress
231      * @param smooth should the image be smoothed
232      * @param aa the resulting long tiledImageOperation
233      */
234     public void decompress(ByteBuffer input, boolean smooth, long[] aa) {
235 
236         LongArrayPointer a = new LongArrayPointer(aa);
237 
238         /* decode the input tiledImageOperation */
239 
240         decode64(input, a);
241 
242         /*
243          * Un-Digitize
244          */
245         undigitize64(a);
246 
247         /*
248          * Inverse H-transform
249          */
250         hinv64(a, smooth);
251 
252     }

254     /**
255      * long a[]; int nx,ny; Array dimensions are [nx][ny] unsigned char nbitplanes[3]; Number of bit planes in quadrants
256      */
257     private int dodecode64(ByteBuffer infile, LongArrayPointer a, byte[] nbitplanes) {
258         int nel = this.nx * this.ny;
259         int nx2 = (this.nx + 1) / 2;
260         int ny2 = (this.ny + 1) / 2;
261         /*
262          * initialize a to zero
263          */
264         for (int i = 0; i < nel; i++) {
265             a.set(i, 0);
266         }
267         /*
268          * Initialize bit input
269          */
270         startInputingBits();
271         /*
272          * read bit planes for each quadrant
273          */
274         qtreeDecode64(infile, a.copy(0), this.ny, nx2, ny2, nbitplanes[0]);
275 
276         qtreeDecode64(infile, a.copy(ny2), this.ny, nx2, this.ny / 2, nbitplanes[1]);
277 
278         qtreeDecode64(infile, a.copy(this.ny * nx2), this.ny, this.nx / 2, ny2, nbitplanes[1]);
279 
280         qtreeDecode64(infile, a.copy(this.ny * nx2 + ny2), this.ny, this.nx / 2, this.ny / 2, nbitplanes[2]);
281 
282         /*
283          * make sure there is an EOF symbol (nybble=0) at end
284          */
285         if (inputNybble(infile) != 0) {
286             throw new RuntimeException("Compression error");
287         }
288         /*
289          * now get the sign bits Re-initialize bit input
290          */
291         startInputingBits();
292         for (int i = 0; i < nel; i++) {
293             if (a.get(i) != 0) {
294                 if (inputBit(infile) != 0) {
295                     a.set(i, -a.get(i));
296                 }
297             }
298         }
299         return 0;
300     }

302     /**
303      * int smooth; 0 for no smoothing, else smooth during inversion int scale; used if smoothing is specified
304      */
305     private int hinv64(LongArrayPointer a, boolean smooth) {
306         int nmax = this.nx > this.ny ? this.nx : this.ny;
307         int log2n = calculateLog2N(nmax);
308         // get temporary storage for shuffling elements
309         long[] tmp = new long[(nmax + 1) / 2];
310         // set up masks, rounding parameters
311         int shift = 1;
312         long bit0 = (long) 1 << log2n - 1;
313         long bit1 = bit0 << 1;
314         long bit2 = bit0 << 2;
315         long mask0 = -bit0;
316         long mask1 = mask0 << 1;
317         long mask2 = mask0 << 2;
318         long prnd0 = bit0 >> 1;
319         long prnd1 = bit1 >> 1;
320         long prnd2 = bit2 >> 1;
321         long nrnd0 = prnd0 - 1;
322         long nrnd1 = prnd1 - 1;
323         long nrnd2 = prnd2 - 1;
324         // round h0 to multiple of bit2
325         a.set(0, a.get(0) + (a.get(0) >= 0 ? prnd2 : nrnd2) & mask2);
326         // do log2n expansions We're indexing a as a 2-D tiledImageOperation
327         // with dimensions
328         // (nx,ny).
329         int nxtop = 1;
330         int nytop = 1;
331         int nxf = this.nx;
332         int nyf = this.ny;
333         int c = 1 << log2n;
334         int i;
335         for (int k = log2n - 1; k >= 0; k--) {
336             // this somewhat cryptic code generates the sequence ntop[k-1] =
337             // (ntop[k]+1)/2, where ntop[log2n] = n
338             c = c >> 1;
339             nxtop = nxtop << 1;
340             nytop = nytop << 1;
341             if (nxf <= c) {
342                 nxtop -= 1;
343             } else {
344                 nxf -= c;
345             }
346             if (nyf <= c) {
347                 nytop -= 1;
348             } else {
349                 nyf -= c;
350             }
351             // double shift and fix nrnd0 (because prnd0=0) on last pass
352             if (k == 0) {
353                 nrnd0 = 0;
354                 shift = 2;
355             }
356             // unshuffle in each dimension to interleave coefficients
357             for (i = 0; i < nxtop; i++) {
358                 unshuffle64(a.copy(this.ny * i), nytop, 1, tmp);
359             }
360             for (int j = 0; j < nytop; j++) {
361                 unshuffle64(a.copy(j), nxtop, this.ny, tmp);
362             }
363             // smooth by interpolating coefficients if SMOOTH != 0
364             if (smooth) {
365                 hsmooth64(a, nxtop, nytop);
366             }
367             int oddx = nxtop % 2;
368             int oddy = nytop % 2;
369             for (i = 0; i < nxtop - oddx; i += 2) {
370                 int s00 = this.ny * i; /* s00 is index of a[i,j] */
371                 int s10 = s00 + this.ny; /* s10 is index of a[i+1,j] */
372                 for (int j = 0; j < nytop - oddy; j += 2) {
373                     long h0 = a.get(s00);
374                     long hx = a.get(s10);
375                     long hy = a.get(s00 + 1);
376                     long hc = a.get(s10 + 1);
377                     // round hx and hy to multiple of bit1, hc to multiple of
378                     // bit0 h0 is already a multiple of bit2
379                     hx = hx + (hx >= 0 ? prnd1 : nrnd1) & mask1;
380                     hy = hy + (hy >= 0 ? prnd1 : nrnd1) & mask1;
381                     hc = hc + (hc >= 0 ? prnd0 : nrnd0) & mask0;
382                     // propagate bit0 of hc to hx,hy
383                     long lowbit0 = hc & bit0;
384                     hx = hx >= 0 ? hx - lowbit0 : hx + lowbit0;
385                     hy = hy >= 0 ? hy - lowbit0 : hy + lowbit0;
386                     // Propagate bits 0 and 1 of hc,hx,hy to h0. This could be
387                     // simplified if we assume h0>0, but then the inversion
388                     // would not be lossless for images with negative pixels.
389                     long lowbit1 = (hc ^ hx ^ hy) & bit1;
390                     h0 = h0 >= 0 ? h0 + lowbit0 - lowbit1 : h0 + (lowbit0 == 0 ? lowbit1 : lowbit0 - lowbit1);
391                     // Divide sums by 2 (4 last time)
392                     a.set(s10 + 1, h0 + hx + hy + hc >> shift);
393                     a.set(s10, h0 + hx - hy - hc >> shift);
394                     a.set(s00 + 1, h0 - hx + hy - hc >> shift);
395                     a.set(s00, h0 - hx - hy + hc >> shift);
396                     s00 += 2;
397                     s10 += 2;
398                 }
399                 if (oddy != 0) {
400                     // do last element in row if row length is odd s00+1, s10+1
401                     // are off edge
402                     long h0 = a.get(s00);
403                     long hx = a.get(s10);
404                     hx = (hx >= 0 ? hx + prnd1 : hx + nrnd1) & mask1;
405                     long lowbit1 = hx & bit1;
406                     h0 = h0 >= 0 ? h0 - lowbit1 : h0 + lowbit1;
407                     a.set(s10, h0 + hx >> shift);
408                     a.set(s00, h0 - hx >> shift);
409                 }
410             }
411             if (oddx != 0) {
412                 // do last row if column length is odd s10, s10+1 are off edge
413                 int s00 = this.ny * i;
414                 for (int j = 0; j < nytop - oddy; j += 2) {
415                     long h0 = a.get(s00);
416                     long hy = a.get(s00 + 1);
417                     hy = (hy >= 0 ? hy + prnd1 : hy + nrnd1) & mask1;
418                     long lowbit1 = hy & bit1;
419                     h0 = h0 >= 0 ? h0 - lowbit1 : h0 + lowbit1;
420                     a.set(s00 + 1, h0 + hy >> shift);
421                     a.set(s00, h0 - hy >> shift);
422                     s00 += 2;
423                 }
424                 if (oddy != 0) {
425                     // do corner element if both row and column lengths are odd
426                     // s00+1, s10, s10+1 are off edge
427                     long h0 = a.get(s00);
428                     a.set(s00, h0 >> shift);
429                 }
430             }
431             // divide all the masks and rounding values by 2
432             bit1 = bit0;
433             bit0 = bit0 >> 1;
434             mask1 = mask0;
435             mask0 = mask0 >> 1;
436             prnd1 = prnd0;
437             prnd0 = prnd0 >> 1;
438             nrnd1 = nrnd0;
439             nrnd0 = prnd0 - 1;
440         }
441         return 0;
442     }

444     /**
445      * long a[]; tiledImageOperation of H-transform coefficients int nxtop,nytop; size of coefficient block to use int
446      * ny; actual 1st dimension of tiledImageOperation int scale; truncation scale factor that was used
447      */
448     private void hsmooth64(LongArrayPointer a, int nxtop, int nytop) {
449         int i, j;
450         int ny2, s10, s00;
451         long hm, h0, hp, hmm, hpm, hmp, hpp, hx2, hy2, diff, dmax, dmin, s, smax, m1, m2;
452 
453         /*
454          * Maximum change in coefficients is determined by scale factor. Since we rounded during division (see
455          * digitize.c), the biggest permitted change is scale/2.
456          */
457         smax = this.scale >> 1;
458         if (smax <= 0) {
459             return;
460         }
461         ny2 = this.ny << 1;
462         /*
463          * We're indexing a as a 2-D tiledImageOperation with dimensions (nxtop,ny) of which only (nxtop,nytop) are
464          * used. The coefficients on the edge of the tiledImageOperation are not adjusted (which is why the loops below
465          * start at 2 instead of 0 and end at nxtop-2 instead of nxtop.)
466          */
467         /*
468          * Adjust x difference hx
469          */
470         for (i = 2; i < nxtop - 2; i += 2) {
471             s00 = this.ny * i; /* s00 is index of a[i,j] */
472             s10 = s00 + this.ny; /* s10 is index of a[i+1,j] */
473             for (j = 0; j < nytop; j += 2) {
474                 /*
475                  * hp is h0 (mean value) in next x zone, hm is h0 in previous x zone
476                  */
477                 hm = a.get(s00 - ny2);
478                 h0 = a.get(s00);
479                 hp = a.get(s00 + ny2);
480                 /*
481                  * diff = 8 * hx slope that would match h0 in neighboring zones
482                  */
483                 diff = hp - hm;
484                 /*
485                  * monotonicity constraints on diff
486                  */
487                 dmax = Math.max(Math.min(hp - h0, h0 - hm), 0) << 2;
488                 dmin = Math.min(Math.max(hp - h0, h0 - hm), 0) << 2;
489                 /*
490                  * if monotonicity would set slope = 0 then don't change hx. note dmax>=0, dmin<=0.
491                  */
492                 if (dmin < dmax) {
493                     diff = Math.max(Math.min(diff, dmax), dmin);
494                     /*
495                      * Compute change in slope limited to range +/- smax. Careful with rounding negative numbers when
496                      * using shift for divide by 8.
497                      */
498                     s = diff - (a.get(s10) << N03);
499                     s = s >= 0 ? s >> N03 : s + N07 >> N03;
500                     s = Math.max(Math.min(s, smax), -smax);
501                     a.set(s10, a.get(s10) + s);
502                 }
503                 s00 += 2;
504                 s10 += 2;
505             }
506         }
507         /*
508          * Adjust y difference hy
509          */
510         for (i = 0; i < nxtop; i += 2) {
511             s00 = this.ny * i + 2;
512             s10 = s00 + this.ny;
513             for (j = 2; j < nytop - 2; j += 2) {
514                 hm = a.get(s00 - 2);
515                 h0 = a.get(s00);
516                 hp = a.get(s00 + 2);
517                 diff = hp - hm;
518                 dmax = Math.max(Math.min(hp - h0, h0 - hm), 0) << 2;
519                 dmin = Math.min(Math.max(hp - h0, h0 - hm), 0) << 2;
520                 if (dmin < dmax) {
521                     diff = Math.max(Math.min(diff, dmax), dmin);
522                     s = diff - (a.get(s00 + 1) << N03);
523                     s = s >= 0 ? s >> N03 : s + N07 >> N03;
524                     s = Math.max(Math.min(s, smax), -smax);
525                     a.set(s00 + 1, a.get(s00 + 1) + s);
526                 }
527                 s00 += 2;
528                 s10 += 2;
529             }
530         }
531         /*
532          * Adjust curvature difference hc
533          */
534         for (i = 2; i < nxtop - 2; i += 2) {
535             s00 = this.ny * i + 2;
536             s10 = s00 + this.ny;
537             for (j = 2; j < nytop - 2; j += 2) {
538                 /*
539                  * ------------------ y | hmp | | hpp | | ------------------ | | | h0 | | | ------------------ -------x
540                  * | hmm | | hpm | ------------------
541                  */
542                 hmm = a.get(s00 - ny2 - 2);
543                 hpm = a.get(s00 + ny2 - 2);
544                 hmp = a.get(s00 - ny2 + 2);
545                 hpp = a.get(s00 + ny2 + 2);
546                 h0 = a.get(s00);
547                 /*
548                  * diff = 64 * hc value that would match h0 in neighboring zones
549                  */
550                 diff = hpp + hmm - hmp - hpm;
551                 /*
552                  * 2 times x,y slopes in this zone
553                  */
554                 hx2 = a.get(s10) << 1;
555                 hy2 = a.get(s00 + 1) << 1;
556                 /*
557                  * monotonicity constraints on diff
558                  */
559                 m1 = Math.min(Math.max(hpp - h0, 0) - hx2 - hy2, Math.max(h0 - hpm, 0) + hx2 - hy2);
560                 m2 = Math.min(Math.max(h0 - hmp, 0) - hx2 + hy2, Math.max(hmm - h0, 0) + hx2 + hy2);
561                 dmax = Math.min(m1, m2) << BITS_OF_1_NYBBLE;
562                 m1 = Math.max(Math.min(hpp - h0, 0) - hx2 - hy2, Math.min(h0 - hpm, 0) + hx2 - hy2);
563                 m2 = Math.max(Math.min(h0 - hmp, 0) - hx2 + hy2, Math.min(hmm - h0, 0) + hx2 + hy2);
564                 dmin = Math.max(m1, m2) << BITS_OF_1_NYBBLE;
565                 /*
566                  * if monotonicity would set slope = 0 then don't change hc. note dmax>=0, dmin<=0.
567                  */
568                 if (dmin < dmax) {
569                     diff = Math.max(Math.min(diff, dmax), dmin);
570                     /*
571                      * Compute change in slope limited to range +/- smax. Careful with rounding negative numbers when
572                      * using shift for divide by 64.
573                      */
574                     s = diff - (a.get(s10 + 1) << N06);
575                     s = s >= 0 ? s >> N06 : s + N63 >> N06;
576                     s = Math.max(Math.min(s, smax), -smax);
577                     a.set(s10 + 1, a.get(s10 + 1) + s);
578                 }
579                 s00 += 2;
580                 s10 += 2;
581             }
582         }
583     }

585     private int inputBit(ByteBuffer infile) {
586         if (this.bitsToGo == 0) { /* Read the next byte if no */
587 
588             this.buffer2 = infile.get() & BYTE_MASK;
589 
590             this.bitsToGo = BITS_OF_1_BYTE;
591         }
592         /*
593          * Return the next bit
594          */
595         this.bitsToGo -= 1;
596         return this.buffer2 >> this.bitsToGo & 1;
597     }

599     /*
600      * Huffman decoding for fixed codes Coded values range from 0-15 Huffman code values (hex): 3e, 00, 01, 08, 02, 09,
601      * 1a, 1b, 03, 1c, 0a, 1d, 0b, 1e, 3f, 0c and number of bits in each code: 6, 3, 3, 4, 3, 4, 5, 5, 3, 5, 4, 5, 4, 5,
602      * 6, 4
603      */
604     private int inputHuffman(ByteBuffer infile) {
605         int c;
606 
607         /*
608          * get first 3 bits to start
609          */
610         c = inputNbits(infile, N03);
611         if (c < N04) {
612             /*
613              * this is all we need return 1,2,4,8 for c=0,1,2,3
614              */
615             return 1 << c;
616         }
617         /*
618          * get the next bit
619          */
620         c = inputBit(infile) | c << 1;
621         if (c < N13) {
622             /*
623              * OK, 4 bits is enough
624              */
625             switch (c) {
626             case N08:
627                 return N03;
628             case N09:
629                 return N05;
630             case N10:
631                 return N10;
632             case N11:
633                 return N12;
634             case N12:
635                 return N15;
636             default:
637             }
638         }
639         /*
640          * get yet another bit
641          */
642         c = inputBit(infile) | c << 1;
643         if (c < N31) {
644             /*
645              * OK, 5 bits is enough
646              */
647             switch (c) {
648             case N26:
649                 return N06;
650             case N27:
651                 return N07;
652             case N28:
653                 return N09;
654             case N29:
655                 return N11;
656             case N30:
657                 return N13;
658             default:
659             }
660         }
661         /*
662          * need the 6th bit
663          */
664         c = inputBit(infile) | c << 1;
665         if (c == N62) {
666             return 0;
667         }
668         return N14;
669     }

671     private int inputNbits(ByteBuffer infile, int n) {
672         if (this.bitsToGo < n) {
673             /*
674              * need another byte's worth of bits
675              */
676 
677             this.buffer2 = this.buffer2 << BITS_OF_1_BYTE | infile.get() & BYTE_MASK;
678             this.bitsToGo += BITS_OF_1_BYTE;
679         }
680         /*
681          * now pick off the first n bits
682          */
683         this.bitsToGo -= n;
684 
685         /* there was a slight gain in speed by replacing the following line */
686         /* return( (buffer2>>bits_to_go) & ((1<<n)-1) ); */
687         return this.buffer2 >> this.bitsToGo & MASKS[n];
688     }

692     private int inputNnybble(ByteBuffer infile, int n, byte[] array) {
693         /*
694          * copy n 4-bit nybbles from infile to the lower 4 bits of tiledImageOperation
695          */
696 
697         int ii, kk, shift1, shift2;
698 
699         /*
700          * forcing byte alignment doesn;t help, and even makes it go slightly slower if (bits_to_go != 8)
701          * input_nbits(infile, bits_to_go);
702          */
703         if (n == 1) {
704             array[0] = (byte) inputNybble(infile);
705             return 0;
706         }
707 
708         if (this.bitsToGo == BITS_OF_1_BYTE) {
709             /*
710              * already have 2 full nybbles in buffer2, so backspace the infile tiledImageOperation to reuse last char
711              */
712             infile.position(infile.position() - 1);
713             this.bitsToGo = 0;
714         }
715 
716         /* bits_to_go now has a value in the range 0 - 7. After adding */
717         /* another byte, bits_to_go effectively will be in range 8 - 15 */
718 
719         shift1 = this.bitsToGo + BITS_OF_1_NYBBLE; /*
720                                                     * shift1 will be in range 4 - 11
721                                                     */
722         shift2 = this.bitsToGo; /* shift2 will be in range 0 - 7 */
723         kk = 0;
724 
725         /* special case */
726         if (this.bitsToGo == 0) {
727             for (ii = 0; ii < n / 2; ii++) {
728                 /*
729                  * refill the buffer with next byte
730                  */
731                 this.buffer2 = this.buffer2 << BITS_OF_1_BYTE | infile.get() & BYTE_MASK;
732                 array[kk] = (byte) (this.buffer2 >> BITS_OF_1_NYBBLE & NYBBLE_MASK);
733                 array[kk + 1] = (byte) (this.buffer2 & NYBBLE_MASK); /*
734                                                                       * no shift required
735                                                                       */
736                 kk += 2;
737             }
738         } else {
739             for (ii = 0; ii < n / 2; ii++) {
740                 /*
741                  * refill the buffer with next byte
742                  */
743                 this.buffer2 = this.buffer2 << BITS_OF_1_BYTE | infile.get() & BYTE_MASK;
744                 array[kk] = (byte) (this.buffer2 >> shift1 & NYBBLE_MASK);
745                 array[kk + 1] = (byte) (this.buffer2 >> shift2 & NYBBLE_MASK);
746                 kk += 2;
747             }
748         }
749 
750         if (ii * 2 != n) { /* have to read last odd byte */
751             array[n - 1] = (byte) inputNybble(infile);
752         }
753 
754         return this.buffer2 >> this.bitsToGo & NYBBLE_MASK;
755     }

757     private int inputNybble(ByteBuffer infile) {
758         if (this.bitsToGo < BITS_OF_1_NYBBLE) {
759             /*
760              * need another byte's worth of bits
761              */
762 
763             this.buffer2 = this.buffer2 << BITS_OF_1_BYTE | infile.get() & BYTE_MASK;
764             this.bitsToGo += BITS_OF_1_BYTE;
765         }
766         /*
767          * now pick off the first 4 bits
768          */
769         this.bitsToGo -= BITS_OF_1_NYBBLE;
770 
771         return this.buffer2 >> this.bitsToGo & NYBBLE_MASK;
772     }

774     /**
775      * Copy 4-bit values from a[(nx+1)/2,(ny+1)/2] to b[nx,ny], expanding each value to 2x2 pixels and inserting into
776      * bitplane BIT of B. A,B may NOT be same tiledImageOperation (it wouldn't make sense to be inserting bits into the
777      * same tiledImageOperation anyway.)
778      */
779     private void qtreeBitins64(byte[] a, int lnx, int lny, LongArrayPointer b, int n, int bit) {
780         int i, j, s00;
781         long planeVal = 1L << bit;
782         // expand each 2x2 block
783         ByteBuffer k = ByteBuffer.wrap(a); /* k is index of a[i/2,j/2] */
784         for (i = 0; i < lnx - 1; i += 2) {
785             s00 = n * i; /* s00 is index of b[i,j] */
786             // Note: this code appears to run very slightly faster on a 32-bit
787             // linux machine using s00+n rather than the s10 intermediate
788             // variable
789             // s10 = s00+n; *//* s10 is index of b[i+1,j]
790             for (j = 0; j < lny - 1; j += 2) {
791                 byte value = k.get();
792                 if ((value & BIT_ONE) != ZERO) {
793                     b.bitOr(s00 + n + 1, planeVal);
794                 }
795                 if ((value & BIT_TWO) != ZERO) {
796                     b.bitOr(s00 + n, planeVal);
797                 }
798                 if ((value & BIT_THREE) != ZERO) {
799                     b.bitOr(s00 + 1, planeVal);
800                 }
801                 if ((value & BIT_FOUR) != ZERO) {
802                     b.bitOr(s00, planeVal);
803                 }
804                 // b.bitOr(s10+1, ((LONGLONG) ( a[k] & 1)) << bit; b.bitOr(s10 ,
805                 // ((((LONGLONG)a[k])>>1) & 1) << bit; b.bitOr(s00+1,
806                 // ((((LONGLONG)a[k])>>2) & 1) << bit; b.bitOr(s00
807                 // ,((((LONGLONG)a[k])>>3) & 1) << bit;
808                 s00 += 2;
809                 /* s10 += 2; */
810             }
811             if (j < lny) {
812                 // row size is odd, do last element in row s00+1, s10+1 are off
813                 // edge
814                 byte value = k.get();
815                 if ((value & BIT_TWO) != ZERO) {
816                     b.bitOr(s00 + n, planeVal);
817                 }
818                 if ((value & BIT_FOUR) != ZERO) {
819                     b.bitOr(s00, planeVal);
820                 }
821                 // b.bitOr(s10 , ((((LONGLONG)a[k])>>1) & 1) << bit; b.bitOr(s00
822                 // , ((((LONGLONG)a[k])>>3) & 1) << bit;
823             }
824         }
825         if (i < lnx) {
826             // column size is odd, do last row s10, s10+1 are off edge
827             s00 = n * i;
828             for (j = 0; j < lny - 1; j += 2) {
829                 byte value = k.get();
830                 if ((value & BIT_THREE) != ZERO) {
831                     b.bitOr(s00 + 1, planeVal);
832                 }
833                 if ((value & BIT_FOUR) != ZERO) {
834                     b.bitOr(s00, planeVal);
835                 } // b.bitOr(s00+1, ((((LONGLONG)a[k])>>2) & 1) << bit;
836                   // b.bitOr(s00 , ((((LONGLONG)a[k])>>3) & 1) << bit;
837                 s00 += 2;
838             }
839             if (j < lny) {
840                 // both row and column size are odd, do corner element s00+1,
841                 // s10, s10+1 are off edge
842                 if ((k.get() & BIT_FOUR) != ZERO) {
843                     b.bitOr(s00, planeVal);
844                 }
845                 // b.bitOr(s00 , ((((LONGLONG)a[k])>>3) & 1) << bit;
846             }
847         }
848     }

850     /**
851      * copy 4-bit values from a[(nx+1)/2,(ny+1)/2] to b[nx,ny], expanding each value to 2x2 pixels a,b may be same
852      * tiledImageOperation
853      */
854     private void qtreeCopy(byte[] a, int lnx, int lny, byte[] b, int n) {
855         int i, j, k, nx2, ny2;
856         int s00, s10;
857         // first copy 4-bit values to b start at end in case a,b are same
858         // tiledImageOperation
859         nx2 = (lnx + 1) / 2;
860         ny2 = (lny + 1) / 2;
861         k = ny2 * (nx2 - 1) + ny2 - 1; /* k is index of a[i,j] */
862         for (i = nx2 - 1; i >= 0; i--) {
863             s00 = 2 * (n * i + ny2 - 1); /* s00 is index of b[2*i,2*j] */
864             for (j = ny2 - 1; j >= 0; j--) {
865                 b[s00] = a[k];
866                 k -= 1;
867                 s00 -= 2;
868             }
869         }
870         for (i = 0; i < lnx - 1; i += 2) { // now expand each 2x2 block
871             // Note: Unlike the case in qtree_bitins, this code runs faster on a
872             // 32-bit linux machine using the s10 intermediate variable, rather
873             // that using s00+n. Go figure!
874             s00 = n * i; // s00 is index of b[i,j]
875             s10 = s00 + n; // s10 is index of b[i+1,j]
876             for (j = 0; j < lny - 1; j += 2) {
877                 b[s10 + 1] = (b[s00] & BIT_ONE) == ZERO ? ZERO : BIT_ONE;
878                 b[s10] = (b[s00] & BIT_TWO) == ZERO ? ZERO : BIT_ONE;
879                 b[s00 + 1] = (b[s00] & BIT_THREE) == ZERO ? ZERO : BIT_ONE;
880                 b[s00] = (b[s00] & BIT_FOUR) == ZERO ? ZERO : BIT_ONE;
881                 s00 += 2;
882                 s10 += 2;
883             }
884             if (j < lny) {
885                 // row size is odd, do last element in row s00+1, s10+1 are off
886                 // edge not worth converting this to use 16 case statements
887                 b[s10] = (byte) (b[s00] >> 1 & 1);
888                 b[s00] = (byte) (b[s00] >> N03 & 1);
889             }
890         }
891         if (i < lnx) {
892             // column size is odd, do last row s10, s10+1 are off edge
893             s00 = n * i;
894             for (j = 0; j < lny - 1; j += 2) {
895                 // not worth converting this to use 16 case statements
896                 b[s00 + 1] = (byte) (b[s00] >> 2 & 1);
897                 b[s00] = (byte) (b[s00] >> N03 & 1);
898                 s00 += 2;
899             }
900             if (j < lny) {
901                 // both row and column size are odd, do corner element s00+1,
902                 // s10, s10+1 are off edge not worth converting this to use 16
903                 // case statements
904                 b[s00] = (byte) (b[s00] >> N03 & 1);
905             }
906         }
907     }

909     /**
910      * char *infile; long a[]; a is 2-D tiledImageOperation with dimensions (n,n) int n; length of full row in a int
911      * nqx; partial length of row to decode int nqy; partial length of column (<=n) int nbitplanes; number of bitplanes
912      * to decode
913      */
914     private int qtreeDecode64(ByteBuffer infile, LongArrayPointer a, int n, int nqx, int nqy, int nbitplanes) {
915         int k, bit, b;
916         int nx2, ny2, nfx, nfy, c;
917         byte[] scratch;
918 
919         /*
920          * log2n is log2 of max(nqx,nqy) rounded up to next power of 2
921          */
922         int nqmax = nqx > nqy ? nqx : nqy;
923         int log2n = calculateLog2N(nqmax);
924         /*
925          * allocate scratch tiledImageOperation for working space
926          */
927         int nqx2 = (nqx + 1) / 2;
928         int nqy2 = (nqy + 1) / 2;
929         scratch = new byte[nqx2 * nqy2];
930 
931         /*
932          * now decode each bit plane, starting at the top A is assumed to be initialized to zero
933          */
934         for (bit = nbitplanes - 1; bit >= 0; bit--) {
935             /*
936              * Was bitplane was quadtree-coded or written directly?
937              */
938             b = inputNybble(infile);
939 
940             if (b == 0) {
941                 /*
942                  * bit map was written directly
943                  */
944                 readBdirect64(infile, a, n, nqx, nqy, scratch, bit);
945             } else if (b != NYBBLE_MASK) {
946                 throw new RuntimeException("Compression error");
947             } else {
948                 /*
949                  * bitmap was quadtree-coded, do log2n expansions read first code
950                  */
951                 scratch[0] = (byte) inputHuffman(infile);
952                 /*
953                  * now do log2n expansions, reading codes from file as necessary
954                  */
955                 nx2 = 1;
956                 ny2 = 1;
957                 nfx = nqx;
958                 nfy = nqy;
959                 c = 1 << log2n;
960                 for (k = 1; k < log2n; k++) {
961                     /*
962                      * this somewhat cryptic code generates the sequence n[k-1] = (n[k]+1)/2 where n[log2n]=nqx or nqy
963                      */
964                     c = c >> 1;
965                     nx2 = nx2 << 1;
966                     ny2 = ny2 << 1;
967                     if (nfx <= c) {
968                         nx2 -= 1;
969                     } else {
970                         nfx -= c;
971                     }
972                     if (nfy <= c) {
973                         ny2 -= 1;
974                     } else {
975                         nfy -= c;
976                     }
977                     qtreeExpand(infile, scratch, nx2, ny2, scratch);
978                 }
979                 /*
980                  * now copy last set of 4-bit codes to bitplane bit of tiledImageOperation a
981                  */
982                 qtreeBitins64(scratch, nqx, nqy, a, n, bit);
983             }
984         }
985         return 0;
986     }

988     /*
989      * do one quadtree expansion step on tiledImageOperation a[(nqx+1)/2,(nqy+1)/2] results put into b[nqx,nqy] (which
990      * may be the same as a)
991      */
992     private void qtreeExpand(ByteBuffer infile, byte[] a, int nx2, int ny2, byte[] b) {
993         int i;
994 
995         /*
996          * first copy a to b, expanding each 4-bit value
997          */
998         qtreeCopy(a, nx2, ny2, b, ny2);
999         /*
1000          * now read new 4-bit values into b for each non-zero element
1001          */
1002         for (i = nx2 * ny2 - 1; i >= 0; i--) {
1003             if (b[i] != 0) {
1004                 b[i] = (byte) inputHuffman(infile);
1005             }
1006         }
1007     }

1009     private void readBdirect64(ByteBuffer infile, LongArrayPointer a, int n, int nqx, int nqy, byte[] scratch,
1010             int bit) {
1011         /*
1012          * read bit image packed 4 pixels/nybble
1013          */
1014         /*
1015          * int i; for (i = 0; i < ((nqx+1)/2) * ((nqy+1)/2); i++) { scratch[i] = input_nybble(infile); }
1016          */
1017         inputNnybble(infile, (nqx + 1) / 2 * ((nqy + 1) / 2), scratch);
1018 
1019         /*
1020          * insert in bitplane BIT of image A
1021          */
1022         qtreeBitins64(scratch, nqx, nqy, a, n, bit);
1023     }

1053     /**
1054      * long a[]; tiledImageOperation to shuffle int n; number of elements to shuffle int n2; second dimension long
1055      * tmp[]; scratch storage
1056      */
1057     private void unshuffle64(LongArrayPointer a, int n, int n2, long[] tmp) {
1058         int i;
1059         int nhalf;
1060         LongArrayPointer p1, p2, pt;
1061 
1062         /*
1063          * copy 2nd half of tiledImageOperation to tmp
1064          */
1065         nhalf = n + 1 >> 1;
1066         pt = new LongArrayPointer(tmp);
1067         p1 = a.copy(n2 * nhalf); /* pointer to a[i] */
1068         for (i = nhalf; i < n; i++) {
1069             pt.set(p1.get());
1070             p1.offset += n2;
1071             pt.offset += 1;
1072         }
1073         /*
1074          * distribute 1st half of tiledImageOperation to even elements
1075          */
1076         p2 = a.copy(n2 * (nhalf - 1)); /* pointer to a[i] */
1077         p1 = a.copy(n2 * (nhalf - 1) << 1); /* pointer to a[2*i] */
1078         for (i = nhalf - 1; i >= 0; i--) {
1079             p1.set(p2.get());
1080             p2.offset -= n2;
1081             p1.offset -= n2 + n2;
1082         }
1083         /*
1084          * now distribute 2nd half of tiledImageOperation (in tmp) to odd elements
1085          */
1086         pt = new LongArrayPointer(tmp);
1087         p1 = a.copy(n2); /* pointer to a[i] */
1088         for (i = 1; i < n; i += 2) {
1089             p1.set(pt.get());
1090             p1.offset += n2 + n2;
1091             pt.offset += 1;
1092         }
1093     }
