66          LongArrayPointer(long[] tmp) {
67              this.a = tmp;
68              this.offset = 0;

72              this.a[this.offset + i] |= planeVal;

74          }

77              LongArrayPointer intAP = new LongArrayPointer(this.a);
78              intAP.offset = this.offset + extraOffset;

93          }

188     /**
189      * log2n is log2 of max(nx,ny) rounded up to next power of 2
190      */
191     private int calculateLog2N(int nmax) {
192         int log2n;
193         log2n = (int) (Math.log(nmax) / Math.log(2.0) + ROUNDING_HALF);
194         if (nmax > 1 << log2n) {
195             log2n += 1;
196         }
197         return log2n;
198     }

200     /**
201      * char *infile; input file long *a; address of output tiledImageOperation
202      * [nx][ny] int *nx,*ny; size of output tiledImageOperation int *scale;
203      * scale factor for digitization
204      *
205      * @param infile
206      * @param a
207      */
208     private void decode64(ByteBuffer infile, LongArrayPointer a) {
209         byte[] nbitplanes = new byte[N03];
210         byte[] tmagic = new byte[2];
211 
212         /*
213          * File starts either with special 2-byte magic code or with FITS
214          * keyword "SIMPLE  ="
215          */
216         infile.get(tmagic);
217         /*
218          * check for correct magic code value
219          */
220         if (tmagic[0] != CODE_MAGIC[0] || tmagic[1] != CODE_MAGIC[1]) {
221             throw new RuntimeException("Compression error");
222         }
223         this.nx = infile.getInt(); /* x size of image */
224         this.ny = infile.getInt(); /* y size of image */
225         this.scale = infile.getInt(); /* scale factor for digitization */
226 
227         /* sum of all pixels */
228         long sumall = infile.getLong();
229         /* # bits in quadrants */
230 
231         infile.get(nbitplanes);
232 
233         dodecode64(infile, a, nbitplanes);
234         /*
235          * put sum of all pixels back into pixel 0
236          */
237         a.set(0, sumall);
238     }

240     /**
241      * decompress the input byte stream using the H-compress algorithm input -
242      * input tiledImageOperation of compressed bytes a - pre-allocated
243      * tiledImageOperation to hold the output uncompressed image nx - returned X
244      * axis size ny - returned Y axis size NOTE: the nx and ny dimensions as
245      * defined within this code are reversed from the usual FITS notation. ny is
246      * the fastest varying dimension, which is usually considered the X axis in
247      * the FITS image display
248      *
249      * @param input
250      *            the input buffer to decompress
251      * @param smooth
252      *            should the image be smoothed
253      * @param aa
254      *            the resulting long tiledImageOperation
255      */
256     public void decompress(ByteBuffer input, boolean smooth, long[] aa) {
257 
258         LongArrayPointer a = new LongArrayPointer(aa);
259 
260         /* decode the input tiledImageOperation */
261 
262         decode64(input, a);
263 
264         /*
265          * Un-Digitize
266          */
267         undigitize64(a);
268 
269         /*
270          * Inverse H-transform
271          */
272         hinv64(a, smooth);
273 
274     }

276     /**
277      * long a[]; int nx,ny; Array dimensions are [nx][ny] unsigned char
278      * nbitplanes[3]; Number of bit planes in quadrants
279      */
280     private int dodecode64(ByteBuffer infile, LongArrayPointer a, byte[] nbitplanes) {
281         int nel = this.nx * this.ny;
282         int nx2 = (this.nx + 1) / 2;
283         int ny2 = (this.ny + 1) / 2;
284         /*
285          * initialize a to zero
286          */
287         for (int i = 0; i < nel; i++) {
288             a.set(i, 0);
289         }
290         /*
291          * Initialize bit input
292          */
293         startInputingBits();
294         /*
295          * read bit planes for each quadrant
296          */
297         qtreeDecode64(infile, a.copy(0), this.ny, nx2, ny2, nbitplanes[0]);
298 
299         qtreeDecode64(infile, a.copy(ny2), this.ny, nx2, this.ny / 2, nbitplanes[1]);
300 
301         qtreeDecode64(infile, a.copy(this.ny * nx2), this.ny, this.nx / 2, ny2, nbitplanes[1]);
302 
303         qtreeDecode64(infile, a.copy(this.ny * nx2 + ny2), this.ny, this.nx / 2, this.ny / 2, nbitplanes[2]);
304 
305         /*
306          * make sure there is an EOF symbol (nybble=0) at end
307          */
308         if (inputNybble(infile) != 0) {
309             throw new RuntimeException("Compression error");
310         }
311         /*
312          * now get the sign bits Re-initialize bit input
313          */
314         startInputingBits();
315         for (int i = 0; i < nel; i++) {
316             if (a.get(i) != 0) {
317                 if (inputBit(infile) != 0) {
318                     a.set(i, -a.get(i));
319                 }
320             }
321         }
322         return 0;
323     }

325     /**
326      * int smooth; 0 for no smoothing, else smooth during inversion int scale;
327      * used if smoothing is specified
328      */
329     private int hinv64(LongArrayPointer a, boolean smooth) {
330         int nmax = this.nx > this.ny ? this.nx : this.ny;
331         int log2n = calculateLog2N(nmax);
332         // get temporary storage for shuffling elements
333         long[] tmp = new long[(nmax + 1) / 2];
334         // set up masks, rounding parameters
335         int shift = 1;
336         long bit0 = (long) 1 << log2n - 1;
337         long bit1 = bit0 << 1;
338         long bit2 = bit0 << 2;
339         long mask0 = -bit0;
340         long mask1 = mask0 << 1;
341         long mask2 = mask0 << 2;
342         long prnd0 = bit0 >> 1;
343         long prnd1 = bit1 >> 1;
344         long prnd2 = bit2 >> 1;
345         long nrnd0 = prnd0 - 1;
346         long nrnd1 = prnd1 - 1;
347         long nrnd2 = prnd2 - 1;
348         // round h0 to multiple of bit2
349         a.set(0, a.get(0) + (a.get(0) >= 0 ? prnd2 : nrnd2) & mask2);
350         // do log2n expansions We're indexing a as a 2-D tiledImageOperation
351         // with dimensions
352         // (nx,ny).
353         int nxtop = 1;
354         int nytop = 1;
355         int nxf = this.nx;
356         int nyf = this.ny;
357         int c = 1 << log2n;
358         int i;
359         for (int k = log2n - 1; k >= 0; k--) {
360             // this somewhat cryptic code generates the sequence ntop[k-1] =
361             // (ntop[k]+1)/2, where ntop[log2n] = n
362             c = c >> 1;
363             nxtop = nxtop << 1;
364             nytop = nytop << 1;
365             if (nxf <= c) {
366                 nxtop -= 1;
367             } else {
368                 nxf -= c;
369             }
370             if (nyf <= c) {
371                 nytop -= 1;
372             } else {
373                 nyf -= c;
374             }
375             // double shift and fix nrnd0 (because prnd0=0) on last pass
376             if (k == 0) {
377                 nrnd0 = 0;
378                 shift = 2;
379             }
380             // unshuffle in each dimension to interleave coefficients
381             for (i = 0; i < nxtop; i++) {
382                 unshuffle64(a.copy(this.ny * i), nytop, 1, tmp);
383             }
384             for (int j = 0; j < nytop; j++) {
385                 unshuffle64(a.copy(j), nxtop, this.ny, tmp);
386             }
387             // smooth by interpolating coefficients if SMOOTH != 0
388             if (smooth) {
389                 hsmooth64(a, nxtop, nytop);
390             }
391             int oddx = nxtop % 2;
392             int oddy = nytop % 2;
393             for (i = 0; i < nxtop - oddx; i += 2) {
394                 int s00 = this.ny * i; /* s00 is index of a[i,j] */
395                 int s10 = s00 + this.ny; /* s10 is index of a[i+1,j] */
396                 for (int j = 0; j < nytop - oddy; j += 2) {
397                     long h0 = a.get(s00);
398                     long hx = a.get(s10);
399                     long hy = a.get(s00 + 1);
400                     long hc = a.get(s10 + 1);
401                     // round hx and hy to multiple of bit1, hc to multiple of
402                     // bit0 h0 is already a multiple of bit2
403                     hx = hx + (hx >= 0 ? prnd1 : nrnd1) & mask1;
404                     hy = hy + (hy >= 0 ? prnd1 : nrnd1) & mask1;
405                     hc = hc + (hc >= 0 ? prnd0 : nrnd0) & mask0;
406                     // propagate bit0 of hc to hx,hy
407                     long lowbit0 = hc & bit0;
408                     hx = hx >= 0 ? hx - lowbit0 : hx + lowbit0;
409                     hy = hy >= 0 ? hy - lowbit0 : hy + lowbit0;
410                     // Propagate bits 0 and 1 of hc,hx,hy to h0. This could be
411                     // simplified if we assume h0>0, but then the inversion
412                     // would not be lossless for images with negative pixels.
413                     long lowbit1 = (hc ^ hx ^ hy) & bit1;
414                     h0 = h0 >= 0 ? h0 + lowbit0 - lowbit1 : h0 + (lowbit0 == 0 ? lowbit1 : lowbit0 - lowbit1);
415                     // Divide sums by 2 (4 last time)
416                     a.set(s10 + 1, h0 + hx + hy + hc >> shift);
417                     a.set(s10, h0 + hx - hy - hc >> shift);
418                     a.set(s00 + 1, h0 - hx + hy - hc >> shift);
419                     a.set(s00, h0 - hx - hy + hc >> shift);
420                     s00 += 2;
421                     s10 += 2;
422                 }
423                 if (oddy != 0) {
424                     // do last element in row if row length is odd s00+1, s10+1
425                     // are off edge
426                     long h0 = a.get(s00);
427                     long hx = a.get(s10);
428                     hx = (hx >= 0 ? hx + prnd1 : hx + nrnd1) & mask1;
429                     long lowbit1 = hx & bit1;
430                     h0 = h0 >= 0 ? h0 - lowbit1 : h0 + lowbit1;
431                     a.set(s10, h0 + hx >> shift);
432                     a.set(s00, h0 - hx >> shift);
433                 }
434             }
435             if (oddx != 0) {
436                 // do last row if column length is odd s10, s10+1 are off edge
437                 int s00 = this.ny * i;
438                 for (int j = 0; j < nytop - oddy; j += 2) {
439                     long h0 = a.get(s00);
440                     long hy = a.get(s00 + 1);
441                     hy = (hy >= 0 ? hy + prnd1 : hy + nrnd1) & mask1;
442                     long lowbit1 = hy & bit1;
443                     h0 = h0 >= 0 ? h0 - lowbit1 : h0 + lowbit1;
444                     a.set(s00 + 1, h0 + hy >> shift);
445                     a.set(s00, h0 - hy >> shift);
446                     s00 += 2;
447                 }
448                 if (oddy != 0) {
449                     // do corner element if both row and column lengths are odd
450                     // s00+1, s10, s10+1 are off edge
451                     long h0 = a.get(s00);
452                     a.set(s00, h0 >> shift);
453                 }
454             }
455             // divide all the masks and rounding values by 2
456             bit1 = bit0;
457             bit0 = bit0 >> 1;
458             mask1 = mask0;
459             mask0 = mask0 >> 1;
460             prnd1 = prnd0;
461             prnd0 = prnd0 >> 1;
462             nrnd1 = nrnd0;
463             nrnd0 = prnd0 - 1;
464         }
465         return 0;
466     }

468     /**
469      * long a[]; tiledImageOperation of H-transform coefficients int
470      * nxtop,nytop; size of coefficient block to use int ny; actual 1st
471      * dimension of tiledImageOperation int scale; truncation scale factor that
472      * was used
473      */
474     private void hsmooth64(LongArrayPointer a, int nxtop, int nytop) {
475         int i, j;
476         int ny2, s10, s00;
477         long hm, h0, hp, hmm, hpm, hmp, hpp, hx2, hy2, diff, dmax, dmin, s, smax, m1, m2;
478 
479         /*
480          * Maximum change in coefficients is determined by scale factor. Since
481          * we rounded during division (see digitize.c), the biggest permitted
482          * change is scale/2.
483          */
484         smax = this.scale >> 1;
485         if (smax <= 0) {
486             return;
487         }
488         ny2 = this.ny << 1;
489         /*
490          * We're indexing a as a 2-D tiledImageOperation with dimensions
491          * (nxtop,ny) of which only (nxtop,nytop) are used. The coefficients on
492          * the edge of the tiledImageOperation are not adjusted (which is why
493          * the loops below start at 2 instead of 0 and end at nxtop-2 instead of
494          * nxtop.)
495          */
496         /*
497          * Adjust x difference hx
498          */
499         for (i = 2; i < nxtop - 2; i += 2) {
500             s00 = this.ny * i; /* s00 is index of a[i,j] */
501             s10 = s00 + this.ny; /* s10 is index of a[i+1,j] */
502             for (j = 0; j < nytop; j += 2) {
503                 /*
504                  * hp is h0 (mean value) in next x zone, hm is h0 in previous x
505                  * zone
506                  */
507                 hm = a.get(s00 - ny2);
508                 h0 = a.get(s00);
509                 hp = a.get(s00 + ny2);
510                 /*
511                  * diff = 8 * hx slope that would match h0 in neighboring zones
512                  */
513                 diff = hp - hm;
514                 /*
515                  * monotonicity constraints on diff
516                  */
517                 dmax = Math.max(Math.min(hp - h0, h0 - hm), 0) << 2;
518                 dmin = Math.min(Math.max(hp - h0, h0 - hm), 0) << 2;
519                 /*
520                  * if monotonicity would set slope = 0 then don't change hx.
521                  * note dmax>=0, dmin<=0.
522                  */
523                 if (dmin < dmax) {
524                     diff = Math.max(Math.min(diff, dmax), dmin);
525                     /*
526                      * Compute change in slope limited to range +/- smax.
527                      * Careful with rounding negative numbers when using shift
528                      * for divide by 8.
529                      */
530                     s = diff - (a.get(s10) << N03);
531                     s = s >= 0 ? s >> N03 : s + N07 >> N03;
532                     s = Math.max(Math.min(s, smax), -smax);
533                     a.set(s10, a.get(s10) + s);
534                 }
535                 s00 += 2;
536                 s10 += 2;
537             }
538         }
539         /*
540          * Adjust y difference hy
541          */
542         for (i = 0; i < nxtop; i += 2) {
543             s00 = this.ny * i + 2;
544             s10 = s00 + this.ny;
545             for (j = 2; j < nytop - 2; j += 2) {
546                 hm = a.get(s00 - 2);
547                 h0 = a.get(s00);
548                 hp = a.get(s00 + 2);
549                 diff = hp - hm;
550                 dmax = Math.max(Math.min(hp - h0, h0 - hm), 0) << 2;
551                 dmin = Math.min(Math.max(hp - h0, h0 - hm), 0) << 2;
552                 if (dmin < dmax) {
553                     diff = Math.max(Math.min(diff, dmax), dmin);
554                     s = diff - (a.get(s00 + 1) << N03);
555                     s = s >= 0 ? s >> N03 : s + N07 >> N03;
556                     s = Math.max(Math.min(s, smax), -smax);
557                     a.set(s00 + 1, a.get(s00 + 1) + s);
558                 }
559                 s00 += 2;
560                 s10 += 2;
561             }
562         }
563         /*
564          * Adjust curvature difference hc
565          */
566         for (i = 2; i < nxtop - 2; i += 2) {
567             s00 = this.ny * i + 2;
568             s10 = s00 + this.ny;
569             for (j = 2; j < nytop - 2; j += 2) {
570                 /*
571                  * ------------------ y | hmp | | hpp | | ------------------ | |
572                  * | h0 | | | ------------------ -------x | hmm | | hpm |
573                  * ------------------
574                  */
575                 hmm = a.get(s00 - ny2 - 2);
576                 hpm = a.get(s00 + ny2 - 2);
577                 hmp = a.get(s00 - ny2 + 2);
578                 hpp = a.get(s00 + ny2 + 2);
579                 h0 = a.get(s00);
580                 /*
581                  * diff = 64 * hc value that would match h0 in neighboring zones
582                  */
583                 diff = hpp + hmm - hmp - hpm;
584                 /*
585                  * 2 times x,y slopes in this zone
586                  */
587                 hx2 = a.get(s10) << 1;
588                 hy2 = a.get(s00 + 1) << 1;
589                 /*
590                  * monotonicity constraints on diff
591                  */
592                 m1 = Math.min(Math.max(hpp - h0, 0) - hx2 - hy2, Math.max(h0 - hpm, 0) + hx2 - hy2);
593                 m2 = Math.min(Math.max(h0 - hmp, 0) - hx2 + hy2, Math.max(hmm - h0, 0) + hx2 + hy2);
594                 dmax = Math.min(m1, m2) << BITS_OF_1_NYBBLE;
595                 m1 = Math.max(Math.min(hpp - h0, 0) - hx2 - hy2, Math.min(h0 - hpm, 0) + hx2 - hy2);
596                 m2 = Math.max(Math.min(h0 - hmp, 0) - hx2 + hy2, Math.min(hmm - h0, 0) + hx2 + hy2);
597                 dmin = Math.max(m1, m2) << BITS_OF_1_NYBBLE;
598                 /*
599                  * if monotonicity would set slope = 0 then don't change hc.
600                  * note dmax>=0, dmin<=0.
601                  */
602                 if (dmin < dmax) {
603                     diff = Math.max(Math.min(diff, dmax), dmin);
604                     /*
605                      * Compute change in slope limited to range +/- smax.
606                      * Careful with rounding negative numbers when using shift
607                      * for divide by 64.
608                      */
609                     s = diff - (a.get(s10 + 1) << N06);
610                     s = s >= 0 ? s >> N06 : s + N63 >> N06;
611                     s = Math.max(Math.min(s, smax), -smax);
612                     a.set(s10 + 1, a.get(s10 + 1) + s);
613                 }
614                 s00 += 2;
615                 s10 += 2;
616             }
617         }
618     }

620     private int inputBit(ByteBuffer infile) {
621         if (this.bitsToGo == 0) { /* Read the next byte if no */
622 
623             this.buffer2 = infile.get() & BYTE_MASK;
624 
625             this.bitsToGo = BITS_OF_1_BYTE;
626         }
627         /*
628          * Return the next bit
629          */
630         this.bitsToGo -= 1;
631         return this.buffer2 >> this.bitsToGo & 1;
632     }

634     /*
635      * Huffman decoding for fixed codes Coded values range from 0-15 Huffman
636      * code values (hex): 3e, 00, 01, 08, 02, 09, 1a, 1b, 03, 1c, 0a, 1d, 0b,
637      * 1e, 3f, 0c and number of bits in each code: 6, 3, 3, 4, 3, 4, 5, 5, 3, 5,
638      * 4, 5, 4, 5, 6, 4
639      */
640     private int inputHuffman(ByteBuffer infile) {
641         int c;
642 
643         /*
644          * get first 3 bits to start
645          */
646         c = inputNbits(infile, N03);
647         if (c < N04) {
648             /*
649              * this is all we need return 1,2,4,8 for c=0,1,2,3
650              */
651             return 1 << c;
652         }
653         /*
654          * get the next bit
655          */
656         c = inputBit(infile) | c << 1;
657         if (c < N13) {
658             /*
659              * OK, 4 bits is enough
660              */
661             switch (c) {
662                 case N08:
663                     return N03;
664                 case N09:
665                     return N05;
666                 case N10:
667                     return N10;
668                 case N11:
669                     return N12;
670                 case N12:
671                     return N15;
672                 default:
673             }
674         }
675         /*
676          * get yet another bit
677          */
678         c = inputBit(infile) | c << 1;
679         if (c < N31) {
680             /*
681              * OK, 5 bits is enough
682              */
683             switch (c) {
684                 case N26:
685                     return N06;
686                 case N27:
687                     return N07;
688                 case N28:
689                     return N09;
690                 case N29:
691                     return N11;
692                 case N30:
693                     return N13;
694                 default:
695             }
696         }
697         /*
698          * need the 6th bit
699          */
700         c = inputBit(infile) | c << 1;
701         if (c == N62) {
702             return 0;
703         }
704         return N14;
705     }

707     private int inputNbits(ByteBuffer infile, int n) {
708         if (this.bitsToGo < n) {
709             /*
710              * need another byte's worth of bits
711              */
712 
713             this.buffer2 = this.buffer2 << BITS_OF_1_BYTE | infile.get() & BYTE_MASK;
714             this.bitsToGo += BITS_OF_1_BYTE;
715         }
716         /*
717          * now pick off the first n bits
718          */
719         this.bitsToGo -= n;
720 
721         /* there was a slight gain in speed by replacing the following line */
722         /* return( (buffer2>>bits_to_go) & ((1<<n)-1) ); */
723         return this.buffer2 >> this.bitsToGo & MASKS[n];
724     }

728     private int inputNnybble(ByteBuffer infile, int n, byte[] array) {
729         /*
730          * copy n 4-bit nybbles from infile to the lower 4 bits of
731          * tiledImageOperation
732          */
733 
734         int ii, kk, shift1, shift2;
735 
736         /*
737          * forcing byte alignment doesn;t help, and even makes it go slightly
738          * slower if (bits_to_go != 8) input_nbits(infile, bits_to_go);
739          */
740         if (n == 1) {
741             array[0] = (byte) inputNybble(infile);
742             return 0;
743         }
744 
745         if (this.bitsToGo == BITS_OF_1_BYTE) {
746             /*
747              * already have 2 full nybbles in buffer2, so backspace the infile
748              * tiledImageOperation to reuse last char
749              */
750             infile.position(infile.position() - 1);
751             this.bitsToGo = 0;
752         }
753 
754         /* bits_to_go now has a value in the range 0 - 7. After adding */
755         /* another byte, bits_to_go effectively will be in range 8 - 15 */
756 
757         shift1 = this.bitsToGo + BITS_OF_1_NYBBLE; /*
758                                                     * shift1 will be in range 4
759                                                     * - 11
760                                                     */
761         shift2 = this.bitsToGo; /* shift2 will be in range 0 - 7 */
762         kk = 0;
763 
764         /* special case */
765         if (this.bitsToGo == 0) {
766             for (ii = 0; ii < n / 2; ii++) {
767                 /*
768                  * refill the buffer with next byte
769                  */
770                 this.buffer2 = this.buffer2 << BITS_OF_1_BYTE | infile.get() & BYTE_MASK;
771                 array[kk] = (byte) (this.buffer2 >> BITS_OF_1_NYBBLE & NYBBLE_MASK);
772                 array[kk + 1] = (byte) (this.buffer2 & NYBBLE_MASK); /*
773                                                                       * no shift
774                                                                       * required
775                                                                       */
776                 kk += 2;
777             }
778         } else {
779             for (ii = 0; ii < n / 2; ii++) {
780                 /*
781                  * refill the buffer with next byte
782                  */
783                 this.buffer2 = this.buffer2 << BITS_OF_1_BYTE | infile.get() & BYTE_MASK;
784                 array[kk] = (byte) (this.buffer2 >> shift1 & NYBBLE_MASK);
785                 array[kk + 1] = (byte) (this.buffer2 >> shift2 & NYBBLE_MASK);
786                 kk += 2;
787             }
788         }
789 
790         if (ii * 2 != n) { /* have to read last odd byte */
791             array[n - 1] = (byte) inputNybble(infile);
792         }
793 
794         return this.buffer2 >> this.bitsToGo & NYBBLE_MASK;
795     }

797     private int inputNybble(ByteBuffer infile) {
798         if (this.bitsToGo < BITS_OF_1_NYBBLE) {
799             /*
800              * need another byte's worth of bits
801              */
802 
803             this.buffer2 = this.buffer2 << BITS_OF_1_BYTE | infile.get() & BYTE_MASK;
804             this.bitsToGo += BITS_OF_1_BYTE;
805         }
806         /*
807          * now pick off the first 4 bits
808          */
809         this.bitsToGo -= BITS_OF_1_NYBBLE;
810 
811         return this.buffer2 >> this.bitsToGo & NYBBLE_MASK;
812     }

814     /**
815      * Copy 4-bit values from a[(nx+1)/2,(ny+1)/2] to b[nx,ny], expanding each
816      * value to 2x2 pixels and inserting into bitplane BIT of B. A,B may NOT be
817      * same tiledImageOperation (it wouldn't make sense to be inserting bits
818      * into the same tiledImageOperation anyway.)
819      */
820     private void qtreeBitins64(byte[] a, int lnx, int lny, LongArrayPointer b, int n, int bit) {
821         int i, j, s00;
822         long planeVal = 1L << bit;
823         // expand each 2x2 block
824         ByteBuffer k = ByteBuffer.wrap(a); /* k is index of a[i/2,j/2] */
825         for (i = 0; i < lnx - 1; i += 2) {
826             s00 = n * i; /* s00 is index of b[i,j] */
827             // Note: this code appears to run very slightly faster on a 32-bit
828             // linux machine using s00+n rather than the s10 intermediate
829             // variable
830             // s10 = s00+n; *//* s10 is index of b[i+1,j]
831             for (j = 0; j < lny - 1; j += 2) {
832                 byte value = k.get();
833                 if ((value & BIT_ONE) != ZERO) {
834                     b.bitOr(s00 + n + 1, planeVal);
835                 }
836                 if ((value & BIT_TWO) != ZERO) {
837                     b.bitOr(s00 + n, planeVal);
838                 }
839                 if ((value & BIT_THREE) != ZERO) {
840                     b.bitOr(s00 + 1, planeVal);
841                 }
842                 if ((value & BIT_FOUR) != ZERO) {
843                     b.bitOr(s00, planeVal);
844                 }
845                 // b.bitOr(s10+1, ((LONGLONG) ( a[k] & 1)) << bit; b.bitOr(s10 ,
846                 // ((((LONGLONG)a[k])>>1) & 1) << bit; b.bitOr(s00+1,
847                 // ((((LONGLONG)a[k])>>2) & 1) << bit; b.bitOr(s00
848                 // ,((((LONGLONG)a[k])>>3) & 1) << bit;
849                 s00 += 2;
850                 /* s10 += 2; */
851             }
852             if (j < lny) {
853                 // row size is odd, do last element in row s00+1, s10+1 are off
854                 // edge
855                 byte value = k.get();
856                 if ((value & BIT_TWO) != ZERO) {
857                     b.bitOr(s00 + n, planeVal);
858                 }
859                 if ((value & BIT_FOUR) != ZERO) {
860                     b.bitOr(s00, planeVal);
861                 }
862                 // b.bitOr(s10 , ((((LONGLONG)a[k])>>1) & 1) << bit; b.bitOr(s00
863                 // , ((((LONGLONG)a[k])>>3) & 1) << bit;
864             }
865         }
866         if (i < lnx) {
867             // column size is odd, do last row s10, s10+1 are off edge
868             s00 = n * i;
869             for (j = 0; j < lny - 1; j += 2) {
870                 byte value = k.get();
871                 if ((value & BIT_THREE) != ZERO) {
872                     b.bitOr(s00 + 1, planeVal);
873                 }
874                 if ((value & BIT_FOUR) != ZERO) {
875                     b.bitOr(s00, planeVal);
876                 } // b.bitOr(s00+1, ((((LONGLONG)a[k])>>2) & 1) << bit;
877                   // b.bitOr(s00 , ((((LONGLONG)a[k])>>3) & 1) << bit;
878                 s00 += 2;
879             }
880             if (j < lny) {
881                 // both row and column size are odd, do corner element s00+1,
882                 // s10, s10+1 are off edge
883                 if ((k.get() & BIT_FOUR) != ZERO) {
884                     b.bitOr(s00, planeVal);
885                 }
886                 // b.bitOr(s00 , ((((LONGLONG)a[k])>>3) & 1) << bit;
887             }
888         }
889     }

891     /**
892      * copy 4-bit values from a[(nx+1)/2,(ny+1)/2] to b[nx,ny], expanding each
893      * value to 2x2 pixels a,b may be same tiledImageOperation
894      */
895     private void qtreeCopy(byte[] a, int lnx, int lny, byte[] b, int n) {
896         int i, j, k, nx2, ny2;
897         int s00, s10;
898         // first copy 4-bit values to b start at end in case a,b are same
899         // tiledImageOperation
900         nx2 = (lnx + 1) / 2;
901         ny2 = (lny + 1) / 2;
902         k = ny2 * (nx2 - 1) + ny2 - 1; /* k is index of a[i,j] */
903         for (i = nx2 - 1; i >= 0; i--) {
904             s00 = 2 * (n * i + ny2 - 1); /* s00 is index of b[2*i,2*j] */
905             for (j = ny2 - 1; j >= 0; j--) {
906                 b[s00] = a[k];
907                 k -= 1;
908                 s00 -= 2;
909             }
910         }
911         for (i = 0; i < lnx - 1; i += 2) { // now expand each 2x2 block
912             // Note: Unlike the case in qtree_bitins, this code runs faster on a
913             // 32-bit linux machine using the s10 intermediate variable, rather
914             // that using s00+n. Go figure!
915             s00 = n * i; // s00 is index of b[i,j]
916             s10 = s00 + n; // s10 is index of b[i+1,j]
917             for (j = 0; j < lny - 1; j += 2) {
918                 b[s10 + 1] = (b[s00] & BIT_ONE) == ZERO ? ZERO : BIT_ONE;
919                 b[s10] = (b[s00] & BIT_TWO) == ZERO ? ZERO : BIT_ONE;
920                 b[s00 + 1] = (b[s00] & BIT_THREE) == ZERO ? ZERO : BIT_ONE;
921                 b[s00] = (b[s00] & BIT_FOUR) == ZERO ? ZERO : BIT_ONE;
922                 s00 += 2;
923                 s10 += 2;
924             }
925             if (j < lny) {
926                 // row size is odd, do last element in row s00+1, s10+1 are off
927                 // edge not worth converting this to use 16 case statements
928                 b[s10] = (byte) (b[s00] >> 1 & 1);
929                 b[s00] = (byte) (b[s00] >> N03 & 1);
930             }
931         }
932         if (i < lnx) {
933             // column size is odd, do last row s10, s10+1 are off edge
934             s00 = n * i;
935             for (j = 0; j < lny - 1; j += 2) {
936                 // not worth converting this to use 16 case statements
937                 b[s00 + 1] = (byte) (b[s00] >> 2 & 1);
938                 b[s00] = (byte) (b[s00] >> N03 & 1);
939                 s00 += 2;
940             }
941             if (j < lny) {
942                 // both row and column size are odd, do corner element s00+1,
943                 // s10, s10+1 are off edge not worth converting this to use 16
944                 // case statements
945                 b[s00] = (byte) (b[s00] >> N03 & 1);
946             }
947         }
948     }

950     /**
951      * char *infile; long a[]; a is 2-D tiledImageOperation with dimensions
952      * (n,n) int n; length of full row in a int nqx; partial length of row to
953      * decode int nqy; partial length of column (<=n) int nbitplanes; number of
954      * bitplanes to decode
955      */
956     private int qtreeDecode64(ByteBuffer infile, LongArrayPointer a, int n, int nqx, int nqy, int nbitplanes) {
957         int k, bit, b;
958         int nx2, ny2, nfx, nfy, c;
959         byte[] scratch;
960 
961         /*
962          * log2n is log2 of max(nqx,nqy) rounded up to next power of 2
963          */
964         int nqmax = nqx > nqy ? nqx : nqy;
965         int log2n = calculateLog2N(nqmax);
966         /*
967          * allocate scratch tiledImageOperation for working space
968          */
969         int nqx2 = (nqx + 1) / 2;
970         int nqy2 = (nqy + 1) / 2;
971         scratch = new byte[nqx2 * nqy2];
972 
973         /*
974          * now decode each bit plane, starting at the top A is assumed to be
975          * initialized to zero
976          */
977         for (bit = nbitplanes - 1; bit >= 0; bit--) {
978             /*
979              * Was bitplane was quadtree-coded or written directly?
980              */
981             b = inputNybble(infile);
982 
983             if (b == 0) {
984                 /*
985                  * bit map was written directly
986                  */
987                 readBdirect64(infile, a, n, nqx, nqy, scratch, bit);
988             } else if (b != NYBBLE_MASK) {
989                 throw new RuntimeException("Compression error");
990             } else {
991                 /*
992                  * bitmap was quadtree-coded, do log2n expansions read first
993                  * code
994                  */
995                 scratch[0] = (byte) inputHuffman(infile);
996                 /*
997                  * now do log2n expansions, reading codes from file as necessary
998                  */
999                 nx2 = 1;
1000                 ny2 = 1;
1001                 nfx = nqx;
1002                 nfy = nqy;
1003                 c = 1 << log2n;
1004                 for (k = 1; k < log2n; k++) {
1005                     /*
1006                      * this somewhat cryptic code generates the sequence n[k-1]
1007                      * = (n[k]+1)/2 where n[log2n]=nqx or nqy
1008                      */
1009                     c = c >> 1;
1010                     nx2 = nx2 << 1;
1011                     ny2 = ny2 << 1;
1012                     if (nfx <= c) {
1013                         nx2 -= 1;
1014                     } else {
1015                         nfx -= c;
1016                     }
1017                     if (nfy <= c) {
1018                         ny2 -= 1;
1019                     } else {
1020                         nfy -= c;
1021                     }
1022                     qtreeExpand(infile, scratch, nx2, ny2, scratch);
1023                 }
1024                 /*
1025                  * now copy last set of 4-bit codes to bitplane bit of
1026                  * tiledImageOperation a
1027                  */
1028                 qtreeBitins64(scratch, nqx, nqy, a, n, bit);
1029             }
1030         }
1031         return 0;
1032     }

1034     /*
1035      * do one quadtree expansion step on tiledImageOperation
1036      * a[(nqx+1)/2,(nqy+1)/2] results put into b[nqx,nqy] (which may be the same
1037      * as a)
1038      */
1039     private void qtreeExpand(ByteBuffer infile, byte[] a, int nx2, int ny2, byte[] b) {
1040         int i;
1041 
1042         /*
1043          * first copy a to b, expanding each 4-bit value
1044          */
1045         qtreeCopy(a, nx2, ny2, b, ny2);
1046         /*
1047          * now read new 4-bit values into b for each non-zero element
1048          */
1049         for (i = nx2 * ny2 - 1; i >= 0; i--) {
1050             if (b[i] != 0) {
1051                 b[i] = (byte) inputHuffman(infile);
1052             }
1053         }
1054     }

1056     private void readBdirect64(ByteBuffer infile, LongArrayPointer a, int n, int nqx, int nqy, byte[] scratch, int bit) {
1057         /*
1058          * read bit image packed 4 pixels/nybble
1059          */
1060         /*
1061          * int i; for (i = 0; i < ((nqx+1)/2) * ((nqy+1)/2); i++) { scratch[i] =
1062          * input_nybble(infile); }
1063          */
1064         inputNnybble(infile, (nqx + 1) / 2 * ((nqy + 1) / 2), scratch);
1065 
1066         /*
1067          * insert in bitplane BIT of image A
1068          */
1069         qtreeBitins64(scratch, nqx, nqy, a, n, bit);
1070     }

1101     /**
1102      * long a[]; tiledImageOperation to shuffle int n; number of elements to
1103      * shuffle int n2; second dimension long tmp[]; scratch storage
1104      */
1105     private void unshuffle64(LongArrayPointer a, int n, int n2, long[] tmp) {
1106         int i;
1107         int nhalf;
1108         LongArrayPointer p1, p2, pt;
1109 
1110         /*
1111          * copy 2nd half of tiledImageOperation to tmp
1112          */
1113         nhalf = n + 1 >> 1;
1114         pt = new LongArrayPointer(tmp);
1115         p1 = a.copy(n2 * nhalf); /* pointer to a[i] */
1116         for (i = nhalf; i < n; i++) {
1117             pt.set(p1.get());
1118             p1.offset += n2;
1119             pt.offset += 1;
1120         }
1121         /*
1122          * distribute 1st half of tiledImageOperation to even elements
1123          */
1124         p2 = a.copy(n2 * (nhalf - 1)); /* pointer to a[i] */
1125         p1 = a.copy(n2 * (nhalf - 1) << 1); /* pointer to a[2*i] */
1126         for (i = nhalf - 1; i >= 0; i--) {
1127             p1.set(p2.get());
1128             p2.offset -= n2;
1129             p1.offset -= n2 + n2;
1130         }
1131         /*
1132          * now distribute 2nd half of tiledImageOperation (in tmp) to odd
1133          * elements
1134          */
1135         pt = new LongArrayPointer(tmp);
1136         p1 = a.copy(n2); /* pointer to a[i] */
1137         for (i = 1; i < n; i += 2) {
1138             p1.set(pt.get());
1139             p1.offset += n2 + n2;
1140             pt.offset += 1;
1141         }
1142     }
