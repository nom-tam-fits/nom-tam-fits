147     private int bufcopy(byte[] a, int n, byte[] buffer, int b, long bmax) {
148         int i;
149 
150         for (i = 0; i < n; i++) {
151             if (a[i] != 0) {
152                 /*
153                  * add Huffman code for a[i] to buffer
154                  */
155                 this.bitbuffer |= CODE[a[i]] << this.bitsToGo3;
156                 this.bitsToGo3 += NCODE[a[i]];
157                 if (this.bitsToGo3 >= BITS_OF_1_BYTE) {
158                     buffer[b] = (byte) (this.bitbuffer & BYTE_MASK);
159                     b += 1;
160                     /*
161                      * return warning code if we fill buffer
162                      */
163                     if (b >= bmax) {
164                         return b;
165                     }
166                     this.bitbuffer >>= BITS_OF_1_BYTE;
167                     this.bitsToGo3 -= BITS_OF_1_BYTE;
168                 }
169             }
170         }
171         return b;
172     }

174     protected void compress(long[] aa, int ny, int nx, int scale, ByteBuffer output) {
175         /*
176          * compress the input image using the H-compress algorithm a - input
177          * image tiledImageOperation nx - size of X axis of image ny - size of Y
178          * axis of image scale - quantization scale factor. Larger values
179          * results in more (lossy) compression scale = 0 does lossless
180          * compression output - pre-allocated tiledImageOperation to hold the
181          * output compressed stream of bytes nbyts - input value = size of the
182          * output buffer; returned value = size of the compressed byte stream,
183          * in bytes NOTE: the nx and ny dimensions as defined within this code
184          * are reversed from the usual FITS notation. ny is the fastest varying
185          * dimension, which is usually considered the X axis in the FITS image
186          * display
187          */
188 
189         /* H-transform */
190         htrans(aa, nx, ny);
191 
192         LongBuffer a = LongBuffer.wrap(aa);
193 
194         /* digitize */
195         digitize(a, 0, nx, ny, scale);
196 
197         /* encode and write to output tiledImageOperation */
198         encode(output, a, nx, ny, scale);
199 
200     }

202     private LongBuffer copy(LongBuffer a, int i) {
203         a.position(i);
204         return a.slice();

207     private void digitize(LongBuffer a, int aOffset, int nx, int ny, long scale) {
208         /*
209          * round to multiple of scale
210          */
211         if (scale <= 1) {
212             return;
213         }
214         long d = (scale + 1L) / 2L - 1L;
215         for (int index = 0; index < a.limit(); index++) {
216             long current = a.get(index);
217             a.put(index, (current > 0 ? current + d : current - d) / scale);
218         }
219     }

221     /**
222      * encode pixels.
223      * 
224      * @param compressedBytes
225      *            compressed data
226      * @param pixels
227      *            pixels to compress
228      * @param nx
229      *            image width dimension
230      * @param ny
231      *            image height dimension
232      * @param nbitplanes
233      *            Number of bit planes in quadrants
234      */
235     private void doEncode(ByteBuffer compressedBytes, LongBuffer pixels, int nx, int ny, byte[] nbitplanes) {
236 
237         int nx2 = (nx + 1) / 2;
238         int ny2 = (ny + 1) / 2;
239         /*
240          * Initialize bit output
241          */
242         startOutputtingBits();
243         /*
244          * write out the bit planes for each quadrant
245          */
246         qtreeEncode(compressedBytes, copy(pixels, 0), ny, nx2, ny2, nbitplanes[0]);
247 
248         qtreeEncode(compressedBytes, copy(pixels, ny2), ny, nx2, ny / 2, nbitplanes[1]);
249 
250         qtreeEncode(compressedBytes, copy(pixels, ny * nx2), ny, nx / 2, ny2, nbitplanes[1]);
251 
252         qtreeEncode(compressedBytes, copy(pixels, ny * nx2 + ny2), ny, nx / 2, ny / 2, nbitplanes[2]);
253         /*
254          * Add zero as an EOF symbol
255          */
256         outputNybble(compressedBytes, 0);
257         doneOutputtingBits(compressedBytes);
258 
259     }

261     private void doneOutputtingBits(ByteBuffer outfile) {
262         if (this.bitsToGo2 < BITS_OF_1_BYTE) {
263             /* putc(buffer2<<bits_to_go2,outfile); */
264 
265             outfile.put((byte) (this.buffer2 << this.bitsToGo2));
266         }
267     }

269     private int encode(ByteBuffer compressedBytes, LongBuffer a, int nx, int ny, int scale) {
270         long[] vmax = new long[N3];
271         byte[] nbitplanes = new byte[N3];
272         // initialize the number of compressed bytes that have been written
273         long noutchar = 0;
274         int nel = nx * ny;
275         /*
276          * write magic value
277          */
278         compressedBytes.put(CODE_MAGIC);
279         compressedBytes.putInt(nx); /* size of image */
280         compressedBytes.putInt(ny);
281         compressedBytes.putInt(scale); /* scale factor for digitization */
282         /*
283          * write first value of A (sum of all pixels -- the only value which
284          * does not compress well)
285          */
286         compressedBytes.putLong(a.get(0));
287 
288         a.put(0, 0);
289         /*
290          * allocate tiledImageOperation for sign bits and save values, 8 per
291          * byte
292          */
293         byte[] signbits = new byte[(nel + BITS_OF_1_BYTE - 1) / BITS_OF_1_BYTE];
294 
295         int nsign = 0;
296         int bitsToGo = BITS_OF_1_BYTE;
297         signbits[0] = 0;
298         for (int i = 0; i < nel; i++) {
299             if (a.get(i) > 0) {
300                 /*
301                  * positive element, put zero at end of buffer
302                  */
303                 signbits[nsign] <<= 1;
304                 bitsToGo -= 1;
305             } else if (a.get(i) < 0) {
306                 /*
307                  * negative element, shift in a one
308                  */
309                 signbits[nsign] <<= 1;
310                 signbits[nsign] |= 1;
311                 bitsToGo -= 1;
312                 /*
313                  * replace a by absolute value
314                  */
315                 a.put(i, -a.get(i));
316             }
317             if (bitsToGo == 0) {
318                 /*
319                  * filled up this byte, go to the next one
320                  */
321                 bitsToGo = BITS_OF_1_BYTE;
322                 nsign += 1;
323                 signbits[nsign] = 0;
324             }
325         }
326         if (bitsToGo != BITS_OF_1_BYTE) {
327             /*
328              * some bits in last element move bits in last byte to bottom and
329              * increment nsign
330              */
331             signbits[nsign] <<= bitsToGo;
332             nsign += 1;
333         }
334         /*
335          * calculate number of bit planes for 3 quadrants quadrant 0=bottom
336          * left, 1=bottom right or top left, 2=top right,
337          */
338         for (int q = 0; q < N3; q++) {
339             vmax[q] = 0;
340         }
341         /*
342          * get maximum absolute value in each quadrant
343          */
344         int nx2 = (nx + 1) / 2;
345         int ny2 = (ny + 1) / 2;
346         int j = 0; /* column counter */
347         int k = 0; /* row counter */
348         for (int i = 0; i < nel; i++) {
349             int q = (j >= ny2 ? 1 : 0) + (k >= nx2 ? 1 : 0);
350             if (vmax[q] < a.get(i)) {
351                 vmax[q] = a.get(i);
352             }
353             if (++j >= ny) {
354                 j = 0;
355                 k += 1;
356             }
357         }
358         /*
359          * now calculate number of bits for each quadrant
360          */
361 
362         /* this is a more efficient way to do this, */
363 
364         for (int q = 0; q < N3; q++) {
365             nbitplanes[q] = 0;
366             while (vmax[q] > 0) {
367                 vmax[q] = vmax[q] >> 1;
368                 nbitplanes[q]++;
369             }
370         }
371 
372         /*
373          * write nbitplanes
374          */
375         compressedBytes.put(nbitplanes, 0, nbitplanes.length);
376 
377         /*
378          * write coded tiledImageOperation
379          */
380         doEncode(compressedBytes, a, nx, ny, nbitplanes);
381         /*
382          * write sign bits
383          */
384 
385         if (nsign > 0) {
386             compressedBytes.put(signbits, 0, nsign);
387         }
388         return (int) noutchar;
389 
390     }

392     private int htrans(long[] a, int nx, int ny) {
393         /*
394          * log2n is log2 of max(nx,ny) rounded up to next power of 2
395          */
396         int nmax = nx > ny ? nx : ny;
397         int log2n = log2n(nmax);
398         if (nmax > 1 << log2n) {
399             log2n += 1;
400         }
401         /*
402          * get temporary storage for shuffling elements
403          */
404         long[] tmp = new long[(nmax + 1) / 2];
405 
406         /*
407          * set up rounding and shifting masks
408          */
409         long shift = 0;
410         long mask = HTRANS_START_MASK;
411         long mask2 = mask << 1;
412         long prnd = 1;
413         long prnd2 = prnd << 1;
414         long nrnd2 = prnd2 - 1;
415         /*
416          * do log2n reductions We're indexing a as a 2-D tiledImageOperation
417          * with dimensions (nx,ny).
418          */
419         int nxtop = nx;
420         int nytop = ny;
421 
422         for (int k = 0; k < log2n; k++) {
423             int oddx = nxtop % 2;
424             int oddy = nytop % 2;
425             int i = 0;
426             for (; i < nxtop - oddx; i += 2) {
427                 int s00 = i * ny; /* s00 is index of a[i,j] */
428                 int s10 = s00 + ny; /* s10 is index of a[i+1,j] */
429                 for (int j = 0; j < nytop - oddy; j += 2) {
430                     /*
431                      * Divide h0,hx,hy,hc by 2 (1 the first time through).
432                      */
433                     long h0 = a[s10 + 1] + a[s10] + a[s00 + 1] + a[s00] >> shift;
434                     long hx = a[s10 + 1] + a[s10] - a[s00 + 1] - a[s00] >> shift;
435                     long hy = a[s10 + 1] - a[s10] + a[s00 + 1] - a[s00] >> shift;
436                     long hc = a[s10 + 1] - a[s10] - a[s00 + 1] + a[s00] >> shift;
437 
438                     /*
439                      * Throw away the 2 bottom bits of h0, bottom bit of hx,hy.
440                      * To get rounding to be same for positive and negative
441                      * numbers, nrnd2 = prnd2 - 1.
442                      */
443                     a[s10 + 1] = hc;
444                     a[s10] = (hx >= 0 ? hx + prnd : hx) & mask;
445                     a[s00 + 1] = (hy >= 0 ? hy + prnd : hy) & mask;
446                     a[s00] = (h0 >= 0 ? h0 + prnd2 : h0 + nrnd2) & mask2;
447                     s00 += 2;
448                     s10 += 2;
449                 }
450                 if (oddy != 0) {
451                     /*
452                      * do last element in row if row length is odd s00+1, s10+1
453                      * are off edge
454                      */
455                     long h0 = a[s10] + a[s00] << 1 - shift;
456                     long hx = a[s10] - a[s00] << 1 - shift;
457                     a[s10] = (hx >= 0 ? hx + prnd : hx) & mask;
458                     a[s00] = (h0 >= 0 ? h0 + prnd2 : h0 + nrnd2) & mask2;
459                     s00 += 1;
460                     s10 += 1;
461                 }
462             }
463             if (oddx != 0) {
464                 /*
465                  * do last row if column length is odd s10, s10+1 are off edge
466                  */
467                 int s00 = i * ny;
468                 for (int j = 0; j < nytop - oddy; j += 2) {
469                     long h0 = a[s00 + 1] + a[s00] << 1 - shift;
470                     long hy = a[s00 + 1] - a[s00] << 1 - shift;
471                     a[s00 + 1] = (hy >= 0 ? hy + prnd : hy) & mask;
472                     a[s00] = (h0 >= 0 ? h0 + prnd2 : h0 + nrnd2) & mask2;
473                     s00 += 2;
474                 }
475                 if (oddy != 0) {
476                     /*
477                      * do corner element if both row and column lengths are odd
478                      * s00+1, s10, s10+1 are off edge
479                      */
480                     long h0 = a[s00] << 2 - shift;
481                     a[s00] = (h0 >= 0 ? h0 + prnd2 : h0 + nrnd2) & mask2;
482                 }
483             }
484             /*
485              * now shuffle in each dimension to group coefficients by order
486              */
487             // achtung eigenlich pointer nach a
488             for (i = 0; i < nxtop; i++) {
489                 shuffle(a, ny * i, nytop, 1, tmp);
490             }
491             for (int j = 0; j < nytop; j++) {
492                 shuffle(a, j, nxtop, ny, tmp);
493             }
494             /*
495              * image size reduced by 2 (round up if odd)
496              */
497             nxtop = nxtop + 1 >> 1;
498             nytop = nytop + 1 >> 1;
499             /*
500              * divisor doubles after first reduction
501              */
502             shift = 1;
503             /*
504              * masks, rounding values double after each iteration
505              */
506             mask = mask2;
507             prnd = prnd2;
508             mask2 = mask2 << 1;
509             prnd2 = prnd2 << 1;
510             nrnd2 = prnd2 - 1;
511         }
512         return 0;
513     }

515     private int log2n(int nqmax) {
516         return (int) (Math.log(nqmax) / Math.log(2.0) + ROUNDING_HALF);
517     }

519     private void outputNbits(ByteBuffer outfile, int bits, int n) {
520         /* AND mask for the right-most n bits */
521 
522         /*
523          * insert bits at end of buffer
524          */
525         this.buffer2 <<= n;
526         /* buffer2 |= ( bits & ((1<<n)-1) ); */
527         this.buffer2 |= bits & BITS_MASK[n];
528         this.bitsToGo2 -= n;
529         if (this.bitsToGo2 <= 0) {
530             /*
531              * buffer2 full, put out top 8 bits
532              */
533 
534             outfile.put((byte) (this.buffer2 >> -this.bitsToGo2 & BYTE_MASK));
535 
536             this.bitsToGo2 += BITS_OF_1_BYTE;
537         }
538     }

540     private void outputNnybble(ByteBuffer outfile, int n, byte[] array) {
541         /*
542          * pack the 4 lower bits in each element of the tiledImageOperation into
543          * the outfile tiledImageOperation
544          */
545 
546         int ii, jj, kk = 0, shift;
547 
548         if (n == 1) {
549             outputNybble(outfile, array[0]);
550             return;
551         }
552         /*
553          * forcing byte alignment doesn;t help, and even makes it go slightly
554          * slower if (bits_to_go2 != 8) output_nbits(outfile, kk, bits_to_go2);
555          */
556         if (this.bitsToGo2 <= BITS_OF_1_NYBBLE) {
557             /* just room for 1 nybble; write it out separately */
558             outputNybble(outfile, array[0]);
559             kk++; /* index to next tiledImageOperation element */
560 
561             if (n == 2) {
562                 // only 1 more nybble to write out
563                 outputNybble(outfile, array[1]);
564                 return;
565             }
566         }
567 
568         /* bits_to_go2 is now in the range 5 - 8 */
569         shift = BITS_OF_1_BYTE - this.bitsToGo2;
570 
571         /*
572          * now write out pairs of nybbles; this does not affect value of
573          * bits_to_go2
574          */
575         jj = (n - kk) / 2;
576 
577         if (this.bitsToGo2 == BITS_OF_1_BYTE) {
578             /* special case if nybbles are aligned on byte boundary */
579             /* this actually seems to make very little difference in speed */
580             this.buffer2 = 0;
581             for (ii = 0; ii < jj; ii++) {
582                 outfile.put((byte) ((array[kk] & NYBBLE_MASK) << BITS_OF_1_NYBBLE | array[kk + 1] & NYBBLE_MASK));
583                 kk += 2;
584             }
585         } else {
586             for (ii = 0; ii < jj; ii++) {
587                 this.buffer2 = this.buffer2 << BITS_OF_1_BYTE | (array[kk] & NYBBLE_MASK) << BITS_OF_1_NYBBLE | array[kk + 1] & NYBBLE_MASK;
588                 kk += 2;
589 
590                 /*
591                  * buffer2 full, put out top 8 bits
592                  */
593 
594                 outfile.put((byte) (this.buffer2 >> shift & BYTE_MASK));
595             }
596         }
597 
598         /* write out last odd nybble, if present */
599         if (kk != n) {
600             outputNybble(outfile, array[n - 1]);
601         }
602     }

604     private void outputNybble(ByteBuffer outfile, int bits) {
605         /*
606          * insert 4 bits at end of buffer
607          */
608         this.buffer2 = this.buffer2 << BITS_OF_1_NYBBLE | bits & NYBBLE_MASK;
609         this.bitsToGo2 -= BITS_OF_1_NYBBLE;
610         if (this.bitsToGo2 <= 0) {
611             /*
612              * buffer2 full, put out top 8 bits
613              */
614 
615             outfile.put((byte) (this.buffer2 >> -this.bitsToGo2 & BYTE_MASK));
616 
617             this.bitsToGo2 += BITS_OF_1_BYTE;
618         }
619     }

621     /**
622      * macros to write out 4-bit nybble, Huffman code for this value
623      */
624     private int qtreeEncode(ByteBuffer outfile, LongBuffer a, int n, int nqx, int nqy, int nbitplanes) {
625 
626         /*
627          * int a[]; int n; physical dimension of row in a int nqx; length of row
628          * int nqy; length of column (<=n) int nbitplanes; number of bit planes
629          * to output
630          */
631 
632         int log2n, i, k, bit, b, nqmax, nqx2, nqy2, nx, ny;
633         long bmax;
634         byte[] scratch, buffer;
635 
636         /*
637          * log2n is log2 of max(nqx,nqy) rounded up to next power of 2
638          */
639         nqmax = nqx > nqy ? nqx : nqy;
640         log2n = log2n(nqmax);
641         if (nqmax > 1 << log2n) {
642             log2n += 1;
643         }
644         /*
645          * initialize buffer point, max buffer size
646          */
647         nqx2 = (nqx + 1) / 2;
648         nqy2 = (nqy + 1) / 2;
649         bmax = (nqx2 * nqy2 + 1) / 2;
650         /*
651          * We're indexing A as a 2-D tiledImageOperation with dimensions
652          * (nqx,nqy). Scratch is 2-D with dimensions (nqx/2,nqy/2) rounded up.
653          * Buffer is used to store string of codes for output.
654          */
655         scratch = new byte[(int) (2 * bmax)];
656         buffer = new byte[(int) bmax];
657 
658         /*
659          * now encode each bit plane, starting with the top
660          */
661         bitplane_done: for (bit = nbitplanes - 1; bit >= 0; bit--) {
662             /*
663              * initial bit buffer
664              */
665             b = 0;
666             this.bitbuffer = 0;
667             this.bitsToGo3 = 0;
668             /*
669              * on first pass copy A to scratch tiledImageOperation
670              */
671             qtreeOnebit(a, n, nqx, nqy, scratch, bit);
672             nx = nqx + 1 >> 1;
673             ny = nqy + 1 >> 1;
674             /*
675              * copy non-zero values to output buffer, which will be written in
676              * reverse order
677              */
678             b = bufcopy(scratch, nx * ny, buffer, b, bmax);
679             if (b >= bmax) {
680                 /*
681                  * quadtree is expanding data, change warning code and just fill
682                  * buffer with bit-map
683                  */
684                 writeBdirect(outfile, a, n, nqx, nqy, scratch, bit);
685                 continue bitplane_done;
686             }
687             /*
688              * do log2n reductions
689              */
690             for (k = 1; k < log2n; k++) {
691                 qtreeReduce(scratch, ny, nx, ny, scratch);
692                 nx = nx + 1 >> 1;
693                 ny = ny + 1 >> 1;
694                 b = bufcopy(scratch, nx * ny, buffer, b, bmax);
695                 if (b >= bmax) {
696                     writeBdirect(outfile, a, n, nqx, nqy, scratch, bit);
697                     continue bitplane_done;
698                 }
699             }
700             /*
701              * OK, we've got the code in buffer Write quadtree warning code,
702              * then write buffer in reverse order
703              */
704             outputNybble(outfile, NYBBLE_MASK);
705             if (b == 0) {
706                 if (this.bitsToGo3 > 0) {
707                     /*
708                      * put out the last few bits
709                      */
710                     outputNbits(outfile, this.bitbuffer & (1 << this.bitsToGo3) - 1, this.bitsToGo3);
711                 } else {
712                     /*
713                      * have to write a zero nybble if there are no 1's in
714                      * tiledImageOperation
715                      */
716                     outputNbits(outfile, CODE[0], NCODE[0]);
717                 }
718             } else {
719                 if (this.bitsToGo3 > 0) {
720                     /*
721                      * put out the last few bits
722                      */
723                     outputNbits(outfile, this.bitbuffer & (1 << this.bitsToGo3) - 1, this.bitsToGo3);
724                 }
725                 for (i = b - 1; i >= 0; i--) {
726                     outputNbits(outfile, buffer[i], BITS_OF_1_BYTE);
727                 }
728             }
729         }
730         return 0;
731     }

733     private void qtreeOnebit(LongBuffer a, int n, int nx, int ny, byte[] b, int bit) {
734         int i, j, k;
735         long b0, b1, b2, b3;
736         int s10, s00;
737 
738         /*
739          * use selected bit to get amount to shift
740          */
741         b0 = 1L << bit;
742         b1 = b0 << 1;
743         b2 = b1 << 1;
744         b3 = b2 << 1;
745         k = 0; /* k is index of b[i/2,j/2] */
746         for (i = 0; i < nx - 1; i += 2) {
747             s00 = n * i; /* s00 is index of a[i,j] */
748             /*
749              * tried using s00+n directly in the statements, but this had no
750              * effect on performance
751              */
752             s10 = s00 + n; /* s10 is index of a[i+1,j] */
753             for (j = 0; j < ny - 1; j += 2) {
754 
755                 b[k] = (byte) ((a.get(s10 + 1) & b0 //
756                         | a.get(s10) << 1 & b1 //
757                         | a.get(s00 + 1) << 2 & b2 //
758                 | a.get(s00) << N3 & b3) >> bit);
759 
760                 k += 1;
761                 s00 += 2;
762                 s10 += 2;
763             }
764             if (j < ny) {
765                 /*
766                  * row size is odd, do last element in row s00+1,s10+1 are off
767                  * edge
768                  */
769                 b[k] = (byte) ((a.get(s10) << 1 & b1 | a.get(s00) << N3 & b3) >> bit);
770                 k += 1;
771             }
772         }
773         if (i < nx) {
774             /*
775              * column size is odd, do last row s10,s10+1 are off edge
776              */
777             s00 = n * i;
778             for (j = 0; j < ny - 1; j += 2) {
779                 b[k] = (byte) ((a.get(s00 + 1) << 2 & b2 | a.get(s00) << N3 & b3) >> bit);
780                 k += 1;
781                 s00 += 2;
782             }
783             if (j < ny) {
784                 /*
785                  * both row and column size are odd, do corner element s00+1,
786                  * s10, s10+1 are off edge
787                  */
788                 b[k] = (byte) ((a.get(s00) << N3 & b3) >> bit);
789                 k += 1;
790             }
791         }
792     }

794     private void qtreeReduce(byte[] a, int n, int nx, int ny, byte[] b) {
795         int i, j, k;
796         int s10, s00;
797 
798         k = 0; /* k is index of b[i/2,j/2] */
799         for (i = 0; i < nx - 1; i += 2) {
800             s00 = n * i; /* s00 is index of a[i,j] */
801             s10 = s00 + n; /* s10 is index of a[i+1,j] */
802             for (j = 0; j < ny - 1; j += 2) {
803                 b[k] = (byte) (b2i(a[s10 + 1] != 0) | b2i(a[s10] != 0) << 1 | b2i(a[s00 + 1] != 0) << 2 | b2i(a[s00] != 0) << N3);
804                 k += 1;
805                 s00 += 2;
806                 s10 += 2;
807             }
808             if (j < ny) {
809                 /*
810                  * row size is odd, do last element in row s00+1,s10+1 are off
811                  * edge
812                  */
813                 b[k] = (byte) (b2i(a[s10] != 0) << 1 | b2i(a[s00] != 0) << N3);
814                 k += 1;
815             }
816         }
817         if (i < nx) {
818             /*
819              * column size is odd, do last row s10,s10+1 are off edge
820              */
821             s00 = n * i;
822             for (j = 0; j < ny - 1; j += 2) {
823                 b[k] = (byte) (b2i(a[s00 + 1] != 0) << 2 | b2i(a[s00] != 0) << N3);
824                 k += 1;
825                 s00 += 2;
826             }
827             if (j < ny) {
828                 /*
829                  * both row and column size are odd, do corner element s00+1,
830                  * s10, s10+1 are off edge
831                  */
832                 b[k] = (byte) (b2i(a[s00] != 0) << N3);
833                 k += 1;
834             }
835         }
836     }

838     private void shuffle(long[] a, int aOffset, int n, int n2, long[] tmp) {
839 
840         /*
841          * int a[]; tiledImageOperation to shuffle int n; number of elements to
842          * shuffle int n2; second dimension int tmp[]; scratch storage
843          */
844 
845         int i;
846         long[] p1, p2, pt;
847         int p1Offset = 0;
848         int ptOffset = 0;
849         int p2Offset = 0;
850         /*
851          * copy odd elements to tmp
852          */
853         pt = tmp;
854         ptOffset = 0;
855         p1 = a;
856         p1Offset = aOffset + n2;
857         for (i = 1; i < n; i += 2) {
858             pt[ptOffset] = p1[p1Offset];
859             ptOffset += 1;
860             p1Offset += n2 + n2;
861         }
862         /*
863          * compress even elements into first half of A
864          */
865         p1 = a;
866         p1Offset = aOffset + n2;
867         p2 = a;
868         p2Offset = aOffset + n2 + n2;
869         for (i = 2; i < n; i += 2) {
870             p1[p1Offset] = p2[p2Offset];
871             p1Offset += n2;
872             p2Offset += n2 + n2;
873         }
874         /*
875          * put odd elements into 2nd half
876          */
877         pt = tmp;
878         ptOffset = 0;
879         for (i = 1; i < n; i += 2) {
880             p1[p1Offset] = pt[ptOffset];
881             p1Offset += n2;
882             ptOffset += 1;
883         }
884     }

891     private void writeBdirect(ByteBuffer outfile, LongBuffer a, int n, int nqx, int nqy, byte[] scratch, int bit) {
892 
893         /*
894          * Write the direct bitmap warning code
895          */
896         outputNybble(outfile, 0x0);
897         /*
898          * Copy A to scratch tiledImageOperation (again!), packing 4 bits/nybble
899          */
900         qtreeOnebit(a, n, nqx, nqy, scratch, bit);
901         /*
902          * write to outfile
903          */
904         /*
905          * int i; for (i = 0; i < ((nqx+1)/2) * ((nqy+1)/2); i++) {
906          * output_nybble(outfile,scratch[i]); }
907          */
908         outputNnybble(outfile, (nqx + 1) / 2 * ((nqy + 1) / 2), scratch);
909 
910     }
