40  /**
41   * (<i>for internal use</i>) The The random seed initialization parameter as recorded in the FITS header.
42   * <p>
43   * The storing of the random seed provides consistent dither implementations. Dithering adds a small amount of noise (at
44   * the level of the quantization level increments) to remove possible systematics biases of the quantization. Since the
45   * quantization (integer representation of floating-point values) is inherently lossy (noisy) the added dithering noise
46   * is generally inconsequential in terms of preseving information in the original image. As such, the image can be
47   * recovered close enough to the original, within the level of the quantization noise, without knowing the exact random
48   * sequence that was used to generate the dither.
49   * </p>
50   * <p>
51   * However, The FITS standard requires that when dithering is used in the compression the same tool (or library) should
52   * also undo the dithering exactly using the same sequence of random numbers as was used when compressing the image.
53   * Furthermore, the standard makes some explicit suggestions on how pseudo-random sequences should be generated, and
54   * offers a specific convention for an algorithm that may be used to provide consistent dithering across tools,
55   * platforms, and library implementations.
56   * </p>
57   * <p>
58   * For the dithering to be reversible, the FITS header should store the integer value, usually between 1 and 10000
59   * inclusive, under the <code>ZDITHER0</code> keyword to indicate the value (random sequence index) that was used for
60   * seeding the random number generator according to the described standard.
61   * </p>
62   * <p>
63   * The nom-tam FITS library implements the suggested convention for the random number generator, and this parameter
64   * deals with recording the corresponding random seed value via the <code>ZDITHER0</code> keyword in the compressed
65   * image headers; as well as retrieving this value to seed the random number generator when decompressing the image, in
66   * a way that is consistent with the FITS standard and recommended convention.
67   * </p>
68   *
69   * @author Attila Kovacs
70   */
71  final class ZDither0Parameter extends CompressHeaderParameter<QuantizeOption> {
72  
73      /**
74       * Creates a new compression parameter that can be used to configure the quantization options when cmpressing /
75       * decompressing image tiles.
76       *
77       * @param quantizeOption The quantization option that will be configured using this particular parameter.
78       */
79      ZDither0Parameter(QuantizeOption quantizeOption) {
80          super(Compression.ZDITHER0.name(), quantizeOption);
81      }
82  
83      @Override
84      public void getValueFromHeader(IHeaderAccess header) {
85          if (getOption() == null) {
86              return;
87          }
88  
89          HeaderCard card = header.findCard(Compression.ZDITHER0);
90          getOption().setSeed(card == null ? 1L : card.getValue(Long.class, 1L));
91      }
92  
93      @Override
94      public void setValueInHeader(IHeaderAccess header) {
95          if (getOption() == null) {
96              return;
97          }
98  
99          header.addValue(Compression.ZDITHER0, (int) getOption().getSeed());
100     }
101 
102     /**
103      * Seeds the random generator for the specific tile
104      *
105      * @param  index                the 0-based tile index.
106      *
107      * @throws NullPointerException if the parameter is no linked to a {@link QuantizeOption} instance (that is
108      *                                  <code>null</code> was specified in the constructor).
109      */
110     void setTileIndex(int index) throws NullPointerException {
111         getOption().setTileIndex(index);
112     }
113 
114 }
