34  import nom.tam.fits.HeaderCard;
35  import nom.tam.fits.compression.algorithm.quant.QuantizeOption;
36  import nom.tam.fits.compression.provider.param.api.IHeaderAccess;
37  import nom.tam.fits.compression.provider.param.base.CompressHeaderParameter;
38  import nom.tam.fits.header.Compression;
39  
40  final class ZQuantizeParameter extends CompressHeaderParameter<QuantizeOption> {
41  
42      /**
43       * @param quantizeOption
44       */
45      ZQuantizeParameter(QuantizeOption quantizeOption) {
46          super(Compression.ZQUANTIZ.name(), quantizeOption);
47      }
48  
49      @Override
50      public void getValueFromHeader(IHeaderAccess header) {
51          HeaderCard card = header.findCard(getName());
52          String value = card != null ? card.getValue() : null;
53          if (Compression.ZQUANTIZ_SUBTRACTIVE_DITHER_2.equals(value)) {
54              getOption().setDither(true);
55              getOption().setDither2(true);
56          } else if (Compression.ZQUANTIZ_SUBTRACTIVE_DITHER_1.equals(value)) {
57              getOption().setDither(true);
58              getOption().setDither2(false);
59          } else {
60              getOption().setDither(false);
61              getOption().setDither2(false);
62          }
63      }
64  
65      @Override
66      public void setValueInHeader(IHeaderAccess header) {
67          String value;
68          if (getOption().isDither2()) {
69              value = Compression.ZQUANTIZ_SUBTRACTIVE_DITHER_2;
70          } else if (getOption().isDither()) {
71              value = Compression.ZQUANTIZ_SUBTRACTIVE_DITHER_1;
72          } else {
73              value = Compression.ZQUANTIZ_NO_DITHER;
74          }
75          header.addValue(Compression.ZQUANTIZ, value);
76      }
77  }
