39  /**
40   * Group of parameters that must be synchronized with the hdu meta data for a
41   * specific compression algorithm.
42   */
43  public interface ICompressParameters {
44  
45      /**
46       * Add the columns that hold the metadata for the parameters that are column
47       * based to the dhu.
48       * 
49       * @param hdu
50       *            the hdu to add the column
51       * @throws FitsException
52       *             if the column could not be added.
53       */
54      void addColumnsToTable(BinaryTableHDU hdu) throws FitsException;
55  
56      /**
57       * create a copy of this parameter for another option (normally a copy of
58       * the current option).
59       * 
60       * @param option
61       *            the new option for the copied parameter
62       * @return this (builder pattern)
63       */
64      ICompressParameters copy(ICompressOption option);
65  
66      /**
67       * extract the option data from the column and set it in the option.
68       * 
69       * @param index
70       *            the index in the column.
71       */
72      void getValuesFromColumn(int index);
73  
74      /**
75       * extract the option values that are represented by headers from the hdu
76       * header.
77       * 
78       * @param header
79       *            the header to extract the option values.
80       */
81      void getValuesFromHeader(IHeaderAccess header);
82  
83      /**
84       * initialize the column based options of the compression algorithm from the
85       * binary table.
86       * 
87       * @param header
88       *            the header of the hdu
89       * @param binaryTable
90       *            the table of the hdu
91       * @param size
92       *            the column size
93       * @throws FitsException
94       *             if the column could not be initialized
95       */
96      void initializeColumns(IHeaderAccess header, BinaryTable binaryTable, int size) throws FitsException;
97  
98      /**
99       * initialize the column based parameter to the specified column length.
100      * 
101      * @param length
102      *            the column length.
103      */
104     void initializeColumns(int length);
105 
106     /**
107      * set the option values, that are column based, into the columns at the
108      * specified index.
109      * 
110      * @param index
111      *            the index in the columns to set.
112      */
113     void setValueFromColumn(int index);
114 
115     /**
116      * set the options values, that are hdu based, into the header.
117      * 
118      * @param header
119      *            the header to set the option value
120      * @throws HeaderCardException
121      *             if the header could not be set.
122      */
123     void setValuesInHeader(IHeaderAccess header) throws HeaderCardException;
124 
125 }
