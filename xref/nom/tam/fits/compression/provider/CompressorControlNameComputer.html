34  import nom.tam.fits.header.Compression;
35  
36  /**
37   * <p>
38   * Computes the name of the tile compressor class name given the algorithm used
39   * to quantize and compress the tile and the type of data the tile contains.
40   * </p>
41   * The name of the class is built of four parts:
42   * <ul>
43   * <li>the capitalized simple name of the base type of the elements in the tile
44   * (like Int, Long etc.);</li>
45   * <li>if a known quantize algorithm is used, the word "Quant", the word
46   * "Unknown" if the quantize algorithm is not recognized, nothing (i.e. the
47   * empty string) if it is null;</li>
48   * <li>the short name of the compression algorithm to use (Rice, PLIO, Gzip
49   * etc.) or the word "Unknown" if the algorithm is not supported;</li>
50   * <li>the suffix "Compressor"</li>
51   * </ul>
52   * <p>
53   * Following exception to above rules exist:
54   * </p>
55   * <ul>
56   * <li>If the primitive type is double or float, the quantize algorithm is
57   * ignored (as if it were specified as null)</li>
58   * </ul>
59   * See the associated unit tests for concrete examples.
60   */
61  public class CompressorControlNameComputer {
62  
63      private static final String COMPRESSOR_CLASS_SUFFIX = "Compressor";
64  
65      private static String standardizeBaseType(String simpleName) {
66          return Character.toUpperCase(simpleName.charAt(0)) + simpleName.substring(1).toLowerCase();
67      }
68  
69      private static String standardizeCompressionAlgorithm(String compressionAlgorithm) {
70          if (Compression.ZCMPTYPE_RICE_1.equalsIgnoreCase(compressionAlgorithm) || //
71                  Compression.ZCMPTYPE_RICE_ONE.equalsIgnoreCase(compressionAlgorithm)) {
72              return "Rice";
73          } else if (Compression.ZCMPTYPE_PLIO_1.equalsIgnoreCase(compressionAlgorithm)) {
74              return "PLIO";
75          } else if (Compression.ZCMPTYPE_HCOMPRESS_1.equalsIgnoreCase(compressionAlgorithm)) {
76              return "H";
77          } else if (Compression.ZCMPTYPE_GZIP_2.equalsIgnoreCase(compressionAlgorithm)) {
78              return "GZip2";
79          } else if (Compression.ZCMPTYPE_GZIP_1.equalsIgnoreCase(compressionAlgorithm)) {
80              return "GZip";
81          } else if (Compression.ZCMPTYPE_NOCOMPRESS.equalsIgnoreCase(compressionAlgorithm)) {
82              return "NoCompress";
83          }
84          return "Unknown";
85      }
86  
87      private static String standardizeQuantAlgorithm(String quantAlgorithm) {
88          if (quantAlgorithm != null) {
89              if (Compression.ZQUANTIZ_NO_DITHER.equalsIgnoreCase(quantAlgorithm) || //
90                      Compression.ZQUANTIZ_SUBTRACTIVE_DITHER_1.equalsIgnoreCase(quantAlgorithm) || //
91                      Compression.ZQUANTIZ_SUBTRACTIVE_DITHER_2.equalsIgnoreCase(quantAlgorithm)) {
92                  return "Quant";
93              } else {
94                  return "Unknown";
95              }
96          }
97          return "";
98      }
99  
100     public CompressorControlNameComputer() {
101         super();
102     }
103 
104     public String createCompressorClassName(String quantAlgorithm, String compressionAlgorithm, Class<?> baseType) {
105         StringBuilder className = new StringBuilder();
106         className.append(standardizeBaseType(baseType.getSimpleName()));
107         if (className.indexOf(Float.class.getSimpleName()) == 0 || className.indexOf(Double.class.getSimpleName()) == 0) {
108             quantAlgorithm = null; // default so not in the className
109         }
110         className.append(standardizeQuantAlgorithm(quantAlgorithm));
111         className.append(standardizeCompressionAlgorithm(compressionAlgorithm));
112         className.append(COMPRESSOR_CLASS_SUFFIX);
113         return className.toString();
114     }
115 }
