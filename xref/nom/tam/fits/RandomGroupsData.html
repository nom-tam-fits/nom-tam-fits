34  import static nom.tam.fits.header.Standard.GCOUNT;
35  import static nom.tam.fits.header.Standard.GROUPS;
36  import static nom.tam.fits.header.Standard.NAXISn;
37  import static nom.tam.fits.header.Standard.PCOUNT;
38  
39  import java.io.EOFException;
40  import java.io.IOException;
41  
42  import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
43  import nom.tam.fits.header.Standard;
44  import nom.tam.util.ArrayDataInput;
45  import nom.tam.util.ArrayDataOutput;
46  import nom.tam.util.ArrayFuncs;
47  
48  /**
49   * This class instantiates FITS Random Groups data. Random groups are
50   * instantiated as a two-dimensional array of objects. The first dimension of
51   * the array is the number of groups. The second dimension is 2. The first
52   * object in every row is a one dimensional parameter array. The second element
53   * is the n-dimensional data array.
54   */
55  public class RandomGroupsData extends Data {
56  
57      private final Object[][] dataArray;
58  
59      /**
60       * Create the equivalent of a null data element.
61       */
62      public RandomGroupsData() {
63          this.dataArray = new Object[0][];
64      }
65  
66      /**
67       * Create a RandomGroupsData object using the specified object to initialize
68       * the data array.
69       * 
70       * @param x
71       *            The initial data array. This should a two-d array of objects
72       *            as described above.
73       */
74      @SuppressFBWarnings(value = "EI_EXPOSE_REP", justification = "intended exposure of mutable data")
75      public RandomGroupsData(Object[][] x) {
76          this.dataArray = x;
77      }
78  
79      @Override
80      protected void fillHeader(Header h) throws FitsException {
81  
82          if (this.dataArray.length <= 0 || this.dataArray[0].length != 2) {
83              throw new FitsException("Data not conformable to Random Groups");
84          }
85          Standard.context(RandomGroupsData.class);
86          Object paraSamp = this.dataArray[0][0];
87          Object dataSamp = this.dataArray[0][1];
88  
89          Class<?> pbase = ArrayFuncs.getBaseClass(paraSamp);
90          Class<?> dbase = ArrayFuncs.getBaseClass(dataSamp);
91  
92          if (pbase != dbase) {
93              throw new FitsException("Data and parameters do not agree in type for random group");
94          }
95  
96          int[] pdims = ArrayFuncs.getDimensions(paraSamp);
97          int[] ddims = ArrayFuncs.getDimensions(dataSamp);
98  
99          if (pdims.length != 1) {
100             throw new FitsException("Parameters are not 1 d array for random groups");
101         }
102 
103         // Got the information we need to build the header.
104 
105         h.setSimple(true);
106         if (dbase == byte.class) {
107             h.setBitpix(BasicHDU.BITPIX_BYTE);
108         } else if (dbase == short.class) {
109             h.setBitpix(BasicHDU.BITPIX_SHORT);
110         } else if (dbase == int.class) {
111             h.setBitpix(BasicHDU.BITPIX_INT);
112         } else if (dbase == long.class) { // Non-standard
113             h.setBitpix(BasicHDU.BITPIX_LONG);
114         } else if (dbase == float.class) {
115             h.setBitpix(BasicHDU.BITPIX_FLOAT);
116         } else if (dbase == double.class) {
117             h.setBitpix(BasicHDU.BITPIX_DOUBLE);
118         } else {
119             throw new FitsException("Data type:" + dbase + " not supported for random groups");
120         }
121 
122         h.setNaxes(ddims.length + 1);
123         h.addValue(NAXISn.n(1), 0);
124         for (int i = 2; i <= ddims.length + 1; i += 1) {
125             h.addValue(NAXISn.n(i), ddims[i - 2]);
126         }
127 
128         h.addValue(GROUPS, true);
129         h.addValue(GCOUNT, this.dataArray.length);
130         h.addValue(PCOUNT, pdims[0]);
131         Standard.context(null);
132     }
133 
134     @Override
135     @SuppressFBWarnings(value = "EI_EXPOSE_REP", justification = "intended exposure of mutable data")
136     public Object getData() {
137         return this.dataArray;
138     }
139 
140     /** Get the size of the actual data element. */
141     @Override
142     protected long getTrueSize() {
143 
144         if (this.dataArray != null && this.dataArray.length > 0) {
145             return (ArrayFuncs.computeLSize(this.dataArray[0][0]) + ArrayFuncs.computeLSize(this.dataArray[0][1])) * this.dataArray.length;
146         } else {
147             return 0;
148         }
149     }
150 
151     /** Read the RandomGroupsData */
152     @Override
153     public void read(ArrayDataInput str) throws FitsException {
154 
155         setFileOffset(str);
156 
157         try {
158             str.readLArray(this.dataArray);
159         } catch (IOException e) {
160             throw new FitsException("IO error reading Random Groups data " + e);
161         }
162         int pad = FitsUtil.padding(getTrueSize());
163         try {
164             str.skipAllBytes(pad);
165         } catch (EOFException e) {
166             throw new PaddingException("EOF reading padding after random groups", this, e);
167         } catch (IOException e) {
168             throw new FitsException("IO error reading padding after random groups", e);
169         }
170     }
171 
172     /** Write the RandomGroupsData */
173     @Override
174     public void write(ArrayDataOutput str) throws FitsException {
175         try {
176             str.writeArray(this.dataArray);
177             FitsUtil.pad(str, getTrueSize());
178         } catch (IOException e) {
179             throw new FitsException("IO error writing random groups data " + e);
180         }
181     }
182 
183 }
