273      *

275      * @param  value               value (can be <code>null</code>, in which case the card type defaults to

284      * @see                        #create(IFitsHeader, Number)
285      * @see                        FitsFactory#setUseExponentD(boolean)
286      */
287     public HeaderCard(String key, Number value, int decimals, String comment) throws HeaderCardException {
288         if (value == null) {
289             set(key, null, comment, Integer.class);
290             return;
291         }
292 
293         try {
294             checkNumber(value);
295         } catch (NumberFormatException e) {
296             throw new HeaderCardException("FITS headers may not contain NaN or Infinite values", e);
297         }
298         set(key, new FlexFormat().setWidth(spaceForValue(key)).setPrecision(decimals).format(value), comment,
299                 value.getClass());
300     }

518     /**
519      * Sets all components of the card to the specified values. For internal use only.
520      *
521      * @param  aKey                Case-sensitive keyword (can be <code>null</code> for an unkeyed comment)
522      * @param  aValue              the serialized value (tailing spaces will be removed), or <code>null</code>
523      * @param  aComment            an optional comment or <code>null</code>.
524      * @param  aType               The Java class from which the value field was derived, or null if it's a
525      *                                 comment-style card.
526      *
527      * @throws HeaderCardException for any invalid keyword or value
528      */
529     private synchronized void set(String aKey, String aValue, String aComment, Class<?> aType) throws HeaderCardException {
530         // TODO we never call with null type and non-null value internally, so this is dead code here...
531         // if (aType == null && aValue != null) {
532         // throw new HeaderCardException("Null type for value: [" + sanitize(aValue) + "]");
533         // }
534 
535         type = aType;
536 
537         // AK: Map null and blank keys to BLANKS.key()
538         // This simplifies things as we won't have to check for null keys separately!
539         if ((aKey == null) || aKey.trim().isEmpty()) {
540             aKey = EMPTY_KEY;
541         }
542 
543         if (aKey.isEmpty() && aValue != null) {
544             throw new HeaderCardException("Blank or null key for value: [" + sanitize(aValue) + "]");
545         }
546 
547         try {
548             validateKey(aKey);
549         } catch (RuntimeException e) {
550             throw new HeaderCardException("Invalid FITS keyword: [" + sanitize(aKey) + "]", e);
551         }
552 
553         key = aKey;
554 
555         try {
556             validateChars(aComment);
557         } catch (IllegalArgumentException e) {
558             throw new HeaderCardException("Invalid FITS comment: [" + sanitize(aComment) + "]", e);
559         }
560 
561         comment = aComment;
562 
563         try {
564             validateChars(aValue);
565         } catch (IllegalArgumentException e) {
566             throw new HeaderCardException("Invalid FITS value: [" + sanitize(aValue) + "]", e);
567         }
568 
569         if (aValue == null) {
570             value = null;
571             return;
572         }
573         if (isStringValue()) {
574             try {
575                 setValue(aValue);
576             } catch (Exception e) {
577                 throw new HeaderCardException("Value too long: [" + sanitize(aValue) + "]", e);
578             }
579         } else {
580             aValue = aValue.trim();
581 
582             // Check that the value fits in the space available for it.
583             if (aValue.length() > spaceForValue()) {
584                 throw new HeaderCardException("Value too long: [" + sanitize(aValue) + "]",
585                         new LongValueException(key, spaceForValue()));
586             }
587 
588             value = aValue;
589         }
590     }

608         if (FitsFactory.isLongStringsEnabled() && isStringValue() && value != null) {

684     /**
685      * <p>
686      * Returns the value cast to the specified type, if possible, or the specified default value if the value is
687      * <code>null</code> or if the value is incompatible with the requested type.
688      * </p>
689      * <p>
690      * For number types and values, if the requested type has lesser range or precision than the number stored in the
691      * FITS header, the value is automatically downcast (i.e. possible rounded and/or truncated) -- the same as if an
692      * explicit cast were used in Java. As long as the header value is a proper decimal value, it will be returned as
693      * any requested number type.
694      * </p>
695      *
696      * @param  asType                   the requested class of the value
697      * @param  defaultValue             the value to use if the card has a null value, or a value that cannot be cast to
698      *                                      the specified type.
699      * @param  <T>                      the generic type of the requested class
700      *
701      * @return                          the value from this card as a specific type, or the specified default value
702      *
703      * @throws IllegalArgumentException if the specified Java type of not one that is supported for use in FITS headers.
704      */
705     public synchronized <T> T getValue(Class<T> asType, T defaultValue) throws IllegalArgumentException {
706 
707         if (value == null) {
708             return defaultValue;
709         }
710         if (String.class.isAssignableFrom(asType)) {
711             return asType.cast(value);
712         }
713         if (value.isEmpty()) {
714             return defaultValue;
715         }
716         if (Boolean.class.isAssignableFrom(asType)) {
717             return asType.cast(getBooleanValue((Boolean) defaultValue));
718         }
719         if (ComplexValue.class.isAssignableFrom(asType)) {
720             return asType.cast(new ComplexValue(value));
721         }
722         if (Number.class.isAssignableFrom(asType)) {
723             try {
724                 BigDecimal big = new BigDecimal(value.toUpperCase().replace('D', 'E'));
725 
726                 if (Byte.class.isAssignableFrom(asType)) {
727                     return asType.cast(big.byteValue());
728                 }
729                 if (Short.class.isAssignableFrom(asType)) {
730                     return asType.cast(big.shortValue());
731                 }
732                 if (Integer.class.isAssignableFrom(asType)) {
733                     return asType.cast(big.intValue());
734                 }
735                 if (Long.class.isAssignableFrom(asType)) {
736                     return asType.cast(big.longValue());
737                 }
738                 if (Float.class.isAssignableFrom(asType)) {
739                     return asType.cast(big.floatValue());
740                 }
741                 if (Double.class.isAssignableFrom(asType)) {
742                     return asType.cast(big.doubleValue());
743                 }
744                 if (BigInteger.class.isAssignableFrom(asType)) {
745                     return asType.cast(big.toBigInteger());
746                 }
747                 // All possibilities have been exhausted, it must be a BigDecimal...
748                 return asType.cast(big);
749             } catch (NumberFormatException e) {
750                 // The value is not a decimal number, so return the default value by contract.
751                 return defaultValue;
752             }
753         }
754 
755         throw new IllegalArgumentException("unsupported class " + asType);
756     }

1188     /**
1189      * Parses a continued long string value and comment for this card, which may occupy one or more consecutive
1190      * 80-character header records.
1191      *
1192      * @param  dis                    the input stream from which to parse the value and comment fields of this card.
1193      * @param  next                   the parser to use for each 80-character record.
1194      *
1195      * @throws IOException            if there was an IO error reading the stream.
1196      * @throws TruncatedFileException if the stream endedc ubnexpectedly in the middle of an 80-character record.
1197      */
1198     @SuppressWarnings("deprecation")
1199     private synchronized void parseLongStringCard(HeaderCardCountingArrayDataInput dis, HeaderCardParser next)
1200             throws IOException, TruncatedFileException {
1201 
1202         StringBuilder longValue = new StringBuilder();
1203         StringBuilder longComment = null;
1204 
1205         while (next != null) {
1206             if (!next.isString()) {
1207                 break;
1208             }
1209             String valuePart = next.getValue();
1210             String untrimmedComment = next.getUntrimmedComment();
1211 
1212             if (valuePart == null) {
1213                 // The card cannot have a null value. If it does it wasn't a string card...
1214                 break;
1215             }
1216 
1217             // The end point of the value
1218             int valueEnd = valuePart.length();
1219 
1220             // Check if there card continues into the next record. The value
1221             // must end with '&' and the next card must be a CONTINUE card.
1222             // If so, remove the '&' from the value part, and parse in the next
1223             // card for the next iteration...
1224             if (!dis.markSupported()) {
1225                 throw new IOException("InputStream does not support mark/reset");
1226             }
1227 
1228             // Peek at the next card.
1229             dis.mark();
1230 
1231             try {
1232                 // Check if we should continue parsing this card...
1233                 next = new HeaderCardParser(readOneHeaderLine(dis));
1234                 if (valuePart.endsWith("&") && CONTINUE.key().equals(next.getKey())) {
1235                     // Remove '& from the value part...
1236                     valueEnd--;
1237                 } else {
1238                     // ok move the input stream one card back.
1239                     dis.reset();
1240                     // Clear the parser also.
1241                     next = null;
1242                 }
1243             } catch (EOFException e) {
1244                 // Nothing left to parse after the current one...
1245                 next = null;
1246             }
1247 
1248             // Append the value part from the record last parsed.
1249             longValue.append(valuePart, 0, valueEnd);
1250 
1251             // Append any comment from the card last parsed.
1252             if (untrimmedComment != null) {
1253                 if (longComment == null) {
1254                     longComment = new StringBuilder(untrimmedComment);
1255                 } else {
1256                     longComment.append(untrimmedComment);
1257                 }
1258             }
1259         }
1260 
1261         comment = longComment == null ? null : longComment.toString().trim();
1262         value = trimEnd(longValue.toString());
1263         type = String.class;
1264     }

1268      * not significant (but leading spaces are). As such we should remove trailing spaces when parsing header string

1433     /**
1434      * <p>
1435      * Creates a new FITS header card from a FITS stream representation of it, which is how the key/value and comment
1436      * are represented inside the FITS file, normally as an 80-character wide entry. The parsing of header 'lines'
1437      * conforms to all FITS standards, and some optional conventions, such as HIERARCH keywords (if
1438      * {@link FitsFactory#setUseHierarch(boolean)} is enabled), COMMENT and HISTORY entries, and OGIP 1.0 long CONTINUE
1439      * lines (if {@link FitsFactory#setLongStringsEnabled(boolean)} is enabled).
1440      * </p>
1441      * <p>
1442      * However, the parsing here is permissive beyond the standards and conventions, and will do its best to support a
1443      * wide range of FITS files, which may deviate from the standard in subtle (or no so subtle) ways.
1444      * </p>
1445      * <p>
1446      * Here is a brief summary of the rules that guide the parsing of keywords, values, and comment 'fields' from the
1447      * single header line:
1448      * </p>
1449      * <p>
1450      * <b>A. Keywords</b>
1451      * </p>
1452      * <ul>
1453      * <li>The standard FITS keyword is the first 8 characters of the line, or up to an equal [=] character, whichever
1454      * comes first, with trailing spaces removed, and always converted to upper-case.</li>
1455      * <li>If {@link FitsFactory#setUseHierarch(boolean)} is enabled, structured longer keywords can be composed after a
1456      * <code>HIERARCH</code> base key, followed by space (and/or dot ].]) separated parts, up to an equal sign [=]. The
1457      * library will represent the same components (including <code>HIERARCH</code>) but separated by single dots [.].
1458      * For example, the header line starting with [<code>HIERARCH SMA OBS TARGET =</code>], will be referred as
1459      * [<code>HIERARCH.SMA.OBS.TARGET</code>] withing this library. The keyword parts can be composed of any ASCII
1460      * characters except dot [.], white spaces, or equal [=].</li>
1461      * <li>By default, all parts of the key are converted to upper-case. Case sensitive HIERARCH keywords can be
1462      * retained after enabling
1463      * {@link nom.tam.fits.header.hierarch.IHierarchKeyFormatter#setCaseSensitive(boolean)}.</li>
1464      * </ul>
1465      * <p>
1466      * <b>B. Values</b>
1467      * </p>
1468      * <p>
1469      * Values are the part of the header line, that is between the keyword and an optional ending comment. Legal header
1470      * values follow the following parse patterns:
1471      * <ul>
1472      * <li>Begin with an equal sign [=], or else come after a CONTINUE keyword.</li>
1473      * <li>Next can be a quoted value such as <code>'hello'</code>, placed inside two single quotes. Or an unquoted
1474      * value, such as <code>123</code>.</li>
1475      * <li>Quoted values must begin with a single quote ['] and and with the next single quote. If there is no end-quote
1476      * in the line, it is not considered a string value but rather a comment, unless
1477      * {@link FitsFactory#setAllowHeaderRepairs(boolean)} is enabled, in which case the entire remaining line after the
1478      * opening quote is assumed to be a malformed value.</li>
1479      * <li>Unquoted values end at the fist [/] character, or else go until the line end.</li>
1480      * <li>Quoted values have trailing spaces removed, s.t. [<code>'  value   '</code>] becomes
1481      * [<code>  value</code>].</li>
1482      * <li>Unquoted values are trimmed, with both leading and trailing spaces removed, e.g. [<code>  123  </code>]
1483      * becomes [<code>123</code>].</li>
1484      * </ul>
1485      * <p>
1486      * <b>C. Comments</b>
1487      * </p>
1488      * <p>
1489      * The following rules guide the parsing of the values component:
1490      * <ul>
1491      * <li>If a value is present (see above), the comment is what comes after it. That is, for quoted values, everything
1492      * that follows the closing quote. For unquoted values, it's what comes after the first [/], with the [/] itself
1493      * removed.</li>
1494      * <li>If a value is not present, then everything following the keyword is considered the comment.</li>
1495      * <li>Comments are trimmed, with both leading and trailing spaces removed.</li>
1496      * </ul>
1497      *
1498      * @return                          a newly created HeaderCard from a FITS card string.
1499      *
1500      * @param  line                     the card image (typically 80 characters if in a FITS file).
1501      *
1502      * @throws IllegalArgumentException if the card was malformed, truncated, or if there was an IO error.
1503      *
1504      * @see                             FitsFactory#setUseHierarch(boolean)
1505      * @see                             nom.tam.fits.header.hierarch.IHierarchKeyFormatter#setCaseSensitive(boolean)
1506      */
1507     public static HeaderCard create(String line) throws IllegalArgumentException {
1508         try (ArrayDataInput in = stringToArrayInputStream(line)) {
1509             return new HeaderCard(in);
1510         } catch (Exception e) {
1511             throw new IllegalArgumentException("card not legal", e);
1512         }
1513     }

1788         byte[] buffer = new byte[FITS_HEADER_CARD_SIZE];
1789

1791

1793             // Read as long as there is more available, even if it comes in a trickle...
1794             while (got < buffer.length) {
1795                 int n = in.read(buffer, got, buffer.length - got);
1796                 if (n < 0) {
1797                     break;
1798                 }
1799                 got += n;

1803         }

1806             // Nothing left to read.
1807             throw new EOFException();
1808         }
1809 
1810         if (got < buffer.length) {
1811             // Got an incomplete header card...
1812             throw new TruncatedFileException(
1813                     "Got only " + got + " of " + buffer.length + " bytes expected for a header card");
1814         }
1815

1844      *
1845      * @return     the space available for the value field in a single record, after the keyword, and the assigmnent
1846      *                 sequence (or equivalent blank space).
1847      */

1849         if (key.length() > MAX_KEYWORD_LENGTH) {
1850             IHierarchKeyFormatter fmt = FitsFactory.getHierarchFormater();
1851             int keyLen = Math.max(key.length(), MAX_KEYWORD_LENGTH) + fmt.getExtraSpaceRequired(key);
1852             return FITS_HEADER_CARD_SIZE - keyLen - fmt.getMinAssignLength();
1853         }
1854         return FITS_HEADER_CARD_SIZE - (Math.max(key.length(), MAX_KEYWORD_LENGTH) + HeaderCardFormatter.getAssignLength());
1855     }

1901     /**
1902      * Replaces illegal characters in the string ith '?' to be suitable for FITS header records. According to the FITS
1903      * standard, headers may only contain ASCII characters in the range 0x20 and 0x7E (inclusive).
1904      *
1905      * @param  str the input string.
1906      *
1907      * @return     the sanitized string for use in a FITS header, with illegal characters replaced by '?'.
1908      *
1909      * @see        #isValidChar(char)
1910      * @see        #validateChars(String)
1911      */
1912     public static String sanitize(String str) {
1913         int nc = str.length();
1914         char[] cbuf = new char[nc];
1915         for (int ic = 0; ic < nc; ic++) {
1916             char c = str.charAt(ic);
1917             cbuf[ic] = isValidChar(c) ? c : '?';
1918         }
1919         return new String(cbuf);
1920     }

1937     /**
1938      * Checks the specified string for characters that are not allowed in FITS headers, and throws an exception if any
1939      * are found. According to the FITS standard, headers may only contain ASCII characters in the range 0x20 and 0x7E
1940      * (inclusive).
1941      *
1942      * @param  text                     the input string
1943      *
1944      * @throws IllegalArgumentException if the unput string contains any characters that cannot be in a FITS header,
1945      *                                      that is characters outside of the 0x20 to 0x7E range.
1946      *
1947      * @since                           1.16
1948      *
1949      * @see                             #isValidChar(char)
1950      * @see                             #sanitize(String)
1951      * @see                             #validateKey(String)
1952      */
1953     public static void validateChars(String text) throws IllegalArgumentException {
1954         if (text == null) {
1955             return;
1956         }
1957 
1958         for (int i = text.length(); --i >= 0;) {
1959             char c = text.charAt(i);
1960             if (c < MIN_VALID_CHAR) {
1961                 throw new IllegalArgumentException(
1962                         "Non-printable character(s), e.g. 0x" + (int) c + ", in [" + sanitize(text) + "].");
1963             }
1964             if (c > MAX_VALID_CHAR) {
1965                 throw new IllegalArgumentException(
1966                         "Extendeed ASCII character(s) in [" + sanitize(text) + "]. Only 0x20 through 0x7E are allowed.");
1967             }
1968         }
1969     }

1971     /**
1972      * Checks if the specified string may be used as a FITS header keyword according to the FITS standard and currently
1973      * settings for supporting extensions to the standard, such as HIERARCH-style keywords.
1974      *
1975      * @param  key                      the proposed keyword string
1976      *
1977      * @throws IllegalArgumentException if the string cannot be used as a FITS keyword with the current settings. The
1978      *                                      exception will contain an informative message describing the issue.
1979      *
1980      * @since                           1.16
1981      *
1982      * @see                             #validateChars(String)
1983      * @see                             FitsFactory#setUseHierarch(boolean)
1984      */
1985     public static void validateKey(String key) throws IllegalArgumentException {
1986         int maxLength = MAX_KEYWORD_LENGTH;
1987         if (isHierarchKey(key)) {
1988             if (!FitsFactory.getUseHierarch()) {
1989                 throw new HierarchNotEnabledException(key);
1990             }
1991             maxLength = MAX_HIERARCH_KEYWORD_LENGTH;
1992             validateHierarchComponents(key);
1993         }
1994 
1995         if (key.length() > maxLength) {
1996             throw new IllegalArgumentException("Keyword is too long: [" + sanitize(key) + "]");
1997         }
1998 
1999         // Check the whole key for non-printable, non-standard ASCII
2000         for (int i = key.length(); --i >= 0;) {
2001             char c = key.charAt(i);
2002             if (c < MIN_VALID_CHAR) {
2003                 throw new IllegalArgumentException(
2004                         "Keyword contains non-printable character 0x" + (int) c + ": [" + sanitize(key) + "].");
2005             }
2006             if (c > MAX_VALID_CHAR) {
2007                 throw new IllegalArgumentException("Keyword contains extendeed ASCII characters: [" + sanitize(key)
2008                         + "]. Only 0x20 through 0x7E are allowed.");
2009             }
2010         }
2011 
2012         // Check if the first 8 characters conform to strict FITS specification...
2013         for (int i = Math.min(MAX_KEYWORD_LENGTH, key.length()); --i >= 0;) {
2014             char c = key.charAt(i);
2015             if ((c >= 'a' && c <= 'z') || (c >= 'A' && c <= 'Z')) {
2016                 continue;
2017             }
2018             if ((c >= '0' && c <= '9') || (c == '-') || (c == '_')) {
2019                 continue;
2020             }
2021             throw new IllegalArgumentException(
2022                     "Keyword [" + sanitize(key) + "] contains invalid characters. Only [A-Z][a-z][0-9][-][_] are allowed.");
2023         }
2024     }

2026     /**
2027      * Additional checks the extended components of the HIEARCH key (in bytes 9-77), to make sure they conform to our
2028      * own standards of storing hierarch keys as a dot-separated list of components. That is, the keyword must not have
2029      * any spaces...
2030      *
2031      * @param  key                      the HIERARCH keyword to check.
2032      *
2033      * @throws IllegalArgumentException if the keyword is not a proper dot-separated set of non-empty hierarchical
2034      *                                      components
2035      */
2036     private static void validateHierarchComponents(String key) throws IllegalArgumentException {
2037         for (int i = key.length(); --i >= 0;) {
2038             if (Character.isSpaceChar(key.charAt(i))) {
2039                 throw new IllegalArgumentException(
2040                         "No spaces allowed in HIERARCH keywords used internally: [" + sanitize(key) + "].");
2041             }
2042         }
2043 
2044         if (key.indexOf("..") >= 0) {
2045             throw new IllegalArgumentException("HIERARCH keywords with empty component: [" + sanitize(key) + "].");
2046         }
2047     }
