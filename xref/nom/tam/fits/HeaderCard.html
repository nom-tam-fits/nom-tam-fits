257      *

263     public HeaderCard(String key, Number value, int decimals, String comment) throws HeaderCardException {
264         if (value == null) {
265             set(key, null, comment, Integer.class);
266             return;
267         }
268         
269         try { 
270             checkNumber(value);        
271         } catch (NumberFormatException e) {
272             throw new HeaderCardException("FITS headers may not contain NaN or Infinite values", e);
273         }
274         set(key, new FlexFormat().setWidth(spaceForValue(key)).setPrecision(decimals).format(value), comment, value.getClass());
275     }

497     /**
498      * Sets all components of the card to the specified values. For internal use only.
499      *
500      * @param aKey       Case-sensitive keyword (can be <code>null</code> for an unkeyed comment)
501      * @param aValue     the serialized value (tailing spaces will be removed), or <code>null</code>
502      * @param aComment   an optional comment or <code>null</code>.
503      * @param aType      The Java class from which the value field was derived, or
504      *                   null if it's a comment-style card.
505      * 
506      * @throws HeaderCardException for any invalid keyword or value
507      */
508     private synchronized void set(String aKey, String aValue, String aComment, Class<?> aType) throws HeaderCardException {
509         // TODO we never call with null type and non-null value internally, so this is dead code here...
510 //        if (aType == null && aValue != null) {
511 //            throw new HeaderCardException("Null type for value: [" + sanitize(aValue) + "]");
512 //        }
513         
514         this.type = aType;
515         
516         // AK: Map null and blank keys to BLANKS.key()
517         // This simplifies things as we won't have to check for null keys separately!
518         if (aKey == null) {
519             aKey = EMPTY_KEY;
520         } else if (aKey.trim().isEmpty()) {
521             aKey = EMPTY_KEY;
522         }
523       
524         if (aKey.isEmpty() && aValue != null) {
525             throw new HeaderCardException("Blank or null key for value: [" + sanitize(aValue) + "]");
526         } 
527         
528         try {
529             validateKey(aKey);
530         } catch (RuntimeException e) {
531             throw new HeaderCardException("Invalid FITS keyword: [" + sanitize(aKey) + "]", e);
532         }
533 
534         this.key = aKey;
535        
536         try {
537             validateChars(aComment);
538         } catch (IllegalArgumentException e) {
539             throw new HeaderCardException("Invalid FITS comment: [" + sanitize(aComment) + "]", e);
540         }
541        
542         this.comment = aComment;
543        
544         try {
545             validateChars(aValue);
546         } catch (IllegalArgumentException e) {
547             throw new HeaderCardException("Invalid FITS value: [" + sanitize(aValue) + "]", e);
548         }
549         
550         if (aValue == null) {
551             this.value = null;
552             return;
553         } else if (isStringValue()) { 
554             // Discard trailing spaces
555             int to = aValue.length();
556             while (--to >= 0) {
557                 if (!Character.isSpaceChar(aValue.charAt(to))) {
558                     break;
559                 }
560             }
561             to++;
562             if (to < aValue.length()) {
563                 aValue = aValue.substring(0, to);
564             }
565 
566             // Remember that quotes get doubled in the value...
567             String printValue = aValue.replace("'", "''");
568 
569             // Check that the value fits in the space available for it.
570             if (!FitsFactory.isLongStringsEnabled() && (printValue.length() + STRING_QUOTES_LENGTH) > spaceForValue()) {
571                 throw new HeaderCardException("value too long: [" + sanitize(aValue) + "]", new LongStringsNotEnabledException(key));
572             }
573 
574         } else {
575             aValue = aValue.trim();
576 
577             // Check that the value fits in the space available for it.
578             if (aValue.length() > spaceForValue()) {
579                 throw new HeaderCardException("Value too long: [" + sanitize(aValue) + "]", new LongValueException(key, spaceForValue()));
580             }
581         }
582         
583         this.value = aValue;
584     }

679     /**
680      * <p>
681      * Returns the value cast to the specified type, if possible, or the specified default value if the
682      * value is <code>null</code> or if the value is incompatible with the requested type. 
683      * </p>
684      * <p>
685      * For number types and values, if the requested type has lesser range or
686      * precision than the number stored in the FITS header, the value is automatically downcast (i.e.
687      * possible rounded and/or truncated) -- the same as if an explicit cast were used in Java. As long
688      * as the header value is a proper decimal value, it will be returned as any requested number type.
689      * </p>
690      * 
691      * @param asType        the requested class of the value
692      * @param defaultValue  the value to use if the card has a null value, or a value that cannot
693      *                      be cast to the specified type.
694      * @param <T>           the generic type of the requested class
695      * 
696      * @return the value from this card as a specific type, or the specified default value
697      * 
698      * @throws IllegalArgumentException     
699      *                      if the specified Java type of not one that is supported for use in 
700      *                      FITS headers.
701      */
702     public synchronized <T> T getValue(Class<T> asType, T defaultValue) throws IllegalArgumentException {
703 
704         if (this.value == null) {
705             return defaultValue;
706         }
707         if (String.class.isAssignableFrom(asType)) {
708             return asType.cast(this.value);
709         }
710         if (this.value.isEmpty()) {
711             return defaultValue;
712         }
713         if (Boolean.class.isAssignableFrom(asType)) {
714             return asType.cast(getBooleanValue((Boolean) defaultValue));
715         }
716         if (ComplexValue.class.isAssignableFrom(asType)) {
717             return asType.cast(new ComplexValue(value)); 
718         }   
719         if (Number.class.isAssignableFrom(asType)) {
720             try {
721                 BigDecimal big = new BigDecimal(value.toUpperCase().replace('D', 'E'));
722                 
723                 if (Byte.class.isAssignableFrom(asType)) {
724                     return asType.cast(big.byteValue());
725                 }
726                 if (Short.class.isAssignableFrom(asType)) {
727                     return asType.cast(big.shortValue());
728                 }
729                 if (Integer.class.isAssignableFrom(asType)) {
730                     return asType.cast(big.intValue());
731                 }
732                 if (Long.class.isAssignableFrom(asType)) {
733                     return asType.cast(big.longValue());
734                 }
735                 if (Float.class.isAssignableFrom(asType)) {
736                     return asType.cast(big.floatValue());
737                 }
738                 if (Double.class.isAssignableFrom(asType)) {
739                     return asType.cast(big.doubleValue());
740                 }
741                 if (BigInteger.class.isAssignableFrom(asType)) {
742                     return asType.cast(big.toBigInteger()); 
743                 }
744                 // All possibilities have been exhausted, it must be a BigDecimal...
745                 return asType.cast(big);
746             } catch (NumberFormatException e) {
747                 // The value is not a decimal number, so return the default value by contract.
748                 return defaultValue;
749             } 
750         }
751 
752         throw new IllegalArgumentException("unsupported class " + asType);
753     }

1120     /**
1121      * Parses a continued long string value and comment for this card, which may occupy one or more
1122      * consecutive 80-character header records.
1123      * 
1124      * @param dis               the input stream from which to parse the value and comment fields of
1125      *                          this card.
1126      * @param next            the parser to use for each 80-character record.
1127      * @throws IOException      if there was an IO error reading the stream.
1128      * @throws TruncatedFileException   if the stream endedc ubnexpectedly in the middle of
1129      *                                  an 80-character record.
1130      */
1131     @SuppressWarnings("deprecation")
1132     private synchronized void parseLongStringCard(HeaderCardCountingArrayDataInput dis, HeaderCardParser next)
1133             throws IOException, TruncatedFileException {
1134 
1135         StringBuilder longValue = new StringBuilder();
1136         StringBuilder longComment = null;       
1137        
1138         
1139         while (next != null) {  
1140             String valuePart = next.getValue();
1141             String untrimmedComment = next.getUntrimmedComment();
1142             
1143             if (valuePart == null) {
1144                 // The card cannot have a null value. If it does it wasn't a string card...
1145                 break;
1146             }
1147             
1148             // The end point of the value
1149             int valueEnd = valuePart.length();
1150             
1151             
1152             // Check if there card continues inot the next record. The value
1153             // must end with '&' and the next card must be a CONTINUE card.
1154             // If so, remove the '&' from the value part, and parse in the next 
1155             // card for the next iteration...
1156             if (!dis.markSupported()) {
1157                 throw new IOException("InputStream does not support mark/reset");
1158             }
1159 
1160             // Peek at the next card.
1161             dis.mark();
1162             
1163             try {
1164                 // Check if we should continue parsing this card...
1165                 next = new HeaderCardParser(readOneHeaderLine(dis));
1166                 if (valuePart.endsWith("&") && CONTINUE.key().equals(next.getKey())) {
1167                     // Remove '& from the value part...
1168                     valueEnd--;
1169                 } else {
1170                     // ok move the input stream one card back.
1171                     dis.reset();
1172                     // Clear the parser also.
1173                     next = null;
1174                 }
1175             } catch (EOFException e) {
1176                 // Nothing left to parse after the current one...
1177                 next = null;
1178             }
1179 
1180             // Append the value part from the record last parsed.
1181             longValue.append(valuePart, 0, valueEnd);
1182             
1183             // Append any comment from the card last parsed.
1184             if (untrimmedComment != null) {
1185                 if (longComment == null) {
1186                     longComment = new StringBuilder(untrimmedComment);
1187                 } else {
1188                     longComment.append(untrimmedComment);
1189                 }
1190             }
1191         }
1192         
1193         this.comment = longComment == null ? null : longComment.toString().trim();
1194         this.value = longValue.toString().trim();
1195         this.type = String.class;
1196     }

1202      * inside. The actual header may add padding (e.g. to ensure the end quote
1203      * does not come before byte 20).

1294     /**
1295      * <p>
1296      * Creates a new FITS header card from a FITS stream representation of it, which is how the key/value and comment
1297      * are represented inside the FITS file, normally as an 80-character wide entry. The parsing of header 'lines'
1298      * conforms to all FITS standards, and some optional conventions, such as HIERARCH keywords (if
1299      * {@link FitsFactory#setUseHierarch(boolean)} is enabled), COMMENT and HISTORY entries, and OGIP 1.0 long CONTINUE
1300      * lines (if {@link FitsFactory#setLongStringsEnabled(boolean)} is enabled).
1301      * </p>
1302      * <p>
1303      * However, the parsing here is permissive beyond the standards and conventions, and will do its best to support a
1304      * wide range of FITS files, which may deviate from the standard in subtle (or no so subtle) ways.
1305      * </p>
1306      * <p>
1307      * Here is a brief summary of the rules that guide the parsing of keywords, values, and comment 'fields' from the
1308      * single header line:
1309      * </p>
1310      * <p>
1311      * <b>A. Keywords</b>
1312      * </p>
1313      * <ul>
1314      * <li>The standard FITS keyword is the first 8 characters of the line, or up to an equal [=] character, whichever
1315      * comes first, with trailing spaces removed, and always converted to upper-case.</li>
1316      * <li>If {@link FitsFactory#setUseHierarch(boolean)} is enabled, structured longer keywords can be composed after a
1317      * <code>HIERARCH</code> base key, followed by space (and/or dot ].]) separated parts, up to an equal sign [=]. The
1318      * library will represent the same components (including <code>HIERARCH</code>) but separated by single dots [.].
1319      * For example, the header line starting with [<code>HIERARCH SMA OBS TARGET =</code>], will be referred as
1320      * [<code>HIERARCH.SMA.OBS.TARGET</code>] withing this library. The keyword parts can be composed of any ASCII
1321      * characters except dot [.], white spaces, or equal [=].</li>
1322      * <li>By default, all parts of the key are converted to upper-case. Case sensitive HIERARCH keywords can be
1323      * retained after enabling
1324      * {@link nom.tam.fits.header.hierarch.IHierarchKeyFormatter#setCaseSensitive(boolean)}.</li>
1325      * </ul>
1326      * <p>
1327      * <b>B. Values</b>
1328      * </p>
1329      * <p>
1330      * Values are the part of the header line, that is between the keyword and an optional ending comment. Legal header
1331      * values follow the following parse patterns:
1332      * <ul>
1333      * <li>Begin with an equal sign [=], or else come after a CONTINUE keyword.</li>
1334      * <li>Next can be a quoted value such as <code>'hello'</code>, placed inside two single quotes. Or an unquoted
1335      * value, such as <code>123</code>.</li>
1336      * <li>Quoted values must begin with a single quote ['] and and with the next single quote. If there is no end-quote
1337      * in the line, it is not considered a string value but rather a comment, unless
1338      * {@link FitsFactory#setAllowHeaderRepairs(boolean)} is enabled, in which case the entire remaining line after the
1339      * opening quote is assumed to be a malformed value.</li>
1340      * <li>Unquoted values end at the fist [/] character, or else go until the line end.</li>
1341      * <li>Quoted values have trailing spaces removed, s.t. [<code>'  value   '</code>] becomes
1342      * [<code>  value</code>].</li>
1343      * <li>Unquoted values are trimmed, with both leading and trailing spaces removed, e.g. [<code>  123  </code>]
1344      * becomes [<code>123</code>].</li>
1345      * </ul>
1346      * <p>
1347      * <b>C. Comments</b>
1348      * </p>
1349      * <p>
1350      * The following rules guide the parsing of the values component:
1351      * <ul>
1352      * <li>If a value is present (see above), the comment is what comes after it. That is, for quoted values, everything
1353      * that follows the closing quote. For unquoted values, it's what comes after the first [/], with the [/] itself
1354      * removed.</li>
1355      * <li>If a value is not present, then everything following the keyword is considered the comment.</li>
1356      * <li>Comments are trimmed, with both leading and trailing spaces removed.</li>
1357      * </ul>
1358      * 
1359      * @return a newly created HeaderCard from a FITS card string.
1360      * 
1361      * @param line the card image (typically 80 characters if in a FITS file).
1362      * 
1363      * @throws IllegalArgumentException     if the card was malformed, truncated, or if there was an IO error.
1364      * 
1365      * @see FitsFactory#setUseHierarch(boolean)
1366      * @see nom.tam.fits.header.hierarch.IHierarchKeyFormatter#setCaseSensitive(boolean)
1367      */
1368     public static HeaderCard create(String line) throws IllegalArgumentException {
1369         try (ArrayDataInput in = stringToArrayInputStream(line)) {
1370             return new HeaderCard(in);
1371         } catch (Exception e) {
1372             throw new IllegalArgumentException("card not legal", e);
1373         }
1374     }

1640      *

1642      * 
1643      * @return a string of exactly 80 characters
1644      *

1650     private static String readOneHeaderLine(HeaderCardCountingArrayDataInput dis) throws IOException, TruncatedFileException {
1651         byte[] buffer = new byte[FITS_HEADER_CARD_SIZE];       
1652        
1653         int got = 0;
1654         
1655         try {
1656             // Read as long as there is more available, even if it comes in a trickle...
1657             while (got < buffer.length) {
1658                 int n = dis.in().read(buffer, got, buffer.length - got);
1659                 if (n <= 0) {
1660                     break;
1661                 }
1662                 got += n;
1663             }
1664         } catch (EOFException e) {
1665             // Just in case read throws EOFException instead of returning -1 by contract.
1666         }
1667         
1668         if (got == 0) {
1669             // Nothing left to read.
1670             throw new EOFException();
1671         }
1672         
1673         if (got < buffer.length) {
1674             // Got an incomplete header card...
1675             throw new TruncatedFileException("Got only " + got + " of " + buffer.length + " bytes expected for a header card"); 
1676         }
1677             
1678         dis.cardRead();
1679         
1680         return AsciiFuncs.asciiString(buffer);

1782     /**
1783      * Checks the specified string for characters that are not allowed in FITS headers, and throws an exception
1784      * if any are found. According to the FITS
1785      * standard, headers may only contain ASCII characters in the range 0x20 and 0x7E (inclusive).
1786      * 
1787      * @param text      the input string
1788      * @throws IllegalArgumentException     if the unput string contains any characters that cannot be
1789      *                                      in a FITS header, that is characters outside of the 0x20 to 0x7E 
1790      *                                      range.
1791      * 
1792      * @since 1.16
1793      * 
1794      * @see #isValidChar(char)
1795      * @see #sanitize(String)
1796      * @see #validateKey(String)
1797      */
1798     public static void validateChars(String text) throws IllegalArgumentException {
1799         if (text == null) {
1800             return;
1801         }
1802         
1803         for (int i = text.length(); --i >= 0;) {
1804             char c = text.charAt(i);
1805             if (c < MIN_VALID_CHAR) {
1806                 throw new IllegalArgumentException(
1807                         "Non-printable character(s), e.g. 0x" + (int) c + ", in [" + sanitize(text) + "].");
1808             }
1809             if (c > MAX_VALID_CHAR) {
1810                 throw new IllegalArgumentException("Extendeed ASCII character(s) in [" + sanitize(text)
1811                         + "]. Only 0x20 through 0x7E are allowed.");
1812             }
1813         }     
1814     }

1816     /**
1817      * Checks if the specified string may be used as a FITS header keyword according to the FITS standard
1818      * and currently settings for supporting extensions to the standard, such as HIERARCH-style keywords.
1819      * 
1820      * @param key           the proposed keyword string
1821      * 
1822      * @throws IllegalArgumentException     
1823      *                      if the string cannot be used as a FITS keyword with the
1824      *                      current settings. The exception will contain an informative
1825      *                      message describing the issue.
1826      * 
1827      * @since 1.16
1828      * 
1829      * @see #validateChars(String)
1830      * @see FitsFactory#setUseHierarch(boolean)
1831      */
1832     public static void validateKey(String key) throws IllegalArgumentException {
1833         int maxLength = MAX_KEYWORD_LENGTH;
1834         if (isHierarchKey(key)) {
1835             if (!FitsFactory.getUseHierarch()) {
1836                 throw new HierarchNotEnabledException(key);
1837             }
1838             
1839             maxLength = MAX_HIERARCH_KEYWORD_LENGTH;
1840             validateHierarchComponents(key);
1841         }
1842 
1843         if (key.length() > maxLength) {
1844             throw new IllegalArgumentException("Keyword is too long: [" + sanitize(key) + "]");
1845         }
1846 
1847         // Check the whole key for non-printable, non-standard ASCII
1848         for (int i = key.length(); --i >= 0;) {
1849             char c = key.charAt(i);
1850             if (c < MIN_VALID_CHAR) {
1851                 throw new IllegalArgumentException(
1852                         "Keyword contains non-printable character 0x" + (int) c + ": [" + sanitize(key) + "].");
1853             }
1854             if (c > MAX_VALID_CHAR) {
1855                 throw new IllegalArgumentException("Keyword contains extendeed ASCII characters: [" + sanitize(key)
1856                         + "]. Only 0x20 through 0x7E are allowed.");
1857             }
1858         }
1859 
1860         // Check if the first 8 characters conform to strict FITS specification...
1861         for (int i = Math.min(MAX_KEYWORD_LENGTH, key.length()); --i >= 0;) {
1862             char c = key.charAt(i);
1863             if (c >= 'a' && c <= 'z') {
1864                 continue;
1865             }
1866             if (c >= 'A' && c <= 'Z') {
1867                 continue;
1868             }
1869             if (c >= '0' && c <= '9') {
1870                 continue;
1871             }
1872             if (c == '-') {
1873                 continue;
1874             }
1875             if (c == '_') {
1876                 continue;
1877             }
1878             throw new IllegalArgumentException("Keyword [" + sanitize(key) + "] contains invalid characters. Only [A-Z][a-z][0-9][-][_] are allowed.");
1879         }
1880     }

1882     /**
1883      * Additional checks the extended components of the HIEARCH key (in bytes 9-77), to make sure they conform to
1884      * our own standards of storing hierarch keys as a dot-separated list of components. That is,
1885      * the keyword must not have any spaces...
1886      * 
1887      * @param key       the HIERARCH keyword to check.
1888      * 
1889      * @throws IllegalArgumentException     if the keyword is not a proper dot-separated set of non-empty 
1890      *                                      hierarchical components
1891      */
1892     private static void validateHierarchComponents(String key) throws IllegalArgumentException {        
1893         for (int i = key.length(); --i >= 0;) {
1894             if (Character.isSpaceChar(key.charAt(i))) {
1895                 throw new IllegalArgumentException("No spaces allowed in HIERARCH keywords used internally: [" + sanitize(key) + "].");
1896             }
1897         }
1898         
1899         if (key.indexOf("..") >= 0) {
1900             throw new IllegalArgumentException("HIERARCH keywords with empty component: [" + sanitize(key) + "].");
1901         }   
1902     }

1905      * Checks that a number value is not NaN or Infinite, since FITS does not have a standard for describing
1906      * those values in the header. If the value is not suitable for the FITS header, an exception is thrown.
1907      * 
1908      * @param value                     The number to check
1909      * @throws NumberFormatException    if the input value is NaN or infinite.
1910      * 
1911      */
1912     private static void checkNumber(Number value) throws NumberFormatException {
1913         if (value instanceof Double) {
1914             if (!Double.isFinite(value.doubleValue())) {
1915                 throw new NumberFormatException("Cannot represent " + value + " in FITS headers.");
1916             }
1917          } else if (value instanceof Float) {
1918              if (!Float.isFinite(value.floatValue())) {
1919                  throw new NumberFormatException("Cannot represent " + value + " in FITS headers.");
1920              }
1921          } 
1922     }
