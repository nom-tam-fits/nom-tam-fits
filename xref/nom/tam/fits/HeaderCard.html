165     /**
166      * Create a string from a double making sure that it's not more than 20
167      * characters long. Probably would be better if we had a way to override
168      * this since we can loose precision for some doubles.
169      */
170     private static String dblString(double input) {
171         String value = String.valueOf(input);
172         if (value.length() > 20) {
173             value = new java.util.Formatter().format("%20.13G", input).out().toString();
174         }
175         return value;
176     }

178     /**
179      * Create a HeaderCard from its component parts
180      * 
181      * @param key
182      *            Keyword (null for a COMMENT)
183      * @param value
184      *            Value
185      * @param comment
186      *            Comment
187      * @param nullable
188      *            Is this a nullable value card?
189      * @exception HeaderCardException
190      *                for any invalid keyword or value
191      */
192     public HeaderCard(String key, String value, String comment, boolean nullable) throws HeaderCardException {
193         if (comment != null && comment.startsWith("ntf::")) {
194             String ckey = comment.substring(5); // Get rid of ntf:: prefix
195             comment = HeaderCommentsMap.getComment(ckey);
196         }
197         if (key == null && value != null) {
198             throw new HeaderCardException("Null keyword with non-null value");
199         }
200 
201         if (key != null && key.length() > MAX_KEYWORD_LENGTH) {
202             if (!FitsFactory.getUseHierarch() || !key.substring(0, 9).equals("HIERARCH.")) {
203                 throw new HeaderCardException("Keyword too long");
204             }
205         }
206 
207         if (value != null) {
208             value = value.replaceAll(" *$", "");
209 
210             if (value.length() > MAX_VALUE_LENGTH) {
211                 throw new HeaderCardException("Value too long");
212             }
213 
214             if (value.startsWith("'")) {
215                 if (value.charAt(value.length() - 1) != '\'') {
216                     throw new HeaderCardException("Missing end quote in string value");
217                 }
218 
219                 value = value.substring(1, value.length() - 1).trim();
220 
221             }
222         }
223 
224         this.key = key;
225         this.value = value;
226         this.comment = comment;
227         this.nullable = nullable;
228         isString = true;
229     }

231     /**
232      * Create a HeaderCard from a FITS card image
233      * 
234      * @param card
235      *            the 80 character card image
236      */
237     public HeaderCard(String card) {
238         key = null;
239         value = null;
240         comment = null;
241         isString = false;
242 
243         if (card.length() > 80) {
244             card = card.substring(0, 80);
245         }
246 
247         if (FitsFactory.getUseHierarch() && card.length() > 9 && card.substring(0, 9).equals("HIERARCH ")) {
248             hierarchCard(card);
249             return;
250         }
251 
252         // We are going to assume that the value has no blanks in
253         // it unless it is enclosed in quotes. Also, we assume that
254         // a / terminates the string (except inside quotes)
255 
256         // treat short lines as special keywords
257         if (card.length() < 9) {
258             key = card;
259             return;
260         }
261 
262         // extract the key
263         key = card.substring(0, 8).trim();
264 
265         // if it is an empty key, assume the remainder of the card is a comment
266         if (key.length() == 0) {
267             key = "";
268             comment = card.substring(8);
269             return;
270         }
271 
272         // Non-key/value pair lines are treated as keyed comments
273         if (key.equals("COMMENT") || key.equals("HISTORY") || !card.substring(8, 10).equals("= ")) {
274             comment = card.substring(8).trim();
275             return;
276         }
277 
278         // extract the value/comment part of the string
279         String valueAndComment = card.substring(10).trim();
280 
281         // If there is no value/comment part, we are done.
282         if (valueAndComment.length() == 0) {
283             value = "";
284             return;
285         }
286 
287         int vend = -1;
288         // If we have a ' then find the matching '.
289         if (valueAndComment.charAt(0) == '\'') {
290 
291             int offset = 1;
292             while (offset < valueAndComment.length()) {
293 
294                 // look for next single-quote character
295                 vend = valueAndComment.indexOf("'", offset);
296 
297                 // if the quote character is the last character on the line...
298                 if (vend == valueAndComment.length() - 1) {
299                     break;
300                 }
301 
302                 // if we did not find a matching single-quote...
303                 if (vend == -1) {
304                     // pretend this is a comment card
305                     key = null;
306                     comment = card;
307                     return;
308                 }
309 
310                 // if this is not an escaped single-quote, we are done
311                 if (valueAndComment.charAt(vend + 1) != '\'') {
312                     break;
313                 }
314 
315                 // skip past escaped single-quote
316                 offset = vend + 2;
317             }
318 
319             // break apart character string
320             value = valueAndComment.substring(1, vend).trim();
321             value = value.replace("''", "'");
322 
323             if (vend + 1 >= valueAndComment.length()) {
324                 comment = null;
325             } else {
326 
327                 comment = valueAndComment.substring(vend + 1).trim();
328                 if (comment.charAt(0) == '/') {
329                     if (comment.length() > 1) {
330                         comment = comment.substring(1);
331                     } else {
332                         comment = "";
333                     }
334                 }
335 
336                 if (comment.length() == 0) {
337                     comment = null;
338                 }
339 
340             }
341             isString = true;
342 
343         } else {
344 
345             // look for a / to terminate the field.
346             int slashLoc = valueAndComment.indexOf('/');
347             if (slashLoc != -1) {
348                 comment = valueAndComment.substring(slashLoc + 1).trim();
349                 value = valueAndComment.substring(0, slashLoc).trim();
350             } else {
351                 value = valueAndComment;
352             }
353         }
354     }

356     /**
357      * Process HIERARCH style cards... HIERARCH LEV1 LEV2 ... = value / comment
358      * The keyword for the card will be "HIERARCH.LEV1.LEV2..." A '/' is assumed
359      * to start a comment.
360      */
361     private void hierarchCard(String card) {
362 
363         String name = "";
364         String token = null;
365         String separator = "";
366         int[] tokLimits;
367         int posit = 0;
368         int commStart = -1;
369 
370         // First get the hierarchy levels
371         while ((tokLimits = getToken(card, posit)) != null) {
372             token = card.substring(tokLimits[0], tokLimits[1]);
373             if (!token.equals("=")) {
374                 name += separator + token;
375                 separator = ".";
376             } else {
377                 tokLimits = getToken(card, tokLimits[1]);
378                 if (tokLimits != null) {
379                     token = card.substring(tokLimits[0], tokLimits[1]);
380                 } else {
381                     key = name;
382                     value = null;
383                     comment = null;
384                     return;
385                 }
386                 break;
387             }
388             posit = tokLimits[1];
389         }
390         key = name;
391 
392         // At the end?
393         if (tokLimits == null) {
394             value = null;
395             comment = null;
396             isString = false;
397             return;
398         }
399 
400         // Really should consolidate the two instances
401         // of this test in this class!
402         if (token.charAt(0) == '\'') {
403             // Find the next undoubled quote...
404             isString = true;
405             if (token.length() > 1 && token.charAt(1) == '\'' && (token.length() == 2 || token.charAt(2) != '\'')) {
406                 value = "";
407                 commStart = tokLimits[0] + 2;
408             } else if (card.length() < tokLimits[0] + 2) {
409                 value = null;
410                 comment = null;
411                 isString = false;
412                 return;
413             } else {
414                 int i;
415                 for (i = tokLimits[0] + 1; i < card.length(); i += 1) {
416                     if (card.charAt(i) == '\'') {
417                         if (i == card.length() - 1) {
418                             value = card.substring(tokLimits[0] + 1, i);
419                             commStart = i + 1;
420                             break;
421                         } else if (card.charAt(i + 1) == '\'') {
422                             // Doubled quotes.
423                             i += 1;
424                             continue;
425                         } else {
426                             value = card.substring(tokLimits[0] + 1, i);
427                             commStart = i + 1;
428                             break;
429                         }
430                     }
431                 }
432             }
433             if (commStart < 0) {
434                 value = null;
435                 comment = null;
436                 isString = false;
437                 return;
438             }
439             for (int i = commStart; i < card.length(); i += 1) {
440                 if (card.charAt(i) == '/') {
441                     comment = card.substring(i + 1).trim();
442                     break;
443                 } else if (card.charAt(i) != ' ') {
444                     comment = null;
445                     break;
446                 }
447             }
448         } else {
449             isString = false;
450             int sl = token.indexOf('/');
451             if (sl == 0) {
452                 value = null;
453                 comment = card.substring(tokLimits[0] + 1);
454             } else if (sl > 0) {
455                 value = token.substring(0, sl);
456                 comment = card.substring(tokLimits[0] + sl + 1);
457             } else {
458                 value = token;
459 
460                 for (int i = tokLimits[1]; i < card.length(); i += 1) {
461                     if (card.charAt(i) == '/') {
462                         comment = card.substring(i + 1).trim();
463                         break;
464                     } else if (card.charAt(i) != ' ') {
465                         comment = null;
466                         break;
467                     }
468                 }
469             }
470         }
471     }

473     /**
474      * Get the next token. Can't use StringTokenizer since we sometimes need to
475      * know the position within the string.
476      */
477     private int[] getToken(String card, int posit) {
478 
479         int i;
480         for (i = posit; i < card.length(); i += 1) {
481             if (card.charAt(i) != ' ') {
482                 break;
483             }
484         }
485 
486         if (i >= card.length()) {
487             return null;
488         }
489 
490         if (card.charAt(i) == '=') {
491             return new int[]{
492                 i,
493                 i + 1
494             };
495         }
496 
497         int j;
498         for (j = i + 1; j < card.length(); j += 1) {
499             if (card.charAt(j) == ' ' || card.charAt(j) == '=') {
500                 break;
501             }
502         }
503         return new int[]{
504             i,
505             j
506         };
507     }

558     /**
559      * Return the 80 character card image
560      */
561     @Override
562     public String toString() {
563         StringBuffer buf = new StringBuffer(80);
564 
565         // start with the keyword, if there is one
566         if (key != null) {
567             if (key.length() > 9 && key.substring(0, 9).equals("HIERARCH.")) {
568                 return hierarchToString();
569             }
570             buf.append(key);
571             if (key.length() < 8) {
572                 buf.append(space80.substring(0, 8 - buf.length()));
573             }
574         }
575 
576         if (value != null || nullable) {
577             buf.append("= ");
578 
579             if (value != null) {
580 
581                 if (isString) {
582                     // left justify the string inside the quotes
583                     buf.append('\'');
584                     buf.append(value.replace("'", "''"));
585                     if (buf.length() < 19) {
586 
587                         buf.append(space80.substring(0, 19 - buf.length()));
588                     }
589                     buf.append('\'');
590                     // Now add space to the comment area starting at column 40
591                     if (buf.length() < 30) {
592                         buf.append(space80.substring(0, 30 - buf.length()));
593                     }
594 
595                 } else {
596 
597                     buf.length();
598                     if (value.length() < 20) {
599                         buf.append(space80.substring(0, 20 - value.length()));
600                     }
601 
602                     buf.append(value);
603 
604                 }
605             } else {
606                 // Pad out a null value.
607                 buf.append(space80.substring(0, 20));
608             }
609 
610             // if there is a comment, add a comment delimiter
611             if (comment != null) {
612                 buf.append(" / ");
613             }
614 
615         } else if (comment != null && comment.startsWith("= ")) {
616             buf.append("  ");
617         }
618 
619         // finally, add any comment
620         if (comment != null) {
621             buf.append(comment);
622         }
623 
624         // make sure the final string is exactly 80 characters long
625         if (buf.length() > 80) {
626             buf.setLength(80);
627 
628         } else {
629 
630             if (buf.length() < 80) {
631                 buf.append(space80.substring(0, 80 - buf.length()));
632             }
633         }
634 
635         return buf.toString();
636     }

638     private String hierarchToString() {
639 
640         StringBuffer b = new StringBuffer(80);
641         int p = 0;
642         String space = "";
643         while (p < key.length()) {
644             int q = key.indexOf('.', p);
645             if (q < 0) {
646                 b.append(space + key.substring(p));
647                 break;
648             } else {
649                 b.append(space + key.substring(p, q));
650             }
651             space = " ";
652             p = q + 1;
653         }
654 
655         if (value != null || nullable) {
656             b.append("= ");
657 
658             if (value != null) {
659                 // Try to align values
660                 int avail = 80 - (b.length() + value.length());
661 
662                 if (isString) {
663                     avail -= 2;
664                 }
665                 if (comment != null) {
666                     avail -= 3 + comment.length();
667                 }
668 
669                 if (avail > 0 && b.length() < 29) {
670                     b.append(space80.substring(0, Math.min(avail, 29 - b.length())));
671                 }
672 
673                 if (isString) {
674                     b.append('\'');
675                 } else if (avail > 0 && value.length() < 10) {
676                     b.append(space80.substring(0, Math.min(avail, 10 - value.length())));
677                 }
678                 b.append(value);
679                 if (isString) {
680                     b.append('\'');
681                 }
682             } else if (b.length() < 30) {
683 
684                 // Pad out a null value
685                 b.append(space80.substring(0, 30 - b.length()));
686             }
687         }
688 
689         if (comment != null) {
690             b.append(" / " + comment);
691         }
692         if (b.length() < 80) {
693             b.append(space80.substring(0, 80 - b.length()));
694         }
695         String card = new String(b);
696         if (card.length() > 80) {
697             card = card.substring(0, 80);
698         }
699         return card;
700     }
