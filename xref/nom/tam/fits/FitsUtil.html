92      /**
93       * @return Convert an array of booleans to bytes.
94       * @param bool
95       *            array of booleans
96       */
97      static byte[] booleanToByte(boolean[] bool) {
98  
99          byte[] byt = new byte[bool.length];
100         for (int i = 0; i < bool.length; i += 1) {
101             byt[i] = bool[i] ? (byte) 'T' : (byte) 'F';
102         }
103         return byt;
104     }

106     /**
107      * @return Convert bytes to Strings.
108      * @param bytes
109      *            byte array to convert
110      * @param maxLen
111      *            the max string length
112      */
113     public static String[] byteArrayToStrings(byte[] bytes, int maxLen) {
114         boolean checking = FitsFactory.getCheckAsciiStrings();
115 
116         // Note that if a String in a binary table contains an internal 0,
117         // the FITS standard says that it is to be considered as terminating
118         // the string at that point, so that software reading the
119         // data back may not include subsequent characters.
120         // No warning of this truncation is given.
121 
122         String[] res = new String[bytes.length / maxLen];
123         for (int i = 0; i < res.length; i += 1) {
124 
125             int start = i * maxLen;
126 
127             // AK: The FITS standard states that a 0 byte terminates the
128             // string before the fixed length, and characters beyond it
129             // are undefined. So, we first check where the string ends.
130             int l = 0;
131             for (; l < maxLen; l++) {
132                 if (bytes[l] == 0) {
133                     break;
134                 }
135             }
136             int end = start + l;
137 
138             // Pre-trim the string to avoid keeping memory
139             // hanging around. (Suggested by J.C. Segovia, ESA).
140 
141             // Note that the FITS standard does not mandate
142             // that we should be trimming the string at all, but
143             // this seems to best meet the desires of the community.
144             for (; start < end; start += 1) {
145                 if (bytes[start] != BYTE_REPRESENTING_BLANK) {
146                     break; // Skip only spaces.
147                 }
148             }
149 
150             for (; end > start; end -= 1) {
151                 if (bytes[end - 1] != BYTE_REPRESENTING_BLANK) {
152                     break;
153                 }
154             }
155 
156             // For FITS binary tables, 0 values are supposed
157             // to terminate strings, a la C. [They shouldn't appear in
158             // any other context.]
159             // Other non-printing ASCII characters
160             // should always be an error which we can check for
161             // if the user requests.
162 
163             // The lack of handling of null bytes was noted by Laurent Bourges.
164             boolean errFound = false;
165             for (int j = start; j < end; j += 1) {
166 
167                 if (bytes[j] == 0) {
168                     end = j;
169                     break;
170                 }
171                 if (checking && (bytes[j] < BYTE_REPRESENTING_BLANK || bytes[j] > BYTE_REPRESENTING_MAX_ASCII_VALUE)) {
172                     errFound = true;
173                     bytes[j] = BYTE_REPRESENTING_BLANK;
174                 }
175             }
176             res[i] = AsciiFuncs.asciiString(bytes, start, end - start);
177             if (errFound && !FitsUtil.wroteCheckingError) {
178                 LOG.log(Level.SEVERE, "Warning: Invalid ASCII character[s] detected in string: " + res[i]
179                         + " Converted to space[s].  Any subsequent invalid characters will be converted silently");
180                 FitsUtil.wroteCheckingError = true;
181             }
182         }
183         return res;
184     }

186     /**
187      * @return Convert an array of bytes to booleans.
188      * @param bytes
189      *            the array of bytes to get the booleans from.
190      */
191     static boolean[] byteToBoolean(byte[] bytes) {
192         boolean[] bool = new boolean[bytes.length];
193 
194         for (int i = 0; i < bytes.length; i += 1) {
195             bool[i] = (bytes[i] == 'T');
196         }
197         return bool;
198     }

200     /**
201      * @return Find out where we are in a random access file .
202      * @param o
203      *            the stream to get the position

205     public static long findOffset(Closeable o) {
206         if (o instanceof RandomAccess) {
207             return ((RandomAccess) o).getFilePointer();
208         }
209         return -1;

212     /**
213      * @return Get a stream to a URL accommodating possible redirections. Note
214      *         that if a redirection request points to a different protocol than
215      *         the original request, then the redirection is not handled
216      *         automatically.
217      * @param url
218      *            the url to get the stream from
219      * @param level
220      *            max levels of redirection
221      * @throws IOException
222      *             if the operation failed
223      */
224     public static InputStream getURLStream(URL url, int level) throws IOException {
225         URLConnection conn = null;
226         int code = -1;
227         try {
228             conn = url.openConnection();
229             if (conn instanceof HttpURLConnection) {
230                 code = ((HttpURLConnection) conn).getResponseCode();
231             }
232             return conn.getInputStream();
233         } catch (ProtocolException e) {
234             LOG.log(Level.WARNING, "could not connect to " + url + (code >= 0 ? " got responce-code" + code : ""), e);
235             throw e;
236         }
237     }

239     /**
240      * @return Get the maximum length of a String in a String array.
241      * @param strings
242      *            array of strings to check
243      * @throws FitsException
244      *             if the operation failed
245      */
246     public static int maxLength(String[] strings) throws FitsException {
247 
248         int max = 0;
249         for (String element : strings) {
250             if (element != null && element.length() > max) {
251                 max = element.length();
252             }
253         }
254         return max;
255     }

268         pad(stream, size, (byte) 0);
269     }

271     /**
272      * Add padding to an output stream.
273      * 
274      * @param stream
275      *            stream to pad
276      * @param size
277      *            the current size
278      * @param fill
279      *            the fill byte to use
280      * @throws FitsException
281      *             if the operation failed
282      */
283     public static void pad(ArrayDataOutput stream, long size, byte fill) throws FitsException {
284         int len = padding(size);
285         if (len > 0) {
286             byte[] buf = new byte[len];
287             Arrays.fill(buf, fill);
288             try {
289                 stream.write(buf);
290                 stream.flush();
291             } catch (Exception e) {
292                 throw new FitsException("Unable to write padding", e);
293             }
294         }
295     }

306     public static int padding(long size) {
307 
308         int mod = (int) (size % FitsFactory.FITS_BLOCK_SIZE);
309         if (mod > 0) {
310             mod = FitsFactory.FITS_BLOCK_SIZE - mod;
311         }
312         return mod;
313     }

315     /**
316      * Reposition a random access stream to a requested offset.
317      * 
318      * @param o
319      *            the closable to reposition
320      * @param offset
321      *            the offset to position it to.
322      * @throws FitsException
323      *             if the operation was failed or not possible
324      */
325     public static void reposition(Closeable o, long offset) throws FitsException {
326         if (o == null) {
327             throw new FitsException("Attempt to reposition null stream");
328         } else if (!(o instanceof RandomAccess) || offset < 0) {
329             throw new FitsException("Invalid attempt to reposition stream " + o + " of type " + o.getClass().getName() + " to " + offset);
330         }
331         try {
332             ((RandomAccess) o).seek(offset);
333         } catch (IOException e) {
334             throw new FitsException("Unable to repostion stream " + o + " of type " + o.getClass().getName() + " to " + offset + ": " + e.getMessage(), e);
335         }
336     }

338     /**
339      * Convert an array of Strings to bytes.
340      * 
341      * @return the resulting bytes
342      * @param stringArray
343      *            the array with Strings
344      * @param maxLen
345      *            the max length (in bytes) of every String
346      */
347     public static byte[] stringsToByteArray(String[] stringArray, int maxLen) {
348         byte[] res = new byte[stringArray.length * maxLen];
349         for (int i = 0; i < stringArray.length; i += 1) {
350             byte[] bstr;
351             if (stringArray[i] == null) {
352                 bstr = new byte[0];
353             } else {
354                 bstr = AsciiFuncs.getBytes(stringArray[i]);
355             }
356             int cnt = bstr.length;
357             if (cnt > maxLen) {
358                 cnt = maxLen;
359             }
360             System.arraycopy(bstr, 0, res, i * maxLen, cnt);
361             for (int j = cnt; j < maxLen; j += 1) {
362                 res[i * maxLen + j] = (byte) ' ';
363             }
364         }
365         return res;
366     }
