100     /**
101      * Converts a boolean array to its FITS representation as an array of bytes.
102      * 
103      * @return      Convert an array of booleans to bytes.
104      *
105      * @param  bool array of booleans
106      */
107     static byte[] booleanToByte(boolean[] bool) {
108 
109         byte[] byt = new byte[bool.length];
110         for (int i = 0; i < bool.length; i++) {
111             byt[i] = FitsEncoder.byteForBoolean(bool[i]);
112         }
113         return byt;
114     }

116     /**
117      * Converts a FITS byte sequence to a Java string array
118      * 
119      * @return        Convert bytes to Strings.
120      *
121      * @param  bytes  byte array to convert
122      * @param  maxLen the max string length
123      */
124     public static String[] byteArrayToStrings(byte[] bytes, int maxLen) {
125         boolean checking = FitsFactory.getCheckAsciiStrings();
126 
127         // Note that if a String in a binary table contains an internal 0,
128         // the FITS standard says that it is to be considered as terminating
129         // the string at that point, so that software reading the
130         // data back may not include subsequent characters.
131         // No warning of this truncation is given.
132 
133         String[] res = new String[bytes.length / maxLen];
134         for (int i = 0; i < res.length; i++) {
135 
136             int start = i * maxLen;
137 
138             // AK: The FITS standard states that a 0 byte terminates the
139             // string before the fixed length, and characters beyond it
140             // are undefined. So, we first check where the string ends.
141             int l = 0;
142             for (; l < maxLen; l++) {
143                 if (bytes[l] == 0) {
144                     break;
145                 }
146             }
147             int end = start + l;
148 
149             // Pre-trim the string to avoid keeping memory
150             // hanging around. (Suggested by J.C. Segovia, ESA).
151 
152             // Note that the FITS standard does not mandate
153             // that we should be trimming the string at all, but
154             // this seems to best meet the desires of the community.
155             for (; start < end; start++) {
156                 if (bytes[start] != BYTE_REPRESENTING_BLANK) {
157                     break; // Skip only spaces.
158                 }
159             }
160 
161             for (; end > start; end--) {
162                 if (bytes[end - 1] != BYTE_REPRESENTING_BLANK) {
163                     break;
164                 }
165             }
166 
167             // For FITS binary tables, 0 values are supposed
168             // to terminate strings, a la C. [They shouldn't appear in
169             // any other context.]
170             // Other non-printing ASCII characters
171             // should always be an error which we can check for
172             // if the user requests.
173 
174             // The lack of handling of null bytes was noted by Laurent Bourges.
175             boolean errFound = false;
176             for (int j = start; j < end; j++) {
177 
178                 if (bytes[j] == 0) {
179                     end = j;
180                     break;
181                 }
182                 if (checking && (bytes[j] < BYTE_REPRESENTING_BLANK || bytes[j] > BYTE_REPRESENTING_MAX_ASCII_VALUE)) {
183                     errFound = true;
184                     bytes[j] = BYTE_REPRESENTING_BLANK;
185                 }
186             }
187             res[i] = AsciiFuncs.asciiString(bytes, start, end - start);
188             if (errFound && !FitsUtil.wroteCheckingError) {
189                 LOG.log(Level.SEVERE, "Warning: Invalid ASCII character[s] detected in string: " + res[i]
190                         + " Converted to space[s].  Any subsequent invalid characters will be converted silently");
191                 FitsUtil.wroteCheckingError = true;
192             }
193         }
194         return res;
195     }

197     /**
198      * Converts a FITS representation of boolean values as bytes to a java boolean array. This implementation does not
199      * handle FITS <code>null</code> values.
200      * 
201      * @return       Convert an array of bytes to booleans.
202      *
203      * @param  bytes the array of bytes to get the booleans from.
204      * 
205      * @see          FitsDecoder#booleanFor(int)
206      */
207     static boolean[] byteToBoolean(byte[] bytes) {
208         boolean[] bool = new boolean[bytes.length];
209 
210         for (int i = 0; i < bytes.length; i++) {
211             bool[i] = FitsDecoder.booleanFor(bytes[i]);
212         }
213         return bool;
214     }

218      * 
219      * @return   The offset from the beginning of file (if random accessible), or -1 otherwise.

226         }

230     /**
231      * Gets and input stream for a given URL resource.
232      * 
233      * @return             Get a stream to a URL accommodating possible redirections. Note that if a redirection request
234      *                         points to a different protocol than the original request, then the redirection is not
235      *                         handled automatically.
236      *
237      * @param  url         the url to get the stream from
238      * @param  level       max levels of redirection
239      *
240      * @throws IOException if the operation failed
241      */
242     public static InputStream getURLStream(URL url, int level) throws IOException {
243         URLConnection conn = null;
244         int code = -1;
245         try {
246             conn = url.openConnection();
247             if (conn instanceof HttpURLConnection) {
248                 code = ((HttpURLConnection) conn).getResponseCode();
249             }
250             return conn.getInputStream();
251         } catch (ProtocolException e) {
252             LOG.log(Level.WARNING, "could not connect to " + url + (code >= 0 ? " got responce-code" + code : ""), e);
253             throw e;
254         }
255     }

257     /**
258      * Returns the maximum String length in an array of Strings.
259      * 
260      * @return               Get the maximum length of a String in a String array.
261      *
262      * @param  strings       array of strings to check
263      *
264      * @throws FitsException if the operation failed
265      */
266     public static int maxLength(String[] strings) throws FitsException {
267 
268         int max = 0;
269         for (String element : strings) {
270             if (element != null && element.length() > max) {
271                 max = element.length();
272             }
273         }
274         return max;
275     }

278      * Adds the necessary amount of padding needed to complete the last FITS block.
279      *
280      * @param  stream        stream to pad
281      * @param  size          the current size of the stream (total number of bytes written to it since the beginning of
282      *                           the FITS).
283      *
284      * @throws FitsException if the operation failed
285      * 
286      * @see                  #pad(ArrayDataOutput, long, byte)
287      */
288     public static void pad(ArrayDataOutput stream, long size) throws FitsException {
289         pad(stream, size, (byte) 0);
290     }

292     /**
293      * Adds the necessary amount of padding needed to complete the last FITS block., usign the designated padding byte
294      * value.
295      *
296      * @param  stream        stream to pad
297      * @param  size          the current size of the stream (total number of bytes written to it since the beginning of
298      *                           the FITS).
299      * @param  fill          the byte value to use for the padding
300      *
301      * @throws FitsException if the operation failed
302      * 
303      * @see                  #pad(ArrayDataOutput, long)
304      */
305     public static void pad(ArrayDataOutput stream, long size, byte fill) throws FitsException {
306         int len = padding(size);
307         if (len > 0) {
308             byte[] buf = new byte[len];
309             Arrays.fill(buf, fill);
310             try {
311                 stream.write(buf);
312                 stream.flush();
313             } catch (Exception e) {
314                 throw new FitsException("Unable to write padding", e);
315             }
316         }
317     }

330     /**
331      * Calculated the amount of padding we need to add given the current size of a FITS file (under construction)
332      * 
333      * @param  size the current size of our FITS file before the padding
334      * 
335      * @return      the number of bytes of padding we need to add at the end to complete the FITS block.
336      * 
337      * @see         #addPadding(long)
338      * @see         #pad(ArrayDataOutput, long)
339      */
340     public static int padding(long size) {
341 
342         int mod = (int) (size % FitsFactory.FITS_BLOCK_SIZE);
343         if (mod > 0) {
344             mod = FitsFactory.FITS_BLOCK_SIZE - mod;
345         }
346         return mod;
347     }

349     /**
350      * Attempts to reposition a FITS input ot output. The call will succeed only if the underlying input or output is
351      * random accessible. Othewise, an exception will be thrown.
352      *
353      * @deprecated               This method wraps an {@link IOException} into a {@link FitsException} for no good
354      *                               reason really. A revision of the API could reduce the visibility of this method,
355      *                               and/or procees the underlying exception instead.
356      *
357      * @param      o             the FITS input or output
358      * @param      offset        the offset to position it to.
359      *
360      * @throws     FitsException if the underlying input/output is not random accessible or if the requested position is
361      *                               invalid.
362      */
363     @Deprecated
364     public static void reposition(FitsIO o, long offset) throws FitsException {
365         // TODO AK: argument should be RandomAccess instead of Closeable, since
366         // that's the only type we actually handle...
367 
368         if (o == null) {
369             throw new FitsException("Attempt to reposition null stream");
370         }
371 
372         if (!(o instanceof RandomAccess) || offset < 0) {
373             throw new FitsException(
374                     "Invalid attempt to reposition stream " + o + " of type " + o.getClass().getName() + " to " + offset);
375         }
376 
377         try {
378             ((RandomAccess) o).seek(offset);
379         } catch (IOException e) {
380             throw new FitsException("Unable to repostion stream " + o + " of type " + o.getClass().getName() + " to "
381                     + offset + ": " + e.getMessage(), e);
382         }
383     }

385     /**
386      * Convert an array of Strings to bytes.
387      *
388      * @return             the resulting bytes
389      *
390      * @param  stringArray the array with Strings
391      * @param  maxLen      the max length (in bytes) of every String
392      */
393     public static byte[] stringsToByteArray(String[] stringArray, int maxLen) {
394         byte[] res = new byte[stringArray.length * maxLen];
395         for (int i = 0; i < stringArray.length; i++) {
396             byte[] bstr;
397             if (stringArray[i] == null) {
398                 bstr = new byte[0];
399             } else {
400                 bstr = AsciiFuncs.getBytes(stringArray[i]);
401             }
402             int cnt = bstr.length;
403             if (cnt > maxLen) {
404                 cnt = maxLen;
405             }
406             System.arraycopy(bstr, 0, res, i * maxLen, cnt);
407             for (int j = cnt; j < maxLen; j++) {
408                 res[i * maxLen + j] = (byte) ' ';
409             }
410         }
411         return res;
412     }
