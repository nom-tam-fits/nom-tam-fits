34  import nom.tam.fits.BasicHDU;
35  import nom.tam.fits.Fits;
36  import nom.tam.util.FitsFile;
37  
38  public final class FitsCopy {
39  
40      private FitsCopy() {
41      }
42  
43      public static void main(String[] args) throws Exception {
44          String file = args[0];
45          try (Fits f = new Fits(file)) {
46              int i = 0;
47              BasicHDU<?> h;
48  
49              do {
50                  h = f.readHDU();
51                  if (h != null) {
52                      if (i == 0) {
53                          System.out.println("\n\nPrimary header:\n");
54                      } else {
55                          System.out.println("\n\nExtension " + i + ":\n");
56                      }
57                      i += 1;
58                      h.info(System.out);
59                  }
60              } while (h != null);
61              f.close();
62              try (FitsFile bf = new FitsFile(args[1], "rw")) {
63                  f.write(bf);
64                  bf.close();
65              }
66          }
67      }
68  }
