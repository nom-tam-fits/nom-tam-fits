32  import nom.tam.fits.*;
33  import nom.tam.util.*;
34  
35  public class FitsCopy {
36  
37      public static void main(String[] args) throws Exception {
38  
39          String file = args[0];
40  
41          Fits f = new Fits(file);
42          int i = 0;
43          BasicHDU h;
44  
45          do {
46              h = f.readHDU();
47              if (h != null) {
48                  if (i == 0) {
49                      System.out.println("\n\nPrimary header:\n");
50                  } else {
51                      System.out.println("\n\nExtension " + i + ":\n");
52                  }
53                  i += 1;
54                  h.info();
55              }
56          } while (h != null);
57          BufferedFile bf = new BufferedFile(args[1], "rw");
58          f.write(bf);
59          bf.close();
60  
61      }
62  }
