36  /**
37   * This class handles the writing of a card line. It keeps track of the position
38   * in the line and will limit it to 80 characters. A write will never cross the
39   * line border but a write when the line is at position 80 will start a new
40   * line.
41   * 
42   * @author Richard van Nieuwenhoven
43   * 
44   * @deprecated Was needed by {@link FitsLineAppender} only, which itself is deprecated in
45   *              favor of the package-level <code>nom.tam.fits.HeaderCardFormatter</code>.
46   */
47  @Deprecated
48  public class FitsLineAppender {
49  
50      /**
51       * A String of 80 spaces to fill up fits card space.
52       */
53      private static final String FULL_CARD_AS_SPACES = String.format("%80s", "");
54  
55      /**
56       * the underlying StringBuilder to which the writing of fits lines happens.
57       */
58      private final StringBuilder buffer;
59  
60      /**
61       * the char current position in the line.
62       */
63      private int charCount;
64  
65      /**
66       * create a new FitsLineAppender that will have space allocated for one
67       * line.
68       */
69      public FitsLineAppender() {
70          this.buffer = new StringBuilder(HeaderCard.FITS_HEADER_CARD_SIZE);
71      }
72  
73      /**
74       * append a character to the fits line.
75       * 
76       * @param character
77       *            the character to append to the line.
78       */
79      public void append(char character) {
80          this.buffer.append(character);
81          this.charCount++;
82      }
83  
84      /**
85       * Append a sub-string to this line.
86       * 
87       * @param stringValue
88       *            the sub string to append.
89       */
90      public void append(FitsSubString stringValue) {
91          stringValue.appendTo(this.buffer);
92          this.charCount += stringValue.length();
93      }
94  
95      /**
96       * append a string to the fits line, but limit the append to the line
97       * length. rest of the string will be silently truncated.
98       * 
99       * @param string
100      *            the string to append
101      */
102     public void append(String string) {
103         this.charCount = this.charCount % HeaderCard.FITS_HEADER_CARD_SIZE;
104         int newLength = this.charCount + string.length();
105         if (newLength > HeaderCard.FITS_HEADER_CARD_SIZE) {
106             this.buffer.append(string, 0, HeaderCard.FITS_HEADER_CARD_SIZE - this.charCount);
107             this.charCount = 0;
108         } else {
109             this.charCount = newLength;
110             this.buffer.append(string);
111         }
112     }
113 
114     /**
115      * append a string to the buffer, replacing all occurrences of a character
116      * with an other.
117      * 
118      * @param key
119      *            the string to write
120      * @param toReplace
121      *            the character to replace
122      * @param with
123      *            the character to replace the toReplace character with.
124      */
125     public void appendReplacing(String key, char toReplace, char with) {
126         int size = key.length();
127         for (int index = 0; index < size; index++) {
128             char character = key.charAt(index);
129             if (character == toReplace) {
130                 this.buffer.append(with);
131             } else {
132                 this.buffer.append(character);
133             }
134         }
135         this.charCount += size;
136     }
137 
138     /**
139      * append a number of spaces to the line, limited to the line length! This
140      * will only be done if the line is already started, so attention when a
141      * line is still empty this method will have no effect on empty lines.
142      * 
143      * @param count
144      *            the number of spaces to write.
145      */
146     public void appendSpacesTo(int count) {
147         this.charCount = this.charCount % HeaderCard.FITS_HEADER_CARD_SIZE;
148         if (this.charCount != 0) {
149             int spaces = count - this.charCount;
150             if (spaces > 0) {
151                 this.buffer.append(FitsLineAppender.FULL_CARD_AS_SPACES, 0, spaces);
152                 this.charCount += spaces;
153             }
154         }
155     }
156 
157     /**
158      * fill the rest of current line with spaces and start a new fits line.
159      */
160     public void completeLine() {
161         int count = HeaderCard.FITS_HEADER_CARD_SIZE - this.charCount % HeaderCard.FITS_HEADER_CARD_SIZE;
162         if (count < HeaderCard.FITS_HEADER_CARD_SIZE) {
163             this.buffer.append(FitsLineAppender.FULL_CARD_AS_SPACES, 0, count);
164         }
165         // line completed start with 0;
166         this.charCount = 0;
167     }
168 
169     /**
170      * @return the character position in the current line.
171      */
172     public int length() {
173         this.charCount = this.charCount % HeaderCard.FITS_HEADER_CARD_SIZE;
174         return this.charCount;
175     }
176 
177     /**
178      * @return the number of characters still available in the current fits
179      *         line.
180      */
181     public int spaceLeftInLine() {
182         this.charCount = this.charCount % HeaderCard.FITS_HEADER_CARD_SIZE;
183         return HeaderCard.FITS_HEADER_CARD_SIZE - this.charCount;
184     }
185 
186     @Override
187     public String toString() {
188         return this.buffer.toString();
189     }
190 
191     public void append(String key, int start, int end) {
192         this.buffer.append(key, start, end);
193         this.charCount += end - start;
194     }
195 }
