53      private static final int CHECKSUM_STRING_SIZE = 16;
54  
55      private static final int CHECKSUM_BLOCK_SIZE = 4;
56  
57      private static final int CHECKSUM_HALF_BLOCK_SIZE = 2;
58  
59      /**
60       * Calculate the Seaman-Pence 32-bit 1's complement checksum over the byte
61       * stream. The option to start from an intermediate checksum accumulated
62       * over another previous byte stream is not implemented. The implementation
63       * accumulates in two 64-bit integer values the two low-order and the two
64       * high-order bytes of adjacent 4-byte groups. A carry-over of bits is never
65       * done within the main loop (only once at the end at reduction to a 32-bit
66       * positive integer) since an overflow of a 64-bit value (signed, with
67       * maximum at 2^63-1) by summation of 16-bit values could only occur after
68       * adding approximately 140G short values (=2^47) (280GBytes) or more. We
69       * assume for now that this routine here is never called to swallow FITS
70       * files of that size or larger. by R J Mathar
71       * {@link nom.tam.fits.header.Checksum#CHECKSUM}
72       *
73       * @param data
74       *            the byte sequence
75       * @return the 32bit checksum in the range from 0 to 2^32-1
76       * @since 2005-10-05
77       */
78      public static long checksum(final byte[] data) {
79          long hi = 0;
80          long lo = 0;
81          final int len = CHECKSUM_HALF_BLOCK_SIZE * (data.length / CHECKSUM_BLOCK_SIZE);
82          // System.out.println(data.length + " bytes") ;
83          final int remain = data.length % CHECKSUM_BLOCK_SIZE;
84          if (remain != 0) {
85              throw new IllegalArgumentException("fits blocks always must be devidable by 4");
86          }
87          /*
88           * a write(2) on Sparc/PA-RISC would write the MSB first, on Linux the
89           * LSB; by some kind of coincidence, we can stay with the byte order
90           * known from the original C version of the algorithm.
91           */
92          for (int i = 0; i < len; i += CHECKSUM_HALF_BLOCK_SIZE) {
93              /*
94               * The four bytes in this block handled by a single 'i' are each
95               * signed (-128 to 127) in Java and need to be masked indivdually to
96               * avoid sign extension /propagation.
97               */
98              int offset = CHECKSUM_HALF_BLOCK_SIZE * i;
99              hi += data[offset++] << FitsIO.BITS_OF_1_BYTE & BYTE_2_OF_LONG_MASK | data[offset++] & BYTE_1_OF_LONG_MASK;
100             lo += data[offset++] << FitsIO.BITS_OF_1_BYTE & BYTE_2_OF_LONG_MASK | data[offset++] & BYTE_1_OF_LONG_MASK;
101         }
102 
103         long hicarry = hi >>> FitsIO.BITS_OF_2_BYTES;
104         long locarry = lo >>> FitsIO.BITS_OF_2_BYTES;
105         while (hicarry != 0 || locarry != 0) {
106             hi = (hi & FitsIO.SHORT_OF_LONG_MASK) + locarry;
107             lo = (lo & FitsIO.SHORT_OF_LONG_MASK) + hicarry;
108             hicarry = hi >>> FitsIO.BITS_OF_2_BYTES;
109             locarry = lo >>> FitsIO.BITS_OF_2_BYTES;
110         }
111         return hi << FitsIO.BITS_OF_2_BYTES | lo;
112     }

114     /**
115      * Encode a 32bit integer according to the Seaman-Pence proposal.
116      *
117      * @see <a
118      *      href="http://heasarc.gsfc.nasa.gov/docs/heasarc/ofwg/docs/general/checksum/node14.html#SECTION00035000000000000000">heasarc
119      *      checksum doc</a>
120      * @param c
121      *            the checksum previously calculated
122      * @param compl
123      *            complement the value
124      * @return the encoded string of 16 bytes.
125      */
126     public static String checksumEnc(final long c, final boolean compl) {
127         byte[] asc = new byte[CHECKSUM_STRING_SIZE];
128         final int[] exclude = {
129             0x3a,
130             0x3b,
131             0x3c,
132             0x3d,
133             0x3e,
134             0x3f,
135             0x40,
136             0x5b,
137             0x5c,
138             0x5d,
139             0x5e,
140             0x5f,
141             0x60
142         };
143         final long[] mask = {
144             BYTE_4_OF_LONG_MASK,
145             BYTE_3_OF_LONG_MASK,
146             BYTE_2_OF_LONG_MASK,
147             BYTE_1_OF_LONG_MASK
148         };
149         final int offset = 0x30; /* ASCII 0 (zero */
150         final long value = compl ? ~c : c;
151         for (int i = 0; i < CHECKSUM_BLOCK_SIZE; i++) {
152             // each byte becomes four
153             final int byt = (int) ((value & mask[i]) >>> FitsIO.BITS_OF_3_BYTES - FitsIO.BITS_OF_1_BYTE * i);
154             final int quotient = byt / CHECKSUM_BLOCK_SIZE + offset;
155             final int remainder = byt % CHECKSUM_BLOCK_SIZE;
156             int[] ch = new int[CHECKSUM_BLOCK_SIZE];
157             for (int j = 0; j < CHECKSUM_BLOCK_SIZE; j++) {
158                 ch[j] = quotient;
159             }
160 
161             ch[0] += remainder;
162             boolean check = true;
163             while (check) {
164                 // avoid ASCII punctuation
165                 check = false;
166                 for (int element : exclude) {
167                     for (int j = 0; j < CHECKSUM_BLOCK_SIZE; j += CHECKSUM_HALF_BLOCK_SIZE) {
168                         if (ch[j] == element || ch[j + 1] == element) {
169                             ch[j]++;
170                             ch[j + 1]--;
171                             check = true;
172                         }
173                     }
174                 }
175             }
176 
177             for (int j = 0; j < CHECKSUM_BLOCK_SIZE; j++) {
178                 // assign the bytes
179                 asc[CHECKSUM_BLOCK_SIZE * j + i] = (byte) ch[j];
180             }
181         }
182         // shift the bytes 1 to the right circularly.
183         String resul = AsciiFuncs.asciiString(asc, CHECKSUM_STRING_SIZE - 1, 1);
184         return resul.concat(AsciiFuncs.asciiString(asc, 0, CHECKSUM_STRING_SIZE - 1));
185     }

187     /**
188      * Add or update the CHECKSUM keyword. by R J Mathar
189      *
190      * @param hdu
191      *            the HDU to be updated.
192      * @throws FitsException
193      *             if the operation failed
194      * @since 2005-10-05
195      */
196     public static void setChecksum(BasicHDU<?> hdu) throws FitsException {
197         try {
198             /*
199              * the next line with the delete is needed to avoid some unexpected
200              * problems with non.tam.fits.Header.checkCard() which otherwise
201              * says it expected PCOUNT and found DATE.
202              */
203             Header hdr = hdu.getHeader();
204             hdr.deleteKey(CHECKSUM);
205             hdr.deleteKey(DATASUM);
206             /*
207              * delete the keys to force rewriting of the ckecksum and update of
208              * the checksum comment
209              */
210             hdr.addValue(CHECKSUM, "0000000000000000");
211             hdr.addValue(DATASUM, "0");
212 
213             // write the header to stream to get the cards sorted. no need to
214             // flush because we will ignore the data.
215             ByteArrayOutputStream hduByteImage = new ByteArrayOutputStream();
216             hdu.getHeader().write(new FitsOutputStream(hduByteImage));
217             hduByteImage.reset();
218 
219             /*
220              * Convert the entire sequence of 2880 byte header cards into a byte
221              * tiledImageOperation. The main benefit compared to the C
222              * implementations is that we do not need to worry about the
223              * particular byte order on machines (Linux/VAX/MIPS vs Hp-UX,
224              * Sparc...) supposed that the correct implementation is in the
225              * write() interface.
226              */
227             FitsOutputStream bdos = new FitsOutputStream(hduByteImage);
228             hdu.getData().write(bdos);
229             bdos.flush();
230             
231             long csd = checksum(hduByteImage.toByteArray());
232             hdu.getHeader().card(DATASUM).value(Long.toString(csd));
233 
234             // We already have the checksum of the data. Lets compute it for
235             // the header.
236             hduByteImage.reset();
237             hdu.getHeader().write(bdos);
238             bdos.flush();
239             
240             long csh = checksum(hduByteImage.toByteArray());
241             
242             long cshdu = csh + csd;
243             // If we had a carry it should go into the
244             // beginning.
245             while ((cshdu & FitsIO.HIGH_INTEGER_MASK) != 0) {
246                 long cshduIntPart = cshdu & FitsIO.INTEGER_MASK;
247                 cshdu = cshduIntPart + 1;
248             }
249             /*
250              * This time we do not use a deleteKey() to ensure that the keyword
251              * is replaced "in place". Note that the value of the checksum is
252              * actually independent to a permutation of the 80-byte records
253              * within the header.
254              */
255             hdr.card(CHECKSUM).value(checksumEnc(cshdu, true));
256         } catch (IOException e) {
257             throw new FitsException("Could not calculate the checksum!", e);
258         }
259     }
