34  /**
35   * This class is a pointer into a part of an other string, it can be manipulated
36   * by changing the position pointers into the "original" string. This class is
37   * aware of the escape quote, two quotes in sequence the respresent a single
38   * quote.
39   * 
40   * @author Richard van Nieuwenhoven
41   */
42  public class FitsSubString {
43  
44      /**
45       * the length of the substring (starting at the offset).
46       */
47      private int length;
48  
49      /**
50       * the offset into the original string where this string starts.
51       */
52      private int offset;
53  
54      /**
55       * the original String.
56       */
57      private final String originalString;
58  
59      /**
60       * constructor for the substring, start by representing the whole string.
61       * 
62       * @param originalString
63       *            the string to represent.
64       */
65      public FitsSubString(String originalString) {
66          this.originalString = originalString == null ? "" : originalString;
67          this.offset = 0;
68          this.length = this.originalString.length();
69      }
70  
71      /**
72       * append the current string representation to the StringBuffer.
73       * 
74       * @param buffer
75       *            the buffer to append to.
76       */
77      public void appendTo(StringBuffer buffer) {
78          buffer.append(this.originalString, this.offset, this.offset + this.length);
79      }
80  
81      /**
82       * get the character at the specified position.
83       * 
84       * @param pos
85       *            the position the get the character from
86       * @return the character at the specified position
87       */
88      public char charAt(int pos) {
89          return this.originalString.charAt(pos + this.offset);
90      }
91  
92      /**
93       * @return get the length of the orginal string from the current offset.
94       */
95      public int fullLength() {
96          return this.originalString.length() - this.offset;
97      }
98  
99      /**
100      * check the string and set it to the maximum length specified. if a escaped
101      * quote is on the boundary the length is reduced in a way that the string
102      * does not separate an escape quote.
103      * 
104      * @param max
105      *            the maximum string legth to set.
106      */
107     public void getAdjustedLength(int max) {
108         if (max <= 0) {
109             this.length = 0;
110         } else if (this.length > max) {
111             int pos = max - 1;
112             while (charAt(pos) == '\'') {
113                 pos--;
114             }
115             // now we are at the start of the quotes step forward in steps of 2
116             pos += (max - 1 - pos) / 2 * 2;
117             this.length = pos + 1;
118         }
119     }
120 
121     /**
122      * @return the string length of this String.
123      */
124     public int length() {
125         return this.length;
126     }
127 
128     /**
129      * shift the sting to the rest of the string, the part of the original
130      * string that is after the part of the string this instance currently
131      * represents.
132      */
133     public void rest() {
134         this.offset += this.length;
135         this.length = this.originalString.length() - this.offset;
136     }
137 
138     /**
139      * skip over the specified number of characters.
140      * 
141      * @param count
142      *            the number of chars to skip
143      */
144     public void skip(int count) {
145         this.offset += count;
146         this.length -= count;
147     }
148 
149     /**
150      * @param string
151      *            the string to check
152      * @return true if the current string starts with the specified string.
153      */
154     public boolean startsWith(String string) {
155         return this.originalString.regionMatches(this.offset, string, 0, string.length());
156     }
157 }
