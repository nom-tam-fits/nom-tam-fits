34  /**
35   * This class is a pointer into a part of an other string, it can be manipulated
36   * by changing the position pointers into the "original" string. This class is
37   * aware of the escape quote, two quotes in sequence the respresent a single
38   * quote.
39   * 
40   * @author Richard van Nieuwenhoven
41   * 
42   * @deprecated Was needed by {@link FitsLineAppender} only, which itself is deprecated in
43   *              favor of the package-level <code>nom.tam.fits.HeaderCardFormatter</code>.
44   */
45  @Deprecated
46  public class FitsSubString {
47  
48      /**
49       * the length of the substring (starting at the offset).
50       */
51      private int length;
52  
53      /**
54       * the offset into the original string where this string starts.
55       */
56      private int offset;
57  
58      /**
59       * the original String.
60       */
61      private final String originalString;
62  
63      /**
64       * constructor for the substring, start by representing the whole string.
65       * 
66       * @param originalString
67       *            the string to represent.
68       */
69      public FitsSubString(String originalString) {
70          this.originalString = originalString == null ? "" : originalString;
71          this.offset = 0;
72          this.length = this.originalString.length();
73      }
74  
75      /**
76       * append the current string representation to the StringBuffer.
77       * 
78       * @param buffer
79       *            the buffer to append to.
80       */
81      public void appendTo(StringBuilder buffer) {
82          buffer.append(this.originalString, this.offset, this.offset + this.length);
83      }
84  
85      /**
86       * get the character at the specified position.
87       * 
88       * @param pos
89       *            the position the get the character from
90       * @return the character at the specified position
91       */
92      public char charAt(int pos) {
93          return this.originalString.charAt(pos + this.offset);
94      }
95  
96      /**
97       * @return get the length of the orginal string from the current offset.
98       */
99      public int fullLength() {
100         return this.originalString.length() - this.offset;
101     }
102 
103     /**
104      * check the string and set it to the maximum length specified. if a escaped
105      * quote is on the boundary the length is reduced in a way that the string
106      * does not separate an escape quote.
107      * 
108      * @param max
109      *            the maximum string legth to set.
110      */
111     public void getAdjustedLength(int max) {
112         if (max <= 0) {
113             this.length = 0;
114         } else if (this.length > max) {
115             int pos = max - 1;
116             while (charAt(pos) == '\'') {
117                 pos--;
118             }
119             // now we are at the start of the quotes step forward in steps of 2
120             pos += (max - 1 - pos) / 2 * 2;
121             this.length = pos + 1;
122         }
123     }
124 
125     /**
126      * @return the string length of this String.
127      */
128     public int length() {
129         return this.length;
130     }
131 
132     /**
133      * shift the sting to the rest of the string, the part of the original
134      * string that is after the part of the string this instance currently
135      * represents.
136      */
137     public void rest() {
138         this.offset += this.length;
139         this.length = this.originalString.length() - this.offset;
140     }
141 
142     /**
143      * skip over the specified number of characters.
144      * 
145      * @param count
146      *            the number of chars to skip
147      */
148     public void skip(int count) {
149         this.offset += count;
150         this.length -= count;
151     }
152 
153     /**
154      * @param string
155      *            the string to check
156      * @return true if the current string starts with the specified string.
157      */
158     public boolean startsWith(String string) {
159         return this.originalString.regionMatches(this.offset, string, 0, string.length());
160     }
161 }
