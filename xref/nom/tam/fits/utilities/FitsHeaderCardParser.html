34  import java.util.regex.Matcher;
35  import java.util.regex.Pattern;
36  
37  /**
38   * A helper utility class to parse header cards for there value (especially
39   * strings) and comments.
40   * 
41   * @author Richard van Nieuwenhoven
42   */
43  public final class FitsHeaderCardParser {
44  
45      /**
46       * value comment pair of the header card.
47       */
48      public static class ParsedValue {
49  
50          /**
51           * the comment specified with the value.
52           */
53          private String comment = null;
54  
55          /**
56           * was the value quoted?
57           */
58          private boolean isString = false;
59  
60          /**
61           * the value of the card. (trimmed)
62           */
63          private String value = null;
64  
65          /**
66           * @return the comment of the card.
67           */
68          public String getComment() {
69              return this.comment;
70          }
71  
72          /**
73           * @return the value of the card.
74           */
75          public String getValue() {
76              return this.value;
77          }
78  
79          /**
80           * @return true if the value was quoted.
81           */
82          public boolean isString() {
83              return this.isString;
84          }
85  
86      }
87  
88      /**
89       * pattern to replace the double quotes from string values to single quotes.
90       */
91      private static final Pattern DOUBLE_QUOTE_PATTERN = Pattern.compile("''");
92  
93      /**
94       * pattern to match FITS keywords, specially to parse hirarchical keywords.
95       */
96      private static final Pattern KEYWORD_PATTERN = Pattern.compile("([A-Z|0-9|_|-]+)([ |\\|.]*=?)");
97  
98      /**
99       * pattern to match a quoted string where 2 quotes are used to escape a
100      * single quote inside the string.
101      */
102     private static final Pattern STRING_PATTERN = Pattern.compile("'(?:[^']|'{2})*'");
103 
104     /**
105      * delete the start and trailing quote from the sting and replace all
106      * (escaped)double quotes with a single quote. Then trim the trailing
107      * blanks.
108      * 
109      * @param quotedString
110      *            the string to un-quote.
111      * @return the unquoted string
112      */
113     private static String deleteQuotes(String quotedString) {
114         Matcher doubleQuoteMatcher = FitsHeaderCardParser.DOUBLE_QUOTE_PATTERN.matcher(quotedString);
115         StringBuffer sb = new StringBuffer();
116         if (doubleQuoteMatcher.find(1)) {
117             do {
118                 doubleQuoteMatcher.appendReplacement(sb, "'");
119             } while (doubleQuoteMatcher.find());
120         }
121         doubleQuoteMatcher.appendTail(sb);
122         sb.deleteCharAt(0);
123         sb.setLength(sb.length() - 1);
124         while (sb.length() > 0 && Character.isWhitespace(sb.charAt(sb.length() - 1))) {
125             sb.setLength(sb.length() - 1);
126         }
127         return sb.toString();
128     }
129 
130     /**
131      * get the not empty comment string from the end of the card. start scanning
132      * from the end of the value.
133      * 
134      * @param stringCard
135      *            the string representing the card
136      * @param startPosition
137      *            the start point for the scan
138      * @return the not empty comment or null if no comment was found.
139      */
140     private static String extractComment(String stringCard, int startPosition) {
141         int startOfComment = stringCard.indexOf('/', startPosition) + 1;
142         if (startOfComment > 0 && stringCard.length() > startOfComment) {
143             return stringCard.substring(startOfComment).trim();
144         }
145         return null;
146     }
147 
148     /**
149      * parse a fits keyword from a card and return it as a dot separated list.
150      * 
151      * @param card
152      *            the card to parse.
153      * @return dot separated key list
154      */
155     public static String parseCardKey(String card) {
156         int indexOfEquals = card.indexOf('=');
157         StringBuilder builder = new StringBuilder();
158         Matcher kewordMatcher = FitsHeaderCardParser.KEYWORD_PATTERN.matcher(card);
159         while (kewordMatcher.find() && kewordMatcher.start() < indexOfEquals) {
160             if (builder.length() != 0) {
161                 builder.append('.');
162             }
163             builder.append(kewordMatcher.group(1));
164             if (kewordMatcher.group(2).endsWith("=")) {
165                 break;
166             }
167         }
168         return builder.toString();
169     }
170 
171     /**
172      * Parse the card for a value and comment. Quoted string values are unquoted
173      * and the {@link ParsedValue#isString} specifies if the value was a quoted
174      * string. non quoted values are trimmed.
175      * 
176      * @param card
177      *            the card to parse.
178      * @return a parsed card or null if no value could be detected.
179      */
180     public static ParsedValue parseCardValue(String card) {
181         ParsedValue value = parseStringValue(card);
182         if (value == null) {
183             // ok no string lets check for an equals.
184             int indexOfEquals = card.indexOf('=');
185             if (indexOfEquals > 0) {
186                 // its no string so the value goes max till the comment
187                 value = new ParsedValue();
188                 int endOfValue = card.length() - 1;
189                 int startOfComment = card.indexOf('/', indexOfEquals);
190                 if (startOfComment > 0) {
191                     endOfValue = startOfComment - 1;
192                     value.comment = extractComment(card, startOfComment);
193                 }
194                 value.value = card.substring(indexOfEquals + 1, endOfValue + 1).trim();
195             }
196         }
197         return value;
198     }
199 
200     /**
201      * lets see if there is a quoted sting in the card.
202      * 
203      * @param card
204      *            the card string to parse.
205      * @return the parsed value with the unquoted string or null if no quoted
206      *         string was found.
207      */
208     private static ParsedValue parseStringValue(String card) {
209         ParsedValue stringValue = null;
210         Matcher matcher = FitsHeaderCardParser.STRING_PATTERN.matcher(card);
211         if (matcher.find()) {
212             int indexOfComment = card.indexOf('/');
213             if (indexOfComment >= 0 && indexOfComment < matcher.start()) {
214                 // ok the string was commented, forget the string.
215                 return null;
216             }
217             stringValue = new ParsedValue();
218             stringValue.isString = true;
219             stringValue.value = deleteQuotes(matcher.group(0));
220             stringValue.comment = extractComment(card, matcher.end());
221         }
222         return stringValue;
223     }
224 
225     /**
226      * utility class will not be instantiated.
227      */
228     private FitsHeaderCardParser() {
229     }
