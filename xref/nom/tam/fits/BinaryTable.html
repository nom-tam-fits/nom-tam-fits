262     public BinaryTable(ColumnTable<?> tabIn) {
263         @SuppressWarnings("unchecked")
264         ColumnTable<SaveState> tab = (ColumnTable<SaveState>) tabIn;
265         // This will throw an error if this isn't the correct type.
266         SaveState extra = tab.getExtraState();
267         this.columnList = new ArrayList<>();
268         for (ColumnDesc col : extra.columns) {
269             ColumnDesc copy = (ColumnDesc) col.clone();
270             copy.column = null;
271             this.columnList.add(copy);
272         }
273         try {
274             this.table = tab.copy();
275         } catch (Exception e) {
276             throw new IllegalStateException("Unexpected Exception", e);
277         }
278         this.heap = extra.heap.copy();
279         this.nRow = tab.getNRows();
280         saveExtraState();
281     }

283     /**
284      * Create a binary table from given header information.
285      *
286      * @param myHeader
287      *            A header describing what the binary table should look like.
288      * @throws FitsException
289      *             if the specified header is not usable for a binary table
290      */
291     public BinaryTable(Header myHeader) throws FitsException {
292         long heapSizeL = myHeader.getLongValue(PCOUNT);
293         long heapOffsetL = myHeader.getLongValue(THEAP);
294         if (heapOffsetL > MAX_INTEGER_VALUE) {
295             throw new FitsException("Heap Offset > 2GB");
296         }
297         if (heapSizeL > MAX_INTEGER_VALUE) {
298             throw new FitsException("Heap size > 2 GB");
299         }
300         if (heapSizeL - heapOffsetL > MAX_INTEGER_VALUE) {
301             throw new FitsException("Unable to allocate heap > 2GB");
302         }
303         this.heapOffset = (int) heapOffsetL;
304         int heapSize = (int) heapSizeL;
305         int rwsz = myHeader.getIntValue(NAXIS1);
306         this.nRow = myHeader.getIntValue(NAXIS2);
307 
308         // Subtract out the size of the regular table from
309         // the heap offset.
310         if (this.heapOffset > 0) {
311             this.heapOffset -= this.nRow * rwsz;
312         }
313 
314         if (this.heapOffset < 0 || this.heapOffset > heapSize) {
315             throw new FitsException("Inconsistent THEAP and PCOUNT");
316         }
317 
318         this.heap = new FitsHeap(heapSize - this.heapOffset);
319         int nCol = myHeader.getIntValue(TFIELDS);
320         this.rowLen = 0;
321         for (int col = 0; col < nCol; col++) {
322             this.rowLen += processCol(myHeader, col);
323         }
324         HeaderCard card = myHeader.findCard(NAXIS1);
325         card.setValue(String.valueOf(this.rowLen));
326         myHeader.updateLine(NAXIS1, card);
327 
328     }

372     /**
373      * Parse the TDIMS value. If the TDIMS value cannot be deciphered a one-d
374      * array with the size given in arrsiz is returned.
375      *
376      * @param tdims
377      *            The value of the TDIMSn card.
378      * @return An int array of the desired dimensions. Note that the order of
379      *         the tdims is the inverse of the order in the TDIMS key.
380      */
381     public static int[] getTDims(String tdims) {
382 
383         // The TDIMs value should be of the form: "(iiii,jjjj,kkk,...)"
384         int[] dims = null;
385         int first = tdims.indexOf('(');
386         int last = tdims.lastIndexOf(')');
387         if (first >= 0 && last > first) {
388 
389             tdims = tdims.substring(first + 1, last - first);
390 
391             java.util.StringTokenizer st = new java.util.StringTokenizer(tdims, ",");
392             int dim = st.countTokens();
393             if (dim > 0) {
394                 dims = new int[dim];
395                 for (int i = dim - 1; i >= 0; i -= 1) {
396                     dims[i] = Integer.parseInt(st.nextToken().trim());
397                 }
398             }
399         }
400         return dims;
401     }

403     /**
404      * Convert a two-d table to a table of columns. Handle String specially.
405      * Every other element of data should be a primitive array of some
406      * dimensionality. Basically the translates a table expressed as objects in
407      * row order to a table with objects in column order.
408      */
409     private static Object[] convertToColumns(Object[][] data) {
410         Object[] row = data[0];
411         int nrow = data.length;
412         Object[] results = new Object[row.length];
413         for (int col = 0; col < row.length; col++) {
414             if (row[col] instanceof String) {
415                 String[] sa = new String[nrow];
416                 for (int irow = 0; irow < nrow; irow++) {
417                     sa[irow] = (String) data[irow][col];
418                 }
419                 results[col] = sa;
420             } else {
421                 Class<?> base = ArrayFuncs.getBaseClass(row[col]);
422                 int[] dims = ArrayFuncs.getDimensions(row[col]);
423 
424                 if (dims.length > 1 || dims[0] > 1) {
425                     int[] xdims = new int[dims.length + 1];
426                     xdims[0] = nrow;
427 
428                     Object[] arr = (Object[]) ArrayFuncs.newInstance(base, xdims);
429                     for (int irow = 0; irow < nrow; irow++) {
430                         arr[irow] = data[irow][col];
431                     }
432                     results[col] = arr;
433                 } else {
434                     Object arr = ArrayFuncs.newInstance(base, nrow);
435                     for (int irow = 0; irow < nrow; irow++) {
436                         System.arraycopy(data[irow][col], 0, arr, irow, 1);
437                     }
438                     results[col] = arr;
439                 }
440 
441             }
442         }
443         return results;
444     }

446     @Override
447     public int addColumn(Object o) throws FitsException {
448         int primeDim = Array.getLength(o);
449         ColumnDesc added = new ColumnDesc();
450         this.columnList.add(added);
451 
452         // A varying length column is a two-d primitive
453         // array where the second index is not constant.
454         // We do not support Q types here, since Java
455         // can't handle the long indices anyway...
456         // This will probably change in some version of Java.
457         if (isVarying(o)) {
458             added.isVarying = true;
459             added.dimens = new int[]{
460                     2
461             };
462         }
463 
464         if (isVaryingComp(o)) {
465             added.isVarying = true;
466             added.isComplex = true;
467             added.dimens = new int[]{
468                     2
469             };
470         }
471 
472         // Flatten out everything but 1-D arrays and the
473         // two-D arrays associated with variable length columns.
474         if (!added.isVarying) {
475 
476             int[] allDim = ArrayFuncs.getDimensions(o);
477             Class<?> base = ArrayFuncs.getBaseClass(o);
478 
479             // Add a dimension for the length of Strings.
480             if (base == String.class) {
481                 int[] xdim = new int[allDim.length + 1];
482 
483                 System.arraycopy(allDim, 0, xdim, 0, allDim.length);
484                 xdim[allDim.length] = -1;
485                 allDim = xdim;
486             }
487 
488             if (allDim.length == 1) {
489                 added.dimens = new int[0];
490             } else {
491                 added.dimens = new int[allDim.length - 1];
492                 System.arraycopy(allDim, 1, added.dimens, 0, allDim.length - 1);
493                 o = ArrayFuncs.flatten(o);
494             }
495         }
496 
497         addFlattenedColumn(o, added.dimens, true);
498         if (this.nRow == 0 && this.columnList.size() == 1) { // Adding the first
499             // column
500             this.nRow = primeDim;
501         }
502         return this.columnList.size();
503 
504     }

522     @Override
523     public int addRow(Object[] o) throws FitsException {
524         ensureData();
525         if (this.columnList.size() == 0 && this.nRow == 0) {
526             for (Object element : o) {
527                 if (element == null) {
528                     throw new FitsException("Cannot add initial rows with nulls");
529                 }
530                 addColumn(encapsulate(element));
531             }
532             createTable();
533 
534         } else {
535             Object[] flatRow = new Object[getNCols()];
536             for (int i = 0; i < getNCols(); i++) {
537                 Object x = ArrayFuncs.flatten(o[i]);
538                 ColumnDesc colDesc = this.columnList.get(i);
539                 flatRow[i] = arrayToColumn(colDesc, x);
540             }
541             this.table.addRow(flatRow);
542             this.nRow++;
543         }
544 
545         return this.nRow;
546     }

548     /**
549      * Delete a set of columns. Note that this does not fix the header, so users
550      * should normally call the routine in TableHDU. * @throws FitsException if
551      * the operation failed
552      */
553     @Override
554     public void deleteColumns(int start, int len) throws FitsException {
555         ensureData();
556         try {
557             this.table.deleteColumns(start, len);
558             // Need to get rid of the column level metadata.
559             for (int i = start + len - 1; i >= start; i -= 1) {
560                 if (i >= 0 && i <= this.columnList.size()) {
561                     ColumnDesc columnDesc = this.columnList.get(i);
562                     this.rowLen -= columnDesc.rowLen();
563                     this.columnList.remove(i);
564                 }
565             }
566         } catch (Exception e) {
567             throw new FitsException("Error deleting columns from BinaryTable", e);
568         }
569     }

587     /**
588      * Update a FITS header to reflect the current state of the data. * @throws
589      * FitsException if the operation failed
590      */
591     @Override
592     public void fillHeader(Header h) throws FitsException {
593         try {
594             Standard.context(BinaryTable.class);
595             h.setXtension(XTENSION_BINTABLE);
596             h.setBitpix(Bitpix.BYTE);
597             h.setNaxes(2);
598             h.setNaxis(1, this.rowLen);
599             h.setNaxis(2, this.nRow);
600             h.addValue(PCOUNT, this.heap.size());
601             h.addValue(GCOUNT, 1);
602             Cursor<String, HeaderCard> iter = h.iterator();
603             iter.setKey(GCOUNT.key());
604             iter.next();
605             iter.add(new HeaderCard(TFIELDS.key(), this.columnList.size(), TFIELDS.comment()));
606             for (int i = 0; i < this.columnList.size(); i++) {
607                 if (i > 0) {
608                     h.positionAfterIndex(TFORMn, i);
609                 }
610                 fillForColumn(h, i, iter);
611             }
612         } finally {
613             Standard.context(null);
614         }
615     }

663     /**
664      * Get a particular element from the table.
665      *
666      * @param i
667      *            The row of the element.
668      * @param j
669      *            The column of the element. * @throws FitsException if the
670      *            operation failed
671      */
672     @Override
673     public Object getElement(int i, int j) throws FitsException {
674 
675         if (!validRow(i) || !validColumn(j)) {
676             throw new FitsException("No such element (" + i + "," + j + ")");
677         }
678 
679         ColumnDesc colDesc = this.columnList.get(j);
680         Object ele;
681         if (colDesc.isVarying) {
682             // Have to read in entire data set.
683             ensureData();
684         }
685 
686         if (this.table == null) {
687             // This is really inefficient.
688             // Need to either save the row, or just read the one element.
689             Object[] row = getRow(i);
690             ele = row[j];
691 
692         } else {
693             ele = this.table.getElement(i, j);
694             ele = columnToArray(colDesc, ele, 1);
695             ele = encurl(ele, j, 1);
696             if (ele instanceof Object[]) {
697                 ele = ((Object[]) ele)[0];
698             }
699         }
700 
701         return ele;
702     }

1000      * operation failed
1001      */
1002     @Override

1006 
1007             this.table.write(os);
1008             if (this.heapOffset > 0) {
1009                 int off = this.heapOffset;
1010                 // Minimize memory usage. This also accommodates
1011                 // the possibility that heapOffset > 2GB.
1012                 // Previous code might have allocated up to 2GB
1013                 // array. [In practice this is always going
1014                 // to be really small though...]
1015                 int arrSiz = MAX_EMPTY_BLOCK_SIZE;
1016                 while (off > 0) {

1018                         arrSiz = off;

1020                     os.write(new byte[arrSiz]);

1022                 }
1023             }
1024

1026             if (this.heap.size() > 0) {
1027                 this.heap.write(os);
1028             }
1029 
1030             FitsUtil.pad(os, getTrueSize());
1031 
1032         } catch (IOException e) {

1037     private Object arrayToVariableColumn(ColumnDesc added, Object o) throws FitsException {
1038         if (added.isBoolean) {
1039             // Handle addRow/addElement
1040             if (o instanceof boolean[]) {
1041                 o = new boolean[][]{
1042                     (boolean[]) o
1043                 };
1044             }
1045             // Convert boolean to byte arrays
1046             boolean[][] to = (boolean[][]) o;
1047             byte[][] xo = new byte[to.length][];
1048             for (int i = 0; i < to.length; i++) {
1049                 xo[i] = FitsUtil.booleanToByte(to[i]);
1050             }
1051             o = xo;
1052         }
1053         
1054         // Write all rows of data onto the heap.
1055         int offset = heap.size();
1056         int elementSize = ArrayFuncs.getBaseLength(o);
1057         if (added.isComplex) {
1058             elementSize *= 2;
1059         }
1060         
1061         heap.putData(o);
1062         
1063         // Handle an addRow of a variable length element.
1064         // In this case we only get a one-d array, but we just
1065         // make is 1 x n to get the second dimension.
1066       
1067         Object[] x = (o instanceof Object[]) ? (Object[]) o : new Object[] {o};
1068 
1069         
1070         // Create the array descriptors
1071         int nrow = Array.getLength(x);
1072         
1073         if (added.isLongVary) {
1074             long[] descrip = new long[2 * nrow];
1075     
1076             // Fill the descriptor for each row.
1077             for (int i = 0; i < nrow; i++) {
1078                 int len = Array.getLength(x[i]);
1079                 int j = i << 1;
1080                 descrip[j++] = len;
1081                 descrip[j] = offset;           
1082                 offset += len * elementSize;
1083             }
1084             return descrip;
1085         } 
1086         
1087         int[] descrip = new int[2 * nrow];
1088   
1089         // Fill the descriptor for each row.
1090         for (int i = 0; i < nrow; i++) {    
1091             int len = Array.getLength(x[i]);
1092             int j = i << 1;
1093             descrip[j++] = len;
1094             descrip[j] = offset;
1095             offset += len * elementSize;
1096         }
1097 
1098         
1099         return descrip;
1100     }

1225     private Object variableColumnToArray(ColumnDesc colDesc, Object o, int rows) throws FitsException {
1226         // A. Kovacs (4/1/08)
1227         // Ensure that the heap has been initialized
1228         if (!this.heapReadFromStream) {
1229             readHeap(this.currInput);
1230         }
1231         int[] descrip;
1232         if (colDesc.isLongVary) {
1233             // Convert longs to int's. This is dangerous.
1234             if (!this.warnedOnVariableConversion) {
1235                 LOG.log(Level.WARNING, "Warning: converting long variable array pointers to int's");
1236                 this.warnedOnVariableConversion = true;
1237 
1238             }
1239             descrip = (int[]) ArrayFuncs.convertArray(o, int.class);
1240         } else {
1241             descrip = (int[]) o;
1242         }
1243         int nrow = descrip.length / 2;
1244         Object[] res; // Res will be the result of extracting from the heap.
1245         
1246         if (colDesc.isComplex) {
1247             // Complex columns have an extra dimension for each row
1248             res = (Object[]) ArrayFuncs.newInstance(colDesc.base, new int[] {nrow, 0, 0});
1249         } else if (colDesc.isString) {
1250             // ---> Added clause by Attila Kovacs (13 July 2007)
1251             // String columns have to read data into a byte array at first
1252             // then do the string conversion later.
1253             res = (Object[]) ArrayFuncs.newInstance(byte.class, new int[] {nrow, 0});
1254         } else {
1255             // Non-complex data has a simple primitive array for each row
1256             res = (Object[]) ArrayFuncs.newInstance(colDesc.base, new int[] {nrow, 0});
1257         }
1258         // Now read in each requested row.
1259         for (int i = 0; i < nrow; i++) {
1260             Object row;
1261             int j = i << 1;
1262             int dim = descrip[j++];
1263             int offset = descrip[j];
1264     
1265             if (colDesc.isComplex) {
1266                 row = ArrayFuncs.newInstance(colDesc.base, new int[] {dim, 2}); 
1267             } else if (colDesc.isString) {
1268                 // ---> Added clause by Attila Kovacs (13 July 2007)
1269                 // Again, String entries read data into a byte array at
1270                 // first
1271                 // then do the string conversion later.
1272                 // For string data, we need to read bytes and convert
1273                 // to strings
1274                 row = ArrayFuncs.newInstance(byte.class, dim);
1275             } else if (colDesc.isBoolean) {
1276                 // For boolean data, we need to read bytes and convert
1277                 // to booleans.
1278                 row = ArrayFuncs.newInstance(byte.class, dim);
1279             } else {
1280                 row = ArrayFuncs.newInstance(colDesc.base, dim);
1281             }  
1282             
1283             this.heap.getData(offset, row);
1284             // Now do the boolean conversion.
1285             
1286             if (colDesc.isBoolean) {
1287                 row = FitsUtil.byteToBoolean((byte[]) row);
1288             }
1289             
1290             res[i] = row;
1291         }
1292         return res;
1293     }

1453      * index is non-zero). In this case it can be a complex varying column.
1454      */
1455     private boolean isVarying(Object o) {
1456         if (o == null || //
1457                 !o.getClass().isArray() || //
1458                 !o.getClass().getComponentType().isArray() || //
1459                 !o.getClass().getComponentType().getComponentType().isPrimitive()) {
1460             return false;
1461         }
1462 
1463         int oLength = Array.getLength(o);
1464         if (oLength < 2) {
1465             return false;
1466         }
1467 
1468         int flen = Array.getLength(Array.get(o, 0));
1469         for (int i = 1; i < oLength; i++) {
1470             if (Array.getLength(Array.get(o, i)) != flen) {
1471                 return true;
1472             }
1473         }
1474         return false;
1475     }

1486     /**
1487      * Process one column from a FITS Header. * @throws FitsException if the
1488      * operation failed
1489      */
1490     private int processCol(Header header, int col) throws FitsException {
1491         String tform = header.getStringValue(TFORMn.n(col + 1));
1492         if (tform == null) {
1493             throw new FitsException("Attempt to process column " + (col + 1) + " but no TFORMn found.");
1494         }
1495         tform = tform.trim();
1496         ColumnDesc colDesc = new ColumnDesc();
1497         String tdims = header.getStringValue(TDIMn.n(col + 1));
1498         if (tdims != null) {
1499             tdims = tdims.trim();
1500         }
1501         char type = getTFORMType(tform);
1502         if (type == 'P' || type == 'Q') {
1503             colDesc.isVarying = true;
1504             colDesc.isLongVary = type == 'Q';
1505             type = getTFORMVarType(tform);
1506         }
1507         int size = getTFORMLength(tform);
1508         // Handle the special size cases.
1509         //
1510         // Bit arrays (8 bits fit in a byte)
1511         if (type == 'X') {
1512             size = (size + FitsIO.BITS_OF_1_BYTE - 1) / FitsIO.BITS_OF_1_BYTE;
1513             // Variable length arrays always have a two-element pointer (offset
1514             // and size)
1515         } else if (colDesc.isVarying) {
1516             size = 2;
1517         }
1518 
1519         // bSize is the number of bytes in the field.
1520         int bSize = size;
1521         int[] dims = null;
1522         // Cannot really handle arbitrary arrays of bits.
1523         if (tdims != null && type != 'X' && !colDesc.isVarying) {
1524             dims = getTDims(tdims);
1525         }
1526         if (dims == null) {
1527             if (size == 1) {
1528                 dims = new int[0]; // Marks this as a scalar column
1529             } else {
1530                 dims = new int[]{
1531                         size
1532                 };
1533             }
1534         }
1535         colDesc.isComplex = type == 'C' || type == 'M';
1536         Class<?> colBase;
1537         switch (type) {
1538         case 'A':
1539             colBase = byte.class;
1540             colDesc.isString = true;
1541             colDesc.base = String.class;
1542             break;
1543         case 'L':
1544             colBase = byte.class;
1545             colDesc.base = boolean.class;
1546             colDesc.isBoolean = true;
1547             break;
1548         case 'X':
1549         case 'B':
1550             colBase = byte.class;
1551             colDesc.base = byte.class;
1552             break;
1553 
1554         case 'I':
1555             colBase = short.class;
1556             colDesc.base = short.class;
1557             bSize *= FitsIO.BYTES_IN_SHORT;
1558             break;
1559 
1560         case 'J':
1561             colBase = int.class;
1562             colDesc.base = int.class;
1563             bSize *= FitsIO.BYTES_IN_INTEGER;
1564             break;
1565 
1566         case 'K':
1567             colBase = long.class;
1568             colDesc.base = long.class;
1569             bSize *= FitsIO.BYTES_IN_LONG;
1570             break;
1571 
1572         case 'E':
1573         case 'C':
1574             colBase = float.class;
1575             colDesc.base = float.class;
1576             bSize *= FitsIO.BYTES_IN_FLOAT;
1577             break;
1578 
1579         case 'D':
1580         case 'M':
1581             colBase = double.class;
1582             colDesc.base = double.class;
1583             bSize *= FitsIO.BYTES_IN_DOUBLE;
1584             break;
1585 
1586         default:
1587             throw new FitsException("Invalid type in column:" + col);
1588         }
1589         if (colDesc.isVarying) {
1590             dims = new int[]{
1591                     2
1592             };
1593             colBase = int.class;
1594             bSize = FitsIO.BYTES_IN_INTEGER * 2;
1595             if (colDesc.isLongVary) {
1596                 colBase = long.class;
1597                 bSize = FitsIO.BYTES_IN_LONG * 2;
1598             }
1599         }
1600         if (!colDesc.isVarying && colDesc.isComplex) {
1601             int[] xdims = new int[dims.length + 1];
1602             System.arraycopy(dims, 0, xdims, 0, dims.length);
1603             xdims[dims.length] = 2;
1604             dims = xdims;
1605             bSize *= 2;
1606             size *= 2;
1607         }
1608         colDesc.model = ArrayFuncs.newInstance(colBase, dims);
1609         colDesc.dimens = dims;
1610         colDesc.size = size;
1611         this.columnList.add(colDesc);
1612 
1613         return bSize;
1614     }

1708     /**
1709      * This function is needed since we had made addFlattenedColumn public so in
1710      * principle a user might have called it directly.
1711      *
1712      * @param o
1713      *            The new column data. This should be a one-dimensional
1714      *            primitive array.
1715      * @param dims
1716      *            The dimensions of one row of the column.
1717      * @param allocated
1718      *            is it already in the columnList?
1719      * @return the new column size
1720      * @throws FitsException
1721      */
1722     int addFlattenedColumn(Object o, int[] dims, boolean allocated) throws FitsException {
1723 
1724         ColumnDesc added;
1725         if (!allocated) {
1726             added = new ColumnDesc();
1727             added.dimens = dims;
1728         } else {
1729             added = this.columnList.get(this.columnList.size() - 1);
1730         }
1731         added.base = ArrayFuncs.getBaseClass(o);
1732         added.isBoolean = added.base == boolean.class;
1733         added.isString = added.base == String.class;
1734 
1735         // Convert to column first in case
1736         // this is a String or variable length array.
1737         o = arrayToColumn(added, o);
1738 
1739         int size = 1;
1740 
1741         for (int dim2 : dims) {
1742             size *= dim2;
1743         }
1744         added.size = size;
1745 
1746         // Check that the number of rows is consistent.
1747         if (size != 0 && this.columnList.size() > 1) {
1748             int xRow = Array.getLength(o) / size;
1749             if (xRow > 0 && xRow != this.nRow) {
1750                 throw new FitsException("Added column does not have correct row count");
1751             }
1752         }
1753 
1754         if (!added.isVarying) {
1755             added.model = ArrayFuncs.newInstance(ArrayFuncs.getBaseClass(o), dims);
1756             this.rowLen += size * ArrayFuncs.getBaseLength(o);
1757         } else if (added.isLongVary) {
1758             added.model = new long[2];
1759             this.rowLen += FitsIO.BYTES_IN_LONG * 2;
1760         } else {
1761             added.model = new int[2];
1762             this.rowLen += FitsIO.BYTES_IN_INTEGER * 2;
1763         }
1764 
1765         // Only add to table if table already exists or if we
1766         // are filling up the last element in columns.
1767         // This way if we allocate a bunch of columns at the beginning
1768         // we only create the column table after we have all the columns
1769         // ready.
1770         added.column = o;
1771         if (this.table != null) {
1772             this.table.addColumn(o, added.size);
1773         }
1774         if (!this.columnList.contains(added)) {
1775             this.columnList.add(added);
1776         }
1777         return this.columnList.size();
1778     }

1780     /**
1781      * Update the header to reflect the details of a given column. 
1782      * 
1783      * @throws FitsException if the operation failed
1784      */
1785     void fillForColumn(Header h, int col, Cursor<String, HeaderCard> iter) throws FitsException {
1786         ColumnDesc colDesc = this.columnList.get(col);
1787 
1788         StringBuffer tform = new StringBuffer();
1789         if (colDesc.isVarying) {
1790             if (colDesc.isLongVary) {
1791                 tform.append("1Q");
1792             } else {
1793                 tform.append("1P");
1794             }
1795         } else {
1796             tform.append(Integer.toString(colDesc.size));
1797         }
1798         if (colDesc.base == int.class) {
1799             tform.append('J');
1800         } else if (colDesc.base == short.class) {
1801             tform.append('I');
1802         } else if (colDesc.base == byte.class) {
1803             tform.append('B');
1804         }  else if (colDesc.base == char.class) {
1805             if (FitsFactory.isUseUnicodeChars()) {
1806                 tform.append('I');
1807                 LOG.warning("char[] will be written as 16-bit integers (type 'I'), not as a FITS character array (type 'A')"
1808                         + " in the binary table. If that is not what you want, you should set FitsFactory.setUseUnicodeChars(false).");
1809                 LOG.warning("Future releases will disable Unicode support by default as it is not FITS standard."
1810                         + " If you do want it still, use FitsFactory.setUseUnicodeChars(true) explicitly to keep the non-standard "
1811                         + " behavior as is.");
1812             } else {
1813                 tform.append('A');
1814             }
1815         } else if (colDesc.base == float.class) {
1816             if (colDesc.isComplex) {
1817                 tform.append('C');
1818             } else {
1819                 tform.append('E');
1820             }
1821         } else if (colDesc.base == double.class) {
1822             if (colDesc.isComplex) {
1823                 tform.append('M');
1824             } else {
1825                 tform.append('D');
1826             }
1827         } else if (colDesc.base == long.class) {
1828             tform.append('K');
1829         } else if (colDesc.base == boolean.class) {
1830             tform.append('L');
1831         } else if (colDesc.base == String.class) {
1832             tform.append('A');
1833         } else {
1834             throw new FitsException("Invalid column data class:" + colDesc.base);
1835         }
1836         IFitsHeader key = TFORMn.n(col + 1);
1837         iter.add(new HeaderCard(key.key(), tform.toString(), key.comment()));
1838 
1839         if (colDesc.dimens.length > 0 && !colDesc.isVarying) {
1840             StringBuffer tdim = new StringBuffer();
1841             char comma = '(';
1842             for (int i = colDesc.dimens.length - 1; i >= 0; i -= 1) {
1843                 tdim.append(comma);
1844                 tdim.append(colDesc.dimens[i]);
1845                 comma = ',';
1846             }            
1847             tdim.append(')');
1848             key = TDIMn.n(col + 1);
1849             iter.add(new HeaderCard(key.key(), tdim.toString(), key.comment()));
1850         }
1851     }

1913     /**
1914      * Convert a column from float/double to float complex/double complex. This
1915      * is only possible for certain columns. The return status indicates if the
1916      * conversion is possible.
1917      *
1918      * @param index
1919      *            The 0-based index of the column to be reset.
1920      * @return Whether the conversion is possible. * 
1921      * 
1922      * @throws FitsException    if the operation failed
1923      */
1924     boolean setComplexColumn(int index) throws FitsException {        
1925         // Currently there is almost no change required to the BinaryTable
1926         // object itself when we convert an eligible column to complex, since
1927         // the internal
1928         // representation of the data is unchanged. We just need
1929         // to set the flag that the column is complex.
1930         // Check that the index is valid,
1931         // the data type is float or double
1932         // the most rapidly changing index in the array has dimension 2.
1933 
1934         if (index < 0 || index >= this.columnList.size()) {
1935             return false;
1936         }
1937         
1938         ColumnDesc colDesc = this.columnList.get(index);
1939         if (colDesc.isComplex) {
1940             return true;
1941         }
1942 
1943         if (colDesc.base != float.class && colDesc.base != double.class) {
1944             return false;
1945         }
1946             
1947         if (colDesc.dimens[colDesc.dimens.length - 1] != 2) { 
1948             return false;
1949         }
1950 
1951         if (!colDesc.isVarying) {
1952             // Set the column to complex
1953             colDesc.isComplex = true;
1954             return true;
1955         }     
1956 
1957         // We need to make sure that for every row, there are
1958         // an even number of elements so that we can
1959         // convert to an integral number of complex numbers.
1960         Object col = getFlattenedColumn(index);
1961         if (col instanceof int[]) {
1962             int[] ptrs = (int[]) col;
1963             for (int i = 1; i < ptrs.length; i += 2) {
1964                 if (ptrs[i] % 2 != 0) {
1965                     return false;
1966                 }
1967             }
1968         } else {
1969             long[] ptrs = (long[]) col;
1970             for (int i = 1; i < ptrs.length; i++) {
1971                 if (ptrs[i] % 2 != 0) {
1972                     return false;
1973                 }
1974             }
1975         }
1976         
1977         // Set the column to complex
1978         colDesc.isComplex = true;
1979         
1980         return true;
1981     }
