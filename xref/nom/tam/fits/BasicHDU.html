71  /**
72   * Abstract base class for all header-data unit (HDU) types. A HDU is a self-contained building block of the FITS files,
73   * which encapsulates information on a particular data object such as an image or table. As the name implies, HDUs
74   * constitute of a header and data entities, which can be accessed separately (via the {@link #getHeader()} and
75   * {@link #getData()} methods respectively). The {@link Header} class provides many functions to add, delete and read
76   * header keywords in HDUs in a variety of formats. The {@link Data} class, and its concrete subclassses provide access
77   * to the specific data object that the HDU encapsulates. It provides basic functionality for an HDU.
78   *
79   * @param <DataClass> the generic type of data contained in this HDU instance.
80   */
81  public abstract class BasicHDU<DataClass extends Data> implements FitsElement {

340     /**
341      * In FITS files the index represented by NAXIS1 is the index that changes most rapidly. This reflectsf the behavior
342      * of Fortran where there are true multidimensional arrays. In Java in a multidimensional array is an array of
343      * arrays and the first index is the index that changes slowest. So at some point a client of the library is going
344      * to have to invert the order. E.g., if I have a FITS file will
345      *
346      * <pre>
347      * BITPIX=16
348      * NAXIS1=10
349      * NAXIS2=20
350      * NAXIS3=30
351      * </pre>
352      *
353      * this will be read into a Java array short[30][20][10] so it makes sense to me at least that the returned
354      * dimensions are 30,20,10
355      *
356      * @return               the dimensions of the axis.
357      *
358      * @throws FitsException if the axis are configured wrong.
359      */
360     public int[] getAxes() throws FitsException {
361         int nAxis = myHeader.getIntValue(NAXIS, 0);
362         if (nAxis < 0) {
363             throw new FitsException("Negative NAXIS value " + nAxis);
364         }
365         if (nAxis > MAX_NAXIS_ALLOWED) {
366             throw new FitsException("NAXIS value " + nAxis + " too large");
367         }
368 
369         if (nAxis == 0) {
370             return null;
371         }
372 
373         int[] axes = new int[nAxis];
374         for (int i = 1; i <= nAxis; i++) {
375             axes[nAxis - i] = myHeader.getIntValue(NAXISn.n(i), 0);
376         }
377 
378         return axes;
379     }

838     /**
839      * Indicate that an HDU is the first element of a FITS file.
840      *
841      * @param  value         value to set
842      *
843      * @throws FitsException if the operation failed
844      */
845     void setPrimaryHDU(boolean value) throws FitsException {
846         if (value && !canBePrimary()) {
847             throw new FitsException("Invalid attempt to make HDU of type:" + this.getClass().getName() + " primary.");
848         }
849 
850         myHeader.setRequiredKeys(value ? null : getCanonicalXtension());
851     }

853     /**
854      * Returns the canonical (expected) value for the XTENSION keywords for this type of HDU. Concrete HDU
855      * implementations should override this method as appropriate. As of FITS version 4, only the following XTENSION
856      * values are recognised: 'IMAGE', 'TABLE', and 'BINTABLE'.
857      *
858      * @return The value to use for the XTENSION keyword.
859      *
860      * @since  1.18
861      */
862     protected String getCanonicalXtension() {
863         // TODO this should become an abstract method for 2.0. Prior to that we provide a default
864         // implementation for API back-compatibility reasons for any 3rd-party HDU implementations.
865         // To warn that this should be ovewritten, we'll log a warning...
866         LOG.warning(getClass().getName() + " should override getCanonicalXtension() method as appropriate.");
867         return "UNKNOWN";
868     }

870     @Override
871     public void write(ArrayDataOutput stream) throws FitsException {
872         if (myHeader == null) {
873             myHeader = new Header();
874         }
875 
876         if (stream instanceof FitsOutput) {
877             boolean isFirst = ((FitsOutput) stream).isAtStart();
878             setPrimaryHDU(canBePrimary() && isFirst);
879         }
880 
881         myHeader.write(stream);
882 
883         if (myData != null) {
884             myData.write(stream);
885         }
886         try {
887             stream.flush();
888         } catch (IOException e) {
889             throw new FitsException("Error flushing at end of HDU", e);
890         }
891     }
