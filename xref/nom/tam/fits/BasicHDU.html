235     /**
236      * In FITS files the index represented by NAXIS1 is the index that changes
237      * most rapidly. This reflects the behavior of Fortran where there are true
238      * multidimensional arrays. In Java in a multidimensional array is an array
239      * of arrays and the first index is the index that changes slowest. So at
240      * some point a client of the library is going to have to invert the order.
241      * E.g., if I have a FITS file will
242      * 
243      * <pre>
244      * BITPIX=16
245      * NAXIS1=10
246      * NAXIS2=20
247      * NAXIS3=30
248      * </pre>
249      * 
250      * this will be read into a Java array short[30][20][10] so it makes sense
251      * to me at least that the returned dimensions are 30,20,10
252      * 
253      * @return the dimensions of the axis.
254      * @throws FitsException
255      *             if the axis are configured wrong.
256      */
257     public int[] getAxes() throws FitsException {
258         int nAxis = this.myHeader.getIntValue(NAXIS, 0);
259         if (nAxis < 0) {
260             throw new FitsException("Negative NAXIS value " + nAxis);
261         }
262         if (nAxis > MAX_NAXIS_ALLOWED) {
263             throw new FitsException("NAXIS value " + nAxis + " too large");
264         }
265 
266         if (nAxis == 0) {
267             return null;
268         }
269 
270         int[] axes = new int[nAxis];
271         for (int i = 1; i <= nAxis; i++) {
272             axes[nAxis - i] = this.myHeader.getIntValue(NAXISn.n(i), 0);
273         }
274 
275         return axes;
276     }

575     /**
576      * Indicate that an HDU is the first element of a FITS file.
577      * 
578      * @param newPrimary
579      *            value to set
580      * @throws FitsException
581      *             if the operation failed
582      */
583     void setPrimaryHDU(boolean newPrimary) throws FitsException {
584 
585         if (newPrimary && !canBePrimary()) {
586             throw new FitsException("Invalid attempt to make HDU of type:" + this.getClass().getName() + " primary.");
587         } 
588         
589         this.isPrimary = newPrimary;
590        
591         // Some FITS readers don't like the PCOUNT and GCOUNT keywords
592         // in a primary array or they EXTEND keyword in extensions.
593 
594         if (this.isPrimary && !this.myHeader.getBooleanValue(GROUPS, false)) {
595             this.myHeader.deleteKey(PCOUNT);
596             this.myHeader.deleteKey(GCOUNT);
597         }
598 
599         if (this.isPrimary) {
600             HeaderCard card = this.myHeader.findCard(EXTEND);
601             if (card == null) {
602                 getAxes(); // Leaves the iterator pointing to the last NAXISn
603                            // card.
604                 this.myHeader.nextCard();
605                 this.myHeader.addValue(EXTEND, true);
606             }
607         }
608 
609         if (!this.isPrimary) {
610 
611             this.myHeader.iterator();
612 
613             int pcount = this.myHeader.getIntValue(PCOUNT, 0);
614             int gcount = this.myHeader.getIntValue(GCOUNT, 1);
615             int naxis = this.myHeader.getIntValue(NAXIS, 0);
616             this.myHeader.deleteKey(EXTEND);
617             HeaderCard pcard = this.myHeader.findCard(PCOUNT);
618             HeaderCard gcard = this.myHeader.findCard(GCOUNT);
619 
620             //this.myHeader.getCard(2 + naxis);
621             this.myHeader.findCard(NAXIS.key() + naxis);
622             
623             if (pcard == null) {
624                 this.myHeader.addValue(PCOUNT, pcount);
625             }
626             if (gcard == null) {
627                 this.myHeader.addValue(GCOUNT, gcount);
628             }
629             this.myHeader.iterator();
630         }
631 
632     }

634     @Override
635     public void write(ArrayDataOutput stream) throws FitsException {
636         if (this.myHeader != null) {
637             this.myHeader.write(stream);
638         }
639         if (this.myData != null) {
640             this.myData.write(stream);
641         }
642         try {
643             stream.flush();
644         } catch (IOException e) {
645             throw new FitsException("Error flushing at end of HDU", e);
646         }
647     }
