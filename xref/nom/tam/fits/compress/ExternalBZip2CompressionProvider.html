34  import java.io.IOException;
35  import java.io.InputStream;
36  import java.util.logging.Logger;
37  
38  import nom.tam.fits.FitsException;
39  
40  public class ExternalBZip2CompressionProvider implements ICompressProvider {
41  
42      private static final int PRIORITY = 10;
43  
44      private static final Logger LOG = Logger.getLogger(ExternalBZip2CompressionProvider.class.getName());
45  
46      private InputStream bunzipper(final InputStream compressed) throws IOException, FitsException {
47          String cmd = getBzip2Cmd();
48          // Allow the user to have already specified the - option.
49          if (cmd.indexOf(" -") < 0) {
50              cmd += " -";
51          }
52          String[] flds = cmd.split(" +");
53          Process proc;
54          try {
55              proc = new ProcessBuilder(flds).start();
56              return new CloseIS(proc, compressed);
57          } catch (Exception e) {
58              ICompressProvider next = CompressionManager.nextCompressionProvider('B', 'Z', this);
59              if (next != null) {
60                  LOG.warning("Error initiating BZIP decompression: " + e.getMessage() + " trieing alternative decompressor");
61                  return next.decompress(compressed);
62              }
63              throw new FitsException("Error initiating BZIP decompression: " + e);
64          }
65      }
66  
67      public String getBzip2Cmd() {
68          return System.getProperty("BZIP_DECOMPRESSOR", System.getenv("BZIP_DECOMPRESSOR"));
69      }
70  
71      @Override
72      public InputStream decompress(InputStream in) throws IOException, FitsException {
73          return bunzipper(in);
74      }
75  
76      @Override
77      public int priority() {
78          return PRIORITY;
79      }
80  
81      @Override
82      public boolean provides(int mag1, int mag2) {
83          return mag1 == 'B' && mag2 == 'Z' && getBzip2Cmd() != null;
84      }
85  }
