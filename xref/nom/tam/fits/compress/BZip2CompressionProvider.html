34  import java.io.IOException;
35  import java.io.InputStream;
36  
37  public class BZip2CompressionProvider implements ICompressProvider {
38  
39      private static final int PRIORITY = 5;
40  
41      @Override
42      public InputStream decompress(InputStream in) throws IOException {
43          try {
44              return CompressionLibLoaderProtection.createBZip2Stream(in);
45          } catch (IOException e) {
46              throw e;
47          } catch (Exception e) {
48              throw new IOException(e);
49          }
50      }
51  
52      @Override
53      public int priority() {
54          return PRIORITY;
55      }
56  
57      @Override
58      public boolean provides(int mag1, int mag2) {
59          return mag1 == 'B' && mag2 == 'Z';
60      }
61  }
