47  public final class CompressionManager {
48  
49      private static final String BZIP2_EXTENTION = ".bz2";
50  
51      private static final String COMPRESS_EXTENTION = ".Z";
52  
53      private static final String GZIP_EXTENTION = ".gz";
54  
55      public static final int ONE_MEGABYTE = 1024 * 1024;
56  
57      /**
58       * logger to log to.
59       */
60      private static final Logger LOG = getLogger(CompressionManager.class);
61  
62      private CompressionManager() {
63      }

65      /**
66       * This method decompresses a compressed input stream. The decompression
67       * method is selected automatically based upon the first two bytes read.
68       * 
69       * @param compressed
70       *            The compressed input stream
71       * @return A stream which wraps the input stream and decompresses it. If the
72       *         input stream is not compressed, a pushback input stream wrapping
73       *         the original stream is returned.
74       * @throws FitsException
75       *             when the stream could not be read or decompressed
76       */
77      public static InputStream decompress(InputStream compressed) throws FitsException {
78  
79          BufferedInputStream pb = new BufferedInputStream(compressed, ONE_MEGABYTE);
80  
81          pb.mark(2);
82          int mag1 = -1;
83          int mag2 = -1;
84  
85          try {
86              mag1 = pb.read();
87              mag2 = pb.read();
88              // Push the data back into the stream
89              pb.reset();
90  
91              ICompressProvider selectedProvider = selectCompressionProvider(mag1, mag2);
92              if (selectedProvider != null) {
93                  return selectedProvider.decompress(pb);
94              } else {
95                  return pb;
96              }
97          } catch (IOException e) {
98              // This is probably a prelude to failure...
99              throw new FitsException("Unable to analyze input stream", e);
100         }
101     }
102 
103     /**
104      * Is a file compressed? (the magic number in the first 2 bytes is used to
105      * detect the compression.
106      * 
107      * @param file
108      *            file to test for compression algorithms
109      * @return true if the file is compressed
110      */
111     public static boolean isCompressed(File file) {
112         InputStream fis = null;
113         try {
114             if (file.exists()) {
115                 fis = new FileInputStream(file);
116                 int mag1 = fis.read();
117                 int mag2 = fis.read();
118                 fis.close();
119                 return selectCompressionProvider(mag1, mag2) != null;
120             }
121 
122         } catch (IOException e) {
123             LOG.log(Level.FINEST, "Error while  checkinf if file " + file + " is compressed", e);
124             // This is probably a prelude to failure...
125             return false;
126 
127         } finally {
128             if (fis != null) {
129                 try {
130                     fis.close();
131                 } catch (IOException e) {
132                     LOG.log(Level.FINEST, "could not close stream", e);
133                 }
134             }
135         }
136         return false;
137     }
138 
139     /**
140      * Is a file compressed? (the magic number in the first 2 bytes is used to
141      * detect the compression.
142      * 
143      * @param filename
144      *            of the file to test for compression algorithms
145      * @return true if the file is compressed
146      */
147     public static boolean isCompressed(String filename) {
148         if (filename == null) {
149             return false;
150         }
151         File test = new File(filename);
152         if (test.exists()) {
153             return isCompressed(test);
154         }
155 
156         int len = filename.length();
157         return len > 2 && (filename.substring(len - GZIP_EXTENTION.length()).equalsIgnoreCase(GZIP_EXTENTION) || //
158                 filename.substring(len - COMPRESS_EXTENTION.length()).equals(COMPRESS_EXTENTION) || //
159                 filename.substring(len - BZIP2_EXTENTION.length()).equals(BZIP2_EXTENTION));
160     }
161 
162     private static ICompressProvider selectCompressionProvider(int mag1, int mag2) {
163         return nextCompressionProvider(mag1, mag2, null);
164     }
165 
166     protected static ICompressProvider nextCompressionProvider(int mag1, int mag2, ICompressProvider old) {
167         ICompressProvider selectedProvider = null;
168         int priority = 0;
169         int maxPriority = Integer.MAX_VALUE;
170         if (old != null) {
171             maxPriority = old.priority();
172         }
173         ServiceLoader<ICompressProvider> compressionProviders = ServiceLoader.load(ICompressProvider.class, Thread.currentThread().getContextClassLoader());
174         for (ICompressProvider provider : compressionProviders) {
175             if (provider.priority() > Math.max(0, priority) && provider.priority() < maxPriority && provider != old && //
176                     provider.provides(mag1, mag2)) {
177                 priority = provider.priority();
178                 selectedProvider = provider;
179             }
180         }
181         return selectedProvider;
182     }
