47  /**
48   * (<i>for internal use</i>) Decompression of compressed FITS files of all supported types (<code>.gz</code>,
49   * <code>.Z</code>, <code>.bz2</code>). It autodetects the type of compression used based on the first 2-bytes of the
50   * compressed input stream. When possible, preference will be given to perform the decompression using a system command
51   * (<code>uncompress</code> or <code>bzip2</code>, which are likely faster for large files). If such a tool is not
52   * available, then the Apache <b>common-compress</b> classes will be used to the same effect.
53   * 
54   * @see GZipCompressionProvider
55   * @see BZip2CompressionProvider
56   * @see ZCompressionProvider
57   */
58  public final class CompressionManager {
59  
60      private static final String BZIP2_EXTENTION = ".bz2";
61  
62      private static final String COMPRESS_EXTENTION = ".Z";
63  
64      private static final String GZIP_EXTENTION = ".gz";
65  
66      /** bytes in a megabyte (MB) */
67      public static final int ONE_MEGABYTE = 1024 * 1024;
68  
69      /**
70       * logger to log to.
71       */
72      private static final Logger LOG = getLogger(CompressionManager.class);
73  
74      private CompressionManager() {
75      }
76  
77      /**
78       * This method decompresses a compressed input stream. The decompression method is selected automatically based upon
79       * the first two bytes read.
80       *
81       * @param  compressed    The compressed input stream
82       *
83       * @return               A stream which wraps the input stream and decompresses it. If the input stream is not
84       *                           compressed, a pushback input stream wrapping the original stream is returned.
85       *
86       * @throws FitsException when the stream could not be read or decompressed
87       */
88      public static InputStream decompress(InputStream compressed) throws FitsException {
89          BufferedInputStream pb = new BufferedInputStream(compressed, ONE_MEGABYTE);
90          pb.mark(2);
91          int mag1 = -1;
92          int mag2 = -1;
93  
94          try {
95              mag1 = pb.read();
96              mag2 = pb.read();
97              // Push the data back into the stream
98              pb.reset();
99              ICompressProvider selectedProvider = selectCompressionProvider(mag1, mag2);
100             if (selectedProvider != null) {
101                 return selectedProvider.decompress(pb);
102             }
103             return pb;
104         } catch (IOException e) {
105             // This is probably a prelude to failure...
106             throw new FitsException("Unable to analyze input stream", e);
107         }
108     }
109 
110     /**
111      * Is a file compressed? (the magic number in the first 2 bytes is used to detect the compression.
112      *
113      * @param  file file to test for compression algorithms
114      *
115      * @return      true if the file is compressed
116      */
117     public static boolean isCompressed(File file) {
118         if (!file.exists()) {
119             return false;
120         }
121 
122         try (InputStream fis = new FileInputStream(file)) {
123             int mag1 = fis.read();
124             int mag2 = fis.read();
125             fis.close();
126             return selectCompressionProvider(mag1, mag2) != null;
127         } catch (IOException e) {
128             LOG.log(Level.FINEST, "Error while checking if file " + file + " is compressed", e);
129         }
130         return false;
131     }
132 
133     /**
134      * <p>
135      * Checks if a file is compressed. If the file by the name exists, it will check the magic number in the first
136      * 2-bytes to see if they matched those of the supported compression algorithms. Otherwise it checks if the file
137      * extension atches one of the standard extensions for supported compressed files (<code>.gz</code>,
138      * <code>.Z</code>, or <code>.bz2</code>).
139      * </p>
140      * <p>
141      * As of 1.18, all file extension are checked in a case insensitive manner
142      * </p>
143      * 
144      * @param  filename of the file to test for compression algorithms
145      *
146      * @return          true if the file is compressed
147      */
148     public static boolean isCompressed(String filename) {
149         if (filename == null) {
150             return false;
151         }
152         File test = new File(filename);
153         if (test.exists()) {
154             return isCompressed(test);
155         }
156 
157         int iExt = filename.lastIndexOf('.');
158         if (iExt < 0) {
159             return false;
160         }
161 
162         String ext = filename.substring(iExt);
163         return ext.equalsIgnoreCase(GZIP_EXTENTION) || ext.equalsIgnoreCase(COMPRESS_EXTENTION)
164                 || ext.equalsIgnoreCase(BZIP2_EXTENTION);
165     }
166 
167     private static ICompressProvider selectCompressionProvider(int mag1, int mag2) {
168         return nextCompressionProvider(mag1, mag2, null);
169     }
170 
171     /**
172      * Returned the next highest priority decompression class, after the one we don't want, for the given type of
173      * compressed file.
174      * 
175      * @param  mag1 the first magic byte at the head of the compressed file
176      * @param  mag2 the second magic byte at the head of the compressed file
177      * @param  old  the last decompression class we tried for this type of file
178      * 
179      * @return      the next lower priority decompression class we might use.
180      */
181     protected static ICompressProvider nextCompressionProvider(int mag1, int mag2, ICompressProvider old) {
182         ICompressProvider selectedProvider = null;
183         int priority = 0;
184         int maxPriority = Integer.MAX_VALUE;
185         if (old != null) {
186             maxPriority = old.priority();
187         }
188         ServiceLoader<ICompressProvider> compressionProviders = ServiceLoader.load(ICompressProvider.class,
189                 Thread.currentThread().getContextClassLoader());
190 
191         for (ICompressProvider provider : compressionProviders) {
192             if (provider.priority() > Math.max(0, priority) && provider.priority() < maxPriority && provider != old && //
193                     provider.provides(mag1, mag2)) {
194                 priority = provider.priority();
195                 selectedProvider = provider;
196             }
197         }
198         return selectedProvider;
199     }
