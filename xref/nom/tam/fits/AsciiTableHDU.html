52  /**
53   * FITS ASCII table header/data unit
54   */
55  public class AsciiTableHDU extends TableHDU<AsciiTable> {
56  
57      /**
58       * The standard column stems for an ASCII table. Note that TBCOL is not
59       * included here -- it needs to be handled specially since it does not
60       * simply shift.
61       */
62      private static final IFitsHeader[] KEY_STEMS = {
63          TFORMn,
64          TZEROn,
65          TNULLn,
66          TTYPEn,
67          TUNITn
68      };
69  
70      /**
71       * Create an ASCII table header/data unit.
72       *
73       * @param h
74       *            the template specifying the ASCII table.
75       * @param d
76       *            the FITS data structure containing the table data.
77       */
78      public AsciiTableHDU(Header h, AsciiTable d) {
79          super(h, d);
80      }
81  
82      /**
83       * @return a ASCII table data structure from an array of objects
84       *         representing the columns.
85       * @param o
86       *            the array of object to create the ASCII table
87       * @throws FitsException
88       *             if the table could not be created.
89       */
90      public static AsciiTable encapsulate(Object o) throws FitsException {
91  
92          Object[] oo = (Object[]) o;
93          AsciiTable d = new AsciiTable();
94          for (Object element : oo) {
95              d.addColumn(element);
96          }
97          return d;
98      }
99  
100     /**
101      * @return true if this data is usable as an ASCII table.
102      * @param o
103      *            object representing the data
104      */
105     public static boolean isData(Object o) {
106 
107         if (o instanceof Object[]) {
108             Object[] oo = (Object[]) o;
109             for (Object element : oo) {
110                 if (!(element instanceof String[]) && //
111                         !(element instanceof int[]) && //
112                         !(element instanceof long[]) && //
113                         !(element instanceof float[]) && //
114                         !(element instanceof double[])) {
115                     return false;
116                 }
117             }
118             return true;
119         }
120         
121         return false;
122     }
123 
124     /**
125      * Check that this is a valid ascii table header.
126      *
127      * @param header
128      *            to validate.
129      * @return <CODE>true</CODE> if this is an ascii table header.
130      */
131     public static boolean isHeader(Header header) {
132         String xtension = header.getStringValue(XTENSION);
133         xtension = xtension == null ? "" : xtension.trim();
134         return "TABLE".equals(xtension);
135     }
136 
137     /**
138      * Create a Data object to correspond to the header description.
139      *
140      * @param hdr
141      *            the header to create the data for
142      * @return An unfilled Data object which can be used to read in the data for
143      *         this HDU.
144      * @throws FitsException
145      *             if the Data object could not be created from this HDU's
146      *             Header
147      */
148     public static Data manufactureData(Header hdr) throws FitsException {
149         return new AsciiTable(hdr);
150     }
151 
152     /**
153      * @return a created header to match the input data.
154      * @param d
155      *            data to create a header for
156      * @throws FitsException
157      *             if the header could not b e created
158      */
159     public static Header manufactureHeader(Data d) throws FitsException {
160         Header hdr = new Header();
161         d.fillHeader(hdr);
162         hdr.iterator();
163         return hdr;
164     }
165 
166     @Override
167     public int addColumn(Object newCol) throws FitsException {
168         Standard.context(AsciiTable.class);
169         this.myData.addColumn(newCol);
170         // Move the iterator to point after all the data describing
171         // the previous column.
172 
173         Cursor<String, HeaderCard> iter = this.myHeader.positionAfterIndex(TBCOLn, this.myData.getNCols());
174 
175         int rowlen = this.myData.addColInfo(getNCols() - 1, iter);
176         int oldRowlen = this.myHeader.getIntValue(NAXIS1);
177         this.myHeader.setNaxis(1, rowlen + oldRowlen);
178 
179         super.addColumn(newCol);
180         Standard.context(null);
181         return getNCols();
182     }
183 
184     @Override
185     protected IFitsHeader[] columnKeyStems() {
186         return KEY_STEMS;
187     }
188 
189     @Override
190     public void info(PrintStream stream) {
191         stream.println("ASCII Table:");
192         stream.println("  Header:");
193         stream.println("    Number of fields:" + this.myHeader.getIntValue(TFIELDS));
194         stream.println("    Number of rows:  " + this.myHeader.getIntValue(NAXIS2));
195         stream.println("    Length of row:   " + this.myHeader.getIntValue(NAXIS1));
196         stream.println("  Data:");
197         Object[] data = (Object[]) getKernel();
198         for (int i = 0; i < getNCols(); i += 1) {
199             stream.println("      " + i + ":" + ArrayFuncs.arrayDescription(data[i]));
200         }
201     }
202 
203     /**
204      * @param row
205      *            row index of the element
206      * @param col
207      *            column index of the element
208      * @return <code>true</code> if an element is null
209      */
210     public boolean isNull(int row, int col) {
211         return this.myData.isNull(row, col);
212     }
213 
214     /**
215      * Mark an entry as null.
216      *
217      * @param row
218      *            row index of the element
219      * @param col
220      *            column index of the element
221      * @param flag
222      *            set to null or not
223      */
224     public void setNull(int row, int col, boolean flag) {
225 
226         if (flag) {
227             String nullStr = this.myHeader.getStringValue(TNULLn.n(col + 1));
228             if (nullStr == null) {
229                 setNullString(col, "NULL");
230             }
231         }
232         this.myData.setNull(row, col, flag);
233     }
234 
235     /**
236      * Set the null string for a column.
237      *
238      * @param col
239      *            the column index
240      * @param newNull
241      *            the String representing null
242      * @throws IllegalArgumentException     
243      *            if the string argument contains characters that are not allowed in FITS
244      *            headers. That is if it contains characters outside the range of 0x20
245      *            thru 0x7E.
246      */
247     public void setNullString(int col, String newNull) throws IllegalArgumentException {
248         this.myHeader.positionAfterIndex(TBCOLn, col + 1);
249         HeaderCard card = HeaderCard.create(TNULLn.n(col + 1), newNull);
250         this.myHeader.deleteKey(card.getKey());
251         this.myHeader.addLine(card);
252         this.myData.setNullString(col, newNull);
253     }
