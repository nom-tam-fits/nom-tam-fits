34  import java.io.IOException;
35  
36  import nom.tam.util.ArrayDataInput;
37  
38  /**
39   * 
40   * <p>
41   * A helper class to keep track of the number of physical cards for a logical
42   * card.
43   * </p>
44   * 
45   * <p>
46   * This class should not have public visibility. And really, the counting should be
47   * completely internalized by HeaderCard. Perhaps remove in a future major release.
48   * </p>
49   * 
50   * @author Richard van Nieuwenhoven
51   */
52  @Deprecated
53  public class HeaderCardCountingArrayDataInput {
54  
55      /**
56       * the input stream.
57       */
58      private final ArrayDataInput input;
59  
60      /**
61       * the number of 80 byte cards read.
62       */
63      private int physicalCardsRead;
64  
65      private int markedPhysicalCardsRead;
66  
67      protected HeaderCardCountingArrayDataInput(ArrayDataInput input) {
68          this.input = input;
69      }
70  
71      /**
72       * @return the number of cards realy read form the stream
73       */
74      protected int getPhysicalCardsRead() {
75          return physicalCardsRead;
76      }
77  
78      /**
79       * @return the stream to read the cards from
80       */
81      protected ArrayDataInput in() {
82          return input;
83      }
84  
85      /**
86       * report a readed card.
87       */
88      public void cardRead() {
89          physicalCardsRead++;
90      }
91  
92      /**
93       * indicate whether mark/reset functionality is supported
94       * 
95       * @return true iff mark/reset will work
96       */
97      public boolean markSupported() {
98          return input.markSupported();
99      }
100 
101     /**
102      * mark the current position in the stream.
103      * 
104      * @throws IOException
105      *             if the underlaying stream does not allow the mark.
106      */
107     public void mark() throws IOException {
108         input.mark(HeaderCard.FITS_HEADER_CARD_SIZE);
109         markedPhysicalCardsRead = physicalCardsRead;
110     }
111 
112     /**
113      * reset the stream th the last marked prosition.
114      * 
115      * @throws IOException
116      *             if the underlaying stream does not allow the mark.
117      */
118     public void reset() throws IOException {
119         input.reset();
120         physicalCardsRead = markedPhysicalCardsRead;
121     }
122 
123 }
