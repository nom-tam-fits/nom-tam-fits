36  import java.io.IOException;
37  import java.util.logging.Level;
38  import java.util.logging.Logger;
39  
40  import nom.tam.util.ArrayDataInput;
41  import nom.tam.util.ArrayDataOutput;
42  import nom.tam.util.RandomAccess;
43  
44  /**
45   * This class provides methods to access the data segment of an HDU.
46   * <p>
47   * This is the object which contains the actual data for the HDU.
48   * </p>
49   * <ul>
50   * <li>For images and primary data this is a simple (but possibly
51   * multi-dimensional) primitive array. When group data is supported it will be a
52   * possibly multidimensional array of group objects.
53   * <li>For ASCII data it is a two dimensional Object array where each of the
54   * constituent objects is a primitive array of length 1.
55   * <li>For Binary data it is a two dimensional Object array where each of the
56   * constituent objects is a primitive array of arbitrary (more or less)
57   * dimensionality.
58   * </ul>
59   */
60  public abstract class Data implements FitsElement {
61  
62      private static final Logger LOG = getLogger(Data.class);
63  
64      private static final int FITS_BLOCK_SIZE_MINUS_ONE = FitsFactory.FITS_BLOCK_SIZE - 1;
65  
66      /**
67       * The starting location of the data when last read
68       */
69      protected long fileOffset = -1;
70  
71      /**
72       * The size of the data when last read
73       */
74      protected long dataSize;
75  
76      /**
77       * The input stream used.
78       */
79      protected RandomAccess input;
80  
81      /**
82       * Modify a header to point to this data, this differs per subclass, they
83       * all need oder provided different informations to the header. Basically
84       * they describe the structure of this data object.
85       * 
86       * @param head
87       *            header to fill with the data from the current data object
88       * @throws FitsException
89       *             if the operation fails
90       */
91      abstract void fillHeader(Header head) throws FitsException;
92  
93      /**
94       * @return the data array object.
95       * @throws FitsException
96       *             if the data could not be gathered .
97       */
98      public abstract Object getData() throws FitsException;
99  
100     /**
101      * @return the file offset
102      */
103     @Override
104     public long getFileOffset() {
105         return this.fileOffset;
106     }
107 
108     /**
109      * @return the non-FITS data object.
110      * @throws FitsException
111      *             if the data could not be gathered .
112      */
113     public Object getKernel() throws FitsException {
114         return getData();
115     }
116 
117     /**
118      * @return the size of the data element in bytes.
119      */
120     @Override
121     public long getSize() {
122         return FitsUtil.addPadding(getTrueSize());
123     }
124 
125     abstract long getTrueSize();
126 
127     @Override
128     public abstract void read(ArrayDataInput in) throws FitsException;
129 
130     @Override
131     public boolean reset() {
132         try {
133             FitsUtil.reposition(this.input, this.fileOffset);
134             return true;
135         } catch (Exception e) {
136             LOG.log(Level.SEVERE, "Unable to reset", e);
137             return false;
138         }
139     }
140 
141     @Override
142     public void rewrite() throws FitsException {
143 
144         if (!rewriteable()) {
145             throw new FitsException("Illegal attempt to rewrite data");
146         }
147 
148         FitsUtil.reposition(this.input, this.fileOffset);
149         write((ArrayDataOutput) this.input);
150         try {
151             ((ArrayDataOutput) this.input).flush();
152         } catch (IOException e) {
153             throw new FitsException("Error in rewrite flush: ", e);
154         }
155     }
156 
157     @Override
158     public boolean rewriteable() {
159         return this.input != null && this.fileOffset >= 0
160                 && (getTrueSize() + FITS_BLOCK_SIZE_MINUS_ONE) / FitsFactory.FITS_BLOCK_SIZE == (this.dataSize + FITS_BLOCK_SIZE_MINUS_ONE) / FitsFactory.FITS_BLOCK_SIZE;
161     }
162 
163     /**
164      * Set the fields needed for a re-read.
165      * 
166      * @param o
167      *            reread information.
168      */
169     protected void setFileOffset(ArrayDataInput o) {
170         if (o instanceof RandomAccess) {
171             this.fileOffset = FitsUtil.findOffset(o);
172             this.dataSize = getTrueSize();
173             this.input = (RandomAccess) o;
174         }
175     }
176 
177     /**
178      * Write the data -- including any buffering needed
179      * 
180      * @param o
181      *            The output stream on which to write the data.
182      */
183     @Override
184     public abstract void write(ArrayDataOutput o) throws FitsException;
185 }
