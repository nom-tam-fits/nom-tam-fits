34  import java.io.IOException;
35  
36  import nom.tam.util.ArrayDataInput;
37  import nom.tam.util.ArrayDataOutput;
38  import nom.tam.util.RandomAccess;
39  
40  /**
41   * This class provides methods to access the data segment of an HDU.
42   */
43  public abstract class Data implements FitsElement {
44  
45      /**
46       * This is the object which contains the actual data for the HDU.
47       * <ul>
48       * <li>For images and primary data this is a simple (but possibly
49       * multi-dimensional) primitive array. When group data is supported it will
50       * be a possibly multidimensional array of group objects.
51       * <li>For ASCII data it is a two dimensional Object array where each of the
52       * constituent objects is a primitive array of length 1.
53       * <li>For Binary data it is a two dimensional Object array where each of
54       * the constituent objects is a primitive array of arbitrary (more or less)
55       * dimensionality.
56       * </ul>
57       */
58      /** The starting location of the data when last read */
59      protected long fileOffset = -1;
60  
61      /** The size of the data when last read */
62      protected long dataSize;
63  
64      /** The inputstream used. */
65      protected RandomAccess input;
66  
67      /** Get the file offset */
68      public long getFileOffset() {
69          return fileOffset;
70      }
71  
72      /** Set the fields needed for a re-read */
73      protected void setFileOffset(Object o) {
74          if (o instanceof RandomAccess) {
75              fileOffset = FitsUtil.findOffset(o);
76              dataSize = getTrueSize();
77              input = (RandomAccess) o;
78          }
79      }
80  
81      /**
82       * Write the data -- including any buffering needed
83       * 
84       * @param o
85       *            The output stream on which to write the data.
86       */
87      public abstract void write(ArrayDataOutput o) throws FitsException;
88  
89      /**
90       * Read a data array into the current object and if needed position to the
91       * beginning of the next FITS block.
92       * 
93       * @param i
94       *            The input data stream
95       */
96      public abstract void read(ArrayDataInput i) throws FitsException;
97  
98      public void rewrite() throws FitsException {
99  
100         if (!rewriteable()) {
101             throw new FitsException("Illegal attempt to rewrite data");
102         }
103 
104         FitsUtil.reposition(input, fileOffset);
105         write((ArrayDataOutput) input);
106         try {
107             ((ArrayDataOutput) input).flush();
108         } catch (IOException e) {
109             throw new FitsException("Error in rewrite flush: " + e);
110         }
111     }
112 
113     public boolean reset() {
114         try {
115             FitsUtil.reposition(input, fileOffset);
116             return true;
117         } catch (Exception e) {
118             return false;
119         }
120     }
121 
122     public boolean rewriteable() {
123         if (input == null || fileOffset < 0 || (getTrueSize() + 2879) / 2880 != (dataSize + 2879) / 2880) {
124             return false;
125         } else {
126             return true;
127         }
128     }
129 
130     abstract long getTrueSize();
131 
132     /** Get the size of the data element in bytes */
133     public long getSize() {
134         return FitsUtil.addPadding(getTrueSize());
135     }
136 
137     /**
138      * Return the data array object.
139      */
140     public abstract Object getData() throws FitsException;
141 
142     /** Return the non-FITS data object */
143     public Object getKernel() throws FitsException {
144         return getData();
145     }
146 
147     /**
148      * Modify a header to point to this data
149      */
150     abstract void fillHeader(Header head) throws FitsException;
151 }
