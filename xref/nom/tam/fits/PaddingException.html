32  /**
33   * This exception is thrown if an error is found reading the padding following a
34   * valid FITS HDU. This padding is required by the FITS standard, but some FITS
35   * writes forego writing it. To access such data users can use something like:
36   * <code>
37   *     Fits f = new Fits("somefile");
38   *     try {
39   *          f.read();
40   *     } catch (PaddingException e) {
41   *          f.addHDU(e.getHDU());
42   *     }
43   * </code> to ensure that a truncated HDU is included in the FITS object.
44   * Generally the FITS file have already added any HDUs prior to the truncatd
45   * one.
46   */
47  public class PaddingException extends FitsException {
48  
49      /**
50       * The HDU where the error happened.
51       */
52      private BasicHDU truncatedHDU;
53  
54      /**
55       * When the error is thrown, the data object being read must be supplied. We
56       * initially create a dummy header for this. If someone is reading the
57       * entire HDU, then they can trap the exception and set the header to the
58       * appropriate value.
59       */
60      public PaddingException(Data datum) throws FitsException {
61          truncatedHDU = FitsFactory.HDUFactory(datum.getKernel());
62          // We want to use the original Data object... so
63          truncatedHDU = FitsFactory.HDUFactory(truncatedHDU.getHeader(), datum);
64      }
65  
66      public PaddingException(String msg, Data datum) throws FitsException {
67          super(msg);
68          truncatedHDU = FitsFactory.HDUFactory(datum.getKernel());
69          truncatedHDU = FitsFactory.HDUFactory(truncatedHDU.getHeader(), datum);
70      }
71  
72      void updateHeader(Header hdr) throws FitsException {
73          truncatedHDU = FitsFactory.HDUFactory(hdr, truncatedHDU.getData());
74      }
75  
76      public BasicHDU getTruncatedHDU() {
77          return truncatedHDU;
78      }
79  }
