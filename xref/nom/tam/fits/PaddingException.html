34  /**
35   * This exception is thrown if an error is found reading the padding following a
36   * valid FITS HDU. This padding is required by the FITS standard, but some FITS
37   * writes forego writing it. To access such data users can use something like:
38   * <code>
39   *     Fits f = new Fits("somefile");
40   *     try {
41   *          f.read();
42   *     } catch (PaddingException e) {
43   *          f.addHDU(e.getHDU());
44   *     }
45   * </code> to ensure that a truncated HDU is included in the FITS object.
46   * Generally the FITS file have already added any HDUs prior to the truncatd
47   * one.
48   */
49  public class PaddingException extends FitsException {
50  
51      /**
52       * serial version id.
53       */
54      private static final long serialVersionUID = 1L;
55  
56      /**
57       * The HDU where the error happened.
58       */
59      private BasicHDU<?> truncatedHDU;
60  
61      /**
62       * When the error is thrown, the data object being read must be supplied. We
63       * initially create a dummy header for this. If someone is reading the
64       * entire HDU, then they can trap the exception and set the header to the
65       * appropriate value.
66       * 
67       * @param data
68       *            the data that was not padded.
69       * @throws FitsException
70       *             if the data could not be used to craete a hdu.
71       */
72      public PaddingException(Data data) throws FitsException {
73          super("data not padded");
74          this.truncatedHDU = FitsFactory.hduFactory(data.getKernel());
75          // We want to use the original Data object... so
76          this.truncatedHDU = FitsFactory.hduFactory(this.truncatedHDU.getHeader(), data);
77      }
78  
79      public PaddingException(String msg, Data data) throws FitsException {
80          super(msg);
81          this.truncatedHDU = FitsFactory.hduFactory(data.getKernel());
82          this.truncatedHDU = FitsFactory.hduFactory(this.truncatedHDU.getHeader(), data);
83      }
84  
85      public BasicHDU<?> getTruncatedHDU() {
86          return this.truncatedHDU;
87      }
88  
89      void updateHeader(Header hdr) throws FitsException {
90          this.truncatedHDU = FitsFactory.hduFactory(hdr, this.truncatedHDU.getData());
91      }
92  }
