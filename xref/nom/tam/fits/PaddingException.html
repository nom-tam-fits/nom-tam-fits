34  /**
35   * This exception is thrown if an error is found reading the padding following a
36   * valid FITS HDU. This padding is required by the FITS standard, but some FITS
37   * writes forego writing it. To access such data users can use something like:
38   * <code>
39   *     Fits f = new Fits("somefile");
40   *     try {
41   *          f.read();
42   *     } catch (PaddingException e) {
43   *          f.addHDU(e.getHDU());
44   *     }
45   * </code> to ensure that a truncated HDU is included in the FITS object.
46   * Generally the FITS file have already added any HDUs prior to the truncated
47   * one.
48   */
49  public class PaddingException extends FitsException {
50  
51  
52      /**
53       * 
54       */
55      private static final long serialVersionUID = 8716484905278318366L;
56      /**
57       * The HDU where the error happened.
58       */
59      private BasicHDU<?> truncatedHDU;
60  
61      public PaddingException(String msg, Data data, Exception cause) throws FitsException {
62          super(msg, cause);
63          try {
64              this.truncatedHDU = FitsFactory.hduFactory(data.getKernel());
65              //this.truncatedHDU = FitsFactory.hduFactory(this.truncatedHDU.getHeader(), data);
66          } catch (Exception e) {
67              // TODO nothing to do
68          }
69      }
70  
71      public BasicHDU<?> getTruncatedHDU() {
72          return this.truncatedHDU;
73      }
74  
75      void updateHeader(Header hdr) throws FitsException {
76          this.truncatedHDU = FitsFactory.hduFactory(hdr, this.truncatedHDU.getData());
77      }
78  }
