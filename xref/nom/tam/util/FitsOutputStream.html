31  package nom.tam.util;
32  
33  import java.io.DataOutput;
34  import java.io.DataOutputStream;
35  import java.io.IOException;
36  import java.io.OutputStream;
37  
38  
39  /**
40   * This class is intended for high performance writing FITS files or 
41   * FITS data blocks.
42   *
43   * <p>
44   * Testing and timing for this class is performed in the
45   * nom.tam.util.test.BufferedFileTester class.
46   * 
47   * <p>
48   * Version 2.0 -- October 30, 2021: Completely overhauled, with new name and 
49   * hierarchy. Performance is 2-4 times better than before. (Attila Kovacs)
50   * 
51   * @see FitsInputStream
52   * @see FitsFile
53   */
54  @SuppressWarnings("deprecation")
55  public class FitsOutputStream extends ArrayOutputStream implements ArrayDataOutput {
56  
57      /** the output, as accessible via the <code>DataInput</code> interface */
58      private final DataOutput data;
59      
60      /**
61       * Use the BufferedOutputStream constructor
62       * 
63       * @param o
64       *            An open output stream.
65       */
66      public FitsOutputStream(OutputStream o) {
67          this(o, FitsIO.DEFAULT_BUFFER_SIZE);
68      }
69  
70      /**
71       * Use the BufferedOutputStream constructor
72       * 
73       * @param o
74       *            An open output stream.
75       * @param bufLength
76       *            The buffer size.
77       */
78      public FitsOutputStream(OutputStream o, int bufLength) {
79          super(o, bufLength);
80          setEncoder(new FitsEncoder(this));
81          if (o instanceof DataOutput) {
82              data = (DataOutput) o;
83          } else {
84              data = new DataOutputStream(o);
85          }
86      }
87      
88      @Override
89      protected FitsEncoder getEncoder() {
90          return (FitsEncoder) super.getEncoder();
91      }
92  
93      @Override
94      public synchronized void write(boolean[] b, int start, int length) throws IOException {
95          getEncoder().write(b, start, length);
96      }
97  
98      @Override
99      public synchronized void write(Boolean[] buf, int offset, int size) throws IOException {
100         getEncoder().write(buf, offset, size);
101     }
102 
103     @Override
104     public synchronized void write(char[] c, int start, int length) throws IOException {
105         getEncoder().write(c, start, length);
106     }
107 
108     @Override
109     public synchronized void write(short[] s, int start, int length) throws IOException {
110         getEncoder().write(s, start, length);
111     }
112 
113     @Override
114     public synchronized void write(int[] i, int start, int length) throws IOException {
115         getEncoder().write(i, start, length);
116     }
117 
118     @Override
119     public synchronized void write(long[] l, int start, int length) throws IOException {
120         getEncoder().write(l, start, length);
121     }
122 
123     @Override
124     public synchronized void write(float[] f, int start, int length) throws IOException {
125         getEncoder().write(f, start, length);
126     }
127 
128     @Override
129     public synchronized void write(double[] d, int start, int length) throws IOException {
130         getEncoder().write(d, start, length);
131     }
132 
133     @Override
134     public synchronized void writeBytes(String s) throws IOException {
135         getEncoder().writeBytes(s);
136     }
137 
138     @Override
139     public synchronized void writeChars(String s) throws IOException {
140         getEncoder().writeChars(s);
141     }
142 
143     @Override
144     public synchronized void writeUTF(String s) throws IOException {
145         data.writeUTF(s);
146     }
147 
148     @Override
149     public synchronized void write(String[] s, int start, int len) throws IOException {
150         getEncoder().write(s, start, len);
151     }
152 
153     @Override
154     public synchronized void writeByte(int b) throws IOException {
155         getEncoder().writeByte(b);        
156     }
157 
158     @Override
159     public synchronized void writeBoolean(boolean b) throws IOException {
160         getEncoder().writeBoolean(b);
161     }
162     
163     @Override
164     public synchronized void writeChar(int c) throws IOException {
165         getEncoder().writeChar(c);
166     }
167 
168     @Override
169     public synchronized void writeShort(int s) throws IOException {
170         getEncoder().writeShort(s);
171     }
172 
173     @Override
174     public synchronized void writeInt(int i) throws IOException {
175         getEncoder().writeInt(i);
176     }
177 
178     @Override
179     public synchronized void writeLong(long l) throws IOException {
180         getEncoder().writeLong(l);
181     }
182 
183     @Override
184     public synchronized void writeFloat(float f) throws IOException {
185         getEncoder().writeFloat(f);
186     }
187 
188     @Override
189     public synchronized void writeDouble(double d) throws IOException {
190         getEncoder().writeDouble(d);
191     }
192 
193     /**
194      * Deprecated use {@link #writeArray(Object)}.
195      * 
196      * @param o
197      *            The object to be written.
198      * @throws IOException
199      *             if one of the underlying write operations failed
200      * @deprecated use {@link #writeArray(Object)} instead
201      */
202     @Deprecated
203     public final synchronized void writePrimitiveArray(Object o) throws IOException {
204         writeArray(o);
205     }
206 }
