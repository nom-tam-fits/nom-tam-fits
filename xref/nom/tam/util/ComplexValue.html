32  package nom.tam.util;
33  
34  import java.util.StringTokenizer;
35  import java.util.logging.Logger;
36  
37  import nom.tam.fits.FitsFactory;
38  import nom.tam.fits.LongValueException;
39  
40  /**
41   * <p>
42   * A no-frills complex value, for representing complex numbers in FITS headers.
43   * It is a non-mutable object that is created with a real and imaginary parts, which can be
44   * retrieved thereafter, and provides string formatting that is suited specifically for
45   * representation in FITS headers.
46   * </p>
47   * 
48   * <p>
49   * Note that binary tables handle complex data differently, with elements of `float[2]` or
50   * `double[2]`.
51   * </p>
52   * 
53   * @author Attila Kovacs
54   *
55   * @since 1.16
56   * 
57   */
58  public class ComplexValue {
59      
60      private static final Logger LOG = Logger.getLogger(ComplexValue.class.getName());
61  
62      /** The complex zero **/
63      public static final ComplexValue ZERO = new ComplexValue(0.0, 0.0);
64      
65      /** The complex unity along the real axis, or (1.0, 0.0) **/
66      public static final ComplexValue ONE = new ComplexValue(1.0, 0.0);
67      
68      /** The unity along the imaginary axis <i>i</i>, or (0.0, 1.0) **/
69      public static final ComplexValue I = new ComplexValue(0.0, 1.0);
70      
71      /** The real and imaginary parts */
72      private double re, im;
73      
74      /** 
75       * The minimum size string needed to represent a complex value with 
76       * even just single digits for the real and imaginary parts. 
77       */
78      private static final int MIN_STRING_LENGTH = 5;     // "(#,#)"
79      
80      /** 
81       * Instantiates a new complex number value with the specified real and imaginary components.
82       * 
83       * @param re    the real part
84       * @param im    thei maginary part
85       */
86      public ComplexValue(double re, double im) {
87          this.re = re;
88          this.im = im;
89      }
90      
91      /**
92       * Returns the real part of this complex value.
93       * 
94       * @return      the real part
95       * 
96       * @see #im()
97       */
98      public final double re() {
99          return re;
100     }
101 
102     /**
103      * Returns the imaginary part of this complex value.
104      * 
105      * @return      the imaginary part
106      * 
107      * @see #re()
108      */
109     public final double im() {
110         return im;
111     }
112     
113     @Override
114     public int hashCode() {
115         return Double.hashCode(re()) ^ Double.hashCode(im());
116     }
117     
118     @Override
119     public boolean equals(Object o) {
120         if (o == this) {
121             return true;
122         }
123         
124         if (!(o instanceof ComplexValue)) {
125             return false;
126         }
127         
128         ComplexValue z = (ComplexValue) o;
129         return z.re() == re() && z.im() == im();        
130     }
131     
132     /**
133      * Checks if the complex value is zero. That is, if both the real or imaginary parts
134      * are zero.
135      * 
136      * @return      <code>true</code>if both the real or imaginary parts are zero.
137      *              Otherwise <code>false</code>.
138      */
139     public final boolean isZero() {
140         return re() == 0.0 && im() == 0.0;
141     }
142     
143     /**
144      * Checks if the complex value is finite. That is, if neither the real or imaginary parts
145      * are NaN or Infinite.
146      * 
147      * @return      <code>true</code>if neither the real or imaginary parts are NaN or Infinite.
148      *              Otherwise <code>false</code>.
149      */
150     public final boolean isFinite() {
151         return Double.isFinite(re()) && Double.isFinite(im());
152     }
153     
154     @Override
155     public String toString() {
156         return "(" + re() + "," + im() + ")";
157     }
158     
159     /**
160      * Converts this complex value to its string representation with up to the specified 
161      * number of decimal places showing after the leading figure, for both the
162      * real and imaginary parts. 
163      * 
164      * @param decimals  the maximum number of decimal places to show.
165      * @return          the string representation with the specified precision, which
166      *                  may be used in a FITS header.
167      *                  
168      * @see FlexFormat
169      */
170     public String toString(int decimals) {
171         FlexFormat f = new FlexFormat().setPrecision(decimals); 
172         return "(" + f.format(re()) + "," + f.format(im()) + ")";
173     }
174     
175 
176     /**
177      * <p>
178      * Instantiates a new complex number value from the string repressentation of it in
179      * a FITS header value. By default, it will parse complex numbers as a comma-separated
180      * pair of real values enclosed in a bracket, such as <code>(1.0, -2.0)</code>, or
181      * standard real values, such as <code>123.456</code> or <code>123</code> (as real-only
182      * values). There can be any number of spaces around the brackets, number components or
183      * the comma.
184      * </p>
185      * <p>
186      * If {@link FitsFactory#setAllowHeaderRepairs(boolean)} is set <code>true</code>, the
187      * parsing becomes more tolerant, working around missing closing brackets, different
188      * number of comma-separated components, and missing empty components. So, for example
189      * <code>(,-1,abc</code> may be parsed assuming it was meant to be -<i>i</i>. 
190      * </p>
191      * 
192      * @param text      The FITS header value representing the complex number, in brackets
193      *                  with the real and imaginary pars separated by a comma. Additional
194      *                  spaces may surround the component parts.
195      * @throws IllegalArgumentException     
196      *                  if the supplied string does not appear to be a FITS standard
197      *                  representation of a complex value.
198      *                  
199      * @see FitsFactory#setAllowHeaderRepairs(boolean)
200      */
201     public ComplexValue(String text) throws IllegalArgumentException {        
202         // Allow the use of 'D' or 'd' to mark the exponent, instead of the standard 'E' or 'e'...
203         text = text.trim().toUpperCase().replace('D', 'E');
204         
205         boolean hasOpeningBracket = text.charAt(0) == '(';
206         boolean hasClosingBracket = text.charAt(text.length() - 1) == ')';
207         
208         if (!(hasOpeningBracket || hasClosingBracket)) {
209             // Use just the real value.
210             re = Double.parseDouble(text);
211             return;
212         } else if (!hasOpeningBracket || !hasClosingBracket) {
213             if (!FitsFactory.isAllowHeaderRepairs()) {
214                 throw new IllegalArgumentException("Missing bracket around complex value: '" + text 
215                         + "'\n\n --> Try FitsFactory.setAllowHeaderRepair(true).\n");
216             }
217             LOG.warning("Ignored missing bracket in '" + text + "'.");
218         }
219         
220         int start = hasOpeningBracket ? 1 : 0;
221         int end = hasClosingBracket ? text.length() - 1 : text.length();
222         StringTokenizer tokens = new StringTokenizer(text.substring(start, end), FitsFactory.isAllowHeaderRepairs() ? ",; \t" : ", ");
223         if (tokens.countTokens() != 2) {
224             if (!FitsFactory.isAllowHeaderRepairs()) {
225                 throw new IllegalArgumentException("Invalid complex value: '" + text 
226                         + "'\n\n --> Try FitsFactory.setAllowHeaderRepair(true).\n");
227             }
228             LOG.warning("Ignored wrong number of components (" + tokens.countTokens() + ") in '" + text + "'.");
229         }
230         
231         if (tokens.hasMoreTokens()) {
232             re = Double.parseDouble(tokens.nextToken());
233         }
234         if (tokens.hasMoreTokens()) {
235             im = Double.parseDouble(tokens.nextToken());
236         }
237     }
238     
239     
240     /**
241      * Converts this comlex value to its string representation using up to the
242      * specified number of characters only. The precision may be reduced as
243      * necessary to ensure that the representation fits in the allotted space.
244      * 
245      * @param maxLength     the maximum length of the returned string representation
246      * @return              the string representation, possibly with reduced
247      *                      precision to fit into the alotted space.
248      * @throws LongValueException
249      *                      if the space was too short to fit the value even
250      *                      with the minimal (1-digit) precision.
251      */
252     public String toBoundedString(int maxLength) throws LongValueException {
253         if (maxLength < MIN_STRING_LENGTH) {
254             throw new LongValueException(maxLength, toString());
255         }
256         
257         String s = toString();
258         if (s.length() <= maxLength) {
259             return s;
260         }
261         
262         int decimals = FlexFormat.DOUBLE_DECIMALS;
263         
264         s = toString(decimals);
265         while (s.length() > maxLength) {
266             // Assume both real and imaginary parts shorten the same amount... 
267             decimals -= (s.length() - maxLength + 1) / 2;
268             
269             if (decimals < 0) {
270                 throw new LongValueException(maxLength, toString());
271             }
272             s = toString(decimals);
273         }
274         
275         return s;
276     }
