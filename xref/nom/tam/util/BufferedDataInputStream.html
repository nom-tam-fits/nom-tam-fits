41  /**
42   * This class is intended for high performance I/O in scientific applications.
43   * It combines the functionality of the BufferedInputStream and the
44   * DataInputStream as well as more efficient handling of arrays. This minimizes
45   * the number of method calls that are required to read data. Informal tests of
46   * this method show that it can be as much as 10 times faster than using a
47   * DataInputStream layered on a BufferedInputStream for writing large arrays.
48   * The performance gain on scalars or small arrays will be less but there should
49   * probably never be substantial degradation of performance.
50   * <p>
51   * Many new read calls are added to allow efficient reading off array data. The
52   * read(Object o) call provides for reading a primitive array of arbitrary type
53   * or dimensionality. There are also reads for each type of one dimensional
54   * array.
55   * <p>
56   * Note that there is substantial duplication of code to minimize method
57   * invocations. E.g., the floating point read routines read the data as integer
58   * values and then convert to float. However the integer code is duplicated
59   * rather than invoked. There has been considerable effort expended to ensure
60   * that these routines are efficient, but they could easily be superceded if an
61   * efficient underlying I/O package were ever delivered as part of the basic
62   * Java libraries. [This has subsequently happened with the NIO package and in
63   * an ideal universe these classes would be rewritten to take advantage of NIO.]
64   * <p>
65   * Testing and timing routines are provided in the
66   * nom.tam.util.test.BufferedFileTester class. Version 1.1: October 12, 2000:
67   * Fixed handling of EOF to return partially read arrays when EOF is detected.
68   * Version 1.2: July 20, 2009: Added handling of very large Object arrays.
69   * Additional work is required to handle very large arrays generally.
70   */

133     /**
134      * For array reads return an EOF if unable to read any data.
135      * 
136      * @param e
137      *            the eof exception that happened.
138      * @param i
139      *            the current index
140      * @param start
141      *            the start index
142      * @param length
143      *            the element length
144      * @return the number of bytes read before the end of file exception.
145      * @throws EOFException
146      *             if no extra bytes could be read
147      */
148     private int eofCheck(EOFException e, int i, int start, int length) throws EOFException {
149 
150         if (i == start) {
151             throw e;
152         } else {
153             return (i - start) * length;
154         }
155     }

167     @Override
168     public int read(byte[] obuf, int offset, int length) throws IOException {
169         int total = 0;
170         while (length > 0) {
171             // Use just the buffered I/O to get needed info.
172             int xlen = super.read(obuf, offset, length);
173             if (xlen <= 0) {
174                 if (total == 0) {
175                     throw new EOFException();
176                 } else {
177                     return total;
178                 }
179             } else {
180                 length -= xlen;
181                 total += xlen;
182                 offset += xlen;
183             }
184         }
185         return total;
186     }

264     private void readBytesIntoSharedBuffer(int bytes) throws IOException, EOFException {
265         this.sharedBuffer.invalidate();
266         if (read(this.sharedBuffer.buffer, 0, bytes) < bytes) {
267             throw new EOFException();
268         }

306     /**
307      * Emulate the deprecated DataInputStream.readLine() method. Originally we
308      * used the method itself, but Alan Brighton suggested using a
309      * java.io.BufferedReader to eliminate the deprecation warning. This was
310      * used for a long time, but more recently we noted that this doesn't work.
311      * We now use a simple method that largely ignores character encodings and
312      * only uses the "\n" as the line separator. This method is slow regardless.
313      * In the current version
314      * 
315      * @return The String read.
316      * @deprecated Use {@link java.io.BufferedReader} methods.
317      */
318     @Deprecated
319     @Override
320     public String readLine() throws IOException {
321         // Punt on this and use BufferedReader routines.
322         StringBuilder b = new StringBuilder("");
323         int chr;
324         while ((chr = read()) >= 0) {
325             if (chr != '\n') {
326                 b.append((char) chr);
327             } else {
328                 return b.toString();
329             }
330         }
331         return b.toString();
332     }

384     @Override
385     public void skipAllBytes(long toSkip) throws IOException {
386         long need = toSkip;
387         while (need > 0) {
388             try {
389                 long got = skip(need);
390                 if (got > 0) {
391                     need -= got;
392                 } else {
393                     break;
394                 }
395             } catch (IOException e) {
396                 // Some input streams (process outputs) don't allow
397                 // skipping. The kludgy solution here is to
398                 // try to do a read when we get an error in the skip....
399                 // Real IO errors will presumably casue an error
400                 // in these reads too.
401                 if (this.skipBuf == null) {
402                     this.skipBuf = new byte[BufferedDataInputStream.SKIP_BUFFER_SIZE];
403                 }
404                 while (need > BufferedDataInputStream.SKIP_BUFFER_SIZE) {
405                     int got = read(this.skipBuf, 0, BufferedDataInputStream.SKIP_BUFFER_SIZE);
406                     if (got <= 0) {
407                         break;
408                     }
409                     need -= got;
410                 }
411                 while (need > 0) {
412                     int got = read(this.skipBuf, 0, (int) need);
413                     if (got <= 0) {
414                         break;
415                     }
416                     need -= got;
417                 }
418             }
419 
420         }
421         if (need > 0) {
422             throw new EOFException();
423         }
424     }

427     public int skipBytes(int toSkip) throws IOException {
428         skipAllBytes(toSkip);
