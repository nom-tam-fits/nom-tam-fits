34  import java.io.IOException;
35  
36  /**
37   * @deprecated  This is a rusty rail implementation only, unsafe for general use.
38   *      Use {@link FitsEncoder} instead, which provides a similar function but in
39   *      a more consistent way and with a less misleading name, or else use {@link ArrayEncoder}
40   *      as a base for implementing efficient custom encoding of binary outputs.
41   *
42   * @see FitsEncoder
43   */
44  @Deprecated
45  public abstract class BufferEncoder extends FitsEncoder {
46      
47      private BufferPointer p;
48      
49      /**
50       * 
51       * @param p     Unused, but the position and length fields are set/reset as to pretend that the
52       *              buffer is perpetually half filled with data, and with position at 0.
53       *              However, at no point will there be any data actually in the buffer of this object.
54       *              You should by all means avoid directly writing data from this buffer to the
55       *              output stream, other than the hopefully untriggered write of an existing 
56       *              <code>needBuffer(int)</code> implementation (and it's safest if you don't override
57       *              or ever call <code>needBuffer(int)</code> from your code!). 
58       */
59      public BufferEncoder(BufferPointer p) {
60          super();
61         
62          this.p = p;
63          
64          pretendHalfPopulated();
65  
66          setOutput(new OutputWriter() {
67               
68              private byte[] b1 = new byte[1];
69             
70              @Override
71              public void write(int b) throws IOException {
72                  b1[0] = (byte) b;
73                  BufferEncoder.this.write(b1, 0, 1);
74              }
75  
76              @Override
77              public void write(byte[] b, int from, int length) throws IOException {
78                  BufferEncoder.this.write(b, from, length);
79              }
80              
81          });
82      }
83      
84      /**
85       * We'll always pretend the buffer to be half populated at pos=0, in order to avoid triggering
86       * a read from the input into the unused buffer of BufferPointer, or a write to the
87       * output from that buffer... If the pointer has no buffer, length will be 0 also.
88       */
89      private void pretendHalfPopulated() {
90          p.pos = 0; 
91          p.length = p.buffer == null ? 0 : p.buffer.length >>> 1;   
92      }
93    
94      /**
95       * @deprecated No longer used internally, kept only for back-compatibility since it used to be a needed abstract method.
96       *             It's safest if you never override or call this method from your code!
97       *             
98       * @param need      the number of consecutive bytes we need available in the conversion buffer
99       * @throws IOException  
100      *                  if the buffer could not be flushed to the output to free up space in the buffer.
101      */ 
102     protected void needBuffer(int need) throws IOException {
103     }
104     
105     @Override
106     void need(int bytes) throws IOException {
107         pretendHalfPopulated();
108         super.need(bytes);
109     }
110 
111     @Override
112     protected void write(byte[] b, int from, int len) throws IOException {
113         throw new UnsupportedOperationException("You need to override this with an implementation that writes to the desired output.");
114     }
115     
116     /**
117      * Writes a single byte to the output, but not before flushing the contents of the conversion buffer. 
118      * The supplied {@link BufferPointer} is not used at all, and is immediately invalidated (which is 
119      * consistent with having flushed all pending output). It's not all that efficient, but then again 
120      * one should be using the new {@link FitsEncoder} instead. This is really just a rusty rail solution. 
121      * Also, since this methods does not throw an exception, and {@link #needBuffer(int)} (which did 
122      * throw an exception) is no longer in use, the duct-tape solution is to convert any IOException 
123      * encountered here into a runtime exception... 
124      * 
125      * @param b     the byte to write
126      * 
127      * @throws IllegalStateException    
128      * 
129      *              if there was an IO error flushing the conversion buffer or writing
130      *              the new byte after it.    
131      */
132     protected void writeUncheckedByte(byte b) {
133         try { 
134             flush();
135             write(b); 
136         } catch (IOException e) {
137             throw new IllegalStateException(e);
138         }
139     }
