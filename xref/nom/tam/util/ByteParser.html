36  /**
37   * @deprecated This class should not be exposed in the public API and should
38   *  be for internal use only in ASCII tables. Also, it may have overlapping 
39   *  functionality with other classes, which should probably be eliminated for
40   *  simplicity's sake (and thus less chance of nasty bugs). 
41   * 
42   * This class provides routines for efficient parsing of data stored in a byte
43   * array. This routine is optimized (in theory at least!) for efficiency rather
44   * than accuracy. The values read in for doubles or floats may differ in the
45   * last bit or so from the standard input utilities, especially in the case
46   * where a float is specified as a very long string of digits (substantially
47   * longer than the precision of the type).
48   * <p>
49   * The get methods generally are available with or without a length parameter
50   * specified. When a length parameter is specified only the bytes with the
51   * specified range from the current offset will be search for the number. If no
52   * length is specified, the entire buffer from the current offset will be
53   * searched.
54   * <p>
55   * The getString method returns a string with leading and trailing white space
56   * left intact. For all other get calls, leading white space is ignored. If
57   * fillFields is set, then the get methods check that only white space follows
58   * valid data and a FormatException is thrown if that is not the case. If
59   * fillFields is not set and valid data is found, then the methods return having
60   * read as much as possible. E.g., for the sequence "T123.258E13", a getBoolean,
61   * getInteger and getFloat call would return true, 123, and 2.58e12 when called
62   * in succession.
63   */
64  @Deprecated
65  public class ByteParser {

128     /**
129      * Find the sign for a number . This routine looks for a sign (+/-) at the
130      * current location and return +1/-1 if one is found, or +1 if not. The
131      * foundSign boolean is set if a sign is found and offset is incremented.
132      */
133     private int checkSign() {
134 
135         this.foundSign = false;
136 
137         if (this.input[this.offset] == '+') {
138             this.foundSign = true;
139             this.offset++;
140             return 1;
141         } else if (this.input[this.offset] == '-') {
142             this.foundSign = true;
143             this.offset++;
144             return -1;
145         }
146 
147         return 1;
148     }

150     /**
151      * Get the integer value starting at the current position. This routine
152      * returns a double rather than an int/long to enable it to read very long
153      * integers (with reduced precision) such as
154      * 111111111111111111111111111111111111111111. Note that this routine does
155      * set numberLength.
156      * 
157      * @param length
158      *            The maximum number of characters to use.
159      */
160     private double getBareInteger(int length) {
161 
162         int startOffset = this.offset;
163         double number = 0;
164 
165         while (length > 0 && this.input[this.offset] >= '0' && this.input[this.offset] <= '9') {
166 
167             number *= ByteParser.NUMBER_BASE;
168             number += this.input[this.offset] - '0';
169             this.offset++;
170             length--;
171         }
172         this.numberLength = this.offset - startOffset;
173         return number;
174     }

185     /**
186      * @return a boolean value from a specified region of the buffer
187      * @param length
188      *            The maximum number of characters used to parse this boolean.
189      * @throws FormatException
190      *             if the double was in an unknown format
191      */
192     public boolean getBoolean(int length) throws FormatException {
193 
194         int startOffset = this.offset;
195         length -= skipWhite(length);
196         if (length == 0) {
197             throw new FormatException("Blank boolean field");
198         }
199 
200         boolean value = false;
201         if (this.input[this.offset] == 'T' || this.input[this.offset] == 't') {
202             value = true;
203         } else if (this.input[this.offset] != 'F' && this.input[this.offset] != 'f') {
204             this.numberLength = 0;
205             this.offset = startOffset;
206             throw new FormatException("Invalid boolean value");
207         }
208         this.offset++;
209         this.numberLength = this.offset - startOffset;
210         return value;
211     }

233     /**
234      * @return a parsed double from the buffer. Leading spaces are ignored.
235      * @param length
236      *            The maximum number of characters used to parse this number. If
237      *            fillFields is specified then exactly only whitespace may
238      *            follow a valid double value.
239      * @throws FormatException
240      *             if the double was in an unknown format
241      */
242     public double getDouble(int length) throws FormatException {
243         int startOffset = this.offset;
244         boolean error = true;
245         double number;
246         // Skip initial blanks.
247         length -= skipWhite(length);
248         if (length == 0) {
249             this.numberLength = this.offset - startOffset;
250             return 0;
251         }
252         double mantissaSign = checkSign();
253         if (this.foundSign) {
254             length--;
255         }
256         // Look for the special strings NaN, Inf,
257         if (isCaseInsensitiv(length, ByteParser.NOT_A_NUMBER_LENGTH, ByteParser.NOT_A_NUMBER_LOWER, ByteParser.NOT_A_NUMBER_UPPER)) {
258             number = Double.NaN;
259             this.offset += ByteParser.NOT_A_NUMBER_LENGTH;
260             // Look for the longer string first then try the shorter.
261         } else if (isCaseInsensitiv(length, ByteParser.INFINITY_LENGTH, ByteParser.INFINITY_LOWER, ByteParser.INFINITY_UPPER)) {
262             number = Double.POSITIVE_INFINITY;
263             this.offset += ByteParser.INFINITY_LENGTH;
264         } else if (isCaseInsensitiv(length, ByteParser.INFINITY_SHORTCUT_LENGTH, ByteParser.INFINITY_LOWER, ByteParser.INFINITY_UPPER)) {
265             number = Double.POSITIVE_INFINITY;
266             this.offset += ByteParser.INFINITY_SHORTCUT_LENGTH;
267         } else {
268             number = getBareInteger(length); // This will update offset
269             length -= this.numberLength; // Set by getBareInteger
270             if (this.numberLength > 0) {
271                 error = false;
272             }
273             // Check for fractional values after decimal
274             if (length > 0 && this.input[this.offset] == '.') {
275                 this.offset++;
276                 length--;
277                 double numerator = getBareInteger(length);
278                 if (numerator > 0) {
279                     number += numerator / Math.pow(ByteParser.NUMBER_BASE_DOUBLE, this.numberLength);
280                 }
281                 length -= this.numberLength;
282                 if (this.numberLength > 0) {
283                     error = false;
284                 }
285             }
286 
287             if (error) {
288                 this.offset = startOffset;
289                 this.numberLength = 0;
290                 throw new FormatException("Invalid real field");
291             }
292 
293             // Look for an exponent ,Our Fortran heritage means that we allow
294             // 'D' for the exponent
295             // indicator.
296             if (length > 0 && (this.input[this.offset] == 'e' || this.input[this.offset] == 'E' || this.input[this.offset] == 'd' || this.input[this.offset] == 'D')) {
297                 this.offset++;
298                 length--;
299                 if (length > 0) {
300                     int sign = checkSign();
301                     if (this.foundSign) {
302                         length--;
303                     }
304 
305                     int exponent = (int) getBareInteger(length);
306 
307                     // For very small numbers we try to miminize
308                     // effects of denormalization.
309                     if (exponent * sign > ByteParser.EXPONENT_DENORMALISATION_CORR_LIMIT) {
310                         number *= Math.pow(ByteParser.NUMBER_BASE_DOUBLE, exponent * sign);
311                     } else {
312                         number =
313                                 ByteParser.EXPONENT_DENORMALISATION_FACTOR
314                                         * (number * Math.pow(ByteParser.NUMBER_BASE_DOUBLE, exponent * sign + ByteParser.EXPONENT_DENORMALISATION_CORR_LIMIT * -1));
315                     }
316                 }
317             }
318         }
319         this.numberLength = this.offset - startOffset;
320         return mantissaSign * number;
321     }

352     /**
353      * @return a region of the buffer to an integer
354      * @param length
355      *            The maximum number of characters used to parse this integer. @throws
356      *            FormatException if the integer was in an unknown format
357      * @throws FormatException
358      *             if the integer was in an unknown format
359      */
360     public int getInt(int length) throws FormatException {
361         int startOffset = this.offset;
362 
363         length -= skipWhite(length);
364         if (length == 0) {
365             this.numberLength = this.offset - startOffset;
366             return 0;
367         }
368 
369         int number = 0;
370         boolean error = true;
371 
372         int sign = checkSign();
373         if (this.foundSign) {
374             length--;
375         }
376 
377         while (length > 0 && this.input[this.offset] >= '0' && this.input[this.offset] <= '9') {
378             number = number * ByteParser.NUMBER_BASE + this.input[this.offset] - '0';
379             this.offset++;
380             length--;
381             error = false;
382         }
383 
384         if (error) {
385             this.numberLength = 0;
386             this.offset = startOffset;
387             throw new FormatException("Invalid Integer");
388         }
389         this.numberLength = this.offset - startOffset;
390         return sign * number;
391     }

393     /**
394      * @return a long in a specified region of the buffer
395      * @param length
396      *            The maximum number of characters used to parse this long.
397      * @throws FormatException
398      *             if the long was in an unknown format
399      */
400     public long getLong(int length) throws FormatException {
401 
402         int startOffset = this.offset;
403 
404         // Skip white space.
405         length -= skipWhite(length);
406         if (length == 0) {
407             this.numberLength = this.offset - startOffset;
408             return 0;
409         }
410 
411         long number = 0;
412         boolean error = true;
413 
414         long sign = checkSign();
415         if (this.foundSign) {
416             length--;
417         }
418 
419         while (length > 0 && this.input[this.offset] >= '0' && this.input[this.offset] <= '9') {
420             number = number * ByteParser.NUMBER_BASE + this.input[this.offset] - '0';
421             error = false;
422             this.offset++;
423             length--;
424         }
425 
426         if (error) {
427             this.numberLength = 0;
428             this.offset = startOffset;
429             throw new FormatException("Invalid long number");
430         }
431         this.numberLength = this.offset - startOffset;
432         return sign * number;
433     }

465     private boolean isCaseInsensitiv(int length, int constantLength, byte[] lowerConstant, byte[] upperConstant) {
466         if (length < constantLength) {
467             return false;
468         }
469         for (int i = 0; i < constantLength; i++) {
470             if (this.input[this.offset + i] != lowerConstant[i] && this.input[this.offset + i] != upperConstant[i]) {
471                 return false;
472             }
473         }
474         return true;
475     }

509     /**
510      * Skip white space. This routine skips with space in the input .
511      * 
512      * @return the number of character skipped. White space is defined as ' ',
513      *         '\t', '\n' or '\r'
514      * @param length
515      *            The maximum number of characters to skip.
516      */
517     public int skipWhite(int length) {
518         int i;
519         for (i = 0; i < length; i++) {
520             if (this.input[this.offset + i] != ' ' && this.input[this.offset + i] != '\t' && this.input[this.offset + i] != '\n' && this.input[this.offset + i] != '\r') {
521                 break;
522             }
523         }
524         this.offset += i;
525         return i;
526     }
