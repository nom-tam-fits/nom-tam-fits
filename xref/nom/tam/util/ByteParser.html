29  /**
30   * This class provides routines for efficient parsing of data stored in a byte
31   * array. This routine is optimized (in theory at least!) for efficiency rather
32   * than accuracy. The values read in for doubles or floats may differ in the
33   * last bit or so from the standard input utilities, especially in the case
34   * where a float is specified as a very long string of digits (substantially
35   * longer than the precision of the type).
36   * <p>
37   * The get methods generally are available with or without a length parameter
38   * specified. When a length parameter is specified only the bytes with the
39   * specified range from the current offset will be search for the number. If no
40   * length is specified, the entire buffer from the current offset will be
41   * searched.
42   * <p>
43   * The getString method returns a string with leading and trailing white space
44   * left intact. For all other get calls, leading white space is ignored. If
45   * fillFields is set, then the get methods check that only white space follows
46   * valid data and a FormatException is thrown if that is not the case. If
47   * fillFields is not set and valid data is found, then the methods return having
48   * read as much as possible. E.g., for the sequence "T123.258E13", a getBoolean,
49   * getInteger and getFloat call would return true, 123, and 2.58e12 when called
50   * in succession.
51   */
52  public class ByteParser {
53  
54      /** Array being parsed */
55      private byte[] input;
56  
57      /** Current offset into input. */
58      private int offset;
59  
60      /** Length of last parsed value */
61      private int numberLength;
62  
63      /** Did we find a sign last time we checked? */
64      private boolean foundSign;
65  
66      /** Do we fill up fields? */
67      private boolean fillFields = false;

140     /**
141      * Look for a double in the buffer. Leading spaces are ignored.
142      * 
143      * @param length
144      *            The maximum number of characters used to parse this number. If
145      *            fillFields is specified then exactly only whitespace may
146      *            follow a valid double value.
147      */
148     public double getDouble(int length) throws FormatException {
149 
150         int startOffset = offset;
151 
152         boolean error = true;
153 
154         double number = 0;
155         int i = 0;
156 
157         // Skip initial blanks.
158         length -= skipWhite(length);
159 
160         if (length == 0) {
161             numberLength = offset - startOffset;
162             return 0;
163         }
164 
165         double mantissaSign = checkSign();
166         if (foundSign) {
167             length -= 1;
168         }
169 
170         // Look for the special strings NaN, Inf,
171         if (length >= 3 && (input[offset] == 'n' || input[offset] == 'N') && (input[offset + 1] == 'a' || input[offset + 1] == 'A')
172                 && (input[offset + 2] == 'n' || input[offset + 2] == 'N')) {
173 
174             number = Double.NaN;
175             length -= 3;
176             offset += 3;
177 
178             // Look for the longer string first then try the shorter.
179         } else if (length >= 8 && (input[offset] == 'i' || input[offset] == 'I') && (input[offset + 1] == 'n' || input[offset + 1] == 'N')
180                 && (input[offset + 2] == 'f' || input[offset + 2] == 'F') && (input[offset + 3] == 'i' || input[offset + 3] == 'I')
181                 && (input[offset + 4] == 'n' || input[offset + 4] == 'N') && (input[offset + 5] == 'i' || input[offset + 5] == 'I')
182                 && (input[offset + 6] == 't' || input[offset + 6] == 'T') && (input[offset + 7] == 'y' || input[offset + 7] == 'Y')) {
183             number = Double.POSITIVE_INFINITY;
184             length -= 8;
185             offset += 8;
186 
187         } else if (length >= 3 && (input[offset] == 'i' || input[offset] == 'I') && (input[offset + 1] == 'n' || input[offset + 1] == 'N')
188                 && (input[offset + 2] == 'f' || input[offset + 2] == 'F')) {
189             number = Double.POSITIVE_INFINITY;
190             length -= 3;
191             offset += 3;
192 
193         } else {
194 
195             number = getBareInteger(length); // This will update offset
196             length -= numberLength; // Set by getBareInteger
197 
198             if (numberLength > 0) {
199                 error = false;
200             }
201 
202             // Check for fractional values after decimal
203             if (length > 0 && input[offset] == '.') {
204 
205                 offset += 1;
206                 length -= 1;
207 
208                 double numerator = getBareInteger(length);
209                 if (numerator > 0) {
210                     number += numerator / Math.pow(10., numberLength);
211                 }
212                 length -= numberLength;
213                 if (numberLength > 0) {
214                     error = false;
215                 }
216             }
217 
218             if (error) {
219                 offset = startOffset;
220                 numberLength = 0;
221                 throw new FormatException("Invalid real field");
222             }
223 
224             // Look for an exponent
225             if (length > 0) {
226 
227                 // Our Fortran heritage means that we allow 'D' for the exponent
228                 // indicator.
229                 if (input[offset] == 'e' || input[offset] == 'E' || input[offset] == 'd' || input[offset] == 'D') {
230 
231                     offset += 1;
232                     length -= 1;
233                     if (length > 0) {
234                         int sign = checkSign();
235                         if (foundSign) {
236                             length -= 1;
237                         }
238 
239                         int exponent = (int) getBareInteger(length);
240 
241                         // For very small numbers we try to miminize
242                         // effects of denormalization.
243                         if (exponent * sign > -300) {
244                             number *= Math.pow(10., exponent * sign);
245                         } else {
246                             number = 1.e-300 * (number * Math.pow(10., exponent * sign + 300));
247                         }
248                         length -= numberLength;
249                     }
250                 }
251             }
252         }
253 
254         if (fillFields && length > 0) {
255 
256             if (isWhite(length)) {
257                 offset += length;
258             } else {
259                 numberLength = 0;
260                 offset = startOffset;
261                 throw new FormatException("Non-blanks following real.");
262             }
263         }
264 
265         numberLength = offset - startOffset;
266         return mantissaSign * number;
267     }

281     /** Convert a region of the buffer to an integer */
282     public int getInt(int length) throws FormatException {
283         int startOffset = offset;
284 
285         length -= skipWhite(length);
286         if (length == 0) {
287             numberLength = offset - startOffset;
288             return 0;
289         }
290 
291         int number = 0;
292         boolean error = true;
293 
294         int sign = checkSign();
295         if (foundSign) {
296             length -= 1;
297         }
298 
299         while (length > 0 && input[offset] >= '0' && input[offset] <= '9') {
300             number = number * 10 + input[offset] - '0';
301             offset += 1;
302             length -= 1;
303             error = false;
304         }
305 
306         if (error) {
307             numberLength = 0;
308             offset = startOffset;
309             throw new FormatException("Invalid Integer");
310         }
311 
312         if (length > 0 && fillFields) {
313             if (isWhite(length)) {
314                 offset += length;
315             } else {
316                 numberLength = 0;
317                 offset = startOffset;
318                 throw new FormatException("Non-white following integer");
319             }
320         }
321 
322         numberLength = offset - startOffset;
323         return sign * number;
324     }

331     /** Look for a long in a specified region of the buffer */
332     public long getLong(int length) throws FormatException {
333 
334         int startOffset = offset;
335 
336         // Skip white space.
337         length -= skipWhite(length);
338         if (length == 0) {
339             numberLength = offset - startOffset;
340             return 0;
341         }
342 
343         long number = 0;
344         boolean error = true;
345 
346         long sign = checkSign();
347         if (foundSign) {
348             length -= 1;
349         }
350 
351         while (length > 0 && input[offset] >= '0' && input[offset] <= '9') {
352             number = number * 10 + input[offset] - '0';
353             error = false;
354             offset += 1;
355             length -= 1;
356         }
357 
358         if (error) {
359             numberLength = 0;
360             offset = startOffset;
361             throw new FormatException("Invalid long number");
362         }
363 
364         if (length > 0 && fillFields) {
365             if (isWhite(length)) {
366                 offset += length;
367             } else {
368                 offset = startOffset;
369                 numberLength = 0;
370                 throw new FormatException("Non-white following long");
371             }
372         }
373         numberLength = offset - startOffset;
374         return sign * number;
375     }

396     /** Get a boolean value from a specified region of the buffer */
397     public boolean getBoolean(int length) throws FormatException {
398 
399         int startOffset = offset;
400         length -= skipWhite(length);
401         if (length == 0) {
402             throw new FormatException("Blank boolean field");
403         }
404 
405         boolean value = false;
406         if (input[offset] == 'T' || input[offset] == 't') {
407             value = true;
408         } else if (input[offset] != 'F' && input[offset] != 'f') {
409             numberLength = 0;
410             offset = startOffset;
411             throw new FormatException("Invalid boolean value");
412         }
413         offset += 1;
414         length -= 1;
415 
416         if (fillFields && length > 0) {
417             if (isWhite(length)) {
418                 offset += length;
419             } else {
420                 numberLength = 0;
421                 offset = startOffset;
422                 throw new FormatException("Non-white following boolean");
423             }
424         }
425         numberLength = offset - startOffset;
426         return value;
427     }

434     /**
435      * Get the integer value starting at the current position. This routine
436      * returns a double rather than an int/long to enable it to read very long
437      * integers (with reduced precision) such as
438      * 111111111111111111111111111111111111111111. Note that this routine does
439      * set numberLength.
440      * 
441      * @param length
442      *            The maximum number of characters to use.
443      */
444     private double getBareInteger(int length) {
445 
446         int startOffset = offset;
447         double number = 0;
448 
449         while (length > 0 && input[offset] >= '0' && input[offset] <= '9') {
450 
451             number *= 10;
452             number += input[offset] - '0';
453             offset += 1;
454             length -= 1;
455         }
456         numberLength = offset - startOffset;
457         return number;
458     }

460     /**
461      * Skip white space. This routine skips with space in the input and returns
462      * the number of character skipped. White space is defined as ' ', '\t',
463      * '\n' or '\r'
464      * 
465      * @param length
466      *            The maximum number of characters to skip.
467      */
468     public int skipWhite(int length) {
469 
470         int i;
471         for (i = 0; i < length; i += 1) {
472             if (input[offset + i] != ' ' && input[offset + i] != '\t' && input[offset + i] != '\n' && input[offset + i] != '\r') {
473                 break;
474             }
475         }
476 
477         offset += i;
478         return i;
479 
480     }
481 
482     /**
483      * Find the sign for a number . This routine looks for a sign (+/-) at the
484      * current location and return +1/-1 if one is found, or +1 if not. The
485      * foundSign boolean is set if a sign is found and offset is incremented.
486      */
487     private int checkSign() {
488 
489         foundSign = false;
490 
491         if (input[offset] == '+') {
492             foundSign = true;
493             offset += 1;
494             return 1;
495         } else if (input[offset] == '-') {
496             foundSign = true;
497             offset += 1;
498             return -1;
499         }
500 
501         return 1;
502     }
503 
504     /**
505      * Is a region blank?
506      * 
507      * @param length
508      *            The length of the region to be tested
509      */
510     private boolean isWhite(int length) {
511         int oldOffset = offset;
512         boolean value = skipWhite(length) == length;
513         offset = oldOffset;
514         return value;
515     }
