25  import java.io.IOException;
26  
27  public interface ArrayDataInput extends java.io.DataInput {
28  
29      /**
30       * Read a generic (possibly multidimensional) primitive array. An Object[]
31       * array is also a legal argument if each element of the array is a legal.
32       * <p>
33       * The ArrayDataInput classes do not support String input since it is
34       * unclear how one would read in an Array of strings.
35       * 
36       * @param o
37       *            A [multidimensional] primitive (or Object) array.
38       * @deprecated See readLArray(Object o).
39       */
40      public int readArray(Object o) throws IOException;
41  
42      /**
43       * Read an array. This version works even if the underlying data is more
44       * than 2 Gigabytes.
45       */
46      public long readLArray(Object o) throws IOException;
47  
48      /* Read a complete primitive array */
49      public int read(byte[] buf) throws IOException;
50  
51      public int read(boolean[] buf) throws IOException;
52  
53      public int read(short[] buf) throws IOException;
54  
55      public int read(char[] buf) throws IOException;
56  
57      public int read(int[] buf) throws IOException;
58  
59      public int read(long[] buf) throws IOException;
60  
61      public int read(float[] buf) throws IOException;
62  
63      public int read(double[] buf) throws IOException;
64  
65      /* Read a segment of a primitive array. */
66      public int read(byte[] buf, int offset, int size) throws IOException;
67  
68      public int read(boolean[] buf, int offset, int size) throws IOException;
69  
70      public int read(char[] buf, int offset, int size) throws IOException;
71  
72      public int read(short[] buf, int offset, int size) throws IOException;
73  
74      public int read(int[] buf, int offset, int size) throws IOException;
75  
76      public int read(long[] buf, int offset, int size) throws IOException;
77  
78      public int read(float[] buf, int offset, int size) throws IOException;
79  
80      public int read(double[] buf, int offset, int size) throws IOException;
81  
82      /* Skip (forward) in a file */
83      public long skip(long distance) throws IOException;
84  
85      /* Skip and require that the data be there. */
86      public long skipBytes(long toSkip) throws IOException;
87  
88      /* Close the file. */
89      public void close() throws IOException;
90  }
