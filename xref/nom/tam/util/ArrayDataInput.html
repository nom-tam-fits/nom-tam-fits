32  import java.io.IOException;
33  
34  public interface ArrayDataInput extends java.io.DataInput {
35  
36      /**
37       * Read a generic (possibly multidimensional) primitive array. An Object[]
38       * array is also a legal argument if each element of the array is a legal.
39       * <p>
40       * The ArrayDataInput classes do not support String input since it is
41       * unclear how one would read in an Array of strings.
42       * 
43       * @param o
44       *            A [multidimensional] primitive (or Object) array.
45       * @deprecated See readLArray(Object o).
46       */
47      public int readArray(Object o) throws IOException;
48  
49      /**
50       * Read an array. This version works even if the underlying data is more
51       * than 2 Gigabytes.
52       */
53      public long readLArray(Object o) throws IOException;
54  
55      /* Read a complete primitive array */
56      public int read(byte[] buf) throws IOException;
57  
58      public int read(boolean[] buf) throws IOException;
59  
60      public int read(short[] buf) throws IOException;
61  
62      public int read(char[] buf) throws IOException;
63  
64      public int read(int[] buf) throws IOException;
65  
66      public int read(long[] buf) throws IOException;
67  
68      public int read(float[] buf) throws IOException;
69  
70      public int read(double[] buf) throws IOException;
71  
72      /* Read a segment of a primitive array. */
73      public int read(byte[] buf, int offset, int size) throws IOException;
74  
75      public int read(boolean[] buf, int offset, int size) throws IOException;
76  
77      public int read(char[] buf, int offset, int size) throws IOException;
78  
79      public int read(short[] buf, int offset, int size) throws IOException;
80  
81      public int read(int[] buf, int offset, int size) throws IOException;
82  
83      public int read(long[] buf, int offset, int size) throws IOException;
84  
85      public int read(float[] buf, int offset, int size) throws IOException;
86  
87      public int read(double[] buf, int offset, int size) throws IOException;
88  
89      /* Skip (forward) in a file */
90      public long skip(long distance) throws IOException;
91  
92      /* Skip and require that the data be there. */
93      public long skipBytes(long toSkip) throws IOException;
94  
95      /* Close the file. */
96      public void close() throws IOException;
97  }
