34  /**
35   * This interface extends the Iterator interface to allow insertion of data and
36   * move to previous entries in a collection.
37   * 
38   * @param <KEY>       the generic type of the keyword element
39   * @param <VALUE>     the generic type of the associated value  
40   */
41  public interface Cursor<KEY, VALUE> extends java.util.Iterator<VALUE> {
42  
43      /**
44       * Add a keyed entry at the current location. The new entry is inserted
45       * before the entry that would be returned in the next invocation of 'next'.
46       * The new element is placed such that it will be called by a prev() call,
47       * but not a next() call.The return value for that call is unaffected. Note:
48       * this method is not in the Iterator interface.
49       * 
50       * @param key
51       *            the key of the value to add
52       * @param reference
53       *            the value to add
54       * @deprecated use key parameter can be omitted, so use {@link #add(Object)}
55       *             instead
56       */
57      @Deprecated
58      void add(KEY key, VALUE reference);
59  
60      /**
61       * Add an unkeyed element to the collection. The new element is placed such
62       * that it will be called by a prev() call, but not a next() call.
63       * 
64       * @param reference
65       *            the value to add
66       */
67      void add(VALUE reference);
68  
69      /**
70       * move to the last element and return that.
71       * 
72       * @return the last element.
73       */
74      VALUE end();
75  
76      /**
77       * @return Is there a previous element in the collection?
78       */
79      boolean hasPrev();
80  
81      /**
82       * Returns the count next element in the iteration.
83       * 
84       * @param count
85       *            the offset
86       * @return the n'th next element in the iteration
87       * @throws java.util.NoSuchElementException
88       *             if the iteration has no more elements
89       */
90      VALUE next(int count);
91  
92      /**
93       * @return the previous element.
94       */
95      VALUE prev();
96  
97      /**
98       * Point the iterator to a particular keyed entry. Point to the end of the
99       * list if the key is not found.This method is not in the Iterator
100      * interface.
101      * 
102      * @param key
103      *            the key to search for
104      */
105     void setKey(KEY key);
106 }
