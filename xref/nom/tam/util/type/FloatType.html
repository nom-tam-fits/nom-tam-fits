34  import java.nio.ByteBuffer;
35  import java.nio.FloatBuffer;
36  
37  class FloatType extends PrimitiveTypeBase<FloatBuffer> {
38  
39      private static final int BIT_PIX = -32;
40  
41      private static final int SIZE = 4;
42  
43      protected FloatType() {
44          super(SIZE, false, float.class, Float.class, FloatBuffer.class, 'F', BIT_PIX);
45      }
46  
47      @Override
48      public void appendBuffer(FloatBuffer buffer, FloatBuffer dataToAppend) {
49          float[] temp = new float[Math.min(COPY_BLOCK_SIZE, dataToAppend.remaining())];
50          while (dataToAppend.hasRemaining()) {
51              int nrObBytes = Math.min(temp.length, dataToAppend.remaining());
52              dataToAppend.get(temp, 0, nrObBytes);
53              buffer.put(temp, 0, nrObBytes);
54          }
55      }
56  
57      @Override
58      public FloatBuffer asTypedBuffer(ByteBuffer buffer) {
59          return buffer.asFloatBuffer();
60      }
61  
62      @Override
63      public void getArray(FloatBuffer buffer, Object array, int offset, int length) {
64          buffer.get((float[]) array, offset, length);
65      }
66  
67      @Override
68      public Object newArray(int length) {
69          return new float[length];
70      }
71  
72      @Override
73      public void putArray(FloatBuffer buffer, Object array, int length) {
74          buffer.put((float[]) array, 0, length);
75      }
76  
77      @Override
78      public FloatBuffer sliceBuffer(FloatBuffer buffer) {
79          return buffer.slice();
80      }
81  
82      @Override
83      public FloatBuffer wrap(Object array) {
84          return FloatBuffer.wrap((float[]) array);
85      }
86  }
