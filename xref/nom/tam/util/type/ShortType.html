34  import java.nio.ByteBuffer;
35  import java.nio.ShortBuffer;
36  
37  class ShortType extends PrimitiveTypeBase<ShortBuffer> {
38  
39      private static final int BIT_PIX = 16;
40  
41      protected ShortType() {
42          super(2, false, short.class, Short.class, ShortBuffer.class, 'S', BIT_PIX);
43      }
44  
45      @Override
46      public void appendBuffer(ShortBuffer buffer, ShortBuffer dataToAppend) {
47          short[] temp = new short[Math.min(COPY_BLOCK_SIZE, dataToAppend.remaining())];
48          while (dataToAppend.hasRemaining()) {
49              int nrObBytes = Math.min(temp.length, dataToAppend.remaining());
50              dataToAppend.get(temp, 0, nrObBytes);
51              buffer.put(temp, 0, nrObBytes);
52          }
53      }
54  
55      @Override
56      public ShortBuffer asTypedBuffer(ByteBuffer buffer) {
57          return buffer.asShortBuffer();
58      }
59  
60      @Override
61      public void getArray(ShortBuffer buffer, Object array, int offset, int length) {
62          buffer.get((short[]) array, offset, length);
63      }
64  
65      @Override
66      public Object newArray(int length) {
67          return new short[length];
68      }
69  
70      @Override
71      public void putArray(ShortBuffer buffer, Object array, int length) {
72          buffer.put((short[]) array, 0, length);
73      }
74  
75      @Override
76      public ShortBuffer sliceBuffer(ShortBuffer buffer) {
77          return buffer.slice();
78      }
79  
80      @Override
81      public ShortBuffer wrap(Object array) {
82          return ShortBuffer.wrap((short[]) array);
83      }
84  }
