34  import java.nio.Buffer;
35  
36  /**
37   * @deprecated  Use equivalent static methods of {@link ElementType} instead.
38   * 
39   */
40  @Deprecated
41  public final class PrimitiveTypeHandler {
42  
43      /**
44       * @deprecated Use {@link ElementType#forDataID(char)} instead.
45       * 
46       * @param type  the Java array type character ID. For example 'J' for long integers.
47       * @return the corresponding FITS element type.
48       * 
49       */
50      public static ElementType<Buffer> valueOf(char type) {
51          return ElementType.forDataID(type);
52      }
53  
54      /**
55       * @deprecated Use {@link ElementType#forClass(Class)} instead.
56       * 
57       * @param clazz  the Java class of an element or an array thereof
58       * @return the corresponding FITS element type.
59       * 
60       * @param <B> the generic tpye of buffer used by the returned element type.
61       */
62      public static <B extends Buffer> ElementType<B> valueOf(Class<?> clazz) {
63          return ElementType.forClass(clazz);
64      }
65  
66      /**
67       * @deprecated Use {@link ElementType#forBitpix(int)} instead.
68       * 
69       * @param bitPix  the FITS BITPIX value
70       * @return the corresponding FITS element type.
71       */
72      public static ElementType<Buffer> valueOf(int bitPix) {
73          return ElementType.forBitpix(bitPix);
74      }
75      
76      /**
77       * @deprecated Use {@link ElementType#forNearestBitpix(int)} instead.
78       * 
79       * @param bitPix  a FITS BITPIX value
80       * @return the nearest corresponding FITS element type.
81       */
82      public static ElementType<Buffer> nearestValueOf(int bitPix) {
83          return ElementType.forNearestBitpix(bitPix);
84      }
85  
86      private PrimitiveTypeHandler() {
87      }
88  }
