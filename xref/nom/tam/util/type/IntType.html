34  import java.nio.ByteBuffer;
35  import java.nio.IntBuffer;
36  
37  import nom.tam.fits.header.Bitpix;
38  
39  /**
40   * A FITS 32-bit integer element (signed).
41   */
42  class IntType extends ElementType<IntBuffer> {
43  
44      private static final int SIZE = 4;
45  
46      protected IntType() {
47          super(SIZE, false, int.class, Integer.class, IntBuffer.class, 'I', Bitpix.VALUE_FOR_INT);
48      }
49  
50      @Override
51      public void appendBuffer(IntBuffer buffer, IntBuffer dataToAppend) {
52          @SuppressWarnings("deprecation")
53          int[] temp = new int[Math.min(COPY_BLOCK_SIZE, dataToAppend.remaining())];
54          while (dataToAppend.hasRemaining()) {
55              int nrObBytes = Math.min(temp.length, dataToAppend.remaining());
56              dataToAppend.get(temp, 0, nrObBytes);
57              buffer.put(temp, 0, nrObBytes);
58          }
59      }
60  
61      @Override
62      public IntBuffer asTypedBuffer(ByteBuffer buffer) {
63          return buffer.asIntBuffer();
64      }
65  
66      @Override
67      public void getArray(IntBuffer buffer, Object array, int offset, int length) {
68          buffer.get((int[]) array, offset, length);
69      }
70  
71      @Override
72      public Object newArray(int length) {
73          return new int[length];
74      }
75  
76      @Override
77      public void putArray(IntBuffer buffer, Object array, int offset, int length) {
78          buffer.put((int[]) array, offset, length);
79      }
80  
81      @Override
82      public IntBuffer sliceBuffer(IntBuffer buffer) {
83          return buffer.slice();
84      }
85  
86      @Override
87      public IntBuffer wrap(Object array) {
88          return IntBuffer.wrap((int[]) array);
89      }
90  }
