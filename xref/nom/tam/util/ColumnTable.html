44  /**
45   * A data table is conventionally considered to consist of rows and columns,
46   * where the structure within each column is constant, but different columns may
47   * have different structures. I.e., structurally columns may differ but rows are
48   * identical. Typically tabular data is usually stored in row order which can
49   * make it extremely difficult to access efficiently using Java. This class
50   * provides efficient access to data which is stored in row order and allows
51   * users to get and set the elements of the table. The table can consist only of
52   * arrays of primitive types. Data stored in column order can be efficiently
53   * read and written using the BufferedDataXputStream classes. The table is
54   * represented entirely as a set of one-dimensional primitive arrays. For a
55   * given column, a row consists of some number of contiguous elements of the
56   * array. Each column is required to have the same number of rows. Information
57   * regarding the dimensionality of columns and possible data pointers is
58   * retained for use by clients which can understand them.
59   * 
60   * @param <T>   the generic type of extra state information associated with this table.
61   * 
62   */
63  public class ColumnTable<T> implements DataTable {
64

331     /**
332      * Add a column .
333      * 
334      * @param newColumn
335      *            the column to add.
336      * @param size
337      *            size for the column
338      * @throws TableException
339      *             if the structure of the new column does not fit the structure
340      *             of the rows/columns
341      */
342     public void addColumn(Object newColumn, int size) throws TableException {
343 
344         String classname = newColumn.getClass().getName();
345         this.nrow = checkColumnConsistency(newColumn, classname, this.nrow, size);
346 
347         int ncol = this.arrays.length;
348 
349         Object[] newArrays = new Object[ncol + 1];
350         int[] newSizes = new int[ncol + 1];
351         Class<?>[] newBases = new Class[ncol + 1];
352         char[] newTypes = new char[ncol + 1];
353 
354         System.arraycopy(this.arrays, 0, newArrays, 0, ncol);
355         System.arraycopy(this.sizes, 0, newSizes, 0, ncol);
356         System.arraycopy(this.bases, 0, newBases, 0, ncol);
357         System.arraycopy(this.types, 0, newTypes, 0, ncol);
358 
359         this.arrays = newArrays;
360         this.sizes = newSizes;
361         this.bases = newBases;
362         this.types = newTypes;
363 
364         this.arrays[ncol] = newColumn;
365         this.sizes[ncol] = size;
366         this.bases[ncol] = ArrayFuncs.getBaseClass(newColumn);
367         this.types[ncol] = classname.charAt(1);
368         addPointer(newColumn);
369     }

371     /**
372      * Add a pointer in the pointer lists.
373      * 
374      * @param data
375      *            data pointer to add
376      * @throws TableException
377      *             if the structure of the specified array does not fit the
378      *             structure of the rows/columns
379      */
380     protected void addPointer(Object data) throws TableException {
381         PointerAccess<Object> accessor = selectPointerAccessor(data);
382         if (accessor == null) {
383             throw new TableException("Invalid type for added column:" + data.getClass().getComponentType());
384         }
385         accessor.set(this, extendArray(accessor.get(this), data));
386     }
387 
388     @SuppressWarnings("unchecked")
389     private PointerAccess<Object> selectPointerAccessor(Object data) {
390         return (PointerAccess<Object>) POINTER_ACCESSORS.get(ElementType.forClass(data.getClass().getComponentType()));
391     }
392 
393     @SuppressWarnings("unchecked")
394     private <ArrayType> ArrayType extendArray(ArrayType originalArray, Object data) {
395         int length = Array.getLength(originalArray);
396         Object xb = Array.newInstance(originalArray.getClass().getComponentType(), length + 1);
397         System.arraycopy(originalArray, 0, xb, 0, length);
398         Array.set(xb, length, data);
399         return (ArrayType) xb;
400     }

402     /**
403      * Add a row to the table. This method is very inefficient for adding
404      * multiple rows and should be avoided if possible.
405      * 
406      * @param row
407      *            the row to add
408      * @throws TableException
409      *             if the structure of the specified array does not fit the
410      *             structure of the rows/columns
411      */
412     public void addRow(Object[] row) throws TableException {
413         if (this.arrays.length == 0) {
414             for (Object element : row) {
415                 addColumn(element, Array.getLength(element));
416             }
417         } else {
418             if (row.length != this.arrays.length) {
419                 throw new TableException("Row length mismatch");
420             }
421             for (int i = 0; i < row.length; i++) {
422                 if (row[i].getClass() != this.arrays[i].getClass() || Array.getLength(row[i]) != this.sizes[i]) {
423                     throw new TableException("Row column mismatch at column:" + i);
424                 }
425                 Object xarray = ArrayFuncs.newInstance(this.bases[i], (this.nrow + 1) * this.sizes[i]);
426                 System.arraycopy(this.arrays[i], 0, xarray, 0, this.nrow * this.sizes[i]);
427                 System.arraycopy(row[i], 0, xarray, this.nrow * this.sizes[i], this.sizes[i]);
428                 this.arrays[i] = xarray;
429             }
430             initializePointers();
431             this.nrow++;
432         }
433     }

435     /**
436      * Check that the columns and sizes are consistent. Inconsistencies include:
437      * <ul>
438      * <li>arrays and sizes have different lengths.
439      * <li>an element of arrays is not a primitive array.
440      * <li>the size of an array is not divisible by the sizes entry.
441      * <li>the number of rows differs for the columns.
442      * </ul>
443      * 
444      * @param newArrays
445      *            The arrays defining the columns.
446      * @param newSizes
447      *            The number of elements in each row for the column.
448      * @throws TableException
449      *             if the table was inconsistent
450      */
451     protected void checkArrayConsistency(Object[] newArrays, int[] newSizes) throws TableException {
452 
453         // This routine throws an error if it detects an inconsistency
454         // between the arrays being read in.
455 
456         // First check that the lengths of the two arrays are the same.
457         if (newArrays.length != newSizes.length) {
458             throw new TableException("readArraysAsColumns: Incompatible arrays and sizes.");
459         }
460 
461         // Now check that that we fill up all of the arrays exactly.
462         int ratio = 0;
463 
464         this.types = new char[newArrays.length];
465         this.bases = new Class[newArrays.length];
466 
467         for (int i = 0; i < newArrays.length; i += 1) {
468 
469             String classname = newArrays[i].getClass().getName();
470 
471             ratio = checkColumnConsistency(newArrays[i], classname, ratio, newSizes[i]);
472 
473             this.types[i] = classname.charAt(1);
474             this.bases[i] = ArrayFuncs.getBaseClass(newArrays[i]);
475         }
476 
477         this.nrow = ratio;
478         this.arrays = newArrays;
479         this.sizes = newSizes;
480     }
481 
482     private int checkColumnConsistency(Object data, String classname, int ratio, int size) throws TableException {
483 
484         if (classname.charAt(0) != '[' || classname.length() != 2) {
485             throw new TableException("Non-primitive array for column");
486         }
487 
488         int thisSize = Array.getLength(data);
489         if (thisSize == 0 && size != 0 && ratio != 0 || thisSize != 0 && size == 0) {
490             throw new TableException("Size mismatch in column: " + thisSize + " != " + size);
491         }
492 
493         // The row size must evenly divide the size of the array.
494         if (size != 0 && thisSize % size != 0) {
495             throw new TableException("Row size does not divide array for column");
496         }
497 
498         // Finally the ratio of sizes must be the same for all columns -- this
499         // is the number of rows in the table.
500         int thisRatio = 0;
501         if (size > 0) {
502             thisRatio = thisSize / size;
503 
504             if (ratio != 0 && thisRatio != ratio) {
505                 throw new TableException("Different number of rows in different columns");
506             }
507         }
508         if (thisRatio > 0) {
509             return thisRatio;
510         }
511         return ratio;
512 
513     }

519     /**
520      * Delete a contiguous set of columns from the table.
521      * 
522      * @param start
523      *            The first column (0-indexed) to be deleted.
524      * @param len
525      *            The number of columns to be deleted.
526      * @throws TableException
527      *             if the request goes outside the boundaries of the table or if
528      *             the length is negative.
529      */
530     public void deleteColumns(int start, int len) throws TableException {
531         int ncol = this.arrays.length;
532         if (start < 0 || len < 0 || start + len > ncol) {
533             throw new TableException("Invalid request to delete columns start: " + start + " length:" + len + " for table with " + ncol + " columns.");
534         }
535         if (len == 0) {
536             return;
537         }
538         int ocol = ncol;
539         ncol -= len;
540 
541         Object[] newArrays = new Object[ncol];
542         int[] newSizes = new int[ncol];
543         Class<?>[] newBases = new Class<?>[ncol];
544         char[] newTypes = new char[ncol];
545 
546         System.arraycopy(this.arrays, 0, newArrays, 0, start);
547         System.arraycopy(this.sizes, 0, newSizes, 0, start);
548         System.arraycopy(this.bases, 0, newBases, 0, start);
549         System.arraycopy(this.types, 0, newTypes, 0, start);
550 
551         int rem = ocol - (start + len);
552 
553         System.arraycopy(this.arrays, start + len, newArrays, start, rem);
554         System.arraycopy(this.sizes, start + len, newSizes, start, rem);
555         System.arraycopy(this.bases, start + len, newBases, start, rem);
556         System.arraycopy(this.types, start + len, newTypes, start, rem);
557 
558         this.arrays = newArrays;
559         this.sizes = newSizes;
560         this.bases = newBases;
561         this.types = newTypes;
562 
563         initializePointers();
564     }

568      * 
569      * @param row

571      * @throws TableException

579     /**
580      * Delete a contiguous set of rows from the table.
581      * 
582      * @param row
583      *            The row (0-indexed) to be deleted.
584      * @param length
585      *            The number of rows to be deleted.
586      * @throws TableException
587      *             if the request goes outside the boundaries of the table or if
588      *             the length is negative.
589      */
590     public void deleteRows(int row, int length) throws TableException {
591 
592         if (row < 0 || length < 0 || row + length > this.nrow) {
593             throw new TableException("Invalid request to delete rows start: " + row + " length:" + length + " for table with " + this.nrow + " rows.");
594         }
595 
596         if (length == 0) {
597             return;
598         }
599 
600         for (int col = 0; col < this.arrays.length; col += 1) {
601 
602             int sz = this.sizes[col];
603             int newSize = sz * (this.nrow - length);
604             Object newArr = ArrayFuncs.newInstance(this.bases[col], newSize);
605 
606             // Copy whatever comes before the deletion
607             System.arraycopy(this.arrays[col], 0, newArr, 0, row * sz);
608 
609             // Copy whatever comes after the deletion
610             System.arraycopy(this.arrays[col], (row + length) * sz, newArr, row * sz, (this.nrow - row - length) * sz);
611             this.arrays[col] = newArr;
612         }
613         this.nrow -= length;
614         initializePointers();
615     }

627     /**
628      * <p>
629      * Gets a particular column in the format that it is stored in the FITS. For most column types
630      * the storage format matches the Java type, but there are exceptions:
631      * </p>
632      * <ul>
633      * <li>Character arrays in FITS are stored as <code>byte[]</code> or <code>short[]</code>, depending
634      * on the {@link nom.tam.fits.FitsFactory#setUseUnicodeChars(boolean)} setting, not unicode Java <code>char[]</code>.
635      * Therefore, this call will return <code>byte[]</code> or <code>short[]</code>, the same as for a byte or 16-bit 
636      * integer array. As a result if a new table is created with the returned data, the new table column will change 
637      * its FITS column type from <code>A</code> to <code>B</code> or <code>I</code>.</li>
638      * <li>Complex values in FITS are stored as <code>float[2]</code> or <code>double[2]</code>, not as a 
639      * {@link ComplexValue} type. Therefore, this call will return <code>float[]</code> or <code>double[]</code>, the 
640      * same as for a float array. As a result if a new table is created with the returned data, the new table column 
641      * will change it's FITS column type from <code>C</code> to <code>F</code>, or from <code>M</code> to 
642      * <code>D</code>,.</li>
643      * </ul>
644      * 
645      * @param col

736     /**
737      * Set the pointer arrays for the eight primitive types to point to the
738      * appropriate elements of arrays.
739      */
740     protected void initializePointers() {
741         int[] columnIndex = new int[MAX_COLUMN_INDEXES];
742         for (int col = 0; col < this.arrays.length; col += 1) {
743             columnIndex[this.types[col]]++;
744         }
745         // Allocate the pointer arrays. Note that many will be
746         // zero-length.
747         this.bytePointers = new byte[columnIndex[ElementType.BYTE.type()]][];
748         this.shortPointers = new short[columnIndex[ElementType.SHORT.type()]][];
749         this.intPointers = new int[columnIndex[ElementType.INT.type()]][];
750         this.longPointers = new long[columnIndex[ElementType.LONG.type()]][];
751         this.floatPointers = new float[columnIndex[ElementType.FLOAT.type()]][];
752         this.doublePointers = new double[columnIndex[ElementType.DOUBLE.type()]][];
753         this.charPointers = new char[columnIndex[ElementType.CHAR.type()]][];
754         this.booleanPointers = new boolean[columnIndex[ElementType.BOOLEAN.type()]][];
755         // Now set the pointers.
756         Arrays.fill(columnIndex, 0);
757         for (int col = 0; col < this.arrays.length; col += 1) {
758             char colType = this.types[col];
759             PointerAccess<?> accessor = POINTER_ACCESSORS_BY_TYPE[colType];
760             Array.set(accessor.get(this), columnIndex[colType], this.arrays[col]);
761             columnIndex[colType]++;
762         }
763     }

765     /**
766      * Read a table.
767      * 
768      * @param is
769      *            The input stream to read from.
770      * @throws IOException
771      *             if the reading failed
772      */
773     public void read(ArrayDataInput is) throws IOException {
774         int[] columnIndex = new int[MAX_COLUMN_INDEXES];
775         // While we have not finished reading the table..
776         for (int row = 0; row < this.nrow; row += 1) {
777             Arrays.fill(columnIndex, 0);
778             // Loop over the columns within the row.
779             for (int col = 0; col < this.arrays.length; col += 1) {
780                 int arrOffset = this.sizes[col] * row;
781                 int size = this.sizes[col];
782                 char colType = this.types[col];
783                 PointerAccess<?> accessor = POINTER_ACCESSORS_BY_TYPE[colType];
784                 accessor.read(this, is, columnIndex[colType], arrOffset, size);
785                 columnIndex[colType] += 1;
786             }
787         }
788     }

790     /**
791      * Set the values in a particular column. The new values must match the old
792      * in length but not necessarily in type.
793      * 
794      * @param col
795      *            The column to modify.
796      * @param newColumn
797      *            The new column data. This should be a primitive array.
798      * @exception TableException
799      *                Thrown when the new data is not commenserable with
800      *                information in the table.
801      */
802     @Override
803     public void setColumn(int col, Object newColumn) throws TableException {
804 
805         boolean reset = newColumn.getClass() != this.arrays[col].getClass() || Array.getLength(newColumn) != Array.getLength(this.arrays[col]);
806         this.arrays[col] = newColumn;
807         if (reset) {
808             setup(this.arrays, this.sizes);
809         } else {
810             // This is required, because otherwise the typed pointer may point
811             // to the old
812             // array, which has been replaced by newColumn. Added by Jeroen de
813             // Jong, 1 Aug 2006
814             initializePointers();
815         }
816     }

818     /**
819      * Modify an element of the table.
820      * 
821      * @param row
822      *            The row containing the element.
823      * @param col
824      *            The column containing the element.
825      * @param x
826      *            The new datum. This should be 1-d primitive array.
827      * @exception TableException
828      *                Thrown when the new data is not of the same type as the
829      *                data it replaces.
830      */
831     @Override
832     public void setElement(int row, int col, Object x) throws TableException {
833 
834         String classname = x.getClass().getName();
835 
836         if (!classname.equals("[" + this.types[col])) {
837             throw new TableException("setElement: Incompatible element type");
838         }
839 
840         if (Array.getLength(x) != this.sizes[col]) {
841             throw new TableException("setElement: Incompatible element size");
842         }
843 
844         System.arraycopy(x, 0, this.arrays[col], this.sizes[col] * row, this.sizes[col]);
845     }

858     /**
859      * Modify a row of data.
860      * 
861      * @param row
862      *            The row to be modified.
863      * @param x
864      *            The data to be modified. This should be an array of objects.
865      *            It is described as an Object here since other table
866      *            implementations may use other methods to store the data (e.g., @see
867      *            nom.tam.util.ColumnTable)
868      */
869     @Override
870     public void setRow(int row, Object x) throws TableException {
871 
872         if (!(x instanceof Object[])) {
873             throw new TableException("setRow: Incompatible row");
874         }
875 
876         for (int col = 0; col < this.arrays.length; col += 1) {
877             setElement(row, col, ((Object[]) x)[col]);
878         }
879     }

882      * Actually perform the initialization.
883      * 
884      * @param newArrays
885      *            An array of one-d primitive arrays.
886      * @param newSizes

888      *            column

890      *             if the structure of the columns is not consistent

901      *            the output stream to write to.

903      *             if the write operation failed

905     public void write(ArrayDataOutput os) throws IOException {
906         int[] columnIndex = new int[MAX_COLUMN_INDEXES];
907         for (int row = 0; row < this.nrow; row += 1) {
908             Arrays.fill(columnIndex, 0);
909             // Loop over the columns within the row.
910             for (int col = 0; col < this.arrays.length; col += 1) {
911

913                 int size = this.sizes[col];
914 
915                 char colType = this.types[col];
916                 POINTER_ACCESSORS_BY_TYPE[colType].write(this, os, columnIndex[colType], arrOffset, size);
917                 columnIndex[colType] += 1;
918             }
919         }

923      * Write a column of a table.
924      * 
925      * @param os
926      *            the output stream to write to.
927      * @param rowStart
928      *            first row to write
929      * @param rowEnd
930      *            row number that should not be written anymore
931      * @param columnNr
932      *            zero based column number to write.
933      * @throws IOException
934      *             if the write operation failed
935      */
936     public void write(ArrayDataOutput os, int rowStart, int rowEnd, int columnNr) throws IOException {
937         int[] columnIndex = new int[MAX_COLUMN_INDEXES];
938         for (int row = 0; row < this.nrow; row += 1) {
939             if (row >= rowStart && row < rowEnd) {
940                 Arrays.fill(columnIndex, 0);
941                 // Loop over the columns within the row.
942                 for (int col = 0; col < this.arrays.length; col += 1) {
943 
944                     int arrOffset = this.sizes[col] * row;
945                     int size = this.sizes[col];
946 
947                     char colType = this.types[col];
948                     if (columnNr == col) {
949                         POINTER_ACCESSORS_BY_TYPE[colType].write(this, os, columnIndex[colType], arrOffset, size);
950                     }
951                     columnIndex[colType] += 1;
952                 }
953             }
954         }
