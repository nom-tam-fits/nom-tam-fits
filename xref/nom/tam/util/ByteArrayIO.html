34  import java.io.EOFException;
35  import java.io.IOException;
36  import java.util.Arrays;
37  
38  import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
39  
40  /**
41   * A class for reading and writing into byte arrays with a stream-like
42   * interface.
43   * 
44   * @author Attila Kovacs
45   *
46   * @since 1.16
47   */
48  public class ByteArrayIO implements ReadWriteAccess {
49      
50      /** Mask for 1-byte */
51      private static final int BYTE_MASK = 0xFF;
52      
53      /** The underlying buffer */
54      private byte[] buf;
55      
56      /** Whether the buffer is allowed to grow as needed to contain more data */
57      private boolean isGrowable;
58      
59      /** The current pointer position, for the next read or write in the buffer */
60      private int pos;
61      
62      /** The current end of the buffer, that is the total number of bytes available for reading from the buffer */
63      private int end;
64     
65      /**
66       * Instantiates a new byte array with an IO interface, with a fixed-size buffer using the
67       * specified array as its backing storage.
68       * 
69       * @param buffer        the fixed buffer.
70       */
71      @SuppressFBWarnings(value = "EI_EXPOSE_REP2", justification = "by design this class provides an IO interface for an accessible array.")
72      public ByteArrayIO(byte[] buffer) {
73          this.buf = buffer;
74          this.end = 0;
75          this.isGrowable = false;
76      }
77      
78      /**
79       * Instantiates a new byte array with an IO interface, with a growable buffer initialized
80       * to the specific size.
81       * 
82       * @param initialCapacity       the number of bytes to contain in the buffer.
83       * @throws IllegalArgumentException
84       *                              if the initial capacity is 0 or negative
85       */
86      public ByteArrayIO(int initialCapacity) throws IllegalArgumentException {
87          if (initialCapacity <= 0) {
88              throw new IllegalArgumentException("Illegal buffer size:" + initialCapacity);
89          }
90          
91          this.buf = new byte[initialCapacity];
92          this.end = 0;
93          this.isGrowable = true;
94      }
95      
96      /**
97       * Returns a copy of this byte array with an IO interface, including a deep copy of
98       * the buffered data.
99       * 
100      * @return      a deep copy of this byte array with an IO interface instance.
101      */
102     public synchronized ByteArrayIO copy() {
103         ByteArrayIO copy = new ByteArrayIO(Arrays.copyOf(buf, buf.length));
104         synchronized (copy) {
105             copy.isGrowable = this.isGrowable;
106             copy.pos = this.pos;
107             copy.end = this.end;
108         }
109         return copy;
110     }
111     
112     /**
113      * Returns the underlying byte array, which is the backing array of this buffer.
114      * 
115      * @return  the backing array of this buffer.
116      */
117     @SuppressFBWarnings(value = "EI_EXPOSE_REP", justification = "by design this class provides an IO interface for an accessible array.")
118     public synchronized byte[] getBuffer() {
119         return buf;
120     }
121     
122     /**
123      * Returns the current capacity of this buffer, that is the total number of
124      * bytes that may be written into the current backing buffer.
125      * 
126      * @return  the current size of the backing array.
127      */
128     public final synchronized int capacity() {
129         return buf.length;
130     }
131     
132     @Override
133     public final synchronized long length() {
134         return end;
135     }
136     
137     /**
138      * Returns the number of bytes available for reading from the current position.
139      * 
140      * @return  the number of bytes that can be read from this buffer from the current position.
141      */
142     public final synchronized int getRemaining() {
143         if (pos >= end) {
144             return 0;
145         }
146         return end - pos;
147     }
148     
149     @Override
150     public final synchronized long position() {
151         return pos;
152     }
153     
154     @Override
155     public synchronized void position(long offset) throws IOException { 
156         if (offset < 0) {
157             throw new EOFException("Negative buffer index: " + offset);
158         }
159         
160         if (offset > buf.length) {
161             if (!isGrowable) {
162                 throw new EOFException("Position " + offset + " beyond fixed buffer size " + buf.length);
163             }
164         }
165         
166         pos = (int) offset;
167     }
168     
169     /**
170      * Changes the length of this buffer. The total number of bytes available
171      * from reading from this buffer will be that of the new length. If the buffer
172      * is truncated and its pointer is positioned beyond the new size, then the pointer
173      * is changed to point to the new buffer end. If the buffer is enlarged beyond
174      * it current capacity, its capacity will grow as necessary provided the
175      * buffer is growable. Otherwise, an EOFException is thrown if the new length
176      * is beyond the fixed buffer capacity. If the new length is larger than the old 
177      * one, the added buffer segment may have undefined contents.
178      * 
179      * @param length        The buffer length, that is number of bytes available for
180      *                      reading.
181      * @throws IllegalArgumentException 
182      *                      if the length is negative or if the new new length 
183      *                      exceeds the capacity of a fixed-type buffer.
184      *                      
185      * @see #length()
186      * @see #capacity()
187      */
188     public synchronized void setLength(int length) throws IllegalArgumentException {
189         if (length < 0) {
190             throw new IllegalArgumentException("Buffer set to negative length: " + length);
191         }
192         
193         if (length > capacity()) {
194             if (!isGrowable) {
195                 throw new IllegalArgumentException("the new length " + length + " is larger than the fixed capacity " + capacity());
196             }
197             grow(length - capacity());
198         }
199         end = length;
200         
201         // If the pointer is beyond the new size, move it back to the new end...
202         if (pos > end) {
203             pos = end;
204         }
205     }
206     
207     /**
208      * Grows the buffer by at least the number of specified bytes. A new buffer is
209      * allocated, and the contents of the previous buffer are copied over.
210      * 
211      * @param need      the minimum number of extra bytes needed beyond the current capacity. 
212      */
213     private synchronized void grow(int need) {
214         int size = capacity() + need;
215         int below = Integer.highestOneBit(size);
216         if (below != size) {
217             size = below << 1;
218         }
219         byte[] newbuf = new byte[size];
220         System.arraycopy(buf, 0, newbuf, 0, buf.length);
221         buf = newbuf;
222     }
223     
224     @Override
225     public final synchronized void write(int b) throws IOException {
226         if (pos + 1 > buf.length) {
227             if (isGrowable) {
228                 grow(pos + 1 - buf.length);
229             } else {
230                 throw new EOFException("buffer is full (size=" + length() + ")");
231             }
232         }
233         buf[pos++] = (byte) b;
234         if (pos > end) {
235             end = pos;
236         }
237     }
238     
239     @Override
240     public final synchronized void write(byte[] b, int from, int length) throws IOException {
241         if (length <= 0) {
242             return;
243         }
244         
245         if (pos > buf.length || (isGrowable && pos + length > buf.length)) {
246             // This may only happen in a growable buffer...
247             grow(buf.length + length - pos);
248         }
249         
250         int l = Math.min(length, buf.length - pos);
251         
252         System.arraycopy(b, from, buf, pos, l);
253         pos += l;
254         if (pos > end) {
255             end = pos;
256         }
257         
258         if (l < length) {
259             throw new EOFException("Incomplete write of " + l + " of " + length + " bytes in buffer of size " + length());
260         }
261     }
262 
263     @Override
264     public final synchronized int read() throws IOException {
265         if (getRemaining() <= 0) {
266             return -1;
267         }
268         return buf[pos++] & BYTE_MASK;
269     }
270 
271     @Override
272     public final synchronized int read(byte[] b, int from, int length) {
273         if (length <= 0) {
274             return 0;
275         }
276         
277         int remaining = getRemaining();
278         
279         if (remaining <= 0) {
280             return -1;
281         }
282         
283         int n = Math.min(remaining, length);
284         System.arraycopy(buf, pos, b, from, n);
285         pos += n;
286         
287         return n;
288     }
