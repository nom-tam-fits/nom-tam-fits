32  package nom.tam.util.array;
33  
34  import java.lang.reflect.Array;
35  
36  /**
37   * Multi-dimensional array iterator.
38   *
39   * @param <BaseArray> 
40   *              the generic type of array at the base of a multi-dimensional array object. For example
41   *              for a <code>float[][][]</code> array the base would be <code>float[]</code>. 
42   */
43  public class MultiArrayIterator<BaseArray> {
44  
45      private final BaseArray baseArray;
46  
47      private final boolean baseIsNoSubArray;
48  
49      private boolean baseNextCalled = false;
50  
51      private final MultiArrayPointer pointer;
52  
53      public MultiArrayIterator(BaseArray baseArray) {
54          this.baseArray = baseArray;
55          this.baseIsNoSubArray = !MultiArrayPointer.isSubArray(this.baseArray);
56          this.pointer = new MultiArrayPointer(this.baseArray);
57      }
58  
59      public Class<?> deepComponentType() {
60          Class<?> clazz = this.baseArray.getClass();
61          while (clazz.isArray()) {
62              clazz = clazz.getComponentType();
63          }
64          return clazz;
65      }
66  
67      public BaseArray next() {
68          if (this.baseIsNoSubArray) {
69              if (this.baseNextCalled) {
70                  return null;
71              }
72              this.baseNextCalled = true;
73              return this.baseArray;
74          }
75          Object result = null;
76          while (result == null || Array.getLength(result) == 0) {
77              result = this.pointer.next();
78              if (result == MultiArrayPointer.END) {
79                  return null;
80              }
81          }
82          return (BaseArray) result;
83      }
84  
85      public void reset() {
86          if (this.baseIsNoSubArray) {
87              this.baseNextCalled = false;
88          } else {
89              this.pointer.reset();
90          }
91      }
92  
93      public int size() {
94          int size = 0;
95          Object next;
96          while ((next = next()) != null) {
97              size += Array.getLength(next);
98          }
99          return size;
100     }
101 }
