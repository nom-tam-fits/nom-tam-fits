32  package nom.tam.util.array;
33  
34  import java.lang.reflect.Array;
35  
36  public final class MultiArrayCopier<Source, Destination> {
37  
38      public static <Source, Destination> void copyInto(Source fromArray, Destination toArray) {
39          new MultiArrayCopier<>(fromArray, toArray).copyInto();
40      }
41  
42      private final MultiArrayIterator<Source> from;
43  
44      private final MultiArrayIterator<Destination> to;
45  
46      private Destination currentToArray;
47  
48      private int currentToArrayOffset;
49  
50      private int currentToArrayLength;
51  
52      private final MultiArrayCopyFactory<Source, Destination> copyFactory;
53  
54      private MultiArrayCopier(Source fromArray, Destination toArray) {
55          this.from = new MultiArrayIterator<>(fromArray);
56          this.to = new MultiArrayIterator<>(toArray);
57          this.copyFactory = (MultiArrayCopyFactory<Source, Destination>) MultiArrayCopyFactory.select(this.from.deepComponentType(), this.to.deepComponentType());
58      }
59  
60      private void copyInto() {
61          Source current = this.from.next();
62          while (current != null) {
63              copyInto(current);
64              current = this.from.next();
65          }
66      }
67  
68      private void copyInto(Source currentFromArray) {
69          int currentFromArrayOffset = 0;
70          int currentFromArrayLength = Array.getLength(currentFromArray);
71          while (currentFromArrayOffset < currentFromArrayLength) {
72              if (this.currentToArray == null || this.currentToArrayOffset >= this.currentToArrayLength) {
73                  this.currentToArray = this.to.next();
74                  this.currentToArrayOffset = 0;
75                  this.currentToArrayLength = Array.getLength(this.currentToArray);
76              }
77              int length = Math.min(this.currentToArrayLength - this.currentToArrayOffset, currentFromArrayLength - currentFromArrayOffset);
78              this.copyFactory.arraycopy(currentFromArray, currentFromArrayOffset, this.currentToArray, this.currentToArrayOffset, length);
79              currentFromArrayOffset += length;
80              this.currentToArrayOffset += length;
81          }
82      }
83  
84  }
