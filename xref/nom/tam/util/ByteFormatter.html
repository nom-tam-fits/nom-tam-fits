25  /**
26   * This class provides mechanisms for efficiently formatting numbers and
27   * Strings. Data is appended to existing byte arrays. Note that the formatting
28   * of real or double values may differ slightly (in the last bit) from the
29   * standard Java packages since this routines are optimized for speed rather
30   * than accuracy.
31   * <p>
32   * The methods in this class create no objects.
33   * <p>
34   * If a number cannot fit into the requested space the truncateOnOverlow flag
35   * controls whether the formatter will attempt to append it using the available
36   * length in the output (a la C or Perl style formats). If this flag is set, or
37   * if the number cannot fit into space left in the buffer it is 'truncated' and
38   * the requested space is filled with a truncation fill character. A
39   * TruncationException may be thrown if the truncationThrow flag is set.
40   * <p>
41   * This class does not explicitly support separate methods for formatting reals
42   * in exponential notation. Real numbers near one are by default formatted in
43   * decimal notation while numbers with large (or very negative) exponents are
44   * formatted in exponential notation. By setting the limits at which these
45   * transitions take place the user can force either exponential or decimal
46   * notation.
47   */
48  public final class ByteFormatter {

76      /**
77       * Powers of 10. We overextend on both sides. These should perhaps be
78       * tabulated rather than computed though it may be faster to calculate them
79       * than to read in the extra bytes in the class file.
80       */
81      private static final double tenpow[];
82  
83      /** What index of tenpow is 10^0 */
84      private static final int zeropow;
85  
86      static { // Static initializer
87  
88          int min = (int) Math.floor((int) (Math.log(Double.MIN_VALUE) * ilog10));
89          int max = (int) Math.floor((int) (Math.log(Double.MAX_VALUE) * ilog10));
90          max += 1;
91  
92          tenpow = new double[(max - min) + 1];
93  
94          for (int i = 0; i < tenpow.length; i += 1) {
95              tenpow[i] = Math.pow(10, i + min);
96          }
97          zeropow = -min;
98      }

182     /**
183      * Format an int into an existing array.
184      * 
185      * @param val
186      *            Integer to be formatted
187      * @param buf
188      *            Buffer in which result is to be stored
189      * @param off
190      *            Offset within buffer
191      * @param len
192      *            Maximum length of integer
193      * @return offset of next unused character in input buffer.
194      */
195     public int format(int val, byte[] buf, int off, int len) throws TruncationException {
196 
197         // Special case
198         if (val == Integer.MIN_VALUE) {
199             if (len > 10 || (!truncateOnOverflow && buf.length - off > 10)) {
200                 return format("-2147483648", buf, off, len);
201             } else {
202                 truncationFiller(buf, off, len);
203                 return off + len;
204             }
205         }
206 
207         int pos = Math.abs(val);
208 
209         // First count the number of characters in the result.
210         // Otherwise we need to use an intermediary buffer.
211 
212         int ndig = 1;
213         int dmax = 10;
214 
215         while (ndig < 10 && pos >= dmax) {
216             ndig += 1;
217             dmax *= 10;
218         }
219 
220         if (val < 0) {
221             ndig += 1;
222         }
223 
224         // Truncate if necessary.
225         if ((truncateOnOverflow && ndig > len) || ndig > buf.length - off) {
226             truncationFiller(buf, off, len);
227             return off + len;
228         }
229 
230         // Right justify if requested.
231         if (align) {
232             off = alignFill(buf, off, len - ndig);
233         }
234 
235         // Now insert the actual characters we want -- backwards
236         // We use a do{} while() to handle the case of 0.
237 
238         off += ndig;
239 
240         int xoff = off - 1;
241         do {
242             buf[xoff] = digits[pos % 10];
243             xoff -= 1;
244             pos /= 10;
245         } while (pos > 0);
246 
247         if (val < 0) {
248             buf[xoff] = (byte) '-';
249         }
250 
251         return off;
252     }

267     /**
268      * Format a long into an existing array.
269      * 
270      * @param val
271      *            Long to be formatted
272      * @param buf
273      *            Buffer in which result is to be stored
274      * @param off
275      *            Offset within buffer
276      * @param len
277      *            Maximum length of integer
278      * @return offset of next unused character in input buffer.
279      */
280     public int format(long val, byte[] buf, int off, int len) throws TruncationException {
281 
282         // Special case
283         if (val == Long.MIN_VALUE) {
284             if (len > 19 || (!truncateOnOverflow && buf.length - off > 19)) {
285                 return format("-9223372036854775808", buf, off, len);
286             } else {
287                 truncationFiller(buf, off, len);
288                 return off + len;
289             }
290         }
291 
292         long pos = Math.abs(val);
293 
294         // First count the number of characters in the result.
295         // Otherwise we need to use an intermediary buffer.
296 
297         int ndig = 1;
298         long dmax = 10;
299 
300         // Might be faster to try to do this partially in ints
301         while (ndig < 19 && pos >= dmax) {
302 
303             ndig += 1;
304             dmax *= 10;
305         }
306 
307         if (val < 0) {
308             ndig += 1;
309         }
310 
311         // Truncate if necessary.
312 
313         if ((truncateOnOverflow && ndig > len) || ndig > buf.length - off) {
314             truncationFiller(buf, off, len);
315             return off + len;
316         }
317 
318         // Right justify if requested.
319         if (align) {
320             off = alignFill(buf, off, len - ndig);
321         }
322 
323         // Now insert the actual characters we want -- backwards.
324 
325         off += ndig;
326         int xoff = off - 1;
327 
328         buf[xoff] = (byte) '0';
329         boolean last = (pos == 0);
330 
331         while (!last) {
332 
333             // Work on ints rather than longs.
334 
335             int giga = (int) (pos % 1000000000L);
336             pos /= 1000000000L;
337 
338             last = (pos == 0);
339 
340             for (int i = 0; i < 9; i += 1) {
341 
342                 buf[xoff] = digits[giga % 10];
343                 xoff -= 1;
344                 giga /= 10;
345                 if (last && giga == 0) {
346                     break;
347                 }
348             }
349         }
350 
351         if (val < 0) {
352             buf[xoff] = (byte) '-';
353         }
354 
355         return off;
356     }

366      * Format a boolean into an existing array

371      *            The buffer in which to format the data.
372      * @param off
373      *            The starting offset within the buffer.

379     public int format(boolean val, byte[] array, int off, int len) {
380         if (align && len > 1) {
381             off = alignFill(array, off, len - 1);
382         }
383 
384         if (len > 0) {
385             if (val) {
386                 array[off] = (byte) 'T';
387             } else {
388                 array[off] = (byte) 'F';
389             }
390             off += 1;
391         }
392         return off;
393     }

400     /**
401      * Insert a String into an existing character array. If the String is longer
402      * than len, then only the the initial len characters will be inserted.
403      * 
404      * @param val
405      *            The string to be inserted. A null string will insert len
406      *            spaces.
407      * @param array
408      *            The buffer in which to insert the string.
409      * @param off
410      *            The starting offset to insert the string.
411      * @param len
412      *            The maximum number of characters to insert.
413      * @return Offset of next available character in buffer.
414      */
415     public int format(String val, byte[] array, int off, int len) {
416 
417         if (val == null) {
418             for (int i = 0; i < len; i += 1) {
419                 array[off + i] = (byte) ' ';
420             }
421             return off + len;
422         }
423 
424         int slen = val.length();
425 
426         if ((truncateOnOverflow && slen > len) || (slen > array.length - off)) {
427             val = val.substring(0, len);
428             slen = len;
429         }
430 
431         if (align && (len > slen)) {
432             off = alignFill(array, off, len - slen);
433         }
434 
435         /**
436          * We should probably require ASCII here, but for the nonce we do not
437          * [TAM 5/11]
438          */
439         System.arraycopy(val.getBytes(), 0, array, off, slen);
440         return off + slen;
441     }

456     /**
457      * Format a float into an existing byteacter array.
458      * <p>
459      * This is hard to do exactly right... The JDK code does stuff with rational
460      * arithmetic and so forth. We use a much simpler algorithm which may give
461      * an answer off in the lowest order bit. Since this is pure Java, it should
462      * still be consistent from machine to machine.
463      * <p>
464      * Recall that the binary representation of the float is of the form
465      * <tt>d = 0.bbbbbbbb x 2<sup>n</sup></tt> where there are up to 24 binary
466      * digits in the binary fraction (including the assumed leading 1 bit for
467      * normalized numbers). We find a value m such that <tt>10<sup>m</su> d</tt>
468      * is between <tt>2<sup>24</sup></tt> and <tt>>2<sup>32</sup></tt>. This
469      * product will be exactly convertible to an int with no loss of precision.
470      * Getting the decimal representation for that is trivial (see
471      * formatInteger). This is a decimal mantissa and we have an exponent (
472      * <tt>-m</tt>). All we have to do is manipulate the decimal point to where
473      * we want to see it. Errors can arise due to roundoff in the scaling
474      * multiplication, but should be very small.
475      * 
476      * @param val
477      *            Float to be formatted
478      * @param buf
479      *            Buffer in which result is to be stored
480      * @param off
481      *            Offset within buffer
482      * @param len
483      *            Maximum length of field
484      * @return Offset of next character in buffer.
485      */
486     public int format(float val, byte[] buf, int off, int len) throws TruncationException {
487 
488         float pos = (float) Math.abs(val);
489 
490         int minlen, actlen;
491 
492         // Special cases
493         if (pos == 0.) {
494             return format("0.0", buf, off, len);
495         } else if (Float.isNaN(val)) {
496             return format("NaN", buf, off, len);
497         } else if (Float.isInfinite(val)) {
498             if (val > 0) {
499                 return format("Infinity", buf, off, len);
500             } else {
501                 return format("-Infinity", buf, off, len);
502             }
503         }
504 
505         int power = (int) Math.floor((Math.log(pos) * ilog10));
506         int shift = 8 - power;
507         float scale;
508         float scale2 = 1;
509 
510         // Scale the number so that we get a number ~ n x 10^8.
511         if (shift < 30) {
512             scale = (float) tenpow[shift + zeropow];
513         } else {
514             // Can get overflow if the original number is
515             // very small, so we break out the shift
516             // into two multipliers.
517             scale2 = (float) tenpow[30 + zeropow];
518             scale = (float) tenpow[shift - 30 + zeropow];
519         }
520 
521         pos = (pos * scale) * scale2;
522 
523         // Parse the float bits.
524 
525         int bits = Float.floatToIntBits(pos);
526 
527         // The exponent should be a little more than 23
528         int exp = ((bits & 0x7F800000) >> 23) - 127;
529 
530         int numb = (bits & 0x007FFFFF);
531 
532         if (exp > -127) {
533             // Normalized....
534             numb |= (0x00800000);
535         } else {
536             // Denormalized
537             exp += 1;
538         }
539 
540         // Multiple this number by the excess of the exponent
541         // over 24. This completes the conversion of float to int
542         // (<<= did not work on Alpha TruUnix)
543 
544         numb = numb << (exp - 23L);
545 
546         // Get a decimal mantissa.
547         boolean oldAlign = align;
548         align = false;
549         int ndig = format(numb, tbuf1, 0, 32);
550         align = oldAlign;
551 
552         // Now format the float.
553 
554         return combineReal(val, buf, off, len, tbuf1, ndig, shift);
555     }

570     /**
571      * Format a double into an existing character array.
572      * <p>
573      * This is hard to do exactly right... The JDK code does stuff with rational
574      * arithmetic and so forth. We use a much simpler algorithm which may give
575      * an answer off in the lowest order bit. Since this is pure Java, it should
576      * still be consistent from machine to machine.
577      * <p>
578      * Recall that the binary representation of the double is of the form
579      * <tt>d = 0.bbbbbbbb x 2<sup>n</sup></tt> where there are up to 53 binary
580      * digits in the binary fraction (including the assumed leading 1 bit for
581      * normalized numbers). We find a value m such that <tt>10<sup>m</su> d</tt>
582      * is between <tt>2<sup>53</sup></tt> and <tt>>2<sup>63</sup></tt>. This
583      * product will be exactly convertible to a long with no loss of precision.
584      * Getting the decimal representation for that is trivial (see formatLong).
585      * This is a decimal mantissa and we have an exponent (<tt>-m</tt>). All we
586      * have to do is manipulate the decimal point to where we want to see it.
587      * Errors can arise due to roundoff in the scaling multiplication, but
588      * should be no more than a single bit.
589      * 
590      * @param val
591      *            Double to be formatted
592      * @param buf
593      *            Buffer in which result is to be stored
594      * @param off
595      *            Offset within buffer
596      * @param len
597      *            Maximum length of integer
598      * @return offset of next unused character in input buffer.
599      */
600     public int format(double val, byte[] buf, int off, int len) throws TruncationException {
601 
602         double pos = Math.abs(val);
603 
604         int minlen, actlen;
605 
606         // Special cases -- It is OK if these get truncated.
607         if (pos == 0.) {
608             return format("0.0", buf, off, len);
609         } else if (Double.isNaN(val)) {
610             return format("NaN", buf, off, len);
611         } else if (Double.isInfinite(val)) {
612             if (val > 0) {
613                 return format("Infinity", buf, off, len);
614             } else {
615                 return format("-Infinity", buf, off, len);
616             }
617         }
618 
619         int power = (int) (Math.log(pos) * ilog10);
620         int shift = 17 - power;
621         double scale;
622         double scale2 = 1;
623 
624         // Scale the number so that we get a number ~ n x 10^17.
625         if (shift < 200) {
626             scale = tenpow[shift + zeropow];
627         } else {
628             // Can get overflow if the original number is
629             // very small, so we break out the shift
630             // into two multipliers.
631             scale2 = tenpow[200 + zeropow];
632             scale = tenpow[shift - 200 + zeropow];
633         }
634 
635         pos = (pos * scale) * scale2;
636 
637         // Parse the double bits.
638 
639         long bits = Double.doubleToLongBits(pos);
640 
641         // The exponent should be a little more than 52.
642         int exp = (int) (((bits & 0x7FF0000000000000L) >> 52) - 1023);
643 
644         long numb = (bits & 0x000FFFFFFFFFFFFFL);
645 
646         if (exp > -1023) {
647             // Normalized....
648             numb |= (0x0010000000000000L);
649         } else {
650             // Denormalized
651             exp += 1;
652         }
653 
654         // Multiple this number by the excess of the exponent
655         // over 52. This completes the conversion of double to long.
656         numb = numb << (exp - 52);
657 
658         // Get a decimal mantissa.
659         boolean oldAlign = align;
660         align = false;
661         int ndig = format(numb, tbuf1, 0, 32);
662         align = oldAlign;
663 
664         // Now format the double.
665 
666         return combineReal(val, buf, off, len, tbuf1, ndig, shift);
667     }

669     /**
670      * This method formats a double given a decimal mantissa and exponent
671      * information.
672      * 
673      * @param val
674      *            The original number
675      * @param buf
676      *            Output buffer
677      * @param off
678      *            Offset into buffer
679      * @param len
680      *            Maximum number of characters to use in buffer.
681      * @param mant
682      *            A decimal mantissa for the number.
683      * @param lmant
684      *            The number of characters in the mantissa
685      * @param shift
686      *            The exponent of the power of 10 that we shifted val to get the
687      *            given mantissa.
688      * @return Offset of next available character in buffer.
689      */
690     int combineReal(double val, byte[] buf, int off, int len, byte[] mant, int lmant, int shift) throws TruncationException {
691 
692         // First get the minimum size for the number
693 
694         double pos = Math.abs(val);
695         boolean simple = false;
696         int minSize;
697         int maxSize;
698 
699         if (pos >= simpleMin && pos <= simpleMax) {
700             simple = true;
701         }
702 
703         int exp = lmant - shift - 1;
704         int lexp = 0;
705 
706         if (!simple) {
707 
708             boolean oldAlign = align;
709             align = false;
710             lexp = format(exp, tbuf2, 0, 32);
711             align = oldAlign;
712 
713             minSize = lexp + 2; // e.g., 2e-12
714             maxSize = lexp + lmant + 2; // add in "." and e
715         } else {
716             if (exp >= 0) {
717                 minSize = exp + 1; // e.g. 32
718 
719                 // Special case. E.g., 99.9 has
720                 // minumum size of 3.
721                 int i;
722                 for (i = 0; i < lmant && i <= exp; i += 1) {
723                     if (mant[i] != (byte) '9') {
724                         break;
725                     }
726                 }
727                 if (i > exp && i < lmant && mant[i] >= (byte) '5') {
728                     minSize += 1;
729                 }
730 
731                 maxSize = lmant + 1; // Add in "."
732                 if (maxSize <= minSize) { // Very large numbers.
733                     maxSize = minSize + 1;
734                 }
735             } else {
736                 minSize = 2;
737                 maxSize = 1 + Math.abs(exp) + lmant;
738             }
739         }
740         if (val < 0) {
741             minSize += 1;
742             maxSize += 1;
743         }
744 
745         // Can the number fit?
746         if ((truncateOnOverflow && minSize > len) || (minSize > buf.length - off)) {
747             truncationFiller(buf, off, len);
748             return off + len;
749         }
750 
751         // Do we need to align it?
752         if (maxSize < len && align) {
753             int nal = len - maxSize;
754             off = alignFill(buf, off, nal);
755             len -= nal;
756         }
757 
758         int off0 = off;
759 
760         // Now begin filling in the buffer.
761         if (val < 0) {
762             buf[off] = (byte) '-';
763             off += 1;
764             len -= 1;
765         }
766 
767         if (simple) {
768             return Math.abs(mantissa(mant, lmant, exp, simple, buf, off, len));
769         } else {
770             off = mantissa(mant, lmant, 0, simple, buf, off, len - lexp - 1);
771             if (off < 0) {
772                 off = -off;
773                 len -= off;
774                 // Handle the expanded exponent by filling
775                 if (exp == 9 || exp == 99) {
776                     // Cannot fit...
777                     if (off + len == minSize) {
778                         truncationFiller(buf, off, len);
779                         return off + len;
780                     } else {
781                         // Steal a character from the mantissa.
782                         off -= 1;
783                     }
784                 }
785                 exp += 1;
786                 lexp = format(exp, tbuf2, 0, 32);
787             }
788             buf[off] = (byte) 'E';
789             off += 1;
790             System.arraycopy(tbuf2, 0, buf, off, lexp);
791             return off + lexp;
792         }
793     }

795     /**
796      * Write the mantissa of the number. This method addresses the subtleties
797      * involved in rounding numbers.
798      */
799     int mantissa(byte[] mant, int lmant, int exp, boolean simple, byte[] buf, int off, int len) {
800 
801         // Save in case we need to extend the number.
802         int off0 = off;
803         int pos = 0;
804 
805         if (exp < 0) {
806             buf[off] = (byte) '0';
807             len -= 1;
808             off += 1;
809             if (len > 0) {
810                 buf[off] = (byte) '.';
811                 off += 1;
812                 len -= 1;
813             }
814             // Leading 0s in small numbers.
815             int cexp = exp;
816             while (cexp < -1 && len > 0) {
817                 buf[off] = (byte) '0';
818                 cexp += 1;
819                 off += 1;
820                 len -= 1;
821             }
822 
823         } else {
824 
825             // Print out all digits to the left of the decimal.
826             while (exp >= 0 && pos < lmant) {
827                 buf[off] = mant[pos];
828                 off += 1;
829                 pos += 1;
830                 len -= 1;
831                 exp -= 1;
832             }
833             // Trust we have enough space for this.
834             for (int i = 0; i <= exp; i += 1) {
835                 buf[off] = (byte) '0';
836                 off += 1;
837                 len -= 1;
838             }
839 
840             // Add in a decimal if we have space.
841             if (len > 0) {
842                 buf[off] = (byte) '.';
843                 len -= 1;
844                 off += 1;
845             }
846         }
847 
848         // Now handle the digits to the right of the decimal.
849         while (len > 0 && pos < lmant) {
850             buf[off] = mant[pos];
851             off += 1;
852             exp -= 1;
853             len -= 1;
854             pos += 1;
855         }
856 
857         // Now handle rounding.
858 
859         if (pos < lmant && mant[pos] >= (byte) '5') {
860             int i;
861 
862             // Increment to the left until we find a non-9
863             for (i = off - 1; i >= off0; i -= 1) {
864 
865                 if (buf[i] == (byte) '.' || buf[i] == (byte) '-') {
866                     continue;
867                 }
868                 if (buf[i] == (byte) '9') {
869                     buf[i] = (byte) '0';
870                 } else {
871                     buf[i] += 1;
872                     break;
873                 }
874             }
875 
876             // Now we handle 99.99 case. This can cause problems
877             // in two cases. If we are not using scientific notation
878             // then we may want to convert 99.9 to 100., i.e.,
879             // we need to move the decimal point. If there is no
880             // decimal point, then we must not be truncating on overflow
881             // but we should be allowed to write it to the
882             // next character (i.e., we are not at the end of buf).
883             //
884             // If we are printing in scientific notation, then we want
885             // to convert 9.99 to 1.00, i.e. we do not move the decimal.
886             // However we need to signal that the exponent should be
887             // incremented by one.
888             //
889             // We cannot have aligned the number, since that requires
890             // the full precision number to fit within the requested
891             // length, and we would have printed out the entire
892             // mantissa (i.e., pos >= lmant)
893 
894             if (i < off0) {
895 
896                 buf[off0] = (byte) '1';
897                 boolean foundDecimal = false;
898                 for (i = off0 + 1; i < off; i += 1) {
899                     if (buf[i] == (byte) '.') {
900                         foundDecimal = true;
901                         if (simple) {
902                             buf[i] = (byte) '0';
903                             i += 1;
904                             if (i < off) {
905                                 buf[i] = (byte) '.';
906                             }
907                         }
908                         break;
909                     }
910                 }
911                 if (simple && !foundDecimal) {
912                     buf[off + 1] = (byte) '0'; // 99 went to 100
913                     off += 1;
914                 }
915 
916                 off = -off; // Signal to change exponent if necessary.
917             }
918 
919         }
920 
921         return off;
922     }

924     /**
925      * Fill the buffer with truncation characters. After filling the buffer, a
926      * TruncationException will be thrown if the appropriate flag is set.
927      */
928     void truncationFiller(byte[] buffer, int offset, int length) throws TruncationException {
929 
930         for (int i = offset; i < offset + length; i += 1) {
931             buffer[i] = truncationFill;
932         }
933         if (truncationThrow) {
934             throw new TruncationException();
935         }
936         return;
937     }

939     /**
940      * Fill the buffer with blanks to align a field.
941      */
942     public int alignFill(byte[] buffer, int offset, int len) {
943         for (int i = offset; i < offset + len; i += 1) {
944             buffer[i] = (byte) ' ';
945         }
946         return offset + len;
947     }
