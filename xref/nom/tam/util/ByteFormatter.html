34  /**
35   * @deprecated  This class should not be exposed in the publi API and is meant
36   *  for internal use only in ASCII tables. Also, it may have overlapping 
37   *  functionality with other classes, which should probably be eliminated for
38   *  simplicity's sake (and thus less chance of nasty bugs).
39   * 
40   * This class provides mechanisms for efficiently formatting numbers and
41   * Strings. Data is appended to existing byte arrays. Note that the formatting
42   * of real or double values may differ slightly (in the last bit) from the
43   * standard Java packages since this routines are optimized for speed rather
44   * than accuracy.
45   * <p>
46   * The methods in this class create no objects.
47   * <p>
48   * If a number cannot fit into the requested space the truncateOnOverlow flag
49   * controls whether the formatter will attempt to append it using the available
50   * length in the output (a la C or Perl style formats). If this flag is set, or
51   * if the number cannot fit into space left in the buffer it is 'truncated' and
52   * the requested space is filled with a truncation fill character. A
53   * TruncationException may be thrown if the truncationThrow flag is set.
54   * <p>
55   * This class does not explicitly support separate methods for formatting reals
56   * in exponential notation. Real numbers near one are by default formatted in
57   * decimal notation while numbers with large (or very negative) exponents are
58   * formatted in exponential notation. By setting the limits at which these
59   * transitions take place the user can force either exponential or decimal
60   * notation.
61   */

201     /**
202      * This method formats a double given a decimal mantissa and exponent
203      * information.
204      * 
205      * @param val
206      *            The original number
207      * @param buf
208      *            Output buffer
209      * @param off
210      *            Offset into buffer
211      * @param len
212      *            Maximum number of characters to use in buffer.
213      * @param mant
214      *            A decimal mantissa for the number.
215      * @param lmant
216      *            The number of characters in the mantissa
217      * @param shift
218      *            The exponent of the power of 10 that we shifted val to get the
219      *            given mantissa.
220      * @return Offset of next available character in buffer.
221      */
222     private int combineReal(double val, byte[] buf, int off, int len, byte[] mant, int lmant, int shift) {
223 
224         // First get the minimum size for the number
225 
226         double pos = Math.abs(val);
227         boolean simple = false;
228         int minSize;
229         int maxSize;
230 
231         if (pos >= ByteFormatter.DEFAULT_SIMPLE_MIN && pos <= ByteFormatter.DEFAULT_SIMPLE_MAX) {
232             simple = true;
233         }
234 
235         int exp = lmant - shift - 1;
236         int lexp = 0;
237 
238         if (!simple) {
239             lexp = format(exp, this.tbuf2, 0, ByteFormatter.TEMP_BUFFER_SIZE);
240 
241             minSize = lexp + 2; // e.g., 2e-12
242             maxSize = lexp + lmant + 2; // add in "." and e
243         } else {
244             if (exp >= 0) {
245                 minSize = exp + 1; // e.g. 32
246 
247                 // Special case. E.g., 99.9 has
248                 // minumum size of 3.
249                 int i;
250                 for (i = 0; i < lmant && i <= exp; i++) {
251                     if (mant[i] != (byte) '9') {
252                         break;
253                     }
254                 }
255                 if (i > exp && i < lmant && mant[i] >= (byte) '5') {
256                     minSize++;
257                 }
258 
259                 maxSize = lmant + 1; // Add in "."
260                 if (maxSize <= minSize) { // Very large numbers.
261                     maxSize = minSize + 1;
262                 }
263             } else {
264                 minSize = 2;
265                 maxSize = 1 + Math.abs(exp) + lmant;
266             }
267         }
268         if (val < 0) {
269             minSize++;
270         }
271 
272         // Can the number fit?
273         if (minSize > len || minSize > buf.length - off) {
274             truncationFiller(buf, off, len);
275             return off + len;
276         }
277 
278         // Now begin filling in the buffer.
279         if (val < 0) {
280             buf[off] = (byte) '-';
281             off++;
282             len--;
283         }
284 
285         if (simple) {
286             return Math.abs(mantissa(mant, lmant, exp, simple, buf, off, len));
287         }
288         off = mantissa(mant, lmant, 0, simple, buf, off, len - lexp - 1);
289         if (off < 0) {
290             off = -off;
291             len -= off;
292             // Handle the expanded exponent by filling
293             if (exp == MAXIMUM_SINGLE_DIGIT_INTEGER || exp == MAXIMUM_TWO_DIGIT_INTEGER) {
294                 // Cannot fit...
295                 if (off + len == minSize) {
296                     truncationFiller(buf, off, len);
297                     return off + len;
298                 }
299                 // Steal a character from the mantissa.
300                 off--;
301             }
302             exp++;
303             lexp = format(exp, this.tbuf2, 0, ByteFormatter.TEMP_BUFFER_SIZE);
304         }
305         buf[off] = (byte) 'E';
306         off++;
307         System.arraycopy(this.tbuf2, 0, buf, off, lexp);
308         return off + lexp;
309     }

363     /**
364      * Format a double into an existing character array.
365      * <p>
366      * This is hard to do exactly right... The JDK code does stuff with rational
367      * arithmetic and so forth. We use a much simpler algorithm which may give
368      * an answer off in the lowest order bit. Since this is pure Java, it should
369      * still be consistent from machine to machine.
370      * <p>
371      * Recall that the binary representation of the double is of the form
372      * <code>d = 0.bbbbbbbb x 2<sup>n</sup></code> where there are up to 53
373      * binary digits in the binary fraction (including the assumed leading 1 bit
374      * for normalized numbers). We find a value m such that
375      * <code>10<sup>m</sup> d</code> is between <code>2<sup>53</sup></code> and
376      * <code>2<sup>63</sup></code>. This product will be exactly convertible to
377      * a long with no loss of precision. Getting the decimal representation for
378      * that is trivial (see formatLong). This is a decimal mantissa and we have
379      * an exponent (<code>-m</code>). All we have to do is manipulate the
380      * decimal point to where we want to see it. Errors can arise due to
381      * roundoff in the scaling multiplication, but should be no more than a
382      * single bit.
383      * 
384      * @param val
385      *            Double to be formatted
386      * @param buf
387      *            Buffer in which result is to be stored
388      * @param off
389      *            Offset within buffer
390      * @param len
391      *            Maximum length of integer
392      * @return offset of next unused character in input buffer. 
393      */
394     public int format(double val, byte[] buf, int off, int len) {
395 
396         double pos = Math.abs(val);
397 
398         // Special cases -- It is OK if these get truncated.
399         if (pos == 0.) {
400             return format("0.0", buf, off, len);
401         } else if (Double.isNaN(val)) {
402             return format(NOT_A_NUMBER, buf, off, len);
403         } else if (Double.isInfinite(val)) {
404             if (val > 0) {
405                 return format(INFINITY, buf, off, len);
406             }
407             return format(NEGATIVE_INFINITY, buf, off, len);
408         }
409 
410         int power = (int) (Math.log(pos) * ByteFormatter.I_LOG_10);
411         int shift = DOUBLE_SHIFT_BASE - power;
412         double scale;
413         double scale2 = 1;
414 
415         // Scale the number so that we get a number ~ n x 10^17.
416         if (shift < DOUBLE_SHIFT_LIMIT) {
417             scale = ByteFormatter.NUMBER_BASE_POWERS[shift + ByteFormatter.ZERO_POW];
418         } else {
419             // Can get overflow if the original number is
420             // very small, so we break out the shift
421             // into two multipliers.
422             scale2 = ByteFormatter.NUMBER_BASE_POWERS[DOUBLE_SHIFT_LIMIT + ByteFormatter.ZERO_POW];
423             scale = ByteFormatter.NUMBER_BASE_POWERS[shift - DOUBLE_SHIFT_LIMIT + ByteFormatter.ZERO_POW];
424         }
425 
426         pos = pos * scale * scale2;
427 
428         // Parse the double bits.
429 
430         long bits = Double.doubleToLongBits(pos);
431 
432         // The exponent should be a little more than 52.
433         int exp = (int) (((bits & DOUBLE_EXPONENT_BIT_MASK) >> DOUBLE_EXPONENT_EXCESS) + DOUBLE_MIN_EXPONENT);
434 
435         long numb = bits & DOUBLE_VALUE_BIT_MASK;
436 
437         // this can never be false because the min exponent of a double is -1022
438         // but lets keep it if we need it for big decimals
439         if (exp > DOUBLE_MIN_EXPONENT) {
440             // Normalized....
441             numb |= DOUBLE_EXPONENT_NORMALIZE_BIT;
442         } else {
443             // Denormalized
444             exp++;
445         }
446 
447         // Multiple this number by the excess of the exponent
448         // over 52. This completes the conversion of double to long.
449         numb = numb << exp - DOUBLE_EXPONENT_EXCESS;
450 
451         // Get a decimal mantissa.
452         int ndig = format(numb, this.tbuf1, 0, ByteFormatter.TEMP_BUFFER_SIZE);
453 
454         // Now format the double.
455 
456         return combineReal(val, buf, off, len, this.tbuf1, ndig, shift);
457     }

472     /**
473      * Format a float into an existing byteacter array.
474      * <p>
475      * This is hard to do exactly right... The JDK code does stuff with rational
476      * arithmetic and so forth. We use a much simpler algorithm which may give
477      * an answer off in the lowest order bit. Since this is pure Java, it should
478      * still be consistent from machine to machine.
479      * <p>
480      * Recall that the binary representation of the float is of the form
481      * <code>d = 0.bbbbbbbb x 2<sup>n</sup></code> where there are up to 24
482      * binary digits in the binary fraction (including the assumed leading 1 bit
483      * for normalized numbers). We find a value m such that
484      * <code>10<sup>m</sup> d</code> is between <code>2<sup>24</sup></code> and
485      * <code>2<sup>32</sup></code>. This product will be exactly convertible to
486      * an int with no loss of precision. Getting the decimal representation for
487      * that is trivial (see formatInteger). This is a decimal mantissa and we
488      * have an exponent ( <code>-m</code>). All we have to do is manipulate the
489      * decimal point to where we want to see it. Errors can arise due to
490      * roundoff in the scaling multiplication, but should be very small.
491      * 
492      * @param val
493      *            Float to be formatted
494      * @param buf
495      *            Buffer in which result is to be stored
496      * @param off
497      *            Offset within buffer
498      * @param len
499      *            Maximum length of field
500      * @return Offset of next character in buffer. @ * if the value was
501      *         truncated
502      */
503     public int format(float val, byte[] buf, int off, int len) {
504 
505         float pos = Math.abs(val);
506 
507         // Special cases
508         if (pos == 0.) {
509             return format("0.0", buf, off, len);
510         } else if (Float.isNaN(val)) {
511             return format(NOT_A_NUMBER, buf, off, len);
512         } else if (Float.isInfinite(val)) {
513             if (val > 0) {
514                 return format("Infinity", buf, off, len);
515             }
516             return format("-Infinity", buf, off, len);
517         }
518 
519         int power = (int) Math.floor(Math.log(pos) * ByteFormatter.I_LOG_10);
520         int shift = FLOAT_SHIFT_BASE - power;
521         float scale;
522         float scale2 = 1;
523 
524         // Scale the number so that we get a number ~ n x 10^8.
525         if (shift < FLOAT_SHIFT_LIMIT) {
526             scale = (float) ByteFormatter.NUMBER_BASE_POWERS[shift + ByteFormatter.ZERO_POW];
527         } else {
528             // Can get overflow if the original number is
529             // very small, so we break out the shift
530             // into two multipliers.
531             scale2 = (float) ByteFormatter.NUMBER_BASE_POWERS[FLOAT_SHIFT_LIMIT + ByteFormatter.ZERO_POW];
532             scale = (float) ByteFormatter.NUMBER_BASE_POWERS[shift - FLOAT_SHIFT_LIMIT + ByteFormatter.ZERO_POW];
533         }
534 
535         pos = pos * scale * scale2;
536 
537         // Parse the float bits.
538 
539         int bits = Float.floatToIntBits(pos);
540 
541         // The exponent should be a little more than 23
542         int exp = ((bits & FLOAT_EXPONENT_BIT_MASK) >> FLOAT_EXPONENT_EXCESS) + FLOAT_MIN_EXPONENT;
543 
544         int numb = bits & FLOAT_VALUE_BIT_MASK;
545 
546         if (exp > FLOAT_MIN_EXPONENT) {
547             // Normalized....
548             numb |= FLOAT_EXPONENT_NORMALIZE_BIT;
549         } else {
550             // Denormalized
551             exp++;
552         }
553 
554         // Multiple this number by the excess of the exponent
555         // over 24. This completes the conversion of float to int
556         // (<<= did not work on Alpha TruUnix)
557 
558         numb = numb << exp - FLOAT_EXPONENT_EXCESS;
559 
560         // Get a decimal mantissa.
561         int ndig = format(numb, this.tbuf1, 0, ByteFormatter.TEMP_BUFFER_SIZE);
562 
563         // Now format the float.
564 
565         return combineReal(val, buf, off, len, this.tbuf1, ndig, shift);
566     }

581     /**
582      * Format an int into an existing array.
583      * 
584      * @param val
585      *            Integer to be formatted
586      * @param buf
587      *            Buffer in which result is to be stored
588      * @param off
589      *            Offset within buffer
590      * @param len
591      *            Maximum length of integer
592      * @return offset of next unused character in input buffer.
593      */
594     public int format(int val, byte[] buf, int off, int len) {
595 
596         // Special case
597         if (val == Integer.MIN_VALUE) {
598             if (len > ByteFormatter.NUMBER_BASE) {
599                 return format("-2147483648", buf, off, len);
600             }
601             truncationFiller(buf, off, len);
602             return off + len;
603         }
604 
605         int pos = Math.abs(val);
606 
607         // First count the number of characters in the result.
608         // Otherwise we need to use an intermediary buffer.
609 
610         int ndig = 1;
611         int dmax = ByteFormatter.NUMBER_BASE;
612 
613         while (ndig < ByteFormatter.NUMBER_BASE && pos >= dmax) {
614             ndig++;
615             dmax *= ByteFormatter.NUMBER_BASE;
616         }
617 
618         if (val < 0) {
619             ndig++;
620         }
621 
622         // Truncate if necessary.
623         if (ndig > len || ndig > buf.length - off) {
624             truncationFiller(buf, off, len);
625             return off + len;
626         }
627 
628         // Now insert the actual characters we want -- backwards
629         // We use a do{} while() to handle the caByteFormatterse of 0.
630 
631         off += ndig;
632 
633         int xoff = off - 1;
634         do {
635             buf[xoff] = ByteFormatter.DIGITS[pos % ByteFormatter.NUMBER_BASE];
636             xoff--;
637             pos /= ByteFormatter.NUMBER_BASE;
638         } while (pos > 0);
639 
640         if (val < 0) {
641             buf[xoff] = (byte) '-';
642         }
643 
644         return off;
645     }

660     /**
661      * Format a long into an existing array.
662      * 
663      * @param val
664      *            Long to be formatted
665      * @param buf
666      *            Buffer in which result is to be stored
667      * @param off
668      *            Offset within buffer
669      * @param len
670      *            Maximum length of integer
671      * @return offset of next unused character in input buffer. @ * if the value
672      *         was truncated
673      */
674     public int format(long val, byte[] buf, int off, int len) {
675         // Special case
676         if (val == Long.MIN_VALUE) {
677             if (len > MAX_LONG_LENGTH) {
678                 return format("-9223372036854775808", buf, off, len);
679             }
680             truncationFiller(buf, off, len);
681             return off + len;
682         }
683         long pos = Math.abs(val);
684         // First count the number of characters in the result.
685         // Otherwise we need to use an intermediary buffer.
686         int ndig = 1;
687         long dmax = ByteFormatter.NUMBER_BASE;
688         // Might be faster to try to do this partially in ints
689         while (ndig < MAX_LONG_LENGTH && pos >= dmax) {
690             ndig++;
691             dmax *= ByteFormatter.NUMBER_BASE;
692         }
693         if (val < 0) {
694             ndig++;
695         }
696         // Truncate if necessary.
697         if (ndig > len || ndig > buf.length - off) {
698             truncationFiller(buf, off, len);
699             return off + len;
700         }
701         // Now insert the actual characters we want -- backwards.
702         off += ndig;
703         int xoff = off - 1;
704         buf[xoff] = (byte) '0';
705         boolean last = pos == 0;
706         while (!last) {
707             // Work on ints rather than longs.
708             int giga = (int) (pos % LONG_TO_INT_MODULO);
709             pos /= LONG_TO_INT_MODULO;
710             last = pos == 0;
711             for (int i = 0; i < MAXIMUM_SINGLE_DIGIT_INTEGER; i++) {
712 
713                 buf[xoff] = ByteFormatter.DIGITS[giga % ByteFormatter.NUMBER_BASE];
714                 xoff--;
715                 giga /= ByteFormatter.NUMBER_BASE;
716                 if (last && giga == 0) {
717                     break;
718                 }
719             }
720         }
721 
722         if (val < 0) {
723             buf[xoff] = (byte) '-';
724         }
725 
726         return off;
727     }

729     /**
730      * Insert a string at the beginning of an array. * @return Offset of next
731      * available character in buffer.
732      * 
733      * @param val
734      *            The string to be inserted. A null string will insert len
735      *            spaces.
736      * @param array
737      *            The buffer in which to insert the string.
738      * @return Offset of next available character in buffer.
739      */
740     public int format(String val, byte[] array) {
741         return format(val, array, 0, array.length);
742     }

744     /**
745      * Insert a String into an existing character array. If the String is longer
746      * than len, then only the the initial len characters will be inserted.
747      * 
748      * @param val
749      *            The string to be inserted. A null string will insert len
750      *            spaces.
751      * @param array
752      *            The buffer in which to insert the string.
753      * @param off
754      *            The starting offset to insert the string.
755      * @param len
756      *            The maximum number of characters to insert.
757      * @return Offset of next available character in buffer.
758      */
759     public int format(String val, byte[] array, int off, int len) {
760 
761         if (val == null) {
762             for (int i = 0; i < len; i++) {
763                 array[off + i] = (byte) ' ';
764             }
765             return off + len;
766         }
767 
768         int slen = val.length();
769 
770         if (slen > len || slen > array.length - off) {
771             val = val.substring(0, len);
772             slen = len;
773         }
774 
775         System.arraycopy(AsciiFuncs.getBytes(val), 0, array, off, slen);
776         return off + slen;
777     }

779     /**
780      * Write the mantissa of the number. This method addresses the subtleties
781      * involved in rounding numbers.
782      */
783     private int mantissa(byte[] mant, int lmant, int exp, boolean simple, byte[] buf, int off, int len) {
784 
785         // Save in case we need to extend the number.
786         int off0 = off;
787         int pos = 0;
788 
789         if (exp < 0) {
790             buf[off] = (byte) '0';
791             len--;
792             off++;
793             if (len > 0) {
794                 buf[off] = (byte) '.';
795                 off++;
796                 len--;
797             }
798             // Leading 0s in small numbers.
799             int cexp = exp;
800             while (cexp < -1 && len > 0) {
801                 buf[off] = (byte) '0';
802                 cexp++;
803                 off++;
804                 len--;
805             }
806 
807         } else {
808 
809             // Print out all digits to the left of the decimal.
810             while (exp >= 0 && pos < lmant) {
811                 buf[off] = mant[pos];
812                 off++;
813                 pos++;
814                 len--;
815                 exp--;
816             }
817             // Trust we have enough space for this.
818             for (int i = 0; i <= exp; i++) {
819                 buf[off] = (byte) '0';
820                 off++;
821                 len--;
822             }
823 
824             // Add in a decimal if we have space.
825             if (len > 0) {
826                 buf[off] = (byte) '.';
827                 len--;
828                 off++;
829             }
830         }
831 
832         // Now handle the digits to the right of the decimal.
833         while (len > 0 && pos < lmant) {
834             buf[off] = mant[pos];
835             off++;
836             exp--;
837             len--;
838             pos++;
839         }
840 
841         // Now handle rounding.
842 
843         if (pos < lmant && mant[pos] >= (byte) '5') {
844             int i;
845 
846             // Increment to the left until we find a non-9
847             for (i = off - 1; i >= off0; i--) {
848 
849                 if (buf[i] == (byte) '.' || buf[i] == (byte) '-') {
850                     continue;
851                 }
852                 if (buf[i] == (byte) '9') {
853                     buf[i] = (byte) '0';
854                 } else {
855                     buf[i]++;
856                     break;
857                 }
858             }
859 
860             // Now we handle 99.99 case. This can cause problems
861             // in two cases. If we are not using scientific notation
862             // then we may want to convert 99.9 to 100., i.e.,
863             // we need to move the decimal point. If there is no
864             // decimal point, then we must not be truncating on overflow
865             // but we should be allowed to write it to the
866             // next character (i.e., we are not at the end of buf).
867             //
868             // If we are printing in scientific notation, then we want
869             // to convert 9.99 to 1.00, i.e. we do not move the decimal.
870             // However we need to signal that the exponent should be
871             // incremented by one.
872             //
873             // We cannot have aligned the number, since that requires
874             // the full precision number to fit within the requested
875             // length, and we would have printed out the entire
876             // mantissa (i.e., pos >= lmant)
877 
878             if (i < off0) {
879 
880                 buf[off0] = (byte) '1';
881                 boolean foundDecimal = false;
882                 for (i = off0 + 1; i < off; i++) {
883                     if (buf[i] == (byte) '.') {
884                         foundDecimal = true;
885                         if (simple) {
886                             buf[i] = (byte) '0';
887                             i++;
888                             if (i < off) {
889                                 buf[i] = (byte) '.';
890                             }
891                         }
892                         break;
893                     }
894                 }
895                 if (simple && !foundDecimal) {
896                     buf[off + 1] = (byte) '0'; // 99 went to 100
897                     off++;
898                 }
899 
900                 off = -off; // Signal to change exponent if necessary.
901             }
902 
903         }
904 
905         return off;
906     }

908     /**
909      * Fill the buffer with truncation characters. After filling the buffer, a
910      * TruncationException will be thrown if the appropriate flag is set.
911      */
912     private void truncationFiller(byte[] buffer, int offset, int length) {
913         for (int i = offset; i < offset + length; i++) {
914             buffer[i] = ByteFormatter.TRUNCATION_FILL;
915         }
916     }
