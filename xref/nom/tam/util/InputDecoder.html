45  /**
46   * Efficient base class for decoding of binary input into Java arrays.
47   * 
48   * @author Attila Kovacs
49   * 
50   * @since 1.16
51   * 
52   * @see OutputEncoder
53   * @see ArrayDataFile
54   * @see ArrayInputStream
55   * @see ArrayOutputStream

57  public abstract class InputDecoder {

59      /** The buffer size for array translation */
60      private static final int BUFFER_SIZE = 8 * FitsFactory.FITS_BLOCK_SIZE;
61  
62      /** bit mask for 1 byte */
63      private static final int BYTE_MASK = 0xFF;
64  
65      /** bit mask for a 16-byte integer (a Java <code>short</code>). */
66      private static final int SHORT_MASK = 0xFFFF;

117     /**
118      * Makes sure that an elements of the specified size is fully available in
119      * the buffer, prompting additional reading of the underlying stream as
120      * appropriate (but not beyond the limit set by
121      * {@link #loadBytes(long, int)}.
122      * 
123      * @param size the number of bytes we need at once from the buffer
124      * 
125      * @return <code>true</code> if the requested number of bytes are, or could
126      *             be made, available. Otherwise <code>false</code>.
127      * 
128      * @throws IOException if there was an underlying IO error, other than the
129      *             end of file, while trying to fetch additional data from the
130      *             underlying input
131      */
132     boolean makeAvailable(int size) throws IOException {
133         // TODO Once the deprecated BufferDecoder is retired, this should become
134         // a private method of InputBuffer (with buf. prefixed removed below).
135         while (buf.buffer.remaining() < size) {
136             if (!buf.fetch()) {
137                 return false;
138             }
139         }
140         return true;
141     }

156     /**
157      * Reads bytes into an array from the input. See the contract of
158      * {@link InputStream#read(byte[], int, int)}.
159      * 
160      * @param b the destination array
161      * @param start the first index in the array to be populated
162      * @param length the number of bytes to read into the array.
163      * 
164      * @return the number of bytes successfully read, or -1 if at the end of the
165      *             file.
166      * 
167      * @throws IOException if an IO error, other than the end-of-file prevented
168      *             the read.
169      */
170     protected synchronized int read(byte[] b, int start, int length) throws IOException {
171         return in.read(b, start, length);
172     }

174     /**
175      * Reads bytes to fill the supplied buffer with the requested number of bytes
176      * from the given starting buffer index. If not enough bytes are avaialable
177      * in the file to deliver the reqauested number of bytes the buffer, an
178      * {@link EOFException} will be thrown.
179      * 
180      * @param b the buffer
181      * @param off the buffer index at which to start reading data
182      * @param len the total number of bytes to read.
183      * 
184      * @throws IOException if there was an IO error before the requested number
185      *             of bytes could all be read.
186      */
187     protected void readFully(byte[] b, int off, int len) throws IOException {
188         while (len > 0) {
189             int n = read(b, off, len);
190             if (n < 0) {
191                 throw new EOFException();
192             }
193             off += n;
194             len -= n;
195         }
196     }

198     /**
199      * Based on {@link #readArray(Object)}, but guaranteeing a complete read of
200      * the supplied object or else an {@link EOFException} is thrown.
201      * 
202      * @param o the array, including multi-dimensional, and heterogeneous arrays
203      *            of arrays.
204      * 
205      * @throws IOException if there was an IO error, uncluding end-of-file (
206      *             {@link EOFException}, before all components of the supplied
207      *             array were populated from the input.
208      * @throws IllegalArgumentException if the argument is not a Java array, or
209      *             is or contains elements that do not have supported conversions
210      *             from binary representation.
211      * 
212      * @see #readArray(Object)
213      * @see #readImage(Object)
214      */
215     public void readArrayFully(Object o) throws IOException, IllegalArgumentException {
216         if (readArray(o) != FitsEncoder.computeSize(o)) {
217             throw new EOFException("Incomplete array read (FITS encoding).");
218         }
219     }

222      * See the contract of {@link ArrayDataInput#readLArray(Object)}.
223      * 
224      * @param o an array, to be populated
225      * 
226      * @return the actual number of bytes read from the input, or -1 if already
227      *             at the end-of-file.
228      * 
229      * @throws IllegalArgumentException if the argument is not an array or if it
230      *             contains an element that is not supported for decoding.
231      * @throws IOException if there was an IO error reading from the input
232      * 
233      * @see #readArrayFully(Object)
234      */
235     public abstract long readArray(Object o) throws IOException, IllegalArgumentException;

237     /**
238      * Like {@link #readArrayFully(Object)} but strictly for numerical types
239      * only.
240      * 
241      * @param o An any-dimensional array containing only numerical types
242      * 
243      * @throws IllegalArgumentException if the argument is not an array or if it
244      *             contains an element that is not supported.
245      * @throws IOException if there was an IO error, uncluding end-of-file (
246      *             {@link EOFException}, before all components of the supplied
247      *             array were populated from the input.
248      * 
249      * @see #readArrayFully(Object)
250      * 
251      * @since 1.18
252      */
253     public void readImage(Object o) throws IOException, IllegalArgumentException {
254         if (o == null) {
255             return;
256         }
257 
258         if (!o.getClass().isArray()) {
259             throw new IllegalArgumentException("Not an array: " + o.getClass().getName());
260         }
261 
262         long size = FitsEncoder.computeSize(o);
263         if (size == 0) {
264             return;
265         }
266 
267         getInputBuffer().loadBytes(size, 1);
268         if (getImage(o) != size) {
269             throw new EOFException("Incomplete image read.");
270         }
271     }
272 
273     private long getImage(Object o) throws IOException, IllegalArgumentException {
274         int length = Array.getLength(o);
275         if (length == 0) {
276             return 0L;
277         }
278 
279         if (o instanceof byte[]) {
280             return buf.get((byte[]) o, 0, length);
281         }
282         if (o instanceof short[]) {
283             return buf.get((short[]) o, 0, length) * Short.BYTES;
284         }
285         if (o instanceof int[]) {
286             return buf.get((int[]) o, 0, length) * Integer.BYTES;
287         }
288         if (o instanceof float[]) {
289             return buf.get((float[]) o, 0, length) * Float.BYTES;
290         }
291         if (o instanceof long[]) {
292             return buf.get((long[]) o, 0, length) * Long.BYTES;
293         }
294         if (o instanceof double[]) {
295             return buf.get((double[]) o, 0, length) * Double.BYTES;
296         }
297         if (!(o instanceof Object[])) {
298             throw new IllegalArgumentException("Not a numerical image type: " + o.getClass().getName());
299         }
300 
301         Object[] array = (Object[]) o;
302         long count = 0L;
303 
304         // Process multidim arrays recursively.
305         for (int i = 0; i < length; i++) {
306             try {
307                 count += getImage(array[i]);
308             } catch (EOFException e) {
309                 return eofCheck(e, count, -1L);
310             }
311         }
312         return count;
313     }

315     /**
316      * Decides what to do when an {@link EOFException} is encountered after
317      * having read some number of bytes from the input. The default behavior is
318      * to re-throw the exception only if no data at all was obtained from the
319      * input, otherwise return the non-zero byte count of data that were
320      * successfully read. Subclass implementations may override this method to
321      * adjust if an when {@link EOFException} is thrown upon an incomplete read.
322      * 
323      * @param e the exception that was thrown, or <code>null</code>.
324      * @param got the number of elements successfully read
325      * @param expected the number of elements expected
326      * 
327      * @return the number of elements successfully read (same as
328      *             <code>got</code>).
329      * 
330      * @throws EOFException the rethrown exception, or a new one, as appropriate
331      */
332     long eofCheck(EOFException e, long got, long expected) throws EOFException {
333         if (got == 0) {
334             if (e == null) {
335                 throw new EOFException();
336             }
337             throw e;
338         }
339         return got;
340     }

342     /**
343      * <p>
344      * The conversion buffer for decoding binary data representation into Java
345      * arrays (objects).
346      * </p>
347      * <p>
348      * The buffering is most efficient, if we fist specify how many bytes of
349      * input maybe be consumed first (buffered from the input), via
350      * {@link #loadBytes(long, int)}. After that, we can call the get routines of
351      * this class to return binary data converted to Java format until we exhaust
352      * the specified alotment of bytes.
353      * </p>
354      * 
355      * <pre>
356      * // The data we want to retrieve
357      * double d;
358      * int i;
359      * short[] shortArray = new short[100];
360      * float[] floaTarray = new float[48];
361      * 
362      * // We convert from the binary format to Java format using
363      * // the local conversion buffer
364      * ConversionBuffer buf = getBuffer();
365      * 
366      * // We can allow the conversion buffer to read enough bytes for all
367      * // data we want to retrieve:
368      * buf.loadBytes(FitsIO.BYTES_IN_DOUBLE + FitsIO.BYTES_IN_INT + FitsIO.BYTES_IN_SHORT * shortArray.length
369      *         + FitsIO.BYTES_IN_FLOAT * floatArray.length);
370      * 
371      * // Now we can get the data with minimal underlying IO calls...
372      * d = buf.getDouble();
373      * i = buf.getInt();
374      * 
375      * for (int i = 0; i &lt; shortArray.length; i++) {
376      *     shortArray[i] = buf.getShort();
377      * }
378      * 
379      * for (int i = 0; i &lt; floatArray.length; i++) {
380      *     floatArray[i] = buf.getFloat();
381      * }
382      * </pre>
383      * <p>
384      * In the special case that one needs just a single element (or a few single
385      * elements) from the input, rather than lots of elements or arrays, one may
386      * use {@link #loadOne(int)} instead of {@link #loadBytes(long, int)} to read
387      * just enough bytes for a single data element from the input before each
388      * conversion. For example:
389      * </p>
390      * 
391      * <pre>
392      * ConversionBuffer buf = getBuffer();
393      * 
394      * buf.loadOne(FitsIO.BYTES_IN_FLOAT);
395      * float f = buf.getFloat();
396      * </pre>
397      * 
398      * @author Attila Kovacs
399      */
400     protected final class InputBuffer {
401 
402         /** the byte array in which to buffer data from the input */
403         private final byte[] data;
404 
405         /** the buffer wrapped for NIO access */
406         private final ByteBuffer buffer;
407 
408         /** The current type-specific view of the buffer or null */
409         private Buffer view;
410 
411         /** the number of bytes requested, but not yet buffered */
412         private long pending = 0;
413 
414         private InputBuffer(int size) {
415             this.data = new byte[size];
416             buffer = ByteBuffer.wrap(data);
417         }
418 
419         /**
420          * Sets the byte order of the binary data representation from which we
421          * are decoding data.
422          * 
423          * @param order the new byte order
424          * 
425          * @see #byteOrder()
426          * @see ByteBuffer#order(ByteOrder)
427          */
428         protected void setByteOrder(ByteOrder order) {
429             buffer.order(order);
430         }
431 
432         /**
433          * Returns the current byte order of the binary data representation from
434          * which we are decoding.
435          * 
436          * @return the byte order
437          * 
438          * @see #setByteOrder(ByteOrder)
439          * @see ByteBuffer#order()
440          */
441         protected ByteOrder byteOrder() {
442             return buffer.order();
443         }
444 
445         private boolean isViewingAs(Class<? extends Buffer> type) {
446             if (view == null) {
447                 return false;
448             }
449             return type.isAssignableFrom(view.getClass());
450         }
451 
452         private void assertView(ElementType<?> type) {
453             if (!isViewingAs(type.bufferClass())) {
454                 view = type.asTypedBuffer(buffer);
455             }
456         }
457 
458         private void rewind() {
459             buffer.rewind();
460             view = null;
461         }
462 
463         /**
464          * Set the number of bytes we can buffer from the input for subsequent
465          * rettrieval from this buffer. The get methods of this class will be
466          * ensured not to fetch data from the input beyond the requested size.
467          * 
468          * @param n the number of elements we can read and buffer from the input
469          * @param size the number of bytes in each elements.
470          */
471         protected void loadBytes(long n, int size) {
472             rewind();
473             buffer.limit(0);
474             this.pending = n * size;
475         }
476 
477         /**
478          * Loads just a single element of the specified byte size. The element
479          * must fit into the conversion buffer, and it is up to the caller to
480          * ensure that. The method itself does not check.
481          * 
482          * @param size The number of bytes in the element
483          * 
484          * @return <code>true</code> if the data was successfully read from the
485          *             uderlying stream or file, otherwise <code>false</code>.
486          * 
487          * @throws IOException if there was an IO error, other than the
488          *             end-of-file.
489          */
490         protected boolean loadOne(int size) throws IOException {
491             this.pending = size;
492             rewind();
493             buffer.limit(0);
494             return makeAvailable(size);
495         }
496 
497         /**
498          * Reads more data into the buffer from the underlying stream, attempting
499          * to fill the buffer if possible.
500          * 
501          * @return <code>true</code> if data was successfully buffered from the
502          *             underlying intput or the buffer is already full. Otherwise
503          *             <code>false</code>.
504          * 
505          * @throws IOException if there as an IO error, other than the end of
506          *             file, while trying to read more data from the underlying
507          *             input into the buffer.
508          */
509         private boolean fetch() throws IOException {
510             int remaining = buffer.remaining();
511 
512             if (remaining > 0) {
513                 System.arraycopy(data, buffer.position(), data, 0, remaining);
514             }
515             rewind();
516 
517             int n = (int) Math.min(pending, data.length - remaining);
518             n = in.read(data, remaining, n);
519             if (n < 0) {
520                 return false;
521             }
522             buffer.limit(remaining + n);
523             pending -= n;
524 
525             return true;
526         }
527 
528         /**
529          * Retrieves a single byte from the buffer. Before data can be retrieved
530          * with this method the should be 'loaded' into the buffer via
531          * {@link #loadOne(int)} or {@link #loadBytes(long, int)}. This method is
532          * appropriate for retrieving one or a fwew bytes at a time. For bulk
533          * input of bytes, you should use
534          * {@link InputDecoder#read(byte[], int, int)} instead for superior
535          * performance.
536          * 
537          * @return the byte value, or -1 if no more data is available from the
538          *             buffer or the underlying input.
539          * 
540          * @throws IOException if there as an IO error, other than the end of
541          *             file, while trying to read more data from the underlying
542          *             input into the buffer.
543          * 
544          * @see #loadOne(int)
545          * @see #loadBytes(long, int)
546          */
547         protected int get() throws IOException {
548             if (makeAvailable(1)) {
549                 view = null;
550                 return buffer.get() & BYTE_MASK;
551             }
552             return -1;
553         }
554 
555         /**
556          * Retrieves a 2-byte unsigned integer from the buffer. Before data can
557          * be retrieved with this method the should be 'loaded' into the buffer
558          * via {@link #loadOne(int)} or {@link #loadBytes(long, int)}.
559          * 
560          * @return the 16-bit integer value, or -1 if no more data is available
561          *             from the buffer or the underlying input.
562          * 
563          * @throws IOException if there as an IO error, other than the end of
564          *             file, while trying to read more data from the underlying
565          *             input into the buffer.
566          * 
567          * @see #loadOne(int)
568          * @see #loadBytes(long, int)
569          */
570         protected int getUnsignedShort() throws IOException {
571             if (makeAvailable(Short.BYTES)) {
572                 view = null;
573                 return buffer.getShort() & SHORT_MASK;
574             }
575             return -1;
576         }

583          * @return the 32-bit integer value.
584          *

658                 view = null;

664         /**
665          * Retrieves a sequence of signed bytes from the buffer. Before data can
666          * be retrieved with this method the should be 'loaded' into the buffer
667          * via {@link #loadBytes(long, int)}.
668          * 
669          * @param dst Java array in which to store the retrieved sequence of
670          *            elements
671          * @param from the array index for storing the first element retrieved
672          * @param n the number of elements to retrieve
673          * 
674          * @return the number of elements successfully retrieved
675          * 
676          * @throws EOFException if already at the end of file.
677          * @throws IOException if there was an IO error before, before requested
678          *             number of bytes could be read
679          * 
680          * @see #loadBytes(long, int)
681          * 
682          * @since 1.18
683          */
684         protected int get(byte[] dst, int from, int n) throws EOFException, IOException {
685             if (n == 1) {
686                 int i = get();
687                 if (i < 0) {
688                     throw new EOFException();
689                 }
690                 dst[from] = (byte) i;
691                 return 1;
692             }
693 
694             view = null;
695             int got = 0;
696 
697             while (got < n) {
698                 if (!makeAvailable(1)) {
699                     return (int) eofCheck(null, got, n);
700                 }
701                 int m = Math.min(n - got, buffer.remaining());
702                 buffer.get(dst, from + got, m);
703                 got += m;
704             }
705 
706             return got;
707         }

709         /**
710          * Retrieves a sequence of big-endian 16-bit signed integers from the
711          * buffer. Before data can be retrieved with this method the should be
712          * 'loaded' into the buffer via {@link #loadBytes(long, int)}.
713          * 
714          * @param dst Java array in which to store the retrieved sequence of
715          *            elements
716          * @param from the array index for storing the first element retrieved
717          * @param n the number of elements to retrieve
718          * 
719          * @return the number of elements successfully retrieved
720          * 
721          * @throws EOFException if already at the end of file.
722          * @throws IOException if there was an IO error before, before requested
723          *             number of bytes could be read
724          * 
725          * @see #loadBytes(long, int)
726          * 
727          * @since 1.18

729         protected int get(short[] dst, int from, int n) throws EOFException, IOException {

853         /**
854          * Retrieves a sequence of values from the buffer. Before data can be
855          * retrieved with this method the should be 'loaded' into the buffer via
856          * {@link #loadBytes(long, int)}.
857          * 
858          * @param dst Java array in which to store the retrieved sequence of
859          *            elements
860          * @param from the array index for storing the first element retrieved
861          * @param n the number of elements to retrieve
862          * 
863          * @return the number of elements successfully retrieved
864          * 
865          * @throws EOFException if already at the end of file.
866          * @throws IOException if there was an IO error before, before requested
867          *             number of bytes could be read
868          * 
869          * @see #loadBytes(long, int)
870          * 
871          * @since 1.18
872          */
873         @SuppressWarnings("unchecked")
874         private <B extends Buffer> int get(ElementType<B> e, Object dst, int from, int n) throws EOFException, IOException {
875             int got = 0;
876 
877             while (got < n) {
878                 if (!makeAvailable(e.size())) {
879                     return (int) eofCheck(null, got, n);
880                 }
881                 assertView(e);
882                 int m = Math.min(n - got, view.remaining());
883                 e.getArray((B) view, dst, from + got, m);
884                 buffer.position(buffer.position() + m * e.size());
885                 got += m;
886             }
887 
888             return got;
889         }
