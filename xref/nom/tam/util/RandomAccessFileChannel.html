46  /**
47   * A random accessible {@link FileChannel} implementation for FITS I/O purposes. It enables file-like access for
48   * nom-tam-fits that goes beyond local files. This class provides a generic implementation, which may be used, for
49   * example, with the NIO SPI file system provider to Amazon S3.
50   *
51   * @author Dustin Jenkins, Attila Kovacs
52   * 
53   * @since  1.21
54   *
55   * @see    <a href="https://github.com/awslabs/aws-java-nio-spi-for-s3">AWS NIO SPI library</a>
56   * @see    FitsFile
57   */
58  public class RandomAccessFileChannel implements RandomAccessFileIO {
59      private static final Logger LOGGER = LoggerHelper.getLogger(RandomAccessFileChannel.class);
60  
61      private final long fileSize;
62      private final FileChannel fileChannel;
63  
64      /**
65       * Constructor for a readable/writable random accessible {@link FileChannel}, which can be used for accessing FITS
66       * files.
67       * 
68       * @param  filePath    The path to the file
69       * 
70       * @throws IOException if the file is not accessible
71       * 
72       * @author             Dustin Jenkins
73       * 
74       * @since              1.21
75       * 
76       * @see                #RandomAccessFileChannel(Path, boolean)
77       */
78      public RandomAccessFileChannel(final Path filePath) throws IOException {
79          this(filePath, true);
80      }
81  
82      /**
83       * Constructor for a random accessible {@link FileChannel}, for reading or for reading and writing. It may be used
84       * for accessing FITS files.
85       * 
86       * @param  filePath    The path to the file
87       * @param  readOnly    <code>true</code> if the file should only be used for reading, otherwise <code>false</code>.
88       * 
89       * @throws IOException if the file is not accessible
90       * 
91       * @author             Dustin Jenkins
92       * 
93       * @since              1.21
94       * 
95       * @see                #RandomAccessFileChannel(Path, boolean)
96       */
97      @SuppressWarnings("resource")
98      public RandomAccessFileChannel(final Path filePath, final boolean readOnly) throws IOException {
99          this(Files.size(filePath), FileChannel.open(filePath, readOnly ? new OpenOption[] {StandardOpenOption.READ} :
100                 new OpenOption[] {StandardOpenOption.READ, StandardOpenOption.WRITE}));
101     }
102 
103     /**
104      * (<i>for internal use</i>) for testing purposes only.
105      *
106      * @param fileSize    The size (length) of the file in bytes.
107      * @param fileChannel The open channel to the file.
108      */
109     RandomAccessFileChannel(long fileSize, FileChannel fileChannel) {
110         this.fileSize = fileSize;
111         this.fileChannel = Objects.requireNonNull(fileChannel, "FileChannel cannot be null.");
112     }
113 
114     // Read UTF string (mock implementation)
115     @Override
116     public String readUTF() {
117         throw new UnsupportedOperationException("readUTF is not implemented for FileChannels.");
118     }
119 
120     @Override
121     public final FileChannel getChannel() {
122         return this.fileChannel;
123     }
124 
125     @Override
126     public FileDescriptor getFD() {
127         throw new UnsupportedOperationException("FileDescriptor is not implemented for FileChannels.");
128     }
129 
130     @Override
131     public void setLength(long l) {
132         throw new UnsupportedOperationException("Setting file length is not implemented for FileChannels.");
133     }
134 
135     @Override
136     public void writeUTF(String s) {
137         throw new UnsupportedOperationException("writeUTF operation is not implemented for FileChannels.");
138     }
139 
140     @Override
141     public void close() throws IOException {
142         fileChannel.close();
143     }
144 
145     @Override
146     public final long position() throws IOException {
147         return fileChannel.position();
148     }
149 
150     @Override
151     public final void position(long l) throws IOException {
152         fileChannel.position(l);
153     }
154 
155     @Override
156     public final long length() {
157         return this.fileSize;
158     }
159 
160     @Override
161     public int read() {
162         throw new UnsupportedOperationException("read is not implemented for FileChannels.");
163     }
164 
165     /**
166      * Read bytes from the underlying file channel.
167      *
168      * @param  bytes       The byte array to read into.
169      * @param  offset      The offset in the FileChannel to start reading.
170      * @param  readLength  The number of bytes to read.
171      * 
172      * @return             The number of bytes read.
173      * 
174      * @throws IOException If an error occurs reading from the FileChannel.
175      */
176     @Override
177     public int read(byte[] bytes, int offset, int readLength) throws IOException {
178         final ByteBuffer buffer = ByteBuffer.allocate(Math.min(bytes.length, readLength));
179         position(offset);
180         final int bytesRead = fileChannel.read(buffer);
181         System.arraycopy(buffer.array(), 0, bytes, 0, bytesRead);
182 
183         LOGGER.log(Level.FINE, "Read {0} bytes from the file channel.", bytesRead);
184         return bytesRead;
185     }
186 
187     @Override
188     public void write(int i) {
189         throw new UnsupportedOperationException("write is not implemented for FileChannels.");
190     }
191 
192     /**
193      * Write bytes to the underlying file channel.
194      *
195      * @param  bytes       The byte array to write.
196      * @param  offset      The offset in the FileChannel to start writing from.
197      * @param  writeLength The number of bytes to write.
198      * 
199      * @throws IOException If an error occurs writing to the FileChannel.
200      */
201     @Override
202     public void write(byte[] bytes, int offset, int writeLength) throws IOException {
203         final ByteBuffer buffer = ByteBuffer.wrap(bytes);
204         position(offset);
205         final int bytesWritten = this.fileChannel.write(buffer);
206         LOGGER.log(Level.FINE, "Wrote {0} bytes to the file channel.", bytesWritten);
207     }
208 }
