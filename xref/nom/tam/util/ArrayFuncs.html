54      /**
55       * Retuens a copy of the input array with the order of elements reversed.
56       * 
57       * @param index     the input array
58       * @return          a copy of the input array in reversed order
59       */
60      public static int[] getReversed(int... index) {
61          int[] rev = new int[index.length];
62          int nm1 = index.length - 1;
63          for (int i = index.length; --i >= 0;) {
64              rev[i] = index[nm1 - i];
65          }
66          return rev;
67      }

110     /**
111      * @return Convert an array to a specified type. This method supports
112      *         conversions only among the primitive numeric types.
113      * @param array
114      *            A possibly multidimensional array to be converted.
115      * @param newType
116      *            The desired output type. This should be one of the class
117      *            descriptors for primitive numeric data, e.g., double.type.
118      */
119     public static Object convertArray(Object array, Class<?> newType) {
120         /*
121          * We break this up into two steps so that users can reuse an array many
122          * times and only allocate a new array when needed.
123          */
124         /* First create the full new array. */
125         Object mimic = mimicArray(array, newType);
126         /* Now copy the info into the new array */
127         copyInto(array, mimic);
128         return mimic;
129     }

132      * @return Convert an array to a specified type. This method supports
133      *         conversions only among the primitive numeric types.
134      * @param array
135      *            A possibly multidimensional array to be converted.
136      * @param newType
137      *            The desired output type. This should be one of the class
138      *            descriptors for primitive numeric data, e.g., double.type.
139      * @param reuse
140      *            If set, and the requested type is the same as the original,
141      *            then the original is returned.

143     public static Object convertArray(Object array, Class<?> newType, boolean reuse) {
144         if (getBaseClass(array) == newType && reuse) {
145             return array;
146         }
147         return convertArray(array, newType);

150     /**
151      * @deprecated No longer used within the library itself.
152      * 
153      * Copy one array into another. This function copies the contents of one
154      * array into a previously allocated array. The arrays must agree in type
155      * and size.
156      * 
157      * @param original
158      *            The array to be copied.
159      * @param copy
160      *            The array to be copied into. This array must already be fully
161      *            allocated.
162      * @throws IllegalArgumentException
163      *            if the two arrays do not match in type or size.
164      */
165     @Deprecated
166     public static void copyArray(Object original, Object copy) throws IllegalArgumentException {
167         Class<? extends Object> cl = original.getClass();
168         if (!cl.isArray()) {
169             throw new IllegalArgumentException("original is not an array");
170         }
171         
172         if (!copy.getClass().equals(cl)) {
173             throw new IllegalArgumentException("mismatch of types: " + cl.getName() + " vs " + copy.getClass().getName());
174         }
175         
176         int length = Array.getLength(original);
177         
178         if (Array.getLength(copy) != length) {
179             throw new IllegalArgumentException("mismatch of sizes: " + length + " vs " + Array.getLength(copy));
180         }
181         
182         if (original instanceof Object[]) {
183             Object[] from = (Object[]) original;
184             Object[] to = (Object[]) copy;
185             for (int index = 0; index < length; index++) {
186                 copyArray(from[index], to[index]);
187             }
188         } else {
189             System.arraycopy(original, 0, copy, 0, length);
190         }
191     }

206     /**
207      * Curl an input array up into a multi-dimensional array.
208      * 
209      * @param input
210      *            The one dimensional array to be curled.
211      * @param dimens
212      *            The desired dimensions
213      * @return The curled array.
214      */
215     public static Object curl(Object input, int[] dimens) {
216         if (input == null) {
217             return null;
218         }
219         if (!input.getClass().isArray()) {
220             throw new RuntimeException("Attempt to curl a non-array");
221         }
222         if (input.getClass().getComponentType().isArray()) {
223             throw new RuntimeException("Attempt to curl non-1D array");
224         }
225         int size = Array.getLength(input);
226         int test = 1;
227         for (int dimen : dimens) {
228             test *= dimen;
229         }
230         if (test != size) {
231             throw new RuntimeException("Curled array does not fit desired dimensions");
232         }
233         Object newArray = ArrayFuncs.newInstance(getBaseClass(input), dimens);
234         MultiArrayCopier.copyInto(input, newArray);
235         return newArray;
236 
237     }

239     /**
240      * @return a deep clone of an Array or a standard clone of a scalar. The
241      *         object may comprise arrays of any primitive type or any Object
242      *         type which implements Cloneable. However, if the Object is some
243      *         kind of collection, e.g., a Vector then only a shallow copy of
244      *         that object is made. I.e., deep refers only to arrays.
245      * @param o
246      *            The object to be copied.
247      */
248     public static Object deepClone(Object o) {
249         if (o == null) {
250             return null;
251         }
252         if (!o.getClass().isArray()) {
253             return genericClone(o);
254         }
255         // Check if this is a 1D primitive array.
256         if (o.getClass().getComponentType().isPrimitive()) {
257             int length = Array.getLength(o);
258             Object result = Array.newInstance(o.getClass().getComponentType(), length);
259             System.arraycopy(o, 0, result, 0, length);
260             return result;
261         }
262         // Get the base type.
263         Class<?> baseClass = getBaseClass(o);
264         // Allocate the array but make all but the first dimension 0.
265         int[] dims = getDimensions(o);
266         Arrays.fill(dims, 1, dims.length, 0);
267         Object copy = ArrayFuncs.newInstance(baseClass, dims);
268         // Now fill in the next level down by recursion.
269         for (int i = 0; i < dims[0]; i++) {
270             Array.set(copy, i, deepClone(Array.get(o, i)));
271         }
272         return copy;
273     }

275     /**
276      * Given an array of arbitrary dimensionality .
277      * 
278      * @return the array flattened into a single dimension.
279      * @param input
280      *            The input array.
281      */
282     public static Object flatten(Object input) {
283         int[] dimens = getDimensions(input);
284         if (dimens.length <= 1) {
285             return input;
286         }
287         int size = 1;
288         for (int dimen : dimens) {
289             size *= dimen;
290         }
291         Object flat = ArrayFuncs.newInstance(getBaseClass(input), size);
292         MultiArrayCopier.copyInto(input, flat);
293         return flat;
294     }

296     /**
297      * Clone an Object if possible. This method returns an Object which is a
298      * clone of the input object. It checks if the method implements the
299      * Cloneable interface and then uses reflection to invoke the clone method.
300      * This can't be done directly since as far as the compiler is concerned the
301      * clone method for Object is protected and someone could implement
302      * Cloneable but leave the clone method protected. The cloning can fail in a
303      * variety of ways which are trapped so that it returns null instead. This
304      * method will generally create a shallow clone. If you wish a deep copy of
305      * an array the method deepClone should be used.
306      * 
307      * @param o
308      *            The object to be cloned.
309      * @return the clone
310      */
311     public static Object genericClone(Object o) {
312         if (o.getClass().isArray()) {
313             return deepClone(o);
314         }
315         if (!(o instanceof Cloneable)) {
316             LOG.log(Level.SEVERE, "generic clone called on a non clonable type");
317             return null;
318         }
319         try {
320             return o.getClass().getMethod("clone").invoke(o);
321         } catch (Exception e) {
322             LOG.log(Level.WARNING, "Implements cloneable, but does not apparently make clone public.", e);
323             return null;
324         }
325     }

328      * This routine returns the base array of a multi-dimensional array. I.e., a
329      * one-d array of whatever the array is composed of. Note that arrays are
330      * not guaranteed to be rectangular, so this returns o[0][0]....
331      * 
332      * @param o
333      *            the multi-dimensional array
334      * @return base array of a multi-dimensional array.
335      */
336     public static Object getBaseArray(Object o) {
337         if (o instanceof Object[]) {
338             return getBaseArray(Array.get(o, 0));
339         }
340         return o;
341     }

343     /**
344      * This routine returns the base class of an object. This is just the class
345      * of the object for non-arrays.
346      * 
347      * @param o
348      *            array to get the base class from
349      * @return the base class of an array
350      */
351     public static Class<?> getBaseClass(Object o) {
352         if (o == null) {
353             return Void.TYPE;
354         }
355         Class<?> clazz = o.getClass();
356         while (clazz.isArray()) {
357             clazz = clazz.getComponentType();
358         }
359         return clazz;
360     }

364      * 
365      * @param o
366      *            The array object whose base length is desired.
367      * @return the size of the object in bytes, 0 if null, or -1 if not a

371         if (o == null) {
372             return 0;
373         }
374         ElementType<?> type = ElementType.forClass(getBaseClass(o));

378     /**
379      * Find the dimensions of an object. This method returns an integer array
380      * with the dimensions of the object o which should usually be an array. It
381      * returns an array of dimension 0 for scalar objects and it returns -1 for
382      * dimension which have not been allocated, e.g., <code>int[][][] x = new
383      * int[100][][];</code> should return [100,-1,-1].
384      * 
385      * @param o
386      *            The object to get the dimensions of.
387      * @return the dimensions of an object
388      */
389     public static int[] getDimensions(Object o) {
390         if (o == null) {
391             return null;
392         }
393         Object object = o;
394         Class<?> clazz = o.getClass();
395         int ndim = 0;
396         while (clazz.isArray()) {
397             clazz = clazz.getComponentType();
398             ndim++;
399         }
400         clazz = o.getClass();
401         int[] dimens = new int[ndim];
402         ndim = 0;
403         while (clazz.isArray()) {
404             dimens[ndim] = -1;
405             if (object != null) {
406                 int length = Array.getLength(object);
407                 if (length > 0) {
408                     dimens[ndim] = length;
409                     object = Array.get(object, 0);
410                 } else {
411                     dimens[ndim] = 0;
412                     object = null;
413                 }
414             }
415             clazz = clazz.getComponentType();
416             ndim++;
417         }
418         return dimens;
419     }

421     /**
422      * Create an array of a type given by new type with the dimensionality given
423      * in array.
424      * 
425      * @return the new array with same dimensions
426      * @param array
427      *            A possibly multidimensional array to be converted.
428      * @param newType
429      *            The desired output type. This should be one of the class
430      *            descriptors for primitive numeric data, e.g., double.type.
431      */
432     public static Object mimicArray(Object array, Class<?> newType) {
433         int dims = 0;
434         Class<?> arrayClass = array.getClass();
435         while (arrayClass.isArray()) {
436             arrayClass = arrayClass.getComponentType();
437             dims += 1;
438         }
439 
440         if (dims <= 1) {
441             return ArrayFuncs.newInstance(newType, Array.getLength(array));
442         }
443         
444         Object[] xarray = (Object[]) array;
445         int[] dimens = new int[dims];
446         dimens[0] = xarray.length; // Leave other dimensions at 0.
447 
448         Object mimic = ArrayFuncs.newInstance(newType, dimens);
449 
450         for (int i = 0; i < xarray.length; i += 1) {
451             Object temp = mimicArray(xarray[i], newType);
452             ((Object[]) mimic)[i] = temp;
453         }
454 
455         return mimic;
456     }

534     /**
535      * Reverse an integer array. This can be especially useful when dealing with
536      * an array of indices in FITS order that you wish to have in Java order.
537      * 
538      * @return the reversed array.
539      * @param indices
540      *            the array to reverse
541      */
542     public static int[] reverseIndices(int[] indices) {
543         int[] result = new int[indices.length];
544         int len = indices.length;
545         for (int i = 0; i < indices.length; i += 1) {
546             result[len - i - 1] = indices[i];
547         }
548         return result;
549     }
