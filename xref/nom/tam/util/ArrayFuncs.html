53      /**
54       * Retuens a copy of the input array with the order of elements reversed.
55       *
56       * @param  index the input array
57       *
58       * @return       a copy of the input array in reversed order
59       */
60      public static int[] getReversed(int... index) {
61          int[] rev = new int[index.length];
62          int nm1 = index.length - 1;
63          for (int i = index.length; --i >= 0;) {
64              rev[i] = index[nm1 - i];
65          }
66          return rev;
67      }

69      /**
70       * Perform an array copy with an API similar to System.arraycopy(), specifying the number of values to jump to the
71       * next read.
72       *
73       * @param src     The source array.
74       * @param srcPos  Starting position in the source array.
75       * @param dest    The destination array.
76       * @param destPos Starting position in the destination data.
77       * @param length  The number of array elements to be read.
78       * @param step    The number of jumps to the next read.
79       *
80       * @since         1.18
81       */
82      public static void copy(Object src, int srcPos, Object dest, int destPos, int length, int step) {
83          if (!dest.getClass().equals(src.getClass())) {
84              throw new IllegalArgumentException("Mismatched types: src " + src.getClass() + ", dst " + dest.getClass());
85          }
86  
87          if (src instanceof Object[]) {
88              final Object[] from = (Object[]) src;
89              final Object[] to = (Object[]) dest;
90              int toIndex = 0;
91              for (int index = srcPos; index < srcPos + length; index += step) {
92                  ArrayFuncs.copy(from[index], srcPos, to[toIndex++], destPos, length, step);
93              }
94          } else if (step == 1) {
95              System.arraycopy(src, srcPos, dest, destPos, length);
96          } else {
97              final int srcLength = Array.getLength(src);
98              final int loopLength = Math.min(srcLength, srcPos + length);
99              for (int i = srcPos; i < loopLength; i += step) {
100                 Array.set(dest, destPos++, Array.get(src, i));
101             }
102         }
103     }

144     /**
145      * Converts a numerical array to a specified element type. This method supports conversions only among the primitive
146      * numeric types.
147      *
148      * @return         a new array with the requested element type.
149      * 
150      * @param  array   a numerical array or one or more dimensions
151      * @param  newType the desired output type. This should be one of the class descriptors for primitive numeric data,
152      *                     e.g., <code>double.class</code>.
153      * 
154      * @see            #convertArray(Object, Class, boolean)
155      */
156     public static Object convertArray(Object array, Class<?> newType) {
157         /*
158          * We break this up into two steps so that users can reuse an array many times and only allocate a new array
159          * when needed.
160          */
161         /* First create the full new array. */
162         Object mimic = mimicArray(array, newType);
163         /* Now copy the info into the new array */
164         copyInto(array, mimic);
165         return mimic;
166     }

189     /**
190      * Copy one array into another. This function copies the contents of one array into a previously allocated array.
191      * The arrays must agree in type and size.
192      * 
193      * @deprecated                          No longer used within the library itself.
194      *
195      * @param      original                 The array to be copied.
196      * @param      copy                     The array to be copied into. This array must already be fully allocated.
197      *
198      * @throws     IllegalArgumentException if the two arrays do not match in type or size.
199      */
200     @Deprecated
201     public static void copyArray(Object original, Object copy) throws IllegalArgumentException {
202         Class<? extends Object> cl = original.getClass();
203         if (!cl.isArray()) {
204             throw new IllegalArgumentException("original is not an array");
205         }
206 
207         if (!copy.getClass().equals(cl)) {
208             throw new IllegalArgumentException("mismatch of types: " + cl.getName() + " vs " + copy.getClass().getName());
209         }
210 
211         int length = Array.getLength(original);
212 
213         if (Array.getLength(copy) != length) {
214             throw new IllegalArgumentException("mismatch of sizes: " + length + " vs " + Array.getLength(copy));
215         }
216 
217         if (original instanceof Object[]) {
218             Object[] from = (Object[]) original;
219             Object[] to = (Object[]) copy;
220             for (int index = 0; index < length; index++) {
221                 copyArray(from[index], to[index]);
222             }
223         } else {
224             System.arraycopy(original, 0, copy, 0, length);
225         }
226     }

239     /**
240      * Curl an input array up into a multi-dimensional array.
241      *
242      * @param  input                 The one dimensional array to be curled.
243      * @param  dimens                The desired dimensions
244      *
245      * @return                       The curled array.
246      * 
247      * @throws IllegalStateException if the size of the input does not match the specified dimensions.
248      */
249     public static Object curl(Object input, int... dimens) throws IllegalStateException {
250         if (input == null) {
251             return null;
252         }
253         if (!input.getClass().isArray()) {
254             throw new IllegalArgumentException("Attempt to curl a non-array: " + input.getClass());
255         }
256         if (input.getClass().getComponentType().isArray()) {
257             throw new IllegalArgumentException("Attempt to curl non-1D array: " + input.getClass());
258         }
259         int size = Array.getLength(input);
260         int test = 1;
261         for (int dimen : dimens) {
262             test *= dimen;
263         }
264         if (test != size) {
265             throw new IllegalStateException("Curled array does not fit desired dimensions: " + size + ", expected " + test
266                     + " (" + getBaseClass(input) + ")");
267         }
268         Object newArray = ArrayFuncs.newInstance(getBaseClass(input), dimens);
269         MultiArrayCopier.copyInto(input, newArray);
270         return newArray;
271 
272     }

274     /**
275      * Returns a deep clone of an array (in one or more dimensions) or a standard clone of a scalar. The object may
276      * comprise arrays of any primitive type or any Object type which implements Cloneable. However, if the Object is
277      * some kind of collection, such as a {@link java.util.List}, then only a shallow copy of that object is made. I.e.,
278      * deep refers only to arrays.
279      *
280      * @return   a new object, with a copy of the original.
281      * 
282      * @param  o The object (usually an array) to be copied.
283      */
284     public static Object deepClone(Object o) {
285         if (o == null) {
286             return null;
287         }
288 
289         if (!o.getClass().isArray()) {
290             return genericClone(o);
291         }
292 
293         if (o instanceof Object[]) {
294             Object[] array = (Object[]) o;
295             Object[] copy = (Object[]) ArrayFuncs.newInstance(array.getClass().getComponentType(), array.length);
296             // Now fill in the next level down by recursion.
297             for (int i = 0; i < array.length; i++) {
298                 copy[i] = deepClone(array[i]);
299             }
300             return copy;
301         }
302 
303         // Must be primitive array...
304         // Check if this is a 1D primitive array.
305         int length = Array.getLength(o);
306         Object copy = Array.newInstance(o.getClass().getComponentType(), length);
307         System.arraycopy(o, 0, copy, 0, length);
308         return copy;
309     }

311     /**
312      * Given an array of arbitrary dimensionality .
313      *
314      * @return       the array flattened into a single dimension.
315      *
316      * @param  input The input array.
317      */
318     public static Object flatten(Object input) {
319         if (input == null) {
320             return null;
321         }
322 
323         if (!input.getClass().isArray()) {
324             return input;
325         }
326 
327         int[] dimens = getDimensions(input);
328         if (dimens.length <= 1) {
329             return input;
330         }
331         int size = 1;
332         for (int dimen : dimens) {
333             size *= dimen;
334         }
335         Object flat = ArrayFuncs.newInstance(getBaseClass(input), size);
336         MultiArrayCopier.copyInto(input, flat);
337         return flat;
338     }

340     /**
341      * Clone an Object if possible. This method returns an Object which is a clone of the input object. It checks if the
342      * method implements the Cloneable interface and then uses reflection to invoke the clone method. This can't be done
343      * directly since as far as the compiler is concerned the clone method for Object is protected and someone could
344      * implement Cloneable but leave the clone method protected. The cloning can fail in a variety of ways which are
345      * trapped so that it returns null instead. This method will generally create a shallow clone. If you wish a deep
346      * copy of an array the method deepClone should be used.
347      *
348      * @param  o The object to be cloned.
349      *
350      * @return   the clone
351      */
352     public static Object genericClone(Object o) {
353         if (o.getClass().isArray()) {
354             return deepClone(o);
355         }
356         if (!(o instanceof Cloneable)) {
357             LOG.log(Level.SEVERE, "generic clone called on a non clonable type");
358             return null;
359         }
360         try {
361             return o.getClass().getMethod("clone").invoke(o);
362         } catch (Exception e) {
363             LOG.log(Level.WARNING, "Implements cloneable, but does not apparently make clone public.", e);
364             return null;
365         }
366     }

385      *
386      * @param  o array to get the base class from
387      *
388      * @return   the base class of an array
389      */
390     public static Class<?> getBaseClass(Object o) {
391         if (o == null) {
392             return Void.TYPE;
393         }
394         Class<?> clazz = o.getClass();
395         while (clazz.isArray()) {
396             clazz = clazz.getComponentType();
397         }
398         return clazz;

416     /**
417      * Find the dimensions of an object. This method returns an integer array with the dimensions of the object o which
418      * should usually be an array. It returns an array of dimension 0 for scalar objects and it returns -1 for dimension
419      * which have not been allocated, e.g., <code>int[][][] x = new
420      * int[100][][];</code> should return [100,-1,-1].
421      *
422      * @param  o The object to get the dimensions of.
423      *
424      * @return   the dimensions of an object
425      */
426     public static int[] getDimensions(Object o) {
427         if (o == null) {
428             return null;
429         }
430         Object object = o;
431         Class<?> clazz = o.getClass();
432         int ndim = 0;
433         while (clazz.isArray()) {
434             clazz = clazz.getComponentType();
435             ndim++;
436         }
437         clazz = o.getClass();
438         int[] dimens = new int[ndim];
439         ndim = 0;
440         while (clazz.isArray()) {
441             dimens[ndim] = -1;
442             if (object != null) {
443                 int length = Array.getLength(object);
444                 if (length > 0) {
445                     dimens[ndim] = length;
446                     object = Array.get(object, 0);
447                 } else {
448                     dimens[ndim] = 0;
449                     object = null;
450                 }
451             }
452             clazz = clazz.getComponentType();
453             ndim++;
454         }
455         return dimens;
456     }

458     /**
459      * Create an array of a type given by new type with the dimensionality given in array.
460      *
461      * @return         the new array with same dimensions
462      *
463      * @param  array   A possibly multidimensional array to be converted.
464      * @param  newType The desired output type. This should be one of the class descriptors for primitive numeric data,
465      *                     e.g., double.type.
466      */
467     public static Object mimicArray(Object array, Class<?> newType) {
468         int dims = 0;
469         Class<?> arrayClass = array.getClass();
470         while (arrayClass.isArray()) {
471             arrayClass = arrayClass.getComponentType();
472             dims++;
473         }
474 
475         if (dims <= 1) {
476             return ArrayFuncs.newInstance(newType, Array.getLength(array));
477         }
478 
479         Object[] xarray = (Object[]) array;
480         int[] dimens = new int[dims];
481         dimens[0] = xarray.length; // Leave other dimensions at 0.
482 
483         Object mimic = ArrayFuncs.newInstance(newType, dimens);
484 
485         for (int i = 0; i < xarray.length; i++) {
486             Object temp = mimicArray(xarray[i], newType);
487             ((Object[]) mimic)[i] = temp;
488         }
489 
490         return mimic;
491     }

590     /**
591      * Checks that an array has a regular structure, with a consistent shape and element types, and returns the regular
592      * array size or else throws an exeption. Optionally, it will also throw an exception if any or all all elements are
593      * <code>null</code>.
594      * 
595      * @param  o                        An array object
596      * @param  allowNulls               If we should tolerate <code>null</code> entries.
597      * 
598      * @return                          the regular shape of the array with sizes along each array dimension.
599      * 
600      * @throws NullPointerException     if the argument is <code>null</code>.
601      * @throws IllegalArgumentException if the array contains mismatched elements in size, or contains <code>null</code>
602      *                                      values.
603      * @throws ClassCastException       if the array contain a heterogeneous collection of different element types.
604      * 
605      * @since                           1.18
606      */
607     public static int[] checkRegularArray(Object o, boolean allowNulls)
608             throws NullPointerException, IllegalArgumentException, ClassCastException {
609         if (!o.getClass().isArray()) {
610             throw new IllegalArgumentException("Not an array: " + o.getClass());
611         }
612 
613         if (o.getClass().getComponentType().isPrimitive()) {
614             return new int[] {Array.getLength(o)};
615         }
616 
617         int[] dim = getDimensions(o);
618         if (dim[0] == 0) {
619             return dim;
620         }
621 
622         Class<?> type = null;
623         Object[] array = (Object[]) o;
624 
625         for (int i = 0; i < dim[0]; i++) {
626             Object e = array[i];
627 
628             if (e == null) {
629                 if (allowNulls) {
630                     continue;
631                 }
632                 throw new IllegalArgumentException("Entry at index " + i + " is null");
633             }
634 
635             if (type == null) {
636                 type = e.getClass();
637             } else if (!e.getClass().equals(type)) {
638                 throw new ClassCastException(
639                         "Mismatched component type at index " + i + ": " + e.getClass() + ", expected " + type);
640             }
641 
642             if (e.getClass().isArray()) {
643                 int[] sub = checkRegularArray(e, allowNulls);
644 
645                 if (sub.length + 1 != dim.length) {
646                     throw new IllegalArgumentException("Mismatched component dimension at index " + i + ": " + sub.length
647                             + ", expected " + (dim.length - 1));
648                 }
649 
650                 if (sub[0] != dim[1]) {
651                     throw new IllegalArgumentException(
652                             "Mismatched component size at index " + i + ": " + sub[0] + ", expected " + dim[0]);
653                 }
654             }
655         }
656 
657         return dim;
658     }

660     /**
661      * Converts complex value(s) count to <code>float[2]</code> or <code>double[2]</code> or arrays thereof, which
662      * maintain the shape of the original input array (if applicable).
663      * 
664      * @param  o                        one of more complex values
665      * @param  decimalType              <code>float.class</code> or <code>double.class</code> (all other values default
666      *                                      to as if <code>double.class</code> was used.
667      * 
668      * @return                          an array of <code>float[2]</code> or <code>double[2]</code>, or arrays thereof.
669      * 
670      * @throws IllegalArgumentException if the argument is not suitable for conversion to complex values.
671      * 
672      * @see                             #decimalsToComplex(Object)
673      * 
674      * @since                           1.18
675      */
676     public static Object complexToDecimals(Object o, Class<?> decimalType) {
677 
678         if (o instanceof ComplexValue) {
679             ComplexValue z = (ComplexValue) o;
680             if (float.class.equals(decimalType)) {
681                 return new float[] {(float) z.re(), (float) z.im()};
682             }
683             return new double[] {z.re(), z.im()};
684         }
685 
686         if (o instanceof ComplexValue[]) {
687             ComplexValue[] z = (ComplexValue[]) o;
688 
689             if (float.class.equals(decimalType)) {
690                 float[][] f = new float[z.length][];
691                 for (int i = 0; i < z.length; i++) {
692                     f[i] = (float[]) complexToDecimals(z[i], decimalType);
693                 }
694                 return f;
695             }
696 
697             double[][] d = new double[z.length][];
698             for (int i = 0; i < z.length; i++) {
699                 d[i] = (double[]) complexToDecimals(z[i], decimalType);
700             }
701             return d;
702         }
703 
704         if (o instanceof Object[]) {
705             Object[] array = (Object[]) o;
706             Object[] z = null;
707 
708             for (int i = 0; i < array.length; i++) {
709                 Object e = complexToDecimals(array[i], decimalType);
710                 if (z == null) {
711                     z = (Object[]) Array.newInstance(e.getClass(), array.length);
712                 }
713                 z[i] = e;
714             }
715 
716             return z;
717         }
718 
719         throw new IllegalArgumentException("Cannot convert to complex values: " + o.getClass().getName());
720     }

722     /**
723      * Converts real-valued arrays of even element count to a {@link ComplexValue} or arrays thereof. The size and shape
724      * is otherwise maintained, apart from coalescing pairs of real values into <code>ComplexValue</code> objects.
725      * 
726      * @param  array                    an array of <code>float</code> or <code>double</code> elements containing an
727      *                                      even number of elements at the last dimension
728      * 
729      * @return                          one of more complex values
730      * 
731      * @throws IllegalArgumentException if the argument is not suitable for conversion to complex values.
732      * 
733      * @see                             #complexToDecimals(Object, Class)
734      * 
735      * @since                           1.18
736      */
737     public static Object decimalsToComplex(Object array) throws IllegalArgumentException {
738         if (array instanceof float[]) {
739             float[] f = (float[]) array;
740             if (f.length % 2 == 1) {
741                 throw new IllegalArgumentException("Odd number floats for complex conversion: " + f.length);
742             }
743             if (f.length == 2) {
744                 return new ComplexValue(f[0], f[1]);
745             }
746             ComplexValue[] z = new ComplexValue[f.length >>> 1];
747             for (int i = 0; i < f.length; i += 2) {
748                 z[i >> 1] = new ComplexValue(f[i], f[i + 1]);
749             }
750             return z;
751         }
752 
753         if (array instanceof double[]) {
754             double[] d = (double[]) array;
755             if (d.length % 2 == 1) {
756                 throw new IllegalArgumentException("Odd number floats for complex conversion: " + d.length);
757             }
758             if (d.length == 2) {
759                 return new ComplexValue(d[0], d[1]);
760             }
761             ComplexValue[] z = new ComplexValue[d.length >>> 1];
762             for (int i = 0; i < d.length; i += 2) {
763                 z[i >> 1] = new ComplexValue(d[i], d[i + 1]);
764             }
765             return z;
766         }
767 
768         if (array instanceof Object[]) {
769             Object[] o = (Object[]) array;
770             Object[] z = null;
771 
772             for (int i = 0; i < o.length; i++) {
773                 Object e = decimalsToComplex(o[i]);
774                 if (z == null) {
775                     z = (Object[]) Array.newInstance(e.getClass(), o.length);
776                 }
777                 z[i] = e;
778             }
779 
780             return z;
781         }
782 
783         throw new IllegalArgumentException("Cannot convert to complex values: " + array.getClass().getName());
784     }
