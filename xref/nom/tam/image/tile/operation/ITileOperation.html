34  /**
35   * (<i>for internal use</i>) Interface for performing parallel operations on
36   * serialized 2D image tiles.
37   * 
38   * @see nom.tam.image.ImageTiler
39   * @see nom.tam.image.tile.operation.buffer.TileBuffer
40   */
41  public interface ITileOperation {
42  
43      /**
44       * Specifies the location of the tile in the serialized buffer.
45       * 
46       * @param dataOffset
47       *            the byte offset at which the tile data begins.
48       * @param width
49       *            the width of the tile in pixels.
50       * @param height
51       *            the tile height in pixels.
52       * @return itself.
53       */
54      ITileOperation setDimensions(int dataOffset, int width, int height);
55  
56      /**
57       * Waits until the parallelized operation is complete
58       */
59      void waitForResult();
60  }
