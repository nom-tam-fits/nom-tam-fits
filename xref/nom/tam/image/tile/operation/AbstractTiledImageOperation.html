34  import java.lang.reflect.Array;
35  import java.nio.Buffer;
36  import java.util.ArrayList;
37  import java.util.Arrays;
38  
39  import nom.tam.fits.FitsException;
40  import nom.tam.util.ArrayFuncs;
41  import nom.tam.util.type.ElementType;
42  
43  public abstract class AbstractTiledImageOperation<OPERATION extends ITileOperation> implements ITiledImageOperation {
44  
45      /** Image sixe in Java array index order (is is last!). */
46      private int[] axes;
47  
48      /**
49       * Interprets the value of the BITPIX keyword in the uncompressed FITS image
50       */
51      private ElementType<Buffer> baseType;
52  
53      /** Tile dimensions in Java array index order (x is last!) */
54      private int[] tileAxes;
55  
56      private OPERATION[] tileOperations;
57  
58      private final Class<OPERATION> operationClass;
59      
60      public AbstractTiledImageOperation(Class<OPERATION> operationClass) {
61          this.operationClass = operationClass;
62      }
63  
64      @Override
65      public ElementType<Buffer> getBaseType() {
66          return baseType;
67      }
68  
69      public int getBufferSize() {
70          int bufferSize = 1;
71          for (int axisValue : axes) {
72              bufferSize *= axisValue;
73          }
74          return bufferSize;
75      }
76  
77      @Override
78      public int getImageWidth() {
79          return axes[axes.length - 1];
80      }
81  
82      @Override
83      public OPERATION getTileOperation(int i) {
84          return tileOperations[i];
85      }
86  
87      /**
88       * Sets the image dimensions, in Java array index order.
89       * 
90       * @param axes  Image dimensions in Java array index order (x is last!).
91       */
92      public void setAxes(int[] axes) {
93          this.axes = Arrays.copyOf(axes, axes.length);
94      }
95  
96      /**
97       * <p>
98       * Sets the tile dimension. Here the dimensions are in Java array index order, that is the
99       * x-dimension (width of tile) is last!
100      * </p>
101      * <p>
102      * Note, that because tile compression is essentially 2D, the tile sizes in higher
103      * dimensions will be forced to 1, even if specified otherwise by the argument (see 
104      * <a href="https://heasarc.gsfc.nasa.gov/docs/software/fitsio/compression.html">FITSIO convention</a>).
105      * </p>
106      * 
107      * @param value     The tile dimensions in Java array index order (x is last!). Only up to
108      *                  the first 2 components are considered. The rest will be assumed to have
109      *                  values equals to 1.
110      * @throws FitsException    If the higher dimensions (above 2) have sizes not equal to 1
111      */
112     public void setTileAxes(int[] value) throws FitsException {
113         tileAxes = Arrays.copyOf(value, value.length);
114         for (int i = value.length - 2; --i >= 0;) {
115             if (value[i] != 1) {
116                 throw new FitsException("Tile sizes in higher dimensions (>2) must be 1 as per the FITSIO convention.");
117             }
118         }
119     }
120     
121     protected boolean hasAxes() {
122         return axes != null;
123     }
124 
125     protected boolean hasTileAxes() {
126         return tileAxes != null;
127     }
128     
129     
130     private int getBufferOffset(int[] index) {
131         int l = 0;
132         int blockSize = 1;
133         for (int i = index.length; --i >= 0;) {
134             l += index[i] * blockSize;
135             blockSize *= axes[i];
136         }
137         return l;
138     }
139 
140     @SuppressWarnings("unchecked")
141     protected void createTiles(ITileOperationInitialisation<OPERATION> init) throws FitsException {
142         int[] offset = new int[axes.length];        // Tile start in image (Java index order)
143         int[] tileSize = new int[2];                // {w, h}
144         int pos = 0;
145         
146         int imLength = 1;
147         for (int i = axes.length; --i >= 0;) {
148             imLength *= axes[i];
149         }
150         
151         tileSize[1] = 1;
152         
153         // If tile is not defined along all axes, pad with 1.
154         if (tileAxes.length < axes.length) {
155             int[] tile = new int[axes.length];
156             System.arraycopy(tileAxes, 0, tile, tile.length - tileAxes.length, tileAxes.length);
157             Arrays.fill(tile, 0, tile.length - tileAxes.length, 1);
158             tileAxes = tile;
159         }
160         
161         ArrayList<OPERATION> opList = new ArrayList<>();
162         
163         // Create 2D tiles to cover image (in N dimensions, where N need not be 2)
164         for (int tileIndex = 0; pos < imLength; tileIndex++) {
165             // Calculate the actual size of the current tile
166             for (int i = Math.min(tileSize.length, tileAxes.length); --i >= 0;) {
167                 int k = axes.length - 1 - i;
168                 tileSize[i] = (offset[k] + tileAxes[k] > axes[k]) ? axes[k] - offset[k] : tileAxes[k];
169             }
170                 
171             // Create the tile at the current buffer offset and tile size
172             OPERATION op = init.createTileOperation(tileIndex, new TileArea().start(ArrayFuncs.getReversed(offset)));
173             op.setDimensions(pos, tileSize[0], tileSize[1]);
174             opList.add(op);
175     
176             // Calculate the image indices where the next tile starts.
177             for (int k = axes.length; --k >= 0;) {
178                 offset[k] += tileAxes[k];   // Try next tile along the current dimension...
179                 if (offset[k] < axes[k]) {
180                     break;              // OK, tile is within image bounds
181                 } else if (k > 0) {
182                     offset[k] = 0;      // Otherwise reset the tile pos in the subarray dimensions
183                 }
184             }
185 
186             // Calculate the buffer position where the next tile starts.
187             pos = getBufferOffset(offset);
188         }
189         
190         tileOperations = (OPERATION[]) Array.newInstance(operationClass, opList.size());
191         opList.toArray(tileOperations);
192         
193         init.tileCount(tileOperations.length);
194 
195         for (OPERATION op : tileOperations) {
196             init.init(op);
197         }
198     }
199 
200     protected int getNAxes() {
201         return this.axes.length;
202     }
203 
204     protected int getNumberOfTileOperations() {
205         return this.tileOperations.length;
206     }
207 
208     /**
209      * Returns the reference to the tile dimensions array. The dimensions are stored in Java array
210      * index order, i.e., the x-dimension (width) is last. 
211      * 
212      * @return      The tile dimensions in Java array index order (x is last!).
213      */
214     protected int[] getTileAxes() {
215         return this.tileAxes;
216     }
217 
218     protected OPERATION[] getTileOperations() {
219         return this.tileOperations;
220     }
221 
222     protected void setBaseType(ElementType<Buffer> baseType) {
223         this.baseType = baseType;
224     }
225 }
