34  import java.nio.Buffer;
35  
36  import nom.tam.util.type.ElementType;
37  
38  /**
39   * <p>
40   * (<i>for internal use</i>) Tiling implementation, for when the tile width matches the width of the image. For example,
41   * for covering a 300 by 250 image with tiles of 300 by 7 pixels.
42   * </p>
43   * 
44   * @deprecated Use {@link TileBufferFactory} instead to construct and access via {@link TileBuffer} interface. Its
45   *                 visiv=bility should be reduced to the package level in the future.
46   */
47  @SuppressWarnings("javadoc")
48  public class TileBufferRowBased extends TileBuffer {
49  
50      public TileBufferRowBased(ElementType<Buffer> baseType, int dataOffset, int width, int height) {
51          super(baseType, dataOffset, width, height);
52      }
53  
54      @Override
55      public Buffer getBuffer() {
56          return getImageBuffer();
57      }
58  
59      @Override
60      public TileBufferRowBased setData(Buffer value) {
61          super.setData(value);
62          getImageBuffer().limit(getPixelSize());
63          return this;
64      }
65  }
