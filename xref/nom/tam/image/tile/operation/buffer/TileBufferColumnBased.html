34  import java.nio.Buffer;
35  
36  import nom.tam.util.type.ElementType;
37  
38  /**
39   * This subclass of the row based view, will abstract the problems that occur
40   * when the tile does not spread over a whole row. in that case the buffer
41   * describing the image does not match the buffer describing the tile. That is
42   * why a temporary buffer is needed to make the buffer continuous.
43   */
44  class TileBufferColumnBased extends TileBuffer {
45  
46      /**
47       * the buffer representing the tile data gap less. this will exist only
48       * between the first getBuffer() and the finish(). This way the memory used
49       * for the data copy is allocates as early as needed and freed as soon as
50       * possible.
51       */
52      private Buffer gapLessBuffer;
53  
54      /**
55       * the width of the image in pixels, that differs from the width of the
56       * tile.
57       */
58      private final int imageWidth;
59  
60      TileBufferColumnBased(ElementType<Buffer> baseType, int dataOffset, int imageWidth, int width, int height) {
61          super(baseType, dataOffset, width, height);
62          this.imageWidth = imageWidth;
63      }
64  
65      @Override
66      public void finish() {
67          desolveGapLessBuffer();
68      }
69  
70      @Override
71      public Buffer getBuffer() {
72          if (this.gapLessBuffer == null) {
73              createGapLessBuffer();
74          }
75          return this.gapLessBuffer;
76      }
77  
78      /**
79       * create the temporary buffer that contains no data gaps.
80       */
81      private void createGapLessBuffer() {
82          final int gap = this.imageWidth - getWidth();
83          final int pixelSizeInData = getPixelSizeInData();
84          Buffer imagebuffer = getImageBuffer();
85          imagebuffer.position(0);
86          imagebuffer.limit(0);
87          ElementType<Buffer> type = elementType();
88          this.gapLessBuffer = type.newBuffer(getPixelSize());
89          while (imagebuffer.limit() < pixelSizeInData) {
90              imagebuffer.limit(imagebuffer.position() + getWidth());
91              type.appendBuffer(this.gapLessBuffer, imagebuffer);
92              imagebuffer.limit(Math.min(pixelSizeInData, imagebuffer.position() + gap));
93              imagebuffer.position(imagebuffer.limit());
94          }
95          this.gapLessBuffer.rewind();
96      }
97  
98      /**
99       * resolve the temporary buffer that contains no data gaps, and put the data
100      * back into the image buffer.
101      */
102     private void desolveGapLessBuffer() {
103         final int gap = this.imageWidth - getWidth();
104         final int pixelSize = getPixelSize();
105         Buffer imagebuffer = getImageBuffer();
106         imagebuffer.limit(getPixelSizeInData());
107         imagebuffer.rewind();
108         this.gapLessBuffer.rewind();
109         this.gapLessBuffer.limit(0);
110         ElementType<Buffer> type = elementType();
111         while (this.gapLessBuffer.limit() < pixelSize) {
112             this.gapLessBuffer.limit(this.gapLessBuffer.position() + getWidth());
113             type.appendBuffer(imagebuffer, this.gapLessBuffer);
114             imagebuffer.position(Math.min(imagebuffer.position() + gap, imagebuffer.limit()));
115         }
116         this.gapLessBuffer = null;
117     }
118 
119     /**
120      * @return size of the tile data inside the image data. normally
121      *         tile-height*image-width but then the data block of the last tile
122      *         would go over the image data limit.
123      */
124     private int getPixelSizeInData() {
125         return (getHeight() - 1) * this.imageWidth + getWidth();
126     }
127 
128     private ElementType<Buffer> elementType() {
129         return ElementType.forBuffer(getImageBuffer());
130     }
131 
132 }
