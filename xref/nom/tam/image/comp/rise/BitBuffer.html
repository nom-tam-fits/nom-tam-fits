34  import java.nio.ByteBuffer;
35  
36  /**
37   * A bit wise reader writer around a bytebuffer.
38   * 
39   * @author Ritchie
40   */
41  public class BitBuffer {
42  
43      private static final int BITS_OF_4_BYTES = 32;
44  
45      private static final int BYTE_MASK = 0xFF;
46  
47      private static final long INTEGER_MASK = 0xFFFFFFFFL;
48  
49      private static final int BITS_OF_1_BYTE = 8;
50  
51      private static final int BITS_OF_2_BYTES = 16;
52  
53      private static final int BITS_OF_3_BYTES = 24;
54  
55      private static final int BYTE_1_OF_INT = 0x000000FF;
56  
57      private static final int BYTE_2_OF_INT = 0x0000FF00;
58  
59      private static final int BYTE_3_OF_INT = 0x00FF0000;
60  
61      private static final int BYTE_4_OF_INT = 0xFF000000;
62  
63      private final ByteBuffer buffer;
64  
65      private long position;
66  
67      public BitBuffer(ByteBuffer writeBuffer) {
68          this.buffer = writeBuffer;
69      }
70  
71      public ByteBuffer asByteBuffer() {
72          return this.buffer;
73      }
74  
75      public int bitbuffer() {
76          return this.buffer.get((int) (this.position / BITS_OF_1_BYTE));
77      }
78  
79      void close() {
80          if (this.position % BITS_OF_1_BYTE != 0) {
81              putByte((byte) 0, (int) (BITS_OF_1_BYTE - this.position % BITS_OF_1_BYTE));
82          }
83          this.buffer.position((int) (this.position / BITS_OF_1_BYTE));
84      }
85  
86      public void limit(int newLimit) {
87          this.buffer.limit(newLimit);
88      }
89  
90      public int missingBitsInCurrentByte() {
91          return (int) (BITS_OF_1_BYTE - this.position % BITS_OF_1_BYTE);
92      }
93  
94      public void movePosition(int i) {
95          this.position += i;
96      }
97  
98      public void putByte(byte byteToAdd) {
99          final int bytePosition = (int) (this.position / BITS_OF_1_BYTE);
100         final int positionInByte = (int) (this.position % BITS_OF_1_BYTE);
101         final byte old = (byte) (this.buffer.get(bytePosition) & (byte) ~(BYTE_MASK >>> positionInByte));
102         final int byteAsInt = byteToAdd & BYTE_MASK;
103         this.buffer.put(bytePosition, (byte) (old | (byte) (byteAsInt >>> positionInByte)));
104         if (positionInByte > 0) {
105             this.buffer.put(bytePosition + 1, (byte) (byteAsInt << BITS_OF_1_BYTE - positionInByte));
106         }
107         this.position += BITS_OF_1_BYTE;
108     }
109 
110     public void putByte(byte byteToAdd, int bits) {
111         final int bytePosition = (int) (this.position / BITS_OF_1_BYTE);
112         final int positionInByte = (int) (this.position % BITS_OF_1_BYTE);
113         final byte old = this.buffer.get(bytePosition);
114         final int byteAsInt = BYTE_MASK & (byteToAdd & BYTE_MASK >>> BITS_OF_1_BYTE - bits) << BITS_OF_1_BYTE - bits;
115         this.buffer.put(bytePosition, (byte) (BYTE_MASK & //
116                 (old & BYTE_MASK << BITS_OF_1_BYTE - positionInByte | byteAsInt >>> positionInByte)));
117         if (BITS_OF_1_BYTE - positionInByte < bits) {
118             this.buffer.put(bytePosition + 1, (byte) (BYTE_MASK & byteAsInt << BITS_OF_1_BYTE - positionInByte));
119         }
120         this.position += bits;
121     }
122 
123     /**
124      * write out int value to the next 4 bytes of the buffer
125      * 
126      * @param i
127      *            integer to write
128      */
129     public void putInt(int i) {
130         putByte((byte) ((i & BYTE_4_OF_INT) >>> BITS_OF_3_BYTES));
131         putByte((byte) ((i & BYTE_3_OF_INT) >>> BITS_OF_2_BYTES));
132         putByte((byte) ((i & BYTE_2_OF_INT) >>> BITS_OF_1_BYTE));
133         putByte((byte) (i & BYTE_1_OF_INT));
134     }
135 
136     public void putInt(int i, int bits) {
137         if (bits == 0) {
138             return;
139         }
140         do {
141             if (bits >= BITS_OF_1_BYTE) {
142                 putByte((byte) (((i & BYTE_MASK << bits - BITS_OF_1_BYTE) >>> bits - BITS_OF_1_BYTE) & BYTE_MASK));
143                 bits -= BITS_OF_1_BYTE;
144             } else {
145                 putByte((byte) (i & BYTE_MASK >> -(bits - BITS_OF_1_BYTE)), bits);
146                 bits = 0;
147             }
148         } while (bits > 0);
149     }
150 
151     public void putLong(long l, int bits) {
152         if (bits == 0) {
153             return;
154         }
155         do {
156             if (bits >= BITS_OF_4_BYTES) {
157                 putInt((int) ((l & INTEGER_MASK << bits - BITS_OF_4_BYTES) >>> bits - BITS_OF_4_BYTES));
158                 bits -= BITS_OF_4_BYTES;
159             } else {
160                 putInt((int) (l & INTEGER_MASK >> -(bits - BITS_OF_4_BYTES)), bits);
161                 bits = 0;
162             }
163         } while (bits > 0);
164     }
