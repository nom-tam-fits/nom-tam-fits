29  import java.io.IOException;
30  import java.io.InputStream;
31  import java.io.OutputStream;
32  import java.util.Map;
33  import nom.tam.fits.FitsException;
34  import nom.tam.fits.Header;
35  
36  /**
37   * @author tmcglynn
38   */
39  public interface CompressionScheme {
40  
41      /** Return the 'name' of the compression scheme */
42      public abstract String name();
43  
44      /**
45       * Initialize the compression scheme with any appropriate parameters.
46       */
47      public abstract void initialize(Map<String, String> params);
48  
49      /** Return a stream which compresses the input. */
50      public abstract byte[] compress(byte[] in) throws IOException;
51  
52      /** Return a stream with decompresses the input. */
53      public abstract byte[] decompress(byte[] in, int length) throws IOException;
54  
55      /**
56       * Update the FITS header and compression parameterswith information about
57       * the compression
58       */
59      public abstract void updateForWrite(Header hdr, Map<String, String> parameters) throws FitsException;
60  
61      /** Get the parameters indicated by a given header */
62      public abstract void getParameters(Map<String, String> params, Header hdr);
63  
64  }
