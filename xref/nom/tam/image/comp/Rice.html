45  /**
46   * Do Rice compression. Integer data in any local region rarely use the full
47   * dynamic range available for the data type. Rice compression looks at the
48   * differences between groups of integers and strips out leading 0's from the
49   * differences. Given a sequence of integers we: (** for steps where we actually
50   * write something)
51   * <ol>
52   * <li>** Write out the value of the first integer
53   * <li>Replace the sequence with the sequence of differences.
54   * <li>Replace positive differences, n, with 2*n and negative differences with
55   * -2*n-1. So 0 -> 0, -1 -> 1, 1 -> 2, -2 -> 3, 2 -> 4,... This means that
56   * differences will have the same number of bits (pretty much) for a given
57   * absolute value.
58   * <li>Loop over the sequence in blocks of 16 or 32 integers. There may be a
59   * shorter terminal block.
60   * <li>For each block:
61   * <ol>
62   * <li>Calculate the sum of the values and get the average of this sum. Find the
63   * number of bits it takes to represent this value, e.g., if the average is
64   * 110.4 then we would need 7 bits to represent this (2^7 = 128 which is >
65   * 110.4). Call this number F.
66   * <li>** Write out (F+1) (using a number of bits appropriate for the data
67   * size). We use F+1 rather than F to cater to the first special case below.
68   * <li>For each value in the block
69   * <ol>
70   * <li>Break up the number into two parts: the top,T, any bits more significant
71   * than the F low order bits, and the bottom, B, the lowest F bits in the value.
72   * To get the top we can divide by 2^F or left shift F bits. To get the bottom
73   * we can do a modulo operation or a bitwise AND with the mask 2^F - 1.
74   * <p>
75   * The top will often be 0 and will always be less than or equal to the number
76   * of values in the block. [In the worst case only one difference in the block
77   * in non-zero, then the average difference is the just that difference over the
78   * block size.]. Typically it will be 0, 1, or 2.
79   * <li>** Write out T 0 bits followed by a single 1 bit.
80   * <li>** Write out the B using just F bits.
81   * <li>In a typical case we will write about B+2 bits for each value. Say we
82   * have 16 bit pixels with values fluctuating typically by 30 units between
83   * adjacent pixels. Then we will write ~8 bits per pixel for a savings of >50%
84   * in storage. Overheads include writing the first value at the beginning of the
85   * file and 4 bits (for shorts) per block giving the value of F for each block.
86   * If we use 32 pixel blocks, a block of uncompressed data will be 512 bits
87   * while the compressed block would be ~240 bits.
88   * </ol>
89   * </ol>
90   * </ol>
91   * There are two special cases for each block.
92   * <ul>
93   * <li>If all the differences are 0, then instead of writing F+1 we just write 0
94   * to mark the block as constant.
95   * <li>If the differences between adjacent pixels are too high, i.e., there are
96   * largely random values in the original data, then splitting into top and
97   * bottom wastes a bit of space and time. This is indicated when F is greater
98   * than some maximum value. When we see this we don't do the splitting we just
99   * write the values directly to the output.
100  * </ul>
101  * 
102  * @author tmcglynn
103  */

159     /**
160      * Compress an input block. While the input is a byte array it may represent
161      * integers of any length. This routine sets up input and output processing
162      * streams.
163      */
164     public byte[] compress(byte[] in) throws IOException {
165         if (!initialized) {
166             throw new IllegalStateException("Rice compressor not initialized");
167         }
168 
169         // The number of elements is just the number of bytes
170         // over the number of bytes per element.
171         int len = in.length / (bitpix / 8);
172 
173         DataInputStream ds = new DataInputStream(new ByteArrayInputStream(in));
174         ByteArrayOutputStream out = new ByteArrayOutputStream(32768);
175         OutputBitStream bo = new OutputBitStream(out);
176         compressStream(ds, bo, len);
177         bo.flush();
178         bo.close();
179         return out.toByteArray();
180     }

182     /**
183      * This routine compresses a stream of input values into the output. It
184      * handles the looping over all blocks.
185      */
186     public void compressStream(DataInputStream ds, OutputBitStream bo, int len) throws IOException {
187 
188         int offset = 0;
189         int word = 0;
190         if (offset < len) {
191             word = getWord(ds);
192             // Write out the first word. We'll be computing
193             // differences starting from this value.
194             bo.writeBits(word, bitpix);
195         }
196 
197         boolean first = true;
198         // Loop over individual compression blocks.
199         while (offset < len) {
200             int thisBlock = Math.min(block, len - offset);
201             word = compressBlock(first, word, thisBlock, ds, bo);
202             first = false;
203             offset += thisBlock;
204         }
205         bo.close();
206     }

208     /**
209      * This routine compresses a single block.
210      * 
211      * @param first
212      *            Is this the first block in the compression stream?
213      * @param lastWord
214      *            What was the last word (not difference) read int from the
215      *            input. If this is the first block it should be the first word
216      *            in the file.
217      * @param len
218      *            The number of words in this block
219      * @param ds
220      *            The input stream from which we are reading data.
221      * @param bo
222      *            The output bit stream on which we are writing results.
223      * @return The last word read (not the last difference).
224      */
225     private int compressBlock(boolean first, int lastWord, int len, DataInputStream ds, OutputBitStream bo) throws IOException {
226 
227         int[] diffs = new int[len];
228 
229         // For the first block we've already read the first word (and
230         // written it out) so we treat it a bit specially.
231         if (!first) {
232             diffs[0] = getWord(ds);
233         } else {
234             diffs[0] = lastWord;
235         }
236         int newLast = diffs[diffs.length - 1];
237 
238         // Fill in the rest of the array.
239         for (int i = 1; i < diffs.length; i += 1) {
240             diffs[i] = getWord(ds);
241         }
242 
243         // Convert the integer values to integer differences
244         // and compute the sum of the differences.
245         long sum = convertToDiffs(lastWord, diffs);
246 
247         // Get the number of noise bits.
248         int fs = getFs(sum, len);
249 
250         if (sum == 0) {
251             // A constant block
252             specialCaseZero(bo);
253         } else if (sum >= fsmax) {
254             // To much variation to comprss
255             specialCaseRandom(bo, diffs);
256         } else {
257 
258             // The normal case.
259             int mask = (1 << fs) - 1;
260             // Write out the number of 'noise' bits.
261             bo.writeBits(fs + 1, fsbits);
262             for (int i = 0; i < len; i += 1) {
263                 // Now write out the encoded differences
264                 emit(diffs[i], fs, bo, mask);
265             }
266         }
267 
268         // Return the last input value which we will need when
269         // computing the differences for the next block.
270         return newLast;
271     }

273     /**
274      * Emit the coding for a single compressed word.
275      * 
276      * @param val
277      *            The input value to be emitted (normally a pixel difference).
278      * @param fs
279      *            The number of 'noise' bits for this block.
280      * @param bo
281      *            The output bit stream.
282      * @param mask
283      *            A mask getting the noise bits (= (1<<fs)-1).
284      */
285     private void emit(int val, int fs, OutputBitStream bo, int mask) throws IOException {
286         int top = val >>> fs;
287         int bot = val & mask;
288         bo.writeBits(1, top + 1);
289         bo.writeBits(bot, fs);
290     }

299     /**
300      * Handle the case where the noise is so large that no compression is
301      * possible
302      */
303     private void specialCaseRandom(OutputBitStream bo, int[] diffs) throws IOException {
304         bo.writeBits(fsmax + 1, fsbits);
305         for (int i = 0; i < diffs.length; i += 1) {
306             bo.writeBits(diffs[i], bitpix);
307         }
308     }

310     /**
311      * Convert an array of integers into an array of differences.
312      * 
313      * @param lastWord
314      *            The last word in the previous output. For the first call this
315      *            should be the same as the first input element of diffs. For
316      *            all later calls it should be the last input value of diffs in
317      *            the previous call.
318      * @param diffs
319      *            On input this contains the original integer values. On output
320      *            it contains the integer differences coded so that all values
321      *            are positive. Positive values go to 2*val Negative values got
322      *            to 2*|val|-1.
323      * @return The sum of the returned elements in the array.
324      */
325     private long convertToDiffs(int lastWord, int[] diffs) {
326         long sum = 0;
327         for (int i = 0; i < diffs.length; i += 1) {
328             int d = diffs[i] - lastWord;
329             lastWord = diffs[i];
330             if (d < 0) {
331                 d = -2 * d - 1;
332             } else {
333                 d *= 2;
334             }
335             sum += d;
336             diffs[i] = d;
337         }
338         return sum;
339     }

341     /**
342      * Get the number of 'noise' bits in the block
343      */
344     private int getFs(long sum, int len) {
345         // We made 1 -> 1. since in original sum was double,
346         // so expression in parens should be double.
347 
348         // We want to compute the 'average' difference. Not
349         // sure why we have the thisBlock/2 - 1 there, but
350         // presumably that comes out in the details.
351         double dpSum = (sum - len / 2 - 1.) / len;
352         if (dpSum < 0) {
353             dpSum = 0;
354         }
355 
356         // How many bits does it take to represent the 'average' difference.
357         int fs;
358         int psum = ((int) dpSum >> 1);
359         for (fs = 0; psum > 0; fs += 1) {
360             psum >>= 1;
361         }
362         return fs;
363     }

365     /**
366      * Read a full pixel from the input where the length of pixels has been set
367      * by the user
368      */
369     private int getWord(DataInputStream is) throws IOException {
370         if (bitpix == 8) {
371             return is.readByte() & 0xFF;
372         } else if (bitpix == 16) {
373             return is.readShort() & 0xFFFF;
374         } else {
375             return is.readInt();
376         }
377     }

379     /**
380      * Decompress input data using the Rice algorithm.
381      * 
382      * @param in
383      *            The compressed data.
384      * @param len
385      *            The number of pixels expected on the input.
386      * @return A byte array representing the uncompressed data. This may need to
387      *         be read through a ByteArrayInputStream to recover the appropriate
388      *         pixel values (which may not be bytes).
389      */
390     public byte[] decompress(byte[] in, int len) throws IOException {
391 
392         if (!initialized) {
393             throw new IllegalStateException("Rice compressor not initialized");
394         }
395 
396         InputBitStream bin = new InputBitStream(new ByteArrayInputStream(in));
397         ByteArrayOutputStream out = new ByteArrayOutputStream(32768);
398         DataOutputStream ds = new DataOutputStream(out);
399 
400         decompressStream(bin, ds, len);
401         bin.close();
402         return out.toByteArray();
403     }

405     /** Decompress an input stream expecting a given number of words */
406 
407     public void decompressStream(InputBitStream bin, DataOutputStream ds, int len) throws IOException {
408 
409         int offset = 0;
410         // Read the first pixel.
411         int word = bin.readBits(bitpix);
412         while (offset < len) {
413             int thisBlock = block;
414             if (offset + block > len) {
415                 thisBlock = len - offset;
416             }
417             decompressBlock(word, bin, ds, thisBlock);
418             offset += thisBlock;
419         }
420     }

422     /** Decompress a single block */
423 
424     private void decompressBlock(int word, InputBitStream bin, DataOutputStream ds, int len) throws IOException {
425 
426         int fs = bin.readBits(fsbits) - 1;
427 
428         if (fs < 0) {
429             decompressConstant(word, ds, len);
430 
431         } else if (fs >= fsmax) {
432             word = decompressRandom(word, bin, ds, len);
433 
434         } else {
435             for (int i = 0; i < len; i += 1) {
436                 word = decodeWord(word, fs, bin);
437                 writeWord(ds, word);
438             }
439         }
440     }
441 
442     /** Decompress a block of constant values */
443     private void decompressConstant(int word, DataOutputStream ds, int len) throws IOException {
444         for (int i = 0; i < len; i += 1) {
445             writeWord(ds, word);
446         }
447     }

449     /**
450      * Write a block of 'random' values, i.e., where the noise is to large to be
451      * usefully compressed.
452      */
453     private int decompressRandom(int word, InputBitStream bin, DataOutputStream ds, int len) throws IOException {
454         for (int i = 0; i < len; i += 1) {
455             int diff = bin.readBits(bbits);
456             if (diff % 2 == 0) {
457                 diff >>= 1;
458             } else {
459                 diff = ~(diff >> 1);
460             }
461             word += diff;
462             writeWord(ds, word);
463         }
464         return word;
465     }

467     /** Reconstruct a single word. */
468     private int decodeWord(int word, int fs, InputBitStream bin) throws IOException {
469         int nbits = 0;
470         int b = 0;
471         // Find the number of 0 bits. That count is the
472         // high order part of the difference.
473         int high = bin.skipBits(false);
474         // Skip the 1 separator bit
475         bin.readBits(1);
476         // Read the fs 'noise' bits.
477         int low = bin.readBits(fs);
478         // We don't worry about sign extension since fs is always < 32.
479         int diff = (high << fs) | low;
480 
481         // Remember that this is not quite the difference, so
482         // convert back to the actual difference and add to the previous
483         // word.
484         if (diff % 2 == 0) {
485             word += diff / 2;
486         } else {
487             word -= (diff + 1) / 2;
488         }
489         return word;
490     }

492     /**
493      * Write a full pixel value to the output decompression stream.
494      */
495     private void writeWord(DataOutputStream ds, int val) throws IOException {
496         if (bitpix == 8) {
497             ds.writeByte(val);
498         } else if (bitpix == 16) {
499             ds.writeShort(val);
500         } else if (bitpix == 32) {
501             ds.writeInt(val);
502         }
503     }

505     public static void main(String[] args) throws Exception {
506         int[] test = new int[100];
507         for (int i = 0; i < test.length; i += 1) {
508             if (i % 2 != 0) {
509                 test[i] = 1000 - 2 * i;
510             } else {
511                 test[i] = 1000 + 2 * i;
512             }
513         }
514         Rice comp = new Rice();
515         Map<String, String> init = new HashMap<String, String>();
516         init.put("bitpix", "32");
517         init.put("block", "32");
518         init.put("length", "100");
519         comp.initialize(init);
520 
521         ByteArrayOutputStream bo = new ByteArrayOutputStream();
522         BufferedDataOutputStream d = new BufferedDataOutputStream(bo);
523         d.write(test);
524         d.close();
525         byte[] input = bo.toByteArray();
526         byte[] result = comp.compress(input);
527         System.out.println("Result len:" + result.length);
528         for (int i = 0; i < result.length; i += 1) {
529             System.out.printf("%d: %3d %2x\n", i, result[i], result[i]);
530         }
531 
532         result = comp.decompress(result, 100);
533         DataInputStream bi = new DataInputStream(new ByteArrayInputStream(result));
534         for (int i = 0; i < 100; i += 1) {
535             System.out.println(i + ": " + bi.readInt());
536         }
537     }

539     public void updateForWrite(Header hdr, Map<String, String> parameters) throws FitsException {
540 
541         int bitpix = hdr.getIntValue("ZBITPIX", -1);
542 
543         int block;
544         if (!parameters.containsKey("block")) {
545             parameters.put("block", "" + 32);
546             block = 32;
547         } else {
548             block = Integer.parseInt(parameters.get("block"));
549         }
550 
551         hdr.addValue("ZNAME1", "BLOCKSIZE", "Compression region size");
552         hdr.addValue("ZVAL1", block, "Compression region size");
553 
554         hdr.addValue("ZNAME2", "BYTEPIX", "Bytes in pixel");
555         if (bitpix > 0) {
556             parameters.put("bitpix", "" + bitpix);
557             hdr.addValue("ZVAL2", bitpix / 8, "Bytes in pixel");
558         } else {
559             parameters.put("bitpix", "32");
560             hdr.addValue("ZVAL2", 4, "Bytes in pixel");
561         }
562     }
