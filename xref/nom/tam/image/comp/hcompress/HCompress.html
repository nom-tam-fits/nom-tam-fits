224     private int bufcopy(byte[] a, int n, byte[] buffer, int b, long bmax) {
225         int i;
226 
227         for (i = 0; i < n; i++) {
228             if (a[i] != 0) {
229                 /*
230                  * add Huffman code for a[i] to buffer
231                  */
232                 this.bitbuffer |= CODE[a[i]] << this.bitsToGo3;
233                 this.bitsToGo3 += NCODE[a[i]];
234                 if (this.bitsToGo3 >= BITS_OF_1_BYTE) {
235                     buffer[b] = (byte) (this.bitbuffer & BYTE_MASK);
236                     b += 1;
237                     /*
238                      * return warning code if we fill buffer
239                      */
240                     if (b >= bmax) {
241                         return b;
242                     }
243                     this.bitbuffer >>= BITS_OF_1_BYTE;
244                     this.bitsToGo3 -= BITS_OF_1_BYTE;
245                 }
246             }
247         }
248         return b;
249     }

251     protected void compress(long[] aa, int ny, int nx, int scale, ByteBuffer output) {
252         /*
253          * compress the input image using the H-compress algorithm a - input
254          * image array nx - size of X axis of image ny - size of Y axis of image
255          * scale - quantization scale factor. Larger values results in more
256          * (lossy) compression scale = 0 does lossless compression output -
257          * pre-allocated array to hold the output compressed stream of bytes
258          * nbyts - input value = size of the output buffer; returned value =
259          * size of the compressed byte stream, in bytes NOTE: the nx and ny
260          * dimensions as defined within this code are reversed from the usual
261          * FITS notation. ny is the fastest varying dimension, which is usually
262          * considered the X axis in the FITS image display
263          */
264 
265         /* H-transform */
266         htrans(aa, nx, ny);
267 
268         LongArrayPointer a = new LongArrayPointer();
269         a.a = aa;
270         a.offset = 0;
271 
272         /* digitize */
273         digitize(a, 0, nx, ny, scale);
274 
275         /* encode and write to output array */
276         encode(output, a, nx, ny, scale);
277 
278     }

282     private void digitize(LongArrayPointer a, int aOffset, int nx, int ny, long scale) {
283         long d;
284         LongArrayPointer p;
285 
286         /*
287          * round to multiple of scale
288          */
289         if (scale <= 1) {
290             return;
291         }
292         d = (scale + 1L) / 2L - 1L;
293         for (int index = 0; index < a.a.length; index++) {
294             long current = a.get(index);
295             a.set(index, (current > 0 ? current + d : current - d) / scale);
296         }
297     }

299     private void doencode(ByteBuffer outfile, LongArrayPointer a, int nx, int ny, byte[] nbitplanes) {
300         /*
301          * char *outfile; output data stream int a[]; Array of values to encode
302          * int nx,ny; Array dimensions [nx][ny] unsigned char nbitplanes[3];
303          * Number of bit planes in quadrants
304          */
305 
306         int nx2, ny2;
307 
308         nx2 = (nx + 1) / 2;
309         ny2 = (ny + 1) / 2;
310         /*
311          * Initialize bit output
312          */
313         startOutputingBits();
314         /*
315          * write out the bit planes for each quadrant
316          */
317         qtreeEncode(outfile, a.copy(), ny, nx2, ny2, nbitplanes[0]);
318 
319         qtreeEncode(outfile, a.copy(ny2), ny, nx2, ny / 2, nbitplanes[1]);
320 
321         qtreeEncode(outfile, a.copy(ny * nx2), ny, nx / 2, ny2, nbitplanes[1]);
322 
323         qtreeEncode(outfile, a.copy(ny * nx2 + ny2), ny, nx / 2, ny / 2, nbitplanes[2]);
324         /*
325          * Add zero as an EOF symbol
326          */
327         outputNybble(outfile, 0);
328         doneOutputingBits(outfile);
329 
330     }

332     private void doneOutputingBits(ByteBuffer outfile) {
333         if (this.bitsToGo2 < BITS_OF_1_BYTE) {
334             /* putc(buffer2<<bits_to_go2,outfile); */
335 
336             outfile.put((byte) (this.buffer2 << this.bitsToGo2));
337         }
338     }

340     private int encode(ByteBuffer outfile, LongArrayPointer a, int nx, int ny, int scale) {
341 
342         /* FILE *outfile; - change outfile to a char array */
343         /*
344          * long * nlength returned length (in bytes) of the encoded array) int
345          * a[]; input H-transform array (nx,ny) int nx,ny; size of H-transform
346          * array int scale; scale factor for digitization
347          */
348         int nel, nx2, ny2, i, j, k, q, nsign, bitsToGo;
349         long[] vmax = new long[N3];
350         byte[] nbitplanes = new byte[N3];
351         byte[] signbits;
352 
353         long noutchar = 0; /*
354                             * initialize the number of compressed bytes that
355                             * have been written
356                             */
357         nel = nx * ny;
358         /*
359          * write magic value
360          */
361         outfile.put(CODE_MAGIC);
362         outfile.putInt(nx); /* size of image */
363         outfile.putInt(ny);
364         outfile.putInt(scale); /* scale factor for digitization */
365         /*
366          * write first value of A (sum of all pixels -- the only value which
367          * does not compress well)
368          */
369         outfile.putLong(a.get());
370 
371         a.set(0);
372         /*
373          * allocate array for sign bits and save values, 8 per byte
374          */
375         signbits = new byte[(nel + BITS_OF_1_BYTE - 1) / BITS_OF_1_BYTE];
376 
377         nsign = 0;
378         bitsToGo = BITS_OF_1_BYTE;
379         signbits[0] = 0;
380         for (i = 0; i < nel; i++) {
381             if (a.get(i) > 0) {
382                 /*
383                  * positive element, put zero at end of buffer
384                  */
385                 signbits[nsign] <<= 1;
386                 bitsToGo -= 1;
387             } else if (a.get(i) < 0) {
388                 /*
389                  * negative element, shift in a one
390                  */
391                 signbits[nsign] <<= 1;
392                 signbits[nsign] |= 1;
393                 bitsToGo -= 1;
394                 /*
395                  * replace a by absolute value
396                  */
397                 a.set(i, -a.get(i));
398             }
399             if (bitsToGo == 0) {
400                 /*
401                  * filled up this byte, go to the next one
402                  */
403                 bitsToGo = BITS_OF_1_BYTE;
404                 nsign += 1;
405                 signbits[nsign] = 0;
406             }
407         }
408         if (bitsToGo != BITS_OF_1_BYTE) {
409             /*
410              * some bits in last element move bits in last byte to bottom and
411              * increment nsign
412              */
413             signbits[nsign] <<= bitsToGo;
414             nsign += 1;
415         }
416         /*
417          * calculate number of bit planes for 3 quadrants quadrant 0=bottom
418          * left, 1=bottom right or top left, 2=top right,
419          */
420         for (q = 0; q < N3; q++) {
421             vmax[q] = 0;
422         }
423         /*
424          * get maximum absolute value in each quadrant
425          */
426         nx2 = (nx + 1) / 2;
427         ny2 = (ny + 1) / 2;
428         j = 0; /* column counter */
429         k = 0; /* row counter */
430         for (i = 0; i < nel; i++) {
431             q = (j >= ny2 ? 1 : 0) + (k >= nx2 ? 1 : 0);
432             if (vmax[q] < a.get(i)) {
433                 vmax[q] = a.get(i);
434             }
435             if (++j >= ny) {
436                 j = 0;
437                 k += 1;
438             }
439         }
440         /*
441          * now calculate number of bits for each quadrant
442          */
443 
444         /* this is a more efficient way to do this, */
445 
446         for (q = 0; q < N3; q++) {
447             for (nbitplanes[q] = 0; vmax[q] > 0; vmax[q] = vmax[q] >> 1, nbitplanes[q]++) {
448                 // noop for later refactoring
449                 "".toString();
450             }
451         }
452 
453         /*
454          * write nbitplanes
455          */
456         outfile.put(nbitplanes, 0, nbitplanes.length);
457 
458         /*
459          * write coded array
460          */
461         doencode(outfile, a, nx, ny, nbitplanes);
462         /*
463          * write sign bits
464          */
465 
466         if (nsign > 0) {
467             outfile.put(signbits, 0, nsign);
468         }
469 
470         return (int) noutchar;
471 
472     }

474     private int htrans(long[] a, int nx, int ny) {
475         int nmax, log2n, nxtop, nytop, i, j, k;
476         int oddx, oddy;
477         int s10, s00;
478         long h0, hx, hy, hc, shift, mask, mask2, prnd, prnd2, nrnd2;
479         long[] tmp;
480 
481         /*
482          * log2n is log2 of max(nx,ny) rounded up to next power of 2
483          */
484         nmax = nx > ny ? nx : ny;
485         log2n = log2n(nmax);
486         if (nmax > 1 << log2n) {
487             log2n += 1;
488         }
489         /*
490          * get temporary storage for shuffling elements
491          */
492         tmp = new long[(nmax + 1) / 2];
493 
494         /*
495          * set up rounding and shifting masks
496          */
497         shift = 0;
498         mask = HTRANS_START_MASK;
499         mask2 = mask << 1;
500         prnd = 1;
501         prnd2 = prnd << 1;
502         nrnd2 = prnd2 - 1;
503         /*
504          * do log2n reductions We're indexing a as a 2-D array with dimensions
505          * (nx,ny).
506          */
507         nxtop = nx;
508         nytop = ny;
509 
510         for (k = 0; k < log2n; k++) {
511             oddx = nxtop % 2;
512             oddy = nytop % 2;
513             for (i = 0; i < nxtop - oddx; i += 2) {
514                 s00 = i * ny; /* s00 is index of a[i,j] */
515                 s10 = s00 + ny; /* s10 is index of a[i+1,j] */
516                 for (j = 0; j < nytop - oddy; j += 2) {
517                     /*
518                      * Divide h0,hx,hy,hc by 2 (1 the first time through).
519                      */
520                     h0 = a[s10 + 1] + a[s10] + a[s00 + 1] + a[s00] >> shift;
521                     hx = a[s10 + 1] + a[s10] - a[s00 + 1] - a[s00] >> shift;
522                     hy = a[s10 + 1] - a[s10] + a[s00 + 1] - a[s00] >> shift;
523                     hc = a[s10 + 1] - a[s10] - a[s00 + 1] + a[s00] >> shift;
524 
525                     /*
526                      * Throw away the 2 bottom bits of h0, bottom bit of hx,hy.
527                      * To get rounding to be same for positive and negative
528                      * numbers, nrnd2 = prnd2 - 1.
529                      */
530                     a[s10 + 1] = hc;
531                     a[s10] = (hx >= 0 ? hx + prnd : hx) & mask;
532                     a[s00 + 1] = (hy >= 0 ? hy + prnd : hy) & mask;
533                     a[s00] = (h0 >= 0 ? h0 + prnd2 : h0 + nrnd2) & mask2;
534                     s00 += 2;
535                     s10 += 2;
536                 }
537                 if (oddy != 0) {
538                     /*
539                      * do last element in row if row length is odd s00+1, s10+1
540                      * are off edge
541                      */
542                     h0 = a[s10] + a[s00] << 1 - shift;
543                     hx = a[s10] - a[s00] << 1 - shift;
544                     a[s10] = (hx >= 0 ? hx + prnd : hx) & mask;
545                     a[s00] = (h0 >= 0 ? h0 + prnd2 : h0 + nrnd2) & mask2;
546                     s00 += 1;
547                     s10 += 1;
548                 }
549             }
550             if (oddx != 0) {
551                 /*
552                  * do last row if column length is odd s10, s10+1 are off edge
553                  */
554                 s00 = i * ny;
555                 for (j = 0; j < nytop - oddy; j += 2) {
556                     h0 = a[s00 + 1] + a[s00] << 1 - shift;
557                     hy = a[s00 + 1] - a[s00] << 1 - shift;
558                     a[s00 + 1] = (hy >= 0 ? hy + prnd : hy) & mask;
559                     a[s00] = (h0 >= 0 ? h0 + prnd2 : h0 + nrnd2) & mask2;
560                     s00 += 2;
561                 }
562                 if (oddy != 0) {
563                     /*
564                      * do corner element if both row and column lengths are odd
565                      * s00+1, s10, s10+1 are off edge
566                      */
567                     h0 = a[s00] << 2 - shift;
568                     a[s00] = (h0 >= 0 ? h0 + prnd2 : h0 + nrnd2) & mask2;
569                 }
570             }
571             /*
572              * now shuffle in each dimension to group coefficients by order
573              */
574             // achtung eigenlich pointer nach a
575             for (i = 0; i < nxtop; i++) {
576                 shuffle(a, ny * i, nytop, 1, tmp);
577             }
578             for (j = 0; j < nytop; j++) {
579                 shuffle(a, j, nxtop, ny, tmp);
580             }
581             /*
582              * image size reduced by 2 (round up if odd)
583              */
584             nxtop = nxtop + 1 >> 1;
585             nytop = nytop + 1 >> 1;
586             /*
587              * divisor doubles after first reduction
588              */
589             shift = 1;
590             /*
591              * masks, rounding values double after each iteration
592              */
593             mask = mask2;
594             prnd = prnd2;
595             mask2 = mask2 << 1;
596             prnd2 = prnd2 << 1;
597             nrnd2 = prnd2 - 1;
598         }
599         return 0;
600     }

602     private int log2n(int nqmax) {
603         return (int) (Math.log(nqmax) / Math.log(2.0) + ROUNDING_HALF);
604     }

606     private void outputNbits(ByteBuffer outfile, int bits, int n) {
607         /* AND mask for the right-most n bits */
608 
609         /*
610          * insert bits at end of buffer
611          */
612         this.buffer2 <<= n;
613         /* buffer2 |= ( bits & ((1<<n)-1) ); */
614         this.buffer2 |= bits & BITS_MASK[n];
615         this.bitsToGo2 -= n;
616         if (this.bitsToGo2 <= 0) {
617             /*
618              * buffer2 full, put out top 8 bits
619              */
620 
621             outfile.put((byte) (this.buffer2 >> -this.bitsToGo2 & BYTE_MASK));
622 
623             this.bitsToGo2 += BITS_OF_1_BYTE;
624         }
625     }

627     private void outputNnybble(ByteBuffer outfile, int n, byte[] array) {
628         /*
629          * pack the 4 lower bits in each element of the array into the outfile
630          * array
631          */
632 
633         int ii, jj, kk = 0, shift;
634 
635         if (n == 1) {
636             outputNybble(outfile, array[0]);
637             return;
638         }
639         /*
640          * forcing byte alignment doesn;t help, and even makes it go slightly
641          * slower if (bits_to_go2 != 8) output_nbits(outfile, kk, bits_to_go2);
642          */
643         if (this.bitsToGo2 <= BITS_OF_1_NYBBLE) {
644             /* just room for 1 nybble; write it out separately */
645             outputNybble(outfile, array[0]);
646             kk++; /* index to next array element */
647 
648             if (n == 2) {
649                 // only 1 more nybble to write out
650                 outputNybble(outfile, array[1]);
651                 return;
652             }
653         }
654 
655         /* bits_to_go2 is now in the range 5 - 8 */
656         shift = BITS_OF_1_BYTE - this.bitsToGo2;
657 
658         /*
659          * now write out pairs of nybbles; this does not affect value of
660          * bits_to_go2
661          */
662         jj = (n - kk) / 2;
663 
664         if (this.bitsToGo2 == BITS_OF_1_BYTE) {
665             /* special case if nybbles are aligned on byte boundary */
666             /* this actually seems to make very little differnece in speed */
667             this.buffer2 = 0;
668             for (ii = 0; ii < jj; ii++) {
669                 outfile.put((byte) ((array[kk] & NYBBLE_MASK) << BITS_OF_1_NYBBLE | array[kk + 1] & NYBBLE_MASK));
670                 kk += 2;
671             }
672         } else {
673             for (ii = 0; ii < jj; ii++) {
674                 this.buffer2 = this.buffer2 << BITS_OF_1_BYTE | (array[kk] & NYBBLE_MASK) << BITS_OF_1_NYBBLE | array[kk + 1] & NYBBLE_MASK;
675                 kk += 2;
676 
677                 /*
678                  * buffer2 full, put out top 8 bits
679                  */
680 
681                 outfile.put((byte) (this.buffer2 >> shift & BYTE_MASK));
682             }
683         }
684 
685         /* write out last odd nybble, if present */
686         if (kk != n) {
687             outputNybble(outfile, array[n - 1]);
688         }
689 
690         return;
691     }

693     private void outputNybble(ByteBuffer outfile, int bits) {
694         /*
695          * insert 4 bits at end of buffer
696          */
697         this.buffer2 = this.buffer2 << BITS_OF_1_NYBBLE | bits & NYBBLE_MASK;
698         this.bitsToGo2 -= BITS_OF_1_NYBBLE;
699         if (this.bitsToGo2 <= 0) {
700             /*
701              * buffer2 full, put out top 8 bits
702              */
703 
704             outfile.put((byte) (this.buffer2 >> -this.bitsToGo2 & BYTE_MASK));
705 
706             this.bitsToGo2 += BITS_OF_1_BYTE;
707         }
708     }

713     private int qtreeEncode(ByteBuffer outfile, LongArrayPointer a, int n, int nqx, int nqy, int nbitplanes) {
714 
715         /*
716          * int a[]; int n; physical dimension of row in a int nqx; length of row
717          * int nqy; length of column (<=n) int nbitplanes; number of bit planes
718          * to output
719          */
720 
721         int log2n, i, k, bit, b, nqmax, nqx2, nqy2, nx, ny;
722         long bmax;
723         byte[] scratch, buffer;
724 
725         /*
726          * log2n is log2 of max(nqx,nqy) rounded up to next power of 2
727          */
728         nqmax = nqx > nqy ? nqx : nqy;
729         log2n = log2n(nqmax);
730         if (nqmax > 1 << log2n) {
731             log2n += 1;
732         }
733         /*
734          * initialize buffer point, max buffer size
735          */
736         nqx2 = (nqx + 1) / 2;
737         nqy2 = (nqy + 1) / 2;
738         bmax = (nqx2 * nqy2 + 1) / 2;
739         /*
740          * We're indexing A as a 2-D array with dimensions (nqx,nqy). Scratch is
741          * 2-D with dimensions (nqx/2,nqy/2) rounded up. Buffer is used to store
742          * string of codes for output.
743          */
744         scratch = new byte[(int) (2 * bmax)];
745         buffer = new byte[(int) bmax];
746 
747         /*
748          * now encode each bit plane, starting with the top
749          */
750         bitplane_done: for (bit = nbitplanes - 1; bit >= 0; bit--) {
751             /*
752              * initial bit buffer
753              */
754             b = 0;
755             this.bitbuffer = 0;
756             this.bitsToGo3 = 0;
757             /*
758              * on first pass copy A to scratch array
759              */
760             qtreeOnebit(a, n, nqx, nqy, scratch, bit);
761             nx = nqx + 1 >> 1;
762             ny = nqy + 1 >> 1;
763             /*
764              * copy non-zero values to output buffer, which will be written in
765              * reverse order
766              */
767             b = bufcopy(scratch, nx * ny, buffer, b, bmax);
768             if (b >= bmax) {
769                 /*
770                  * quadtree is expanding data, change warning code and just fill
771                  * buffer with bit-map
772                  */
773                 writeBdirect(outfile, a, n, nqx, nqy, scratch, bit);
774                 continue bitplane_done;
775             }
776             /*
777              * do log2n reductions
778              */
779             for (k = 1; k < log2n; k++) {
780                 qtreeReduce(scratch, ny, nx, ny, scratch);
781                 nx = nx + 1 >> 1;
782                 ny = ny + 1 >> 1;
783                 b = bufcopy(scratch, nx * ny, buffer, b, bmax);
784                 if (b >= bmax) {
785                     writeBdirect(outfile, a, n, nqx, nqy, scratch, bit);
786                     continue bitplane_done;
787                 }
788             }
789             /*
790              * OK, we've got the code in buffer Write quadtree warning code,
791              * then write buffer in reverse order
792              */
793             outputNybble(outfile, NYBBLE_MASK);
794             if (b == 0) {
795                 if (this.bitsToGo3 > 0) {
796                     /*
797                      * put out the last few bits
798                      */
799                     outputNbits(outfile, this.bitbuffer & (1 << this.bitsToGo3) - 1, this.bitsToGo3);
800                 } else {
801                     /*
802                      * have to write a zero nybble if there are no 1's in array
803                      */
804                     outputNbits(outfile, CODE[0], NCODE[0]);
805                 }
806             } else {
807                 if (this.bitsToGo3 > 0) {
808                     /*
809                      * put out the last few bits
810                      */
811                     outputNbits(outfile, this.bitbuffer & (1 << this.bitsToGo3) - 1, this.bitsToGo3);
812                 }
813                 for (i = b - 1; i >= 0; i--) {
814                     outputNbits(outfile, buffer[i], BITS_OF_1_BYTE);
815                 }
816             }
817         }
818         return 0;
819     }

821     private void qtreeOnebit(LongArrayPointer a, int n, int nx, int ny, byte[] b, int bit) {
822         int i, j, k;
823         long b0, b1, b2, b3;
824         int s10, s00;
825 
826         /*
827          * use selected bit to get amount to shift
828          */
829         b0 = 1L << bit;
830         b1 = b0 << 1;
831         b2 = b1 << 1;
832         b3 = b2 << 1;
833         k = 0; /* k is index of b[i/2,j/2] */
834         for (i = 0; i < nx - 1; i += 2) {
835             s00 = n * i; /* s00 is index of a[i,j] */
836             /*
837              * tried using s00+n directly in the statements, but this had no
838              * effect on performance
839              */
840             s10 = s00 + n; /* s10 is index of a[i+1,j] */
841             for (j = 0; j < ny - 1; j += 2) {
842 
843                 b[k] = (byte) ((a.get(s10 + 1) & b0 //
844                         | a.get(s10) << 1 & b1 //
845                         | a.get(s00 + 1) << 2 & b2 //
846                 | a.get(s00) << N3 & b3) >> bit);
847 
848                 k += 1;
849                 s00 += 2;
850                 s10 += 2;
851             }
852             if (j < ny) {
853                 /*
854                  * row size is odd, do last element in row s00+1,s10+1 are off
855                  * edge
856                  */
857                 b[k] = (byte) ((a.get(s10) << 1 & b1 | a.get(s00) << N3 & b3) >> bit);
858                 k += 1;
859             }
860         }
861         if (i < nx) {
862             /*
863              * column size is odd, do last row s10,s10+1 are off edge
864              */
865             s00 = n * i;
866             for (j = 0; j < ny - 1; j += 2) {
867                 b[k] = (byte) ((a.get(s00 + 1) << 2 & b2 | a.get(s00) << N3 & b3) >> bit);
868                 k += 1;
869                 s00 += 2;
870             }
871             if (j < ny) {
872                 /*
873                  * both row and column size are odd, do corner element s00+1,
874                  * s10, s10+1 are off edge
875                  */
876                 b[k] = (byte) ((a.get(s00) << N3 & b3) >> bit);
877                 k += 1;
878             }
879         }
880     }

882     private void qtreeReduce(byte[] a, int n, int nx, int ny, byte[] b) {
883         int i, j, k;
884         int s10, s00;
885 
886         k = 0; /* k is index of b[i/2,j/2] */
887         for (i = 0; i < nx - 1; i += 2) {
888             s00 = n * i; /* s00 is index of a[i,j] */
889             s10 = s00 + n; /* s10 is index of a[i+1,j] */
890             for (j = 0; j < ny - 1; j += 2) {
891                 b[k] = (byte) (b2i(a[s10 + 1] != 0) | b2i(a[s10] != 0) << 1 | b2i(a[s00 + 1] != 0) << 2 | b2i(a[s00] != 0) << N3);
892                 k += 1;
893                 s00 += 2;
894                 s10 += 2;
895             }
896             if (j < ny) {
897                 /*
898                  * row size is odd, do last element in row s00+1,s10+1 are off
899                  * edge
900                  */
901                 b[k] = (byte) (b2i(a[s10] != 0) << 1 | b2i(a[s00] != 0) << N3);
902                 k += 1;
903             }
904         }
905         if (i < nx) {
906             /*
907              * column size is odd, do last row s10,s10+1 are off edge
908              */
909             s00 = n * i;
910             for (j = 0; j < ny - 1; j += 2) {
911                 b[k] = (byte) (b2i(a[s00 + 1] != 0) << 2 | b2i(a[s00] != 0) << N3);
912                 k += 1;
913                 s00 += 2;
914             }
915             if (j < ny) {
916                 /*
917                  * both row and column size are odd, do corner element s00+1,
918                  * s10, s10+1 are off edge
919                  */
920                 b[k] = (byte) (b2i(a[s00] != 0) << N3);
921                 k += 1;
922             }
923         }
924     }

926     private void shuffle(long[] a, int aOffet, int n, int n2, long[] tmp) {
927 
928         /*
929          * int a[]; array to shuffle int n; number of elements to shuffle int
930          * n2; second dimension int tmp[]; scratch storage
931          */
932 
933         int i;
934         long[] p1, p2, pt;
935         int p1Offset = 0;
936         int ptOffset = 0;
937         int p2Offset = 0;
938         /*
939          * copy odd elements to tmp
940          */
941         pt = tmp;
942         ptOffset = 0;
943         p1 = a;
944         p1Offset = aOffet + n2;
945         for (i = 1; i < n; i += 2) {
946             pt[ptOffset] = p1[p1Offset];
947             ptOffset += 1;
948             p1Offset += n2 + n2;
949         }
950         /*
951          * compress even elements into first half of A
952          */
953         p1 = a;
954         p1Offset = aOffet + n2;
955         p2 = a;
956         p2Offset = aOffet + n2 + n2;
957         for (i = 2; i < n; i += 2) {
958             p1[p1Offset] = p2[p2Offset];
959             p1Offset += n2;
960             p2Offset += n2 + n2;
961         }
962         /*
963          * put odd elements into 2nd half
964          */
965         pt = tmp;
966         ptOffset = 0;
967         for (i = 1; i < n; i += 2) {
968             p1[p1Offset] = pt[ptOffset];
969             p1Offset += n2;
970             ptOffset += 1;
971         }
972     }

979     private void writeBdirect(ByteBuffer outfile, LongArrayPointer a, int n, int nqx, int nqy, byte[] scratch, int bit) {
980 
981         /*
982          * Write the direct bitmap warning code
983          */
984         outputNybble(outfile, 0x0);
985         /*
986          * Copy A to scratch array (again!), packing 4 bits/nybble
987          */
988         qtreeOnebit(a, n, nqx, nqy, scratch, bit);
989         /*
990          * write to outfile
991          */
992         /*
993          * int i; for (i = 0; i < ((nqx+1)/2) * ((nqy+1)/2); i++) {
994          * output_nybble(outfile,scratch[i]); }
995          */
996         outputNnybble(outfile, (nqx + 1) / 2 * ((nqy + 1) / 2), scratch);
997 
998     }
