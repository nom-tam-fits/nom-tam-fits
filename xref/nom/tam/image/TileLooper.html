38  import java.util.Iterator;
39  
40  /**
41   * @author tmcglynn
42   */
43  
44  public class TileLooper implements Iterable<TileDescriptor> {
45  
46      private int[] tileIndices;
47  
48      private int[] imageSize;
49  
50      private int[] tileSize;
51  
52      private int[] nTiles;
53  
54      private int[] tilesCorner;
55  
56      private int[] tilesCount;
57  
58      private int dim;
59  
60      private class TileIterator implements Iterator<TileDescriptor> {
61  
62          boolean first = true;
63  
64          public boolean hasNext() {
65              return nextCandidate() != null;
66          }
67  
68          private int[] nextCandidate() {
69  
70              if (first) {
71                  return tilesCorner.clone();
72              }
73  
74              int[] candidate = tileIndices.clone();
75              boolean found = false;
76              for (int i = 0; i < dim; i += 1) {
77                  int len = tileSize[i];
78                  int lastIndex = tileIndices[i] + 1;
79                  if (lastIndex < (tilesCorner[i] + tilesCount[i])) {
80                      candidate[i] = lastIndex;
81                      found = true;
82                      break;
83                  } else {
84                      candidate[i] = tilesCorner[i];
85                  }
86              }
87  
88              if (found) {
89                  return candidate;
90              } else {
91                  return null;
92              }
93          }
94  
95          public TileDescriptor next() {
96              int[] cand = nextCandidate();
97              if (cand == null) {
98                  return null;
99              }
100             tileIndices = cand.clone();
101             first = false;
102 
103             int[] corner = new int[dim];
104             int[] size = new int[dim];
105             for (int i = 0; i < dim; i += 1) {
106                 int offset = cand[i] * tileSize[i];
107                 int len = Math.min(imageSize[i] - offset, tileSize[i]);
108                 corner[i] = offset;
109                 size[i] = len;
110             }
111             TileDescriptor t = new TileDescriptor();
112             t.corner = corner;
113             t.size = size;
114             boolean notFirst = false;
115             t.count = 0;
116             // Compute the index of the tile. Note that we
117             // are using FITS indexing, so that first element
118             // changes fastest.
119             for (int i = dim - 1; i >= 0; i -= 1) {
120                 if (notFirst) {
121                     t.count *= nTiles[i];
122                 }
123                 t.count += tileIndices[i];
124                 notFirst = true;
125             }
126             return t;
127         }
128 
129         public void remove() {
130             throw new UnsupportedOperationException("Can't delete tile descriptors.");
131         }
132     }
133 
134     /**
135      * Loop over tiles.
136      * 
137      * @param imageSize
138      *            Dimensions of the image.
139      * @param tileSize
140      *            Dimensions of a single tile. The last tile in a given
141      *            dimension may be smaller.
142      */
143     public TileLooper(int[] imageSize, int[] tileSize) {
144         this(imageSize, tileSize, null, null);
145     }
146 
147     /**
148      * Loop over tiles.
149      * 
150      * @param imageSize
151      *            Dimensions of the image.
152      * @param tileSize
153      *            Dimensions of a single tile. The last tile in each dimension
154      *            may be truncated.
155      * @param tilesCorner
156      *            The indices (with tile space) of the first tile we want. If
157      *            null then it will be set to [0,0,...]
158      * @param tilesCount
159      *            The number of tiles we want in each dimension. If null then it
160      *            will be set to [nTx,nTy, ...] where nTx is the total number of
161      *            tiles available in that dimension.
162      */
163     public TileLooper(int[] imageSize, int[] tileSize, int[] tilesCorner, int[] tilesCount) {
164 
165         this.imageSize = imageSize.clone();
166         this.tileSize = tileSize.clone();
167         this.nTiles = new int[imageSize.length];
168 
169         if (imageSize == null || tileSize == null) {
170             throw new IllegalArgumentException("Invalid null argument");
171         }
172 
173         if (imageSize.length != tileSize.length) {
174             throw new IllegalArgumentException("Image and tiles must have same dimensionality");
175         }
176 
177         dim = imageSize.length;
178         for (int i = 0; i < dim; i += 1) {
179             if (imageSize[i] <= 0 || tileSize[i] <= 0) {
180                 throw new IllegalArgumentException("Negative or 0 dimension specified");
181             }
182             nTiles[i] = (imageSize[i] + tileSize[i] - 1) / tileSize[i];
183         }
184         // This initializes tileIndices to 0.
185         if (tilesCorner == null) {
186             tilesCorner = new int[tileSize.length];
187         } else {
188             for (int i = 0; i < tilesCorner.length; i += 1) {
189                 if (tilesCorner[i] >= nTiles[i]) {
190                     throw new IllegalArgumentException("Tile corner outside tile array");
191                 }
192             }
193         }
194         if (tilesCount == null) {
195             tilesCount = nTiles.clone();
196         } else {
197             for (int i = 0; i < tilesCount.length; i += 1) {
198                 if (tilesCorner[i] + tilesCount[i] > nTiles[i]) {
199                     throw new IllegalArgumentException("Tile range extends outside tile array");
200                 }
201             }
202         }
203         this.tilesCorner = tilesCorner.clone();
204         this.tilesCount = tilesCount.clone();
205 
206         // The first tile is at the specified corner (which is 0,0... if
207         // the user didn't specify it.
208 
209         this.tileIndices = tilesCorner.clone();
210     }
211 
212     public Iterator<TileDescriptor> iterator() {
213         return new TileIterator();
214     }
215 
216     public static void main(String[] args) {
217         if (args.length == 0) {
218             args = new String[]{
219                 "512",
220                 "600",
221                 "100",
222                 "50"
223             };
224         }
225         int nx = Integer.parseInt(args[0]);
226         int ny = Integer.parseInt(args[1]);
227         int tx = Integer.parseInt(args[2]);
228         int ty = Integer.parseInt(args[3]);
229         int[] img = new int[]{
230             nx,
231             ny
232         };
233         int[] tile = new int[]{
234             tx,
235             ty
236         };
237         TileLooper tl = new TileLooper(img, tile);
238         for (TileDescriptor td : tl) {
239             System.err.println("Corner:" + td.corner[0] + "," + td.corner[1]);
240             System.err.println("  Size:" + td.size[0] + "," + td.size[1]);
241         }
242     }
