117     private static void setNullEntries(Object column, Object defaultValue) {
118         if (column != null) {
119             for (int index = 0; index < Array.getLength(column); index++) {
120                 if (Array.get(column, index) == null) {
121                     Array.set(column, index, defaultValue);
122                 }
123             }
124         }
125     }

143     @Override
144     public synchronized ICompressOption compressOptions() {
145         if (compressorControl == null) {
146             getCompressorControl();
147             compressOptions = this.compressorControl.option();
148             if (this.quantAlgorithm != null) {
149                 this.compressOptions.getCompressionParameters()
150                         .getValuesFromHeader(new HeaderCardAccess(ZQUANTIZ, this.quantAlgorithm));
151             }
152             compressOptions.getCompressionParameters().initializeColumns(getNumberOfTileOperations());
153         }
154         return this.compressOptions;
155     }

157     public Buffer decompress() {
158         Buffer decompressedWholeArea = getBaseType().newBuffer(getBufferSize());
159         for (TileCompressionOperation tileOperation : getTileOperations()) {
160             tileOperation.setWholeImageBuffer(decompressedWholeArea);
161         }
162         processAllTiles();
163         decompressedWholeArea.rewind();
164         return decompressedWholeArea;
165     }

167     public void forceNoLoss(int x, int y, int width, int heigth) {
168         TileArea tileArea = new TileArea().start(x, y).end(x + width, y + heigth);
169         for (TileCompressionOperation operation : getTileOperations()) {
170             if (operation.getArea().intersects(tileArea)) {
171                 operation.forceNoLoss(true);
172             }
173         }
174     }

181     @Override
182     public synchronized ICompressorControl getCompressorControl() {
183         if (this.compressorControl == null) {
184             this.compressorControl = CompressorProvider.findCompressorControl(this.quantAlgorithm, compressAlgorithm,
185                     getBaseType().primitiveClass());
186             if (this.compressorControl == null) {
187                 throw new IllegalStateException(
188                         "Found no compressor control for compression algorithm:" + this.compressAlgorithm + //
189                                 " (quantize algorithm = " + this.quantAlgorithm + ", base type = "
190                                 + getBaseType().primitiveClass() + ")");
191             }
192         }
193         return this.compressorControl;
194     }

205     public TiledImageCompressionOperation prepareUncompressedData(final Buffer buffer) throws FitsException {
206         this.compressedWholeArea = ByteBuffer.wrap(new byte[getBaseType().size() * getBufferSize()]);
207         createTiles(new TileCompressorInitialisation(this, buffer));

209         return this;

212     /**
213      * preserve null values, where the value representing null is specified as a parameter. This parameter is ignored
214      * for floating point values where NaN is used as null value.
215      *
216      * @param nullValue the value representing null for byte/short and integer pixel values
217      * @param compressionAlgorithm compression algorithm to use for the null pixel mask
218      * 
219      * @return the created null pixel mask
220      */
221     public ImageNullPixelMask preserveNulls(long nullValue, String compressionAlgorithm) {
222         this.imageNullPixelMask = new ImageNullPixelMask(getTileOperations().length, nullValue, compressionAlgorithm);
223         for (TileCompressionOperation tileOperation : getTileOperations()) {
224             tileOperation.createImageNullPixelMask(getImageNullPixelMask());
225         }
226         return this.imageNullPixelMask;
227     }

229     private synchronized void setQuantAlgorithm(final Header header) {
230         setQuantAlgorithm(header.findCard(ZQUANTIZ));
231 
232         if (quantAlgorithm != null) {
233             return;
234         }
235 
236         // AK: If no ZQUANTIZ keyword, but has ZSCALE and ZZERO columns, then use NO_DITHER quantiz...
237         boolean hasScale = false;
238         boolean hasZero = false;
239 
240         int nFields = header.getIntValue(TFIELDS);
241 
242         for (int i = 1; i <= nFields; i++) {
243             String type = header.getStringValue(TTYPEn.n(i));
244 
245             if (ZSCALE_COLUMN.equals(type)) {
246                 hasScale = true;
247             } else if (ZZERO_COLUMN.equals(type)) {
248                 hasZero = true;
249             } else {
250                 continue;
251             }
252 
253             if (hasScale && hasZero) {
254                 setQuantAlgorithm(HeaderCard.create(ZQUANTIZ, Compression.ZQUANTIZ_NO_DITHER));
255                 break;
256             }
257         }
258     }

260     public TiledImageCompressionOperation read(final Header header) throws FitsException {
261         readPrimaryHeaders(header);
262         setCompressAlgorithm(header.findCard(ZCMPTYPE));
263         setQuantAlgorithm(header);
264 
265         createTiles(new TileDecompressorInitialisation(this, //
266                 getNullableColumn(header, Object[].class, UNCOMPRESSED_DATA_COLUMN), //
267                 getNullableColumn(header, Object[].class, COMPRESSED_DATA_COLUMN), //
268                 getNullableColumn(header, Object[].class, GZIP_COMPRESSED_DATA_COLUMN), //
269                 new HeaderAccess(header)));
270         byte[][] nullPixels = getNullableColumn(header, byte[][].class, NULL_PIXEL_MASK_COLUMN);
271         if (nullPixels != null) {
272             preserveNulls(0L, header.getStringValue(ZMASKCMP)).setColumn(nullPixels);
273         }
274         readCompressionHeaders(header);
275         return this;
276     }

278     public void readPrimaryHeaders(Header header) throws FitsException {
279         readBaseType(header);

281         readTileAxis(header);

284     /**
285      * Sets the compression algorithm, via a <code>ZCMPTYPE</code> header card or equivalent. The card must contain one
286      * of the values recognized by the FITS standard. If not, <code>null</code> will be set instead.
287      * 
288      * @param compressAlgorithmCard The header card that specifies the compression algorithm, with one of the standard
289      *            recognized values such as {@link Compression#ZCMPTYPE_GZIP_1}, or <code>null</code>.
290      * 
291      * @return itself
292      * 
293      * @see #getCompressAlgorithm()
294      * @see #setQuantAlgorithm(HeaderCard)
295      */
296     public TiledImageCompressionOperation setCompressAlgorithm(HeaderCard compressAlgorithmCard) {
297         this.compressAlgorithm = null;
298 
299         if (compressAlgorithmCard == null) {
300             return this;
301         }
302 
303         String algo = compressAlgorithmCard.getValue().toUpperCase(Locale.US);
304         if (algo.equals(Compression.ZCMPTYPE_GZIP_1) || algo.equals(Compression.ZCMPTYPE_GZIP_2)
305                 || algo.equals(Compression.ZCMPTYPE_RICE_1) || algo.equals(Compression.ZCMPTYPE_PLIO_1)
306                 || algo.equals(Compression.ZCMPTYPE_HCOMPRESS_1) || algo.equals(Compression.ZCMPTYPE_NOCOMPRESS)) {
307             this.compressAlgorithm = algo;
308         } else {
309             Logger.getLogger(Header.class.getName()).warning("Ignored invalid ZCMPTYPE value: " + algo);
310         }
311 
312         return this;
313     }

315     /**
316      * Sets the quantization algorithm, via a <code>ZQUANTIZ</code> header card or equivalent. The card must contain one
317      * of the values recognized by the FITS standard. If not, <code>null</code> will be set instead.
318      * 
319      * @param quantAlgorithmCard The header card that specifies the compression algorithm, with one of the standard
320      *            recognized values such as {@link Compression#ZQUANTIZ_NO_DITHER}, or <code>null</code>.
321      * 
322      * @return itself
323      * 
324      * @see #getQuantAlgorithm()
325      * @see #setCompressAlgorithm(HeaderCard)
326      */
327     public synchronized TiledImageCompressionOperation setQuantAlgorithm(HeaderCard quantAlgorithmCard) {
328         this.quantAlgorithm = null;
329 
330         if (quantAlgorithmCard == null) {
331             return this;
332         }
333 
334         String algo = quantAlgorithmCard.getValue().toUpperCase();
335 
336         if (algo.equals(Compression.ZQUANTIZ_NO_DITHER) || algo.equals(Compression.ZQUANTIZ_SUBTRACTIVE_DITHER_1)
337                 || algo.equals(Compression.ZQUANTIZ_SUBTRACTIVE_DITHER_2)) {
338             this.quantAlgorithm = algo;
339         } else {
340             Logger.getLogger(Header.class.getName()).warning("Ignored invalid ZQUANTIZ value: " + algo);
341         }
342 
343         return this;
344     }

378     private <T> T getNullableColumn(Header header, Class<T> class1, String columnName) throws FitsException {
379         for (int i = 1; i <= this.binaryTable.getNCols(); i++) {
380             String val = header.getStringValue(TTYPEn.n(i));
381             if (val != null && val.trim().equals(columnName)) {
382                 return class1.cast(this.binaryTable.getColumn(i - 1));
383             }
384         }
385         return null;
386     }

388     private void processAllTiles() {
389         compressOptions();
390         ExecutorService threadPool = FitsFactory.threadPool();
391         for (TileCompressionOperation tileOperation : getTileOperations()) {
392             tileOperation.execute(threadPool);
393         }
394         for (TileCompressionOperation tileOperation : getTileOperations()) {
395             tileOperation.waitForResult();
396         }
397     }

399     private void readAxis(Header header) throws FitsException {
400         if (hasAxes()) {
401             return;
402         }
403         int naxis = header.getIntValue(ZNAXIS);
404         int[] axes = new int[naxis];
405         for (int i = 1; i <= naxis; i++) {
406             int axisValue = header.getIntValue(ZNAXISn.n(i), -1);
407             if (axisValue == -1) {
408                 throw new FitsException("Required ZNAXISn not found");
409             }
410             axes[naxis - i] = axisValue;
411         }
412         setAxes(axes);
413     }
414 
415     private void readBaseType(Header header) {
416         if (getBaseType() == null) {
417             int zBitPix = header.getIntValue(ZBITPIX);
418             ElementType<Buffer> elementType = ElementType.forNearestBitpix(zBitPix);
419             if (elementType == ElementType.UNKNOWN) {
420                 throw new IllegalArgumentException("illegal value for ZBITPIX " + zBitPix);
421             }
422             setBaseType(elementType);
423         }
424     }

426     private void readCompressionHeaders(Header header) {
427         compressOptions().getCompressionParameters().getValuesFromHeader(new HeaderAccess(header));
428     }

430     private void readTileAxis(Header header) throws FitsException {
431         if (hasTileAxes()) {
432             return;
433         }
434 
435         int naxes = getNAxes();
436         int[] tileAxes = new int[naxes];
437         // TODO
438         // The FITS default is to tile by row (Pence, W., et al. 2000, ASPC, 216, 551)
439         // However, this library defaulted to full image size by default...
440         for (int i = 1; i <= naxes; i++) {
441             tileAxes[naxes - i] = header.getIntValue(ZTILEn.n(i), i == 1 ? header.getIntValue(ZNAXISn.n(1)) : 1);
442         }
443 
444         setTileAxes(tileAxes);
445     }

447     private <T> Object setInColumn(Object column, boolean predicate, TileCompressionOperation tileOperation,
448             Class<T> clazz, T value) {
449         if (predicate) {
450             if (column == null) {
451                 column = Array.newInstance(clazz, getNumberOfTileOperations());
452             }
453             Array.set(column, tileOperation.getTileIndex(), value);
454         }
455         return column;
456     }

458     private synchronized void writeColumns(BinaryTableHDU hdu) throws FitsException {
459         Object compressedColumn = null;
460         Object uncompressedColumn = null;
461         Object gzipColumn = null;
462         for (TileCompressionOperation tileOperation : getTileOperations()) {
463             TileCompressionType compression = tileOperation.getCompressionType();
464             byte[] compressedData = tileOperation.getCompressedData();
465 
466             compressedColumn = setInColumn(compressedColumn, compression == COMPRESSED, tileOperation, byte[].class,
467                     compressedData);
468             gzipColumn = setInColumn(gzipColumn, compression == GZIP_COMPRESSED, tileOperation, byte[].class,
469                     compressedData);
470             uncompressedColumn = setInColumn(uncompressedColumn, compression == UNCOMPRESSED, tileOperation,
471                     byte[].class, compressedData);
472         }
473         setNullEntries(compressedColumn, new byte[0]);
474         setNullEntries(gzipColumn, new byte[0]);
475         setNullEntries(uncompressedColumn, new byte[0]);
476         addColumnToTable(hdu, compressedColumn, COMPRESSED_DATA_COLUMN);
477         addColumnToTable(hdu, gzipColumn, GZIP_COMPRESSED_DATA_COLUMN);
478         addColumnToTable(hdu, uncompressedColumn, UNCOMPRESSED_DATA_COLUMN);
479         if (this.imageNullPixelMask != null) {
480             addColumnToTable(hdu, this.imageNullPixelMask.getColumn(), NULL_PIXEL_MASK_COLUMN);
481         }
482         this.compressOptions.getCompressionParameters().addColumnsToTable(hdu);
483         hdu.getData().fillHeader(hdu.getHeader());
484     }
485 
486     private void writeHeader(Header header) throws FitsException {
487         HeaderCardBuilder cardBuilder = header//
488                 .card(ZBITPIX).value(getBaseType().bitPix())//
489                 .card(ZCMPTYPE).value(this.compressAlgorithm);
490         int[] tileAxes = getTileAxes();
491         int naxes = tileAxes.length;
492         for (int i = 1; i <= naxes; i++) {
493             cardBuilder.card(ZTILEn.n(i)).value(tileAxes[naxes - i]);
494         }
495         compressOptions().getCompressionParameters().setValuesInHeader(new HeaderAccess(header));
496         if (this.imageNullPixelMask != null) {
497             cardBuilder.card(ZMASKCMP).value(this.imageNullPixelMask.getCompressAlgorithm());
498         }
499     }
