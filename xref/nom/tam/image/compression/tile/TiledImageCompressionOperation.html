74  /**
75   * This class represents a complete tiledImageOperation of tileOperations
76   * describing an image ordered from left to right and top down. the
77   * tileOperations all have the same geometry only the tileOperations at the
78   * right side and the bottom side can have different sizes.
79   */

112     private static void setNullEntries(Object column, Object defaultValue) {
113         if (column != null) {
114             for (int index = 0; index < Array.getLength(column); index++) {
115                 if (Array.get(column, index) == null) {
116                     Array.set(column, index, defaultValue);
117                 }
118             }
119         }

145     public Buffer decompress() {
146         Buffer decompressedWholeArea = getBaseType().newBuffer(getBufferSize());
147         for (TileCompressionOperation tileOperation : getTileOperations()) {
148             tileOperation.setWholeImageBuffer(decompressedWholeArea);
149         }
150         processAllTiles();
151         decompressedWholeArea.rewind();
152         return decompressedWholeArea;
153     }

155     public void forceNoLoss(int x, int y, int width, int heigth) {
156         TileArea tileArea = new TileArea().start(x, y).end(x + width, y + heigth);
157         for (TileCompressionOperation operation : getTileOperations()) {
158             if (operation.getArea().intersects(tileArea)) {
159                 operation.forceNoLoss(true);
160             }
161         }
162     }

183     public TiledImageCompressionOperation prepareUncompressedData(final Buffer buffer) throws FitsException {
184         this.compressedWholeArea = ByteBuffer.wrap(new byte[getBaseType().size() * getBufferSize()]);
185         createTiles(new TileCompressorInitialisation(this, buffer));
186         this.compressedWholeArea.rewind();
187         return this;

190     /**
191      * preserve null values, where the value representing null is specified as a
192      * parameter. This parameter is ignored for floating point values where NaN
193      * is used as null value.
194      *
195      * @param nullValue
196      *            the value representing null for byte/short and integer pixel
197      *            values
198      * @param compressionAlgorithm
199      *            compression algorithm to use for the null pixel mask
200      * @return the created null pixel mask
201      */
202     public ImageNullPixelMask preserveNulls(long nullValue, String compressionAlgorithm) {
203         this.imageNullPixelMask = new ImageNullPixelMask(getTileOperations().length, nullValue, compressionAlgorithm);
204         for (TileCompressionOperation tileOperation : getTileOperations()) {
205             tileOperation.createImageNullPixelMask(getImageNullPixelMask());
206         }
207         return this.imageNullPixelMask;
208     }

210     public TiledImageCompressionOperation read(final Header header) throws FitsException {
211         readPrimaryHeaders(header);
212         setCompressAlgorithm(header.findCard(ZCMPTYPE));
213         setQuantAlgorithm(header.findCard(ZQUANTIZ));
214         createTiles(new TileDecompressorInitialisation(this, //
215                 getNullableColumn(header, Object[].class, UNCOMPRESSED_DATA_COLUMN), //
216                 getNullableColumn(header, Object[].class, COMPRESSED_DATA_COLUMN), //
217                 getNullableColumn(header, Object[].class, GZIP_COMPRESSED_DATA_COLUMN), //
218                 new HeaderAccess(header)));
219         byte[][] nullPixels = getNullableColumn(header, byte[][].class, NULL_PIXEL_MASK_COLUMN);
220         if (nullPixels != null) {
221             preserveNulls(0L, header.getStringValue(ZMASKCMP)).setColumn(nullPixels);
222         }
223         readCompressionHeaders(header);
224         return this;
225     }

247     private <T> T getNullableColumn(Header header, Class<T> class1, String columnName) throws FitsException {
248         for (int i = 1; i <= this.binaryTable.getNCols(); i++) {
249             String val = header.getStringValue(TTYPEn.n(i));
250             if (val != null && val.trim().equals(columnName)) {
251                 return class1.cast(this.binaryTable.getColumn(i - 1));
252             }
253         }
254         return null;
255     }

257     private void initializeCompressionControl() {
258         if (this.compressorControl == null) {
259             this.compressorControl = CompressorProvider.findCompressorControl(this.quantAlgorithm, this.compressAlgorithm, getBaseType().primitiveClass());
260             if (this.compressorControl == null) {
261                 throw new IllegalStateException("Found no compressor control for compression algorithm:" + this.compressAlgorithm + //
262                         " (quantize algorithm = " + this.quantAlgorithm + ", base type = " + getBaseType().primitiveClass() + ")");
263             }
264             initImageOptions();
265         }
266     }
267 
268     private void initImageOptions() {
269         this.imageOptions = this.compressorControl.option();
270         initializeQuantAlgorithm();
271         this.imageOptions.getCompressionParameters().initializeColumns(getNumberOfTileOperations());
272     }

274     private void processAllTiles() {
275         ExecutorService threadPool = FitsFactory.threadPool();
276         for (TileCompressionOperation tileOperation : getTileOperations()) {
277             tileOperation.execute(threadPool);
278         }
279         for (TileCompressionOperation tileOperation : getTileOperations()) {
280             tileOperation.waitForResult();
281         }
282     }

284     private void readAxis(Header header) throws FitsException {
285         if (areAxesUndefined()) {
286             int naxis = header.getIntValue(ZNAXIS);
287             int[] axes = new int[naxis];
288             for (int i = 1; i <= naxis; i++) {
289                 int axisValue = header.getIntValue(ZNAXISn.n(i), -1);
290                 axes[i - 1] = axisValue;
291                 if (axes[i - 1] == -1) {
292                     throw new FitsException("Required ZNAXISn not found");
293                 }
294             }
295             setAxes(axes);
296         }
297     }
298 
299     private void readBaseType(Header header) {
300         if (getBaseType() == null) {
301             int zBitPix = header.getIntValue(ZBITPIX);
302             ElementType<Buffer> elementType = ElementType.forNearestBitpix(zBitPix);
303             if (elementType == ElementType.UNKNOWN) {
304                 throw new IllegalArgumentException("illegal value for ZBITPIX " + zBitPix);
305             }
306             setBaseType(elementType);
307         }
308     }
309 
310     private void readCompressionHeaders(Header header) {
311         compressOptions().getCompressionParameters().getValuesFromHeader(new HeaderAccess(header));
312     }
313 
314     private void readTileAxis(Header header) throws FitsException {
315         if (areTileAxesUndefined()) {
316             int[] tileAxes = new int[getNAxes()];
317             Arrays.fill(tileAxes, 1);
318             tileAxes[0] = -1;
319             for (int i = 1; i <= tileAxes.length; i++) {
320                 HeaderCard card = header.findCard(ZTILEn.n(i));
321                 if (card != null) {
322                     tileAxes[i - 1] = card.getValue(Integer.class, -1);
323                 }
324             }
325             setTileAxes(tileAxes);
326         }
327     }

329     private <T> Object setInColumn(Object column, boolean predicate, TileCompressionOperation tileOperation, Class<T> clazz, T value) {
330         if (predicate) {
331             if (column == null) {
332                 column = Array.newInstance(clazz, getNumberOfTileOperations());
333             }
334             Array.set(column, tileOperation.getTileIndex(), value);
335         }
336         return column;
337     }

339     private void writeColumns(BinaryTableHDU hdu) throws FitsException {
340         Object compressedColumn = null;
341         Object uncompressedColumn = null;
342         Object gzipColumn = null;
343         for (TileCompressionOperation tileOperation : getTileOperations()) {
344             TileCompressionType compression = tileOperation.getCompressionType();
345             byte[] compressedData = tileOperation.getCompressedData();
346 
347             compressedColumn = setInColumn(compressedColumn, compression == COMPRESSED, tileOperation, byte[].class, compressedData);
348             gzipColumn = setInColumn(gzipColumn, compression == GZIP_COMPRESSED, tileOperation, byte[].class, compressedData);
349             uncompressedColumn = setInColumn(uncompressedColumn, compression == UNCOMPRESSED, tileOperation, byte[].class, compressedData);
350         }
351         setNullEntries(compressedColumn, new byte[0]);
352         setNullEntries(gzipColumn, new byte[0]);
353         setNullEntries(uncompressedColumn, new byte[0]);
354         addColumnToTable(hdu, compressedColumn, COMPRESSED_DATA_COLUMN);
355         addColumnToTable(hdu, gzipColumn, GZIP_COMPRESSED_DATA_COLUMN);
356         addColumnToTable(hdu, uncompressedColumn, UNCOMPRESSED_DATA_COLUMN);
357         if (this.imageNullPixelMask != null) {
358             addColumnToTable(hdu, this.imageNullPixelMask.getColumn(), NULL_PIXEL_MASK_COLUMN);
359         }
360         this.imageOptions.getCompressionParameters().addColumnsToTable(hdu);
361         hdu.getData().fillHeader(hdu.getHeader());
362     }
363 
364     private void writeHeader(Header header) throws FitsException {
365         HeaderCardBuilder cardBuilder = header//
366                 .card(ZBITPIX).value(getBaseType().bitPix())//
367                 .card(ZCMPTYPE).value(this.compressAlgorithm);
368         int[] tileAxes = getTileAxes();
369         for (int i = 1; i <= tileAxes.length; i++) {
370             cardBuilder.card(ZTILEn.n(i)).value(tileAxes[i - 1]);
371         }
372         compressOptions().getCompressionParameters().setValuesInHeader(new HeaderAccess(header));
373         if (this.imageNullPixelMask != null) {
374             cardBuilder.card(ZMASKCMP).value(this.imageNullPixelMask.getCompressAlgorithm());
375         }
376     }
