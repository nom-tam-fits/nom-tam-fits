34  import java.lang.reflect.Array;
35  import java.nio.ByteBuffer;
36  
37  import nom.tam.fits.compression.algorithm.api.ICompressOption;
38  import nom.tam.fits.compression.algorithm.api.ICompressorControl;
39  import nom.tam.image.tile.operation.AbstractTileOperation;
40  import nom.tam.image.tile.operation.TileArea;
41  import nom.tam.util.type.PrimitiveType;
42  import nom.tam.util.type.PrimitiveTypeHandler;
43  
44  /**
45   * abstract information holder about the a tile that represents a rectangular
46   * part of the image. Will be sub classed for compression and decompression
47   * variants.
48   */
49  abstract class TileCompressionOperation extends AbstractTileOperation<TileCompressionOperation, TiledImageCompressionOperation> {
50  
51      protected ByteBuffer compressedData;
52  
53      protected int compressedOffset;
54  
55      protected TileCompressionType compressionType;
56  
57      protected ICompressOption tileOptions;
58  
59      protected TileCompressionOperation(TiledImageCompressionOperation operation, int tileIndex, TileArea area) {
60          super(operation, tileIndex, area);
61      }
62  
63      @Override
64      public String toString() {
65          return getClass().getSimpleName() + "(" + getTileIndex() + "," + this.compressionType + "," + this.compressedOffset + ")";
66      }
67  
68      private ByteBuffer convertToBuffer(Object data) {
69          return PrimitiveTypeHandler.valueOf(data.getClass().getComponentType()).convertToByteBuffer(data);
70      }
71  
72      /**
73       * should the data of this tile be forced to case no data loss. This
74       * information is not relevant in all cases that it is ignored by default.
75       *
76       * @param value
77       *            the value to set.
78       */
79      protected void forceNoLoss(boolean value) {
80      }
81  
82      protected byte[] getCompressedData() {
83          byte[] data = new byte[this.compressedData.limit()];
84          this.compressedData.rewind();
85          PrimitiveType.BYTE.getArray(this.compressedData, data);
86          return data;
87      }
88  
89      protected ByteBuffer getCompressedWholeArea() {
90          return getTiledImageOperation().getCompressedWholeArea();
91      }
92  
93      protected TileCompressionType getCompressionType() {
94          return this.compressionType;
95      }
96  
97      protected ICompressorControl getCompressorControl() {
98          return getTiledImageOperation().getCompressorControl();
99      }
100 
101     protected ICompressorControl getGzipCompressorControl() {
102         return getTiledImageOperation().getGzipCompressorControl();
103     }
104 
105     protected ICompressOption getTileOptions() {
106         return this.tileOptions;
107     }
108 
109     protected TileCompressionOperation initTileOptions() {
110         ICompressOption compressOptions = getTiledImageOperation().compressOptions();
111         this.tileOptions = compressOptions.copy() //
112                 .setTileWidth(getTileBuffer().getWidth()) //
113                 .setTileHeight(getTileBuffer().getHeight());
114         return this;
115     }
116 
117     protected TileCompressionOperation setCompressed(Object data, TileCompressionType type) {
118         if (data != null && Array.getLength(data) > 0) {
119             this.compressionType = type;
120             this.compressedData = convertToBuffer(data);
121             this.compressedOffset = 0;
122         }
123         return this;
124     }
125 
126     protected TileCompressionOperation setCompressedOffset(int value) {
127         this.compressedOffset = value;
128         return this;
129     }
130 
131     @Override
132     protected TileCompressionOperation setDimensions(int dataOffset, int width, int height) {
133         super.setDimensions(dataOffset, width, height);
134         return this;
135     }
136 
137     /**
138      * set the buffer that describes the whole compressed image and let the tile
139      * create a slice of it from the position where the tile starts in the whole
140      * image. Attention this method is not thread-safe because it changes the
141      * position of the buffer parameter. This buffer is just as big as the image
142      * buffer but will be reduced to the needed size as a last step of the
143      * Compression.
144      *
145      * @param compressed
146      *            the buffer that describes the whole image.
147      */
148     protected void setWholeImageCompressedBuffer(ByteBuffer compressed) {
149         compressed.position(this.compressedOffset * getBaseType().size());
150         this.compressedData = compressed.slice();
151         this.compressedOffset = 0;
152         // we do not limit this buffer but is expected not to write more than
153         // the uncompressed size.
154     }
155 }
