34  import java.nio.ByteBuffer;
35  import java.nio.DoubleBuffer;
36  import java.nio.FloatBuffer;
37  import java.nio.IntBuffer;
38  import java.nio.LongBuffer;
39  import java.nio.ShortBuffer;
40  
41  import nom.tam.fits.compression.algorithm.api.ICompressorControl;
42  import nom.tam.image.tile.operation.buffer.TileBuffer;
43  import nom.tam.util.type.ElementType;
44  
45  /**
46   * This class overwrites the pixels specified in the mask with null values.
47   * Where the null value can be defined separately.
48   */
49  public class NullPixelMaskPreserver extends AbstractNullPixelMask {
50  
51      public NullPixelMaskPreserver(TileBuffer tileBuffer, int tileIndex, long nullValue, ICompressorControl compressorControl) {
52          super(tileBuffer, tileIndex, nullValue, compressorControl);
53      }
54  
55      public void preserveNull() {
56          if (getTileBuffer().getBaseType().is(ElementType.DOUBLE)) {
57              preserveNullDoubles();
58          } else if (getTileBuffer().getBaseType().is(ElementType.FLOAT)) {
59              preserveNullFloats();
60          } else if (getTileBuffer().getBaseType().is(ElementType.LONG)) {
61              preserveNullLongs();
62          } else if (getTileBuffer().getBaseType().is(ElementType.INT)) {
63              preserveNullInts();
64          } else if (getTileBuffer().getBaseType().is(ElementType.SHORT)) {
65              preserveNullShorts();
66          } else if (getTileBuffer().getBaseType().is(ElementType.BYTE)) {
67              preserveNullBytes();
68          }
69          if (getMask() != null) {
70              ByteBuffer compressed = ByteBuffer.allocate(getTileBuffer().getPixelSize());
71              if (!getCompressorControl().compress(getMask(), compressed, getCompressorControl().option())) {
72                  throw new IllegalStateException("could not compress the null pixel mask");
73              }
74              setMask(compressed);
75          }
76      }
77  
78      private void preserveNullBytes() {
79          ByteBuffer buffer = (ByteBuffer) getTileBuffer().getBuffer();
80          byte nullValue = (byte) getNullValue();
81          int size = buffer.remaining();
82          for (int index = 0; index < size; index++) {
83              if (nullValue == buffer.get(index)) {
84                  initializedMask(size).put(index, NULL_INDICATOR);
85              }
86          }
87      }
88  
89      private void preserveNullDoubles() {
90          DoubleBuffer buffer = (DoubleBuffer) getTileBuffer().getBuffer();
91          int size = getTileBuffer().getPixelSize();
92          for (int index = 0; index < size; index++) {
93              if (Double.isNaN(buffer.get(index))) {
94                  initializedMask(size).put(index, NULL_INDICATOR);
95              }
96          }
97      }
98  
99      private void preserveNullFloats() {
100         FloatBuffer buffer = (FloatBuffer) getTileBuffer().getBuffer();
101         int size = getTileBuffer().getPixelSize();
102         for (int index = 0; index < size; index++) {
103             if (Float.isNaN(buffer.get(index))) {
104                 initializedMask(size).put(index, NULL_INDICATOR);
105             }
106         }
107     }
108 
109     private void preserveNullInts() {
110         IntBuffer buffer = (IntBuffer) getTileBuffer().getBuffer();
111         int nullValue = (int) getNullValue();
112         int size = getTileBuffer().getPixelSize();
113         for (int index = 0; index < size; index++) {
114             if (nullValue == buffer.get(index)) {
115                 initializedMask(size).put(index, NULL_INDICATOR);
116             }
117         }
118     }
119 
120     private void preserveNullLongs() {
121         LongBuffer buffer = (LongBuffer) getTileBuffer().getBuffer();
122         long nullValue = getNullValue();
123         int size = getTileBuffer().getPixelSize();
124         for (int index = 0; index < size; index++) {
125             if (nullValue == buffer.get(index)) {
126                 initializedMask(size).put(index, NULL_INDICATOR);
127             }
128         }
129     }
130 
131     private void preserveNullShorts() {
132         ShortBuffer buffer = (ShortBuffer) getTileBuffer().getBuffer();
133         short nullValue = (short) getNullValue();
134         int size = getTileBuffer().getPixelSize();
135         for (int index = 0; index < size; index++) {
136             if (nullValue == buffer.get(index)) {
137                 initializedMask(size).put(index, NULL_INDICATOR);
138             }
139         }
140     }
