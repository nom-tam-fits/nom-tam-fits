176     private final IFitsHeader compressedHeaderKey;
177 
178     private final IFitsHeader uncompressedHeaderKey;
179 
180     public static void backup(HeaderCard card, Cursor<String, HeaderCard> headerIterator) throws HeaderCardException {
181         BackupRestoreUnCompressedHeaderCard mapping = selectMapping(CompressedImageHDU.UNCOMPRESSED_HEADER_MAPPING, card);
182         mapping.backupCard(card, headerIterator);
183     }
184 
185     public static void restore(HeaderCard card, Cursor<String, HeaderCard> headerIterator) throws HeaderCardException {
186         BackupRestoreUnCompressedHeaderCard mapping = selectMapping(CompressedImageHDU.COMPRESSED_HEADER_MAPPING, card);
187         mapping.restoreCard(card, headerIterator);
188     }
189 
190     protected static BackupRestoreUnCompressedHeaderCard selectMapping(Map<IFitsHeader, BackupRestoreUnCompressedHeaderCard> mappings, HeaderCard card) {
191         IFitsHeader key = GenericKey.lookup(card.getKey());
192         if (key != null) {
193             BackupRestoreUnCompressedHeaderCard mapping = mappings.get(key);
194             if (mapping != null) {
195                 return mapping;
196             }
197         }
198         return MAP_ANY;
199     }
200 
201     BackupRestoreUnCompressedHeaderCard(IFitsHeader header) {
202         this.compressedHeaderKey = header;
203         if (header instanceof Compression) {
204             this.uncompressedHeaderKey = ((Compression) this.compressedHeaderKey).getUncompressedKey();
205 
206         } else {
207             this.uncompressedHeaderKey = null;
208         }
209         CompressedImageHDU.UNCOMPRESSED_HEADER_MAPPING.put(header, this);
210         if (this.uncompressedHeaderKey != null) {
211             CompressedImageHDU.COMPRESSED_HEADER_MAPPING.put(this.uncompressedHeaderKey, this);
212         }
213     }
214 
215     private void addHeaderCard(HeaderCard card, Cursor<String, HeaderCard> headerIterator, IFitsHeader targetKey) throws HeaderCardException {
216         if (targetKey != null) {
217             if (targetKey.valueType() == VALUE.INTEGER) {
218                 headerIterator.add(new HeaderCard(targetKey.key(), card.getValue(Integer.class, 0), card.getComment()));
219             } else if (targetKey.valueType() == VALUE.STRING) {
220                 headerIterator.add(new HeaderCard(targetKey.key(), card.getValue(), card.getComment()));
221             } else if (targetKey.valueType() == VALUE.LOGICAL) {
222                 headerIterator.add(new HeaderCard(targetKey.key(), card.getValue(Boolean.class, false), card.getComment()));
223             }
224         }
225     }

227     /**
228      * default behaviour is to ignore the card and by that to exclude it from
229      * the uncompressed header if it does not have a uncompressed equivalent..
230      *
231      * @param card
232      *            the card from the compressed header
233      * @param headerIterator
234      *            the iterator for the uncompressed header.
235      * @throws HeaderCardException
236      *             if the card could not be copied
237      */
238     protected void backupCard(HeaderCard card, Cursor<String, HeaderCard> headerIterator) throws HeaderCardException {
239         IFitsHeader uncompressedKey = this.uncompressedHeaderKey;
240         addHeaderCard(card, headerIterator, uncompressedKey);
241     }
242 
243     protected IFitsHeader compressedHeaderKey() {
244         return this.compressedHeaderKey;
245     }
246 
247     protected void restoreCard(HeaderCard card, Cursor<String, HeaderCard> headerIterator) throws HeaderCardException {
248         addHeaderCard(card, headerIterator, this.compressedHeaderKey);
249     }
250 
251     protected IFitsHeader uncompressedHeaderKey() {
252         return this.uncompressedHeaderKey;
253     }
