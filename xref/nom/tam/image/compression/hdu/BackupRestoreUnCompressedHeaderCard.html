175     private final IFitsHeader compressedHeaderKey;
176 
177     private final IFitsHeader uncompressedHeaderKey;
178 
179     public static void backup(HeaderCard card, Cursor<String, HeaderCard> headerIterator) throws HeaderCardException {
180         BackupRestoreUnCompressedHeaderCard mapping = selectMapping(CompressedImageHDU.UNCOMPRESSED_HEADER_MAPPING, card);
181         mapping.backupCard(card, headerIterator);
182     }
183 
184     public static void restore(HeaderCard card, Cursor<String, HeaderCard> headerIterator) throws HeaderCardException {
185         BackupRestoreUnCompressedHeaderCard mapping = selectMapping(CompressedImageHDU.COMPRESSED_HEADER_MAPPING, card);
186         mapping.restoreCard(card, headerIterator);
187     }
188 
189     protected static BackupRestoreUnCompressedHeaderCard selectMapping(Map<IFitsHeader, BackupRestoreUnCompressedHeaderCard> mappings, HeaderCard card) {
190         IFitsHeader key = GenericKey.lookup(card.getKey());
191         if (key != null) {
192             BackupRestoreUnCompressedHeaderCard mapping = mappings.get(key);
193             if (mapping != null) {
194                 return mapping;
195             }
196         }
197         return MAP_ANY;
198     }
199 
200     BackupRestoreUnCompressedHeaderCard(IFitsHeader header) {
201         this.compressedHeaderKey = header;
202         if (header instanceof Compression) {
203             this.uncompressedHeaderKey = ((Compression) this.compressedHeaderKey).getUncompressedKey();
204 
205         } else {
206             this.uncompressedHeaderKey = null;
207         }
208         CompressedImageHDU.UNCOMPRESSED_HEADER_MAPPING.put(header, this);
209         if (this.uncompressedHeaderKey != null) {
210             CompressedImageHDU.COMPRESSED_HEADER_MAPPING.put(this.uncompressedHeaderKey, this);
211         }
212     }
213 
214     private void addHeaderCard(HeaderCard card, Cursor<String, HeaderCard> headerIterator, IFitsHeader targetKey) throws HeaderCardException {
215         if (targetKey != null) {
216             if (targetKey.valueType() == VALUE.INTEGER) {
217                 headerIterator.add(new HeaderCard(targetKey.key(), card.getValue(Integer.class, 0), card.getComment()));
218             } else if (targetKey.valueType() == VALUE.STRING) {
219                 headerIterator.add(new HeaderCard(targetKey.key(), card.getValue(), card.getComment()));
220             } else if (targetKey.valueType() == VALUE.LOGICAL) {
221                 headerIterator.add(new HeaderCard(targetKey.key(), card.getValue(Boolean.class, false), card.getComment()));
222             }
223         }
224     }

226     /**
227      * default behaviour is to ignore the card and by that to exclude it from
228      * the uncompressed header if it does not have a uncompressed equivalent..
229      *
230      * @param card
231      *            the card from the compressed header
232      * @param headerIterator
233      *            the iterator for the uncompressed header.
234      * @throws HeaderCardException
235      *             if the card could not be copied
236      */
237     protected void backupCard(HeaderCard card, Cursor<String, HeaderCard> headerIterator) throws HeaderCardException {
238         IFitsHeader uncompressedKey = this.uncompressedHeaderKey;
239         addHeaderCard(card, headerIterator, uncompressedKey);
240     }
241 
242     protected IFitsHeader compressedHeaderKey() {
243         return this.compressedHeaderKey;
244     }
245 
246     protected void restoreCard(HeaderCard card, Cursor<String, HeaderCard> headerIterator) throws HeaderCardException {
247         addHeaderCard(card, headerIterator, this.compressedHeaderKey);
248     }
249 
250     protected IFitsHeader uncompressedHeaderKey() {
251         return this.uncompressedHeaderKey;
252     }
