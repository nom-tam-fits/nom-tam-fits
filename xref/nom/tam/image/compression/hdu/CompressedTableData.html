52  public class CompressedTableData extends BinaryTable {
53  
54      private int rowsPerTile;
55  
56      private List<BinaryTableTile> tiles;
57  
58      private String[] columnCompressionAlgorithms;
59  
60      public CompressedTableData() {
61      }
62  
63      public CompressedTableData(Header header) throws FitsException {
64          super(header);
65      }
66  
67      public void compress(Header header) throws FitsException {
68          for (BinaryTableTile binaryTableTile : this.tiles) {
69              binaryTableTile.execute(FitsFactory.threadPool());
70          }
71          for (BinaryTableTile binaryTableTile : this.tiles) {
72              binaryTableTile.waitForResult();
73              binaryTableTile.fillHeader(header);
74          }
75          fillHeader(header);
76      }
77  
78      @Override
79      public void fillHeader(Header h) throws FitsException {
80          super.fillHeader(h);
81          h.setNaxis(2, getData().getNRows());
82          h.addValue(Compression.ZTABLE.key(), true, "this is a compressed table");
83          long ztilelenValue = this.rowsPerTile > 0 ? this.rowsPerTile : h.getIntValue(Standard.NAXIS2);
84          h.addValue(Compression.ZTILELEN.key(), ztilelenValue, "number of rows in each tile");
85      }
86  
87      public void prepareUncompressedData(ColumnTable<SaveState> data) throws FitsException {
88          int nrows = data.getNRows();
89          int ncols = data.getNCols();
90          if (this.rowsPerTile <= 0) {
91              this.rowsPerTile = nrows;
92          }
93          if (this.columnCompressionAlgorithms.length < ncols) {
94              this.columnCompressionAlgorithms = Arrays.copyOfRange(this.columnCompressionAlgorithms, 0, ncols);
95          }
96          this.tiles = new ArrayList<>();
97          for (int column = 0; column < ncols; column++) {
98              addByteVaryingColumn();
99              int tileIndex = 1;
100             for (int rowStart = 0; rowStart < nrows; rowStart += this.rowsPerTile) {
101                 addRow(new byte[ncols][0]);
102                 this.tiles.add(new BinaryTableTileCompressor(this, data,
103                         tile()//
104                                 .rowStart(rowStart)//
105                                 .rowEnd(rowStart + this.rowsPerTile)//
106                                 .column(column)//
107                                 .tileIndex(tileIndex++)//
108                                 .compressionAlgorithm(this.columnCompressionAlgorithms[column])));
109             }
110         }
111     }
112 
113     protected BinaryTable asBinaryTable(BinaryTable dataToFill, Header compressedHeader, Header targetHeader) throws FitsException {
114         int nrows = targetHeader.getIntValue(Standard.NAXIS2);
115         int ncols = compressedHeader.getIntValue(TFIELDS);
116         this.rowsPerTile = compressedHeader.getIntValue(Compression.ZTILELEN, nrows);
117         this.tiles = new ArrayList<>();
118         BinaryTable.createColumnDataFor(dataToFill);
119         for (int column = 0; column < ncols; column++) {
120             int tileIndex = 1;
121             String compressionAlgorithm = compressedHeader.getStringValue(Compression.ZCTYPn.n(column + 1));
122             for (int rowStart = 0; rowStart < nrows; rowStart += this.rowsPerTile) {
123                 BinaryTableTileDecompressor binaryTableTile = new BinaryTableTileDecompressor(this, dataToFill.getData(),
124                         tile()//
125                                 .rowStart(rowStart)//
126                                 .rowEnd(rowStart + this.rowsPerTile)//
127                                 .column(column)//
128                                 .tileIndex(tileIndex++)//
129                                 .compressionAlgorithm(compressionAlgorithm));
130                 this.tiles.add(binaryTableTile);
131                 binaryTableTile.execute(FitsFactory.threadPool());
132             }
133         }
134         for (BinaryTableTile binaryTableTile : this.tiles) {
135             binaryTableTile.waitForResult();
136         }
137         return dataToFill;
138     }
139 
140     protected int getRowsPerTile() {
141         return this.rowsPerTile;
142     }
143 
144     protected void setColumnCompressionAlgorithms(String[] columnCompressionAlgorithms) {
145         this.columnCompressionAlgorithms = columnCompressionAlgorithms;
146     }
147 
148     protected CompressedTableData setRowsPerTile(int value) {
149         this.rowsPerTile = value;
150         return this;
151     }
152 }
