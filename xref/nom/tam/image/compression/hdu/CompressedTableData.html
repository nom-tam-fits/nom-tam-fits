52  /**
53   * FITS representation of a compressed binary table. It itself is a binary table, but one in which each row represents
54   * the compressed image of one or more rows of the original table.
55   * 
56   * @see CompressedTableHDU
57   */
58  @SuppressWarnings("deprecation")
59  public class CompressedTableData extends BinaryTable {
60  
61      private int rowsPerTile;
62  
63      private List<BinaryTableTile> tiles;
64  
65      private String[] columnCompressionAlgorithms;
66  
67      /**
68       * Creates a new empty compressed table data to be initialized at a later point
69       */
70      public CompressedTableData() {
71      }
72  
73      /**
74       * Creates a new compressed table data based on the prescription of the supplied header.
75       * 
76       * @param  header        The header that describes the compressed table
77       * 
78       * @throws FitsException If the header is invalid or could not be accessed.
79       */
80      public CompressedTableData(Header header) throws FitsException {
81          super(header);
82      }
83  
84      /**
85       * (<i>for internal use</i>) This should only be called by {@link CompressedTableHDU}, and should have reduced
86       * visibility accordingly.
87       */
88      @SuppressWarnings("javadoc")
89      public void compress(Header header) throws FitsException {
90          for (BinaryTableTile binaryTableTile : tiles) {
91              binaryTableTile.execute(FitsFactory.threadPool());
92          }
93          for (BinaryTableTile binaryTableTile : tiles) {
94              binaryTableTile.waitForResult();
95              binaryTableTile.fillHeader(header);
96          }
97          fillHeader(header);
98      }
99  
100     @Override
101     public void fillHeader(Header h) throws FitsException {
102         super.fillHeader(h);
103         h.setNaxis(2, getData().getNRows());
104         h.addValue(Compression.ZTABLE.key(), true, "this is a compressed table");
105         long ztilelenValue = rowsPerTile > 0 ? rowsPerTile : h.getIntValue(Standard.NAXIS2);
106         h.addValue(Compression.ZTILELEN.key(), ztilelenValue, "number of rows in each tile");
107     }
108 
109     /**
110      * (<i>for internal use</i>) This should only be called by {@link CompressedTableHDU}, and should have reduced
111      * visibility accordingly.
112      */
113     @SuppressWarnings("javadoc")
114     public void prepareUncompressedData(ColumnTable<SaveState> data) throws FitsException {
115         int nrows = data.getNRows();
116         int ncols = data.getNCols();
117         if (rowsPerTile <= 0) {
118             rowsPerTile = nrows;
119         }
120         if (columnCompressionAlgorithms.length < ncols) {
121             columnCompressionAlgorithms = Arrays.copyOfRange(columnCompressionAlgorithms, 0, ncols);
122         }
123         tiles = new ArrayList<>();
124         for (int column = 0; column < ncols; column++) {
125             addByteVaryingColumn();
126             int tileIndex = 1;
127             for (int rowStart = 0; rowStart < nrows; rowStart += rowsPerTile) {
128                 addRow(new byte[ncols][0]);
129                 tiles.add(new BinaryTableTileCompressor(this, data, tile()//
130                         .rowStart(rowStart)//
131                         .rowEnd(rowStart + rowsPerTile)//
132                         .column(column)//
133                         .tileIndex(tileIndex++)//
134                         .compressionAlgorithm(columnCompressionAlgorithms[column])));
135             }
136         }
137     }
138 
139     /**
140      * This should only be called by {@link CompressedTableHDU}.
141      */
142     @SuppressWarnings("javadoc")
143     protected BinaryTable asBinaryTable(BinaryTable dataToFill, Header compressedHeader, Header targetHeader)
144             throws FitsException {
145         int nrows = targetHeader.getIntValue(Standard.NAXIS2);
146         int ncols = compressedHeader.getIntValue(TFIELDS);
147         rowsPerTile = compressedHeader.getIntValue(Compression.ZTILELEN, nrows);
148         tiles = new ArrayList<>();
149         BinaryTable.createColumnDataFor(dataToFill);
150         for (int column = 0; column < ncols; column++) {
151             int tileIndex = 1;
152             String compressionAlgorithm = compressedHeader.getStringValue(Compression.ZCTYPn.n(column + 1));
153             for (int rowStart = 0; rowStart < nrows; rowStart += rowsPerTile) {
154                 BinaryTableTileDecompressor binaryTableTile = new BinaryTableTileDecompressor(this, dataToFill.getData(),
155                         tile()//
156                                 .rowStart(rowStart)//
157                                 .rowEnd(rowStart + rowsPerTile)//
158                                 .column(column)//
159                                 .tileIndex(tileIndex++)//
160                                 .compressionAlgorithm(compressionAlgorithm));
161                 tiles.add(binaryTableTile);
162                 binaryTableTile.execute(FitsFactory.threadPool());
163             }
164         }
165         for (BinaryTableTile binaryTableTile : tiles) {
166             binaryTableTile.waitForResult();
167         }
168         return dataToFill;
169     }
170 
171     /**
172      * Returns the number of original (uncompressed) table rows that are cmopressed as a block into a single compressed
173      * table row.
174      * 
175      * @return the number of table rows compressed together as a block.
176      */
177     protected int getRowsPerTile() {
178         return rowsPerTile;
179     }
180 
181     /**
182      * This should only be called by {@link CompressedTableHDU}.
183      */
184     @SuppressWarnings("javadoc")
185     protected void setColumnCompressionAlgorithms(String[] columnCompressionAlgorithms) {
186         this.columnCompressionAlgorithms = columnCompressionAlgorithms;
187     }
188 
189     /**
190      * This should only be called by {@link CompressedTableHDU}.
191      */
192     @SuppressWarnings("javadoc")
193     protected CompressedTableData setRowsPerTile(int value) {
194         rowsPerTile = value;
195         return this;
196     }
