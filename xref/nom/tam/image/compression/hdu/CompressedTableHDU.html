47  /**
48   * A header-data unit (HDU) containing a compressed binary table. A compressed table is still a binary table but with
49   * some additional constraints. The original table is divided into groups of rows (tiles) and each tile is compressed on
50   * its own. The compressed data is then stored in the 3 data columns of this binary table (compressed, gzipped and
51   * uncompressed) depending on the compression type used in the tile. Additional data columns may contain specific
52   * compression options for each tile (i.e. compressed table row) individually. Table keywords, which conflict with those
53   * in the original table are 'saved' under standard alternative names, so they may be restored with the original table
54   * as appropriate.
55   * <p>
56   * Compressing a table HDU is typically a two-step process:
57   * </p>
58   * <ol>
59   * <li>Create a <code>CompressedTableHDU</code>, e.g. with {@link #fromBinaryTableHDU(BinaryTableHDU, int, String...)},
60   * using the specified number of table rows per compressed block, and compression algorithm(s)</li>
61   * <li>Perform the compression via {@link #compress()}</li>
62   * </ol>
63   * <p>
64   * For example to compress a binary table:
65   * </p>
66   * 
67   * <pre>
68   *   BinaryTableHDU table = ...
69   *   
70   *   // 1. Create compressed HDU with the
71   *   CompressedTableHDU compressed = CompressedTableHDU.fromBinaryTableHDU(table, 4, Compression.ZCMPTYPE_RICE_1);
72   *   
73   *   // 2. Perform the compression.
74   *   compressed.compress();
75   * </pre>
76   * <p>
77   * which of course you can compact into a single line as:
78   * </p>
79   * 
80   * <pre>
81   * CompressedTableHDU compressed = CompressedTableHDU.fromBinaryTableHDU(table, 4, Compression.ZCMPTYPE_RICE_1).compress();
82   * </pre>
83   * <p>
84   * The two step process (as opposed to a single-step one) was probably chosen because it mimics that of
85   * {@link CompressedImageHDU}, where further configuration steps may be inserted in-between. After the compression, the
86   * compressed table HDU can be handled just like any other HDU, and written to a file or stream, for example.
87   * </p>
88   * <p>
89   * The reverse process is simply via the {@link #asBinaryTableHDU()} method. E.g.:
90   * </p>
91   * 
92   * <pre>
93   *    CompressedTableHDU compressed = ...
94   *    BinaryTableHDU table = compressed.asBinaryTableHDU();
95   * </pre>
96   *
97   * @see CompressedTableData
98   */
99  @SuppressWarnings("deprecation")
100 public class CompressedTableHDU extends BinaryTableHDU {
101 
102     /**
103      * Prepare a compressed binary table HDU for the specified binary table. When the tile row size is specified with
104      * -1, the value will be set ti the number of rows in the table. The table will be compressed in "rows" that are
105      * defined by the tile size. Next step would be to set the compression options into the HDU and then compress it.
106      *
107      * @param  binaryTableHDU              the binary table to compress
108      * @param  tileRows                    the number of rows that should be compressed per tile.
109      * @param  columnCompressionAlgorithms the compression algorithms to use for the columns (optional default
110      *                                         compression will be used if a column has no compression specified). You
111      *                                         should typically use one or more of the enum values defined in
112      *                                         {@link Compression}.
113      *
114      * @return                             the prepared compressed binary table HDU.
115      *
116      * @throws FitsException               if the binary table could not be used to create a compressed binary table.
117      */
118     public static CompressedTableHDU fromBinaryTableHDU(BinaryTableHDU binaryTableHDU, int tileRows,
119             String... columnCompressionAlgorithms) throws FitsException {
120         Header header = new Header();
121         CompressedTableData compressedData = new CompressedTableData();
122 
123         int rowsPerTile = tileRows > 0 ? tileRows : binaryTableHDU.getData().getNRows();
124         compressedData.setRowsPerTile(rowsPerTile);
125         compressedData.fillHeader(header);
126 
127         Cursor<String, HeaderCard> headerIterator = header.iterator();
128         Cursor<String, HeaderCard> imageIterator = binaryTableHDU.getHeader().iterator();
129         while (imageIterator.hasNext()) {
130             HeaderCard card = imageIterator.next();
131             CompressedCard.restore(card, headerIterator);
132         }
133         CompressedTableHDU compressedImageHDU = new CompressedTableHDU(header, compressedData);
134         compressedData.setColumnCompressionAlgorithms(columnCompressionAlgorithms);
135         compressedData.prepareUncompressedData(binaryTableHDU.getData().getData());
136         return compressedImageHDU;
137     }

174     /**
175      * Restores the original binary table HDU by decompressing the data contained in this compresed table HDU.
176      * 
177      * @return               The uncompressed binary table HDU.
178      * 
179      * @throws FitsException If there was an issue with the decompression.
180      * 
181      * @see                  #asBinaryTableHDU(int, int)
182      * @see                  #fromBinaryTableHDU(BinaryTableHDU, int, String...)
183      */
184     public BinaryTableHDU asBinaryTableHDU() throws FitsException {
185         Header header = getTableHeader();
186         BinaryTable data = BinaryTableHDU.manufactureData(header);
187         BinaryTableHDU tableHDU = new BinaryTableHDU(header, data);
188         getData().asBinaryTable(data, getHeader(), header);
189         return tableHDU;
190     }

192     /**
193      * Returns the number of table rows that are compressed in each table tile. This may be useful for figuring out what
194      * tiles to decompress, e.g. via {@link #asBinaryTableHDU(int, int)}, when wanting to access select table rows only.
195      * This value is stored under the FITS keyword ZTILELEN in the compressed header. Thus, this method simply provides
196      * a user-friendly way to access it. Note that the last tile may contain fewer rows than the value indicated by this
197      * 
198      * @return               the number of table rows that are compressed into a tile.
199      * 
200      * @throws FitsException if the compressed header does not contain the required ZTILELEN keyword, or it is &lt;= 0.
201      * 
202      * @see                  #asBinaryTableHDU(int, int)
203      * 
204      * @since                1.19
205      */
206     public int getTileRows() throws FitsException {
207         int n = getHeader().getIntValue(Compression.ZTILELEN, -1);
208         if (n <= 0) {
209             throw new FitsException("imnvalid or missing ZTILELEN header keyword");
210         }
211         return n;
212     }

228     /**
229      * Restores a section of the original binary table HDU by decompressing a selected range of compressed table tiles.
230      * The returned section will start at row index <code>fromTile * getTileRows()</code> of the full table.
231      * 
232      * @param  fromTile                 Java index of first tile to decompress
233      * @param  toTile                   Java index of last tile to decompress
234      * 
235      * @return                          The uncompressed binary table HDU from the selected compressed tiles.
236      * 
237      * @throws IllegalArgumentException If the tile range is out of bounds
238      * @throws FitsException            If there was an issue with the decompression.
239      * 
240      * @see                             #getTileRows()
241      * @see                             #getTileCount()
242      * @see                             #asBinaryTableHDU()
243      * @see                             #fromBinaryTableHDU(BinaryTableHDU, int, String...)
244      */
245     public BinaryTableHDU asBinaryTableHDU(int fromTile, int toTile) throws FitsException, IllegalArgumentException {
246         Header header = getTableHeader();
247         int tileSize = getTileRows();
248 
249         if (fromTile < 0 || toTile > getTileCount() || toTile <= fromTile) {
250             throw new IllegalArgumentException(
251                     "illegal tile range [" + fromTile + ", " + toTile + "] for " + getTileCount() + " tiles");
252         }
253 
254         // Set the correct number of rows
255         int rows = getHeader().getIntValue(Compression.ZNAXISn.n(2));
256         header.addValue(Standard.NAXIS2, Integer.min(rows, toTile * tileSize) - fromTile * tileSize);
257 
258         BinaryTable data = BinaryTableHDU.manufactureData(header);
259         BinaryTableHDU tableHDU = new BinaryTableHDU(header, data);
260         getData().asBinaryTable(data, getHeader(), header, fromTile);
261 
262         return tableHDU;
263     }

322     /**
323      * Performs the actual compression with the selected algorithm(s) and options. When creating a compressed table HDU,
324      * e.g using the {@link #fromBinaryTableHDU(BinaryTableHDU, int, String...)} method, the HDU is merely prepared but
325      * without actually performing the compression, and this method will have to be called to actually perform the
326      * compression. The design would allow for setting options between creation and compressing, but in this case there
327      * is really nothing of the sort.
328      * 
329      * @return               itself
330      * 
331      * @throws FitsException if the compression could not be performed
332      * 
333      * @see                  #fromBinaryTableHDU(BinaryTableHDU, int, String...)
334      */
335     public CompressedTableHDU compress() throws FitsException {
336         getData().compress(getHeader());
337         return this;
338     }

340     /**
341      * Obtain a header representative of a decompressed TableHDU.
342      *
343      * @return                     Header with decompressed cards.
344      *
345      * @throws HeaderCardException if the card could not be copied
346      *
347      * @since                      1.18
348      */
349     public Header getTableHeader() throws HeaderCardException {
350         Header header = new Header();
351 
352         header.addValue(Standard.XTENSION, Standard.XTENSION_BINTABLE);
353         header.addValue(Standard.BITPIX, ElementType.BYTE.bitPix());
354         header.addValue(Standard.NAXIS, 2);
355 
356         Cursor<String, HeaderCard> tableIterator = header.iterator();
357         Cursor<String, HeaderCard> iterator = getHeader().iterator();
358 
359         while (iterator.hasNext()) {
360             CompressedCard.backup(iterator.next(), tableIterator);
361         }
362 
363         return header;
364     }
