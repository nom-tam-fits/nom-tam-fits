45  public class CompressedTableHDU extends BinaryTableHDU {
46  
47      /**
48       * Prepare a compressed binary table HDU for the specified binary table.
49       * When the tile row size is specified with -1, the value will be set ti the
50       * number of rows in the table. The table will be compressed in "rows" that
51       * are defined by the tile size. Next step would be to set the compression
52       * options into the HDU and then compress it.
53       *
54       * @param binaryTableHDU
55       *            the binary table to compress
56       * @param tileRows
57       *            the number of rows that should be compressed per tile.
58       * @param columnCompressionAlgorithms
59       *            the compression algorithms to use for the columns (optional
60       *            default compression will be used if a column has no
61       *            compression specified)
62       * @return the prepared compressed binary table HDU.
63       * @throws FitsException
64       *             if the binary table could not be used to create a compressed
65       *             binary table.
66       */
67      public static CompressedTableHDU fromBinaryTableHDU(BinaryTableHDU binaryTableHDU, int tileRows, String... columnCompressionAlgorithms) throws FitsException {
68          Header header = new Header();
69          CompressedTableData compressedData = new CompressedTableData();
70  
71          int rowsPerTile = tileRows > 0 ? tileRows : binaryTableHDU.getData().getNRows();
72          compressedData.setRowsPerTile(rowsPerTile);
73          compressedData.fillHeader(header);
74  
75          Cursor<String, HeaderCard> headerIterator = header.iterator();
76          Cursor<String, HeaderCard> imageIterator = binaryTableHDU.getHeader().iterator();
77          while (imageIterator.hasNext()) {
78              HeaderCard card = imageIterator.next();
79              BackupRestoreUnCompressedHeaderCard.restore(card, headerIterator);
80          }
81          CompressedTableHDU compressedImageHDU = new CompressedTableHDU(header, compressedData);
82          compressedData.setColumnCompressionAlgorithms(columnCompressionAlgorithms);
83          compressedData.prepareUncompressedData(binaryTableHDU.getData().getData());
84          return compressedImageHDU;
85      }
86  
87      /**
88       * Check that this HDU has a valid header for this type.
89       *
90       * @param hdr
91       *            header to check
92       * @return <CODE>true</CODE> if this HDU has a valid header.
93       */
94      public static boolean isHeader(Header hdr) {
95          return hdr.getBooleanValue(ZTABLE, false);
96      }
97  
98      public static CompressedTableData manufactureData(Header hdr) throws FitsException {
99          return new CompressedTableData(hdr);
100     }
101 
102     public CompressedTableHDU(Header hdr, CompressedTableData datum) {
103         super(hdr, datum);
104     }
105 
106     public BinaryTableHDU asBinaryTableHDU() throws FitsException {
107         Header header = new Header();
108         header.addValue(Standard.XTENSION, Standard.XTENSION_BINTABLE);
109         header.addValue(Standard.BITPIX, ElementType.BYTE.bitPix());
110         header.addValue(Standard.NAXIS, 2);
111         Cursor<String, HeaderCard> headerIterator = header.iterator();
112         Cursor<String, HeaderCard> iterator = getHeader().iterator();
113         while (iterator.hasNext()) {
114             HeaderCard card = iterator.next();
115             BackupRestoreUnCompressedHeaderCard.backup(card, headerIterator);
116         }
117         BinaryTable data = BinaryTableHDU.manufactureData(header);
118         BinaryTableHDU tableHDU = new BinaryTableHDU(header, data);
119         getData().asBinaryTable(data, getHeader(), header);
120         return tableHDU;
121     }
122 
123     public CompressedTableHDU compress() throws FitsException {
124         getData().compress(getHeader());
125         return this;
126     }
127 
128     @Override
129     public CompressedTableData getData() {
130         return (CompressedTableData) super.getData();
131     }
