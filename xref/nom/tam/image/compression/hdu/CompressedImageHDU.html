56  /**
57   * A compressed image is a normal binary table with a defined structure. The
58   * image is split in tiles and each tile is compressed on its own. The
59   * compressed data is then stored in the 3 data columns of this binary table
60   * (compressed, gzipped and uncompressed) depending on the compression type used
61   * in the tile.
62   */
63  public class CompressedImageHDU extends BinaryTableHDU {
64  
65      /**
66       * keys that are only valid in tables and should not go into the
67       * uncompressed image.
68       */
69      static final List<IFitsHeader> TABLE_COLUMN_KEYS = Collections.unmodifiableList(Arrays.asList(binaryTableColumnKeyStems()));
70  
71      static final Map<IFitsHeader, BackupRestoreUnCompressedHeaderCard> COMPRESSED_HEADER_MAPPING = new HashMap<>();
72  
73      static final Map<IFitsHeader, BackupRestoreUnCompressedHeaderCard> UNCOMPRESSED_HEADER_MAPPING = new HashMap<>();
74  
75      /**
76       * Prepare a compressed image hdu for the specified image. the tile axis
77       * that are specified with -1 are set to the corresponding axis of the
78       * image. The image will be compressed in "sqaures" that ar defined by the
79       * tile size. Next step would be to set the compression options into the hdu
80       * and then compress it.
81       *
82       * @param imageHDU
83       *            the image to compress
84       * @param tileAxis
85       *            the axis of the tiles in the image
86       * @return the prepared compressed image hdu.
87       * @throws FitsException
88       *             if the image could not be used to create a compressed image.
89       */
90      public static CompressedImageHDU fromImageHDU(ImageHDU imageHDU, int... tileAxis) throws FitsException {
91          Header header = new Header();
92          CompressedImageData compressedData = new CompressedImageData();
93          compressedData.setAxis(imageHDU.getAxes());
94          if (tileAxis.length > 0) {
95              compressedData.setTileSize(tileAxis);
96          }
97          compressedData.fillHeader(header);
98          Cursor<String, HeaderCard> iterator = header.iterator();
99          Cursor<String, HeaderCard> imageIterator = imageHDU.getHeader().iterator();
100         while (imageIterator.hasNext()) {
101             HeaderCard card = imageIterator.next();
102             BackupRestoreUnCompressedHeaderCard.restore(card, iterator);
103         }
104         CompressedImageHDU compressedImageHDU = new CompressedImageHDU(header, compressedData);
105         compressedData.prepareUncompressedData(imageHDU.getData().getData(), header);
106         return compressedImageHDU;
107     }

128     public ImageHDU asImageHDU() throws FitsException {
129         Header header = new Header();
130         Cursor<String, HeaderCard> imageIterator = header.iterator();
131         Cursor<String, HeaderCard> iterator = getHeader().iterator();
132         while (iterator.hasNext()) {
133             HeaderCard card = iterator.next();
134             if (!TABLE_COLUMN_KEYS.contains(GenericKey.lookup(card.getKey()))) {
135                 BackupRestoreUnCompressedHeaderCard.backup(card, imageIterator);
136             }
137         }
138         ImageData data = (ImageData) ImageHDU.manufactureData(header);
139         ImageHDU imageHDU = new ImageHDU(header, data);
140         data.setBuffer(getUncompressedData());
141         return imageHDU;
142     }

148     /**
149      * Specify an areaWithin the image that will not undergo a lossy
150      * compression. This will only have affect it the selected compression
151      * (including the options) is a lossy compression. All tiles touched by this
152      * region will be handled so that there is no loss of any data, the
153      * reconstruction will be exact.
154      *
155      * @param x
156      *            the x position in the image
157      * @param y
158      *            the y position in the image
159      * @param width
160      *            the width of the area
161      * @param heigth
162      *            the height of the area
163      * @return this
164      */
165     public CompressedImageHDU forceNoLoss(int x, int y, int width, int heigth) {
166         getData().forceNoLoss(x, y, width, heigth);
167         return this;
168     }

193     /**
194      * preserve the null values in the image even if the compression algorithm
195      * is lossy. I the image that will be compressed a BLANK header should be
196      * available if the pixel value is one of the integer types.
197      *
198      * @param compressionAlgorithm
199      *            compression algorithm to use for the null pixel mask
200      * @return this
201      */
202     public CompressedImageHDU preserveNulls(String compressionAlgorithm) {
203         long nullValue = getHeader().getLongValue(BLANK, Long.MIN_VALUE);
204         getData().preserveNulls(nullValue, compressionAlgorithm);
205         return this;
206     }

208     public CompressedImageHDU setCompressAlgorithm(String compressAlgorithm) throws FitsException {
209         HeaderCard compressAlgorithmCard = getHeader().card(Compression.ZCMPTYPE).value(compressAlgorithm).card();
210         getData().setCompressAlgorithm(compressAlgorithmCard);
211         return this;
212     }
213 
214     public CompressedImageHDU setQuantAlgorithm(String quantAlgorithm) throws FitsException {
215         if (quantAlgorithm != null && !quantAlgorithm.isEmpty()) {
216             HeaderCard quantAlgorithmCard = getHeader().card(Compression.ZQUANTIZ).value(quantAlgorithm).card();
217             getData().setQuantAlgorithm(quantAlgorithmCard);
218         } else {
219             getData().setQuantAlgorithm(null);
220         }
221         return this;
222     }
