34  import java.io.IOException;
35  import java.nio.ByteBuffer;
36  
37  import nom.tam.fits.FitsException;
38  import nom.tam.image.compression.hdu.CompressedTableData;
39  import nom.tam.util.ArrayDataOutput;
40  import nom.tam.util.FitsOutputStream;
41  import nom.tam.util.ByteBufferOutputStream;
42  import nom.tam.util.ColumnTable;
43  
44  public class BinaryTableTileCompressor extends BinaryTableTile {
45  
46      private static final int FACTOR_15 = 15;
47  
48      private static final int FACTOR_10 = 10;
49  
50      private static final int FACTOR_11 = 11;
51  
52      private static final int MINIMUM_EXTRA_SPACE = 1024;
53  
54      private final CompressedTableData binData;
55  
56      public BinaryTableTileCompressor(CompressedTableData binData, ColumnTable<?> columnTable, BinaryTableTileDescription description) {
57          super(columnTable, description);
58          this.binData = binData;
59      }
60  
61      @Override
62      public void run() {
63          ByteBuffer buffer = ByteBuffer.wrap(new byte[getUncompressedSizeInBytes()]);
64          try (ArrayDataOutput os = new FitsOutputStream(new ByteBufferOutputStream(buffer))) {
65              this.data.write(os, this.rowStart, this.rowEnd, this.column);
66          } catch (IOException e) {
67              throw new IllegalStateException("could not write compressed data", e);
68          }
69          buffer.rewind();
70          int spaceForCompression = getUncompressedSizeInBytes();
71          // give the compression 10% more space and a minimum of 1024 bytes
72          spaceForCompression = Math.max(spaceForCompression * FACTOR_11 / FACTOR_10, spaceForCompression + MINIMUM_EXTRA_SPACE);
73          ByteBuffer compressedBuffer = ByteBuffer.wrap(new byte[spaceForCompression]);
74          if (!getCompressorControl().compress(type.asTypedBuffer(buffer), compressedBuffer, null)) {
75              // very bad case lets try again with 50% more space
76              spaceForCompression = spaceForCompression * FACTOR_15 / FACTOR_10;
77              compressedBuffer = ByteBuffer.wrap(new byte[spaceForCompression]);
78              if (!getCompressorControl().compress(type.asTypedBuffer(buffer), compressedBuffer, null)) {
79                  throw new IllegalStateException("could not compress the tile with the requested algorithem!");
80              }
81          }
82          byte[] compressedBytes = new byte[compressedBuffer.position()];
83          compressedBuffer.rewind();
84          compressedBuffer.get(compressedBytes);
85          try {
86              synchronized (this.binData) {
87                  this.binData.setElement(getTileIndex() - 1, this.column, compressedBytes);
88              }
89          } catch (FitsException e) {
90              throw new IllegalStateException("could not include compressed data into the table", e);
91          }
92      }
93  }
