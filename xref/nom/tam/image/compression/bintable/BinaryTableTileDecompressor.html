34  import java.io.ByteArrayInputStream;
35  import java.io.IOException;
36  import java.nio.ByteBuffer;
37  
38  import nom.tam.fits.FitsException;
39  import nom.tam.image.compression.hdu.CompressedTableData;
40  import nom.tam.util.ArrayDataInput;
41  import nom.tam.util.FitsInputStream;
42  import nom.tam.util.ColumnTable;
43  
44  public class BinaryTableTileDecompressor extends BinaryTableTile {
45  
46      private final ByteBuffer compressedBytes;
47  
48      private ArrayDataInput is;
49  
50      public BinaryTableTileDecompressor(CompressedTableData binData, ColumnTable<?> columnTable, BinaryTableTileDescription description) throws FitsException {
51          super(columnTable, description);
52          this.compressedBytes = ByteBuffer.wrap((byte[]) binData.getElement(this.rowStart, this.column));
53      }
54  
55      @Override
56      public void run() {
57          if (this.is == null) {
58              ByteBuffer unCompressedBytes = ByteBuffer.wrap(new byte[getUncompressedSizeInBytes()]);
59              getCompressorControl().decompress(this.compressedBytes, type.asTypedBuffer(unCompressedBytes), null);
60              this.is = new FitsInputStream(new ByteArrayInputStream(unCompressedBytes.array()));
61          }
62          try {
63              this.data.read(this.is, this.rowStart, this.rowEnd, this.column);
64          } catch (IOException e) {
65              throw new IllegalStateException("could not read compressed data", e);
66          }
67      }
68  
69  }
