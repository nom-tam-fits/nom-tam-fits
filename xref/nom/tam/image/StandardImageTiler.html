40  /**
41   * This class provides a subset of an N-dimensional image. Modified May 2, 2000
42   * by T. McGlynn to permit tiles that go off the edge of the image.
43   */
44  public abstract class StandardImageTiler implements ImageTiler {
45  
46      /**
47       * @return the offset of a given position.
48       * @param dims
49       *            The dimensions of the array.
50       * @param pos
51       *            The index requested.
52       */
53      public static long getOffset(int[] dims, int[] pos) {
54  
55          long offset = 0;
56          for (int i = 0; i < dims.length; i += 1) {
57              if (i > 0) {
58                  offset *= dims[i];
59              }
60              offset += pos[i];
61          }
62          return offset;
63      }
64  
65      /**
66       * Increment the offset within the position array. Note that we never look
67       * at the last index since we copy data a block at a time and not byte by
68       * byte.
69       * 
70       * @param start
71       *            The starting corner values.
72       * @param current
73       *            The current offsets.
74       * @param lengths
75       *            The desired dimensions of the subset.
76       * @return <code>true</code> if the current array was changed
77       */
78      protected static boolean incrementPosition(int[] start, int[] current, int[] lengths) {
79  
80          for (int i = start.length - 2; i >= 0; i -= 1) {
81              if (current[i] - start[i] < lengths[i] - 1) {
82                  current[i] += 1;
83                  for (int j = i + 1; j < start.length - 1; j += 1) {
84                      current[j] = start[j];
85                  }
86                  return true;
87              }
88          }
89          return false;
90      }

100     /**
101      * Create a tiler.
102      * 
103      * @param f
104      *            The random access device from which image data may be read.
105      *            This may be null if the tile information is available from
106      *            memory.
107      * @param fileOffset
108      *            The file offset within the RandomAccess device at which the
109      *            data begins.
110      * @param dims
111      *            The actual dimensions of the image.
112      * @param base
113      *            The base class (should be a primitive type) of the image.
114      */
115     @SuppressFBWarnings(value = "EI_EXPOSE_REP", justification = "intended exposure of mutable data")
116     public StandardImageTiler(RandomAccess f, long fileOffset, int[] dims, Class<?> base) {
117         this.randomAccessFile = f;
118         this.fileOffset = fileOffset;
119         this.dims = dims;
120         this.base = base;
121     }

123     /**
124      * File a tile segment from a file.
125      * 
126      * @param output
127      *            The output tile.
128      * @param delta
129      *            The offset from the beginning of the image in bytes.
130      * @param outputOffset
131      *            The index into the output array.
132      * @param segment
133      *            The number of elements to be read for this segment.
134      * @throws IOException
135      *             if the underlying stream failed
136      */
137     @SuppressFBWarnings(value = "RR_NOT_CHECKED", justification = "this read will never return less than the requested length")
138     protected void fillFileData(Object output, long delta, int outputOffset, int segment) throws IOException {
139 
140         this.randomAccessFile.seek(this.fileOffset + delta);
141 
142         if (this.base == float.class) {
143             this.randomAccessFile.read((float[]) output, outputOffset, segment);
144         } else if (this.base == int.class) {
145             this.randomAccessFile.read((int[]) output, outputOffset, segment);
146         } else if (this.base == short.class) {
147             this.randomAccessFile.read((short[]) output, outputOffset, segment);
148         } else if (this.base == double.class) {
149             this.randomAccessFile.read((double[]) output, outputOffset, segment);
150         } else if (this.base == byte.class) {
151             this.randomAccessFile.read((byte[]) output, outputOffset, segment);
152         } else if (this.base == long.class) {
153             this.randomAccessFile.read((long[]) output, outputOffset, segment);
154         } else {
155             throw new IOException("Invalid type for tile array");
156         }
157     }

159     /**
160      * Fill a single segment from memory. This routine is called recursively to
161      * handle multi-dimensional arrays. E.g., if data is three-dimensional, this
162      * will recurse two levels until we get a call with a single dimensional
163      * datum. At that point the appropriate data will be copied into the output.
164      * 
165      * @param data
166      *            The in-memory image data.
167      * @param posits
168      *            The current position for which data is requested.
169      * @param length
170      *            The size of the segments.
171      * @param output
172      *            The output tile.
173      * @param outputOffset
174      *            The current offset into the output tile.
175      * @param dim
176      *            The current dimension being
177      */
178     protected void fillMemData(Object data, int[] posits, int length, Object output, int outputOffset, int dim) {
179 
180         if (data instanceof Object[]) {
181 
182             Object[] xo = (Object[]) data;
183             fillMemData(xo[posits[dim]], posits, length, output, outputOffset, dim + 1);
184 
185         } else {
186 
187             // Adjust the spacing for the actual copy.
188             int startFrom = posits[dim];
189             int startTo = outputOffset;
190             int copyLength = length;
191 
192             if (posits[dim] < 0) {
193                 startFrom -= posits[dim];
194                 startTo -= posits[dim];
195                 copyLength += posits[dim];
196             }
197             if (posits[dim] + length > this.dims[dim]) {
198                 copyLength -= posits[dim] + length - this.dims[dim];
199             }
200 
201             System.arraycopy(data, startFrom, output, startTo, copyLength);
202         }
203     }

205     /**
206      * Fill the subset.
207      * 
208      * @param data
209      *            The memory-resident data image. This may be null if the image
210      *            is to be read from a file. This should be a multi-dimensional
211      *            primitive array.
212      * @param o
213      *            The tile to be filled. This is a simple primitive array.
214      * @param newDims
215      *            The dimensions of the full image.
216      * @param corners
217      *            The indices of the corner of the image.
218      * @param lengths
219      *            The dimensions of the subset.
220      * @throws IOException
221      *             if the underlying stream failed
222      */
223     protected void fillTile(Object data, Object o, int[] newDims, int[] corners, int[] lengths) throws IOException {
224 
225         int n = newDims.length;
226         int[] posits = new int[n];
227         int baseLength = ArrayFuncs.getBaseLength(o);
228         int segment = lengths[n - 1];
229 
230         System.arraycopy(corners, 0, posits, 0, n);
231         long currentOffset = 0;
232         if (data == null) {
233             currentOffset = this.randomAccessFile.getFilePointer();
234         }
235 
236         int outputOffset = 0;
237 
238         do {
239 
240             // This implies there is some overlap
241             // in the last index (in conjunction
242             // with other tests)
243 
244             int mx = newDims.length - 1;
245             boolean validSegment = posits[mx] + lengths[mx] >= 0 && posits[mx] < newDims[mx];
246 
247             // Don't do anything for the current
248             // segment if anything but the
249             // last index is out of range.
250 
251             if (validSegment) {
252                 for (int i = 0; i < mx; i += 1) {
253                     if (posits[i] < 0 || posits[i] >= newDims[i]) {
254                         validSegment = false;
255                         break;
256                     }
257                 }
258             }
259 
260             if (validSegment) {
261                 if (data != null) {
262                     fillMemData(data, posits, segment, o, outputOffset, 0);
263                 } else {
264                     long offset = getOffset(newDims, posits) * baseLength;
265 
266                     // Point to offset at real beginning
267                     // of segment
268                     int actualLen = segment;
269                     long actualOffset = offset;
270                     int actualOutput = outputOffset;
271                     if (posits[mx] < 0) {
272                         actualOffset -= posits[mx] * baseLength;
273                         actualOutput -= posits[mx];
274                         actualLen += posits[mx];
275                     }
276                     if (posits[mx] + segment > newDims[mx]) {
277                         actualLen -= posits[mx] + segment - newDims[mx];
278                     }
279                     fillFileData(o, actualOffset, actualOutput, actualLen);
280                 }
281             }
282             outputOffset += segment;
283 
284         } while (incrementPosition(corners, posits, lengths));
285         if (data == null) {
286             this.randomAccessFile.seek(currentOffset);
287         }
288     }

290     /**
291      * Read the entire image into a multidimensional array.
292      * 
293      * @throws IOException
294      *             if the underlying stream failed
295      */
296     @Override
297     public Object getCompleteImage() throws IOException {
298 
299         if (this.randomAccessFile == null) {
300             throw new IOException("Attempt to read from null file");
301         }
302         long currentOffset = this.randomAccessFile.getFilePointer();
303         Object o = ArrayFuncs.newInstance(this.base, this.dims);
304         this.randomAccessFile.seek(this.fileOffset);
305         this.randomAccessFile.readArrayFully(o);
306         this.randomAccessFile.seek(currentOffset);
307         return o;
308     }

318     /**
319      * Get a subset of the image. An image tile is returned as a one-dimensional
320      * array although the image will normally be multi-dimensional.
321      * 
322      * @param corners
323      *            The starting corner (using 0 as the start) for the image.
324      * @param lengths
325      *            The length requested in each dimension.
326      * @throws IOException
327      *             if the underlying stream failed
328      */
329     @Override
330     public Object getTile(int[] corners, int[] lengths) throws IOException {
331 
332         if (corners.length != this.dims.length || lengths.length != this.dims.length) {
333             throw new IOException("Inconsistent sub-image request");
334         }
335 
336         int arraySize = 1;
337         for (int i = 0; i < this.dims.length; i += 1) {
338 
339             if (corners[i] < 0 || lengths[i] < 0 || corners[i] + lengths[i] > this.dims[i]) {
340                 throw new IOException("Sub-image not within image");
341             }
342 
343             arraySize *= lengths[i];
344         }
345 
346         Object outArray = ArrayFuncs.newInstance(this.base, arraySize);
347 
348         getTile(outArray, corners, lengths);
349         return outArray;
350     }

352     /**
353      * Get a tile, filling in a prespecified array. This version does not check
354      * that the user hase entered a valid set of corner and length arrays.
355      * ensure that out matches the length implied by the lengths array.
356      * 
357      * @param outArray
358      *            The output tile array. A one-dimensional array. Data not
359      *            within the valid limits of the image will be left unchanged.
360      *            The length of this array should be the product of lengths.
361      * @param corners
362      *            The corners of the tile.
363      * @param lengths
364      *            The dimensions of the tile.
365      * @throws IOException
366      *             if the underlying stream failed
367      */
368     @Override
369     public void getTile(Object outArray, int[] corners, int[] lengths) throws IOException {
370 
371         Object data = getMemoryImage();
372 
373         if (data == null && this.randomAccessFile == null) {
374             throw new IOException("No data source for tile subset");
375         }
376         fillTile(data, outArray, this.dims, corners, lengths);
377     }
