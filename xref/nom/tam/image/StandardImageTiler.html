45   * This class provides a subset of an N-dimensional image. Modified May 2, 2000

48  public abstract class StandardImageTiler implements ImageTiler {
49      /**
50       * @param dims The dimensions of the array.
51       * @param pos The index requested.
52       * 
53       * @return the offset of a given position.
54       */
55      public static long getOffset(int[] dims, int[] pos) {
56  
57          long offset = 0;
58          for (int i = 0; i < dims.length; i += 1) {
59              if (i > 0) {
60                  offset *= dims[i];
61              }
62              offset += pos[i];
63          }
64          return offset;
65      }
66  
67      /**
68       * Increment the offset within the position array. Note that we never look at
69       * the last index since we copy data a block at a time and not byte by byte.
70       *
71       * @param start The starting corner values.
72       * @param current The current offsets.
73       * @param lengths The desired dimensions of the subset.
74       * 
75       * @return <code>true</code> if the current array was changed
76       */
77      protected static boolean incrementPosition(int[] start, int[] current, int[] lengths) {
78          final int[] steps = new int[start.length];
79          Arrays.fill(steps, 1);
80          return StandardImageTiler.incrementPosition(start, current, lengths, steps);
81      }
82  
83      /**
84       * Increment the offset within the position array. Note that we never look at
85       * the last index since we copy data a block at a time and not byte by byte.
86       *
87       * @param start The starting corner values.
88       * @param current The current offsets.
89       * @param lengths The desired dimensions of the subset.
90       * @param steps The desired number of steps to take until the next position.
91       * 
92       * @return <code>true</code> if the current array was changed
93       */
94      protected static boolean incrementPosition(int[] start, int[] current, int[] lengths, int[] steps) {
95          for (int i = start.length - 2; i >= 0; i -= 1) {
96              if (current[i] - start[i] < lengths[i] - steps[i]) {
97                  current[i] += steps[i];
98                  if (start.length - 1 - (i + 1) >= 0) {
99                      System.arraycopy(start, i + 1, current, i + 1, start.length - 1 - (i + 1));
100                 }
101                 return true;
102             }
103         }
104         return false;
105     }

150     /**
151      * File a tile segment from a file, jumping each step number of values to the
152      * next read.
153      *
154      * @param output The output to send data. This can be an ArrayDataOutput to
155      *            stream data to and prevent memory consumption of a tile being
156      *            in memory.
157      * @param delta The offset from the beginning of the image in bytes.
158      * @param outputOffset The index into the output array.
159      * @param segment The number of elements to be read for this segment.
160      * @param step The number of jumps until the next read. Only works for
161      *            streaming out data.
162      * 
163      * @throws IOException if the underlying stream failed
164      */
165     @SuppressFBWarnings(value = "RR_NOT_CHECKED", justification = "this read will never return less than the requested length")
166     protected void fillFileData(Object output, long delta, int outputOffset, int segment, int step) throws IOException {
167         if (output instanceof ArrayDataOutput) {
168             this.fillFileData((ArrayDataOutput) output, delta, segment, step);
169         } else {
170             this.randomAccessFile.seek(this.fileOffset + delta);
171 
172             if (this.base == float.class) {
173                 this.randomAccessFile.read((float[]) output, outputOffset, segment);
174             } else if (this.base == int.class) {
175                 this.randomAccessFile.read((int[]) output, outputOffset, segment);
176             } else if (this.base == short.class) {
177                 this.randomAccessFile.read((short[]) output, outputOffset, segment);
178             } else if (this.base == double.class) {
179                 this.randomAccessFile.read((double[]) output, outputOffset, segment);
180             } else if (this.base == byte.class) {
181                 this.randomAccessFile.read((byte[]) output, outputOffset, segment);
182             } else if (this.base == long.class) {
183                 this.randomAccessFile.read((long[]) output, outputOffset, segment);
184             } else {
185                 throw new IOException("Invalid type for tile array");
186             }
187         }
188     }

207     /**
208      * File a tile segment from a file into the given stream. This will deal only
209      * with bytes to avoid having to check the base type and calling a specific
210      * method. Converting the base type to a byte is a simple multiplication
211      * operation anyway.
212      *
213      * @param output The output stream.
214      * @param delta The offset from the beginning of the image in bytes.
215      * @param segment The number of elements to be read for this segment.
216      * @param step The number of elements until the next read.
217      * 
218      * @throws IOException if the underlying stream failed
219      *
220      * @since 1.18
221      */
222     @SuppressFBWarnings(value = "RR_NOT_CHECKED", justification = "this read will never return less than the requested length")
223     protected void fillFileData(ArrayDataOutput output, long delta, int segment, int step) throws IOException {
224         final int byteSize = ElementType.forClass(this.base).size();
225 
226         // Subtract one from the step since when we read from a stream, an actual
227         // "step" only exists if it's greater
228         // than 1.
229         final int stepSize = (step - 1) * byteSize;
230         this.randomAccessFile.seek(this.fileOffset + delta);
231 
232         // One value at a time
233         final byte[] buffer = new byte[byteSize];
234         long seekOffset = this.randomAccessFile.position();
235         int bytesRead = 0;
236 
237         // This is the byte count that will be read.
238         final int expectedBytes = segment * byteSize;
239         while (bytesRead < expectedBytes) {
240             // Prepare for the next read by seeking to the next step
241             this.randomAccessFile.seek(seekOffset);
242             final int currReadByteCount = this.randomAccessFile.read(buffer, 0, buffer.length);
243 
244             // Stop if there is no more to read.
245             if (currReadByteCount < 0) {
246                 break;
247             } else {
248                 output.write(buffer, 0, currReadByteCount);
249                 seekOffset = this.randomAccessFile.position() + stepSize;
250                 bytesRead += currReadByteCount + stepSize;
251             }
252         }
253 
254         output.flush();
255     }

257     /**
258      * Fill a single segment from memory. This routine is called recursively to
259      * handle multidimensional arrays. E.g., if data is three-dimensional, this
260      * will recurse two levels until we get a call with a single dimensional
261      * datum. At that point the appropriate data will be copied into the output.
262      * Uses a default value for striding (1).
263      *
264      * @param data The in-memory image data.
265      * @param posits The current position for which data is requested.
266      * @param length The size of the segments.
267      * @param output The output tile.
268      * @param outputOffset The current offset into the output tile.
269      * @param dim The current dimension being
270      * 
271      * @throws IOException If the output is a stream and there is an I/O error.
272      */
273     protected void fillMemData(Object data, int[] posits, int length, Object output, int outputOffset, int dim)
274             throws IOException {
275         fillMemData(data, posits, length, output, outputOffset, dim, 1);
276     }

278     /**
279      * Fill a single segment from memory. This routine is called recursively to
280      * handle multidimensional arrays. E.g., if data is three-dimensional, this
281      * will recurse two levels until we get a call with a single dimensional
282      * datum. At that point the appropriate data will be copied into the output,
283      * jumping the number of step values.
284      *
285      * @param data The in-memory image data.
286      * @param posits The current position for which data is requested.
287      * @param length The size of the segments.
288      * @param output The output tile.
289      * @param outputOffset The current offset into the output tile.
290      * @param dim The current dimension being
291      * @param step The number of jumps to the next value.
292      * 
293      * @throws IOException If the output is a stream and there is an I/O error.
294      * 
295      * @since 1.18
296      */
297     protected void fillMemData(Object data, int[] posits, int length, Object output, int outputOffset, int dim, int step)
298             throws IOException {
299 
300         if (data instanceof Object[]) {
301 
302             Object[] xo = (Object[]) data;
303             fillMemData(xo[posits[dim]], posits, length, output, outputOffset, dim + 1, step);
304 
305         } else {
306 
307             // Adjust the spacing for the actual copy.
308             int startFrom = posits[dim];
309             int startTo = outputOffset;
310             int copyLength = length;
311 
312             if (posits[dim] < 0) {
313                 startFrom -= posits[dim];
314                 startTo -= posits[dim];
315                 copyLength += posits[dim];
316             }
317             if (posits[dim] + length > this.dims[dim]) {
318                 copyLength -= posits[dim] + length - this.dims[dim];
319             }
320 
321             if (output instanceof ArrayDataOutput) {
322                 // Intentionally missing char and boolean here as they are not
323                 // valid BITPIX values.
324                 final ArrayDataOutput arrayDataOutput = ((ArrayDataOutput) output);
325                 for (int i = startFrom; i < startFrom + copyLength; i += step) {
326                     if (this.base == float.class) {
327                         arrayDataOutput.writeFloat(Array.getFloat(data, i));
328                     } else if (this.base == int.class) {
329                         arrayDataOutput.writeInt(Array.getInt(data, i));
330                     } else if (this.base == double.class) {
331                         arrayDataOutput.writeDouble(Array.getDouble(data, i));
332                     } else if (this.base == long.class) {
333                         arrayDataOutput.writeLong(Array.getLong(data, i));
334                     } else if (this.base == short.class) {
335                         arrayDataOutput.writeShort(Array.getShort(data, i));
336                     } else if (this.base == byte.class) {
337                         arrayDataOutput.writeByte(Array.getByte(data, i));
338                     }
339                 }
340 
341                 arrayDataOutput.flush();
342             } else {
343                 ArrayFuncs.copy(data, startFrom, output, startTo, copyLength, step);
344             }
345         }
346     }

368     /**
369      * Fill the subset, jumping each step value to the next read.
370      *
371      * @param data The memory-resident data image. This may be null if the image
372      *            is to be read from a file. This should be a multidimensional
373      *            primitive array.
374      * @param o The tile to be filled. This is a simple primitive array, or an
375      *            ArrayDataOutput instance.
376      * @param newDims The dimensions of the full image.
377      * @param corners The indices of the corner of the image.
378      * @param lengths The dimensions of the subset.
379      * @param steps The number of steps to take until the next read in each axis.
380      *
381      * @throws IOException if the underlying stream failed
382      */
383     protected void fillTile(Object data, Object o, int[] newDims, int[] corners, int[] lengths, int[] steps)
384             throws IOException {
385 
386         int n = newDims.length;
387         int[] posits = new int[n];
388         final boolean isStreaming = (o instanceof ArrayDataOutput);
389 
390         // TODO: When streaming out to an ArrayDataOutput, use this tiler's base
391         // class to determine the element size.
392         // TODO: If that is not sufficient, then maybe it needs to be passed in?
393         // TODO: jenkinsd 2022.12.21
394         //
395         final int baseLength = isStreaming ? ElementType.forClass(this.base).size() : ArrayFuncs.getBaseLength(o);
396 
397         int segment = lengths[n - 1];
398         int segmentStep = steps[n - 1];
399 
400         System.arraycopy(corners, 0, posits, 0, n);
401         long currentOffset = 0;
402         if (data == null) {
403             currentOffset = this.randomAccessFile.getFilePointer();
404         }
405 
406         int outputOffset = 0;
407 
408         // Flag to indicate something was written out. This is only relevant if
409         // the output is an ArrayDataOutput.
410         boolean hasNoOverlap = true;
411 
412         do {
413 
414             // This implies there is some overlap
415             // in the last index (in conjunction
416             // with other tests)
417 
418             int mx = newDims.length - 1;
419             boolean validSegment = posits[mx] + lengths[mx] >= 0 && posits[mx] < newDims[mx];
420 
421             // Don't do anything for the current
422             // segment if anything but the
423             // last index is out of range.
424 
425             if (validSegment) {
426                 for (int i = 0; i < mx; i += 1) {
427                     if (posits[i] < 0 || posits[i] >= newDims[i]) {
428                         validSegment = false;
429                         break;
430                     }
431                 }
432             }
433 
434             if (validSegment) {
435                 hasNoOverlap = false;
436                 if (data != null) {
437                     fillMemData(data, posits, segment, o, outputOffset, 0, segmentStep);
438                 } else {
439                     long offset = getOffset(newDims, posits) * baseLength;
440 
441                     // Point to offset at real beginning
442                     // of segment
443                     int actualLen = segment;
444                     long actualOffset = offset;
445                     int actualOutput = outputOffset;
446                     if (posits[mx] < 0) {
447                         actualOffset -= (long) posits[mx] * baseLength;
448                         actualOutput -= posits[mx];
449                         actualLen += posits[mx];
450                     }
451                     if (posits[mx] + segment > newDims[mx]) {
452                         actualLen -= posits[mx] + segment - newDims[mx];
453                     }
454                     fillFileData(o, actualOffset, actualOutput, actualLen, segmentStep);
455                 }
456             }
457             if (!isStreaming) {
458                 outputOffset += segment;
459             }
460 
461         } while (incrementPosition(corners, posits, lengths, steps));
462         if (data == null) {
463             this.randomAccessFile.seek(currentOffset);
464         }
465 
466         if (isStreaming && hasNoOverlap) {
467             throw new IOException("Sub-image not within image");
468         }
469     }

471     /**
472      * Read the entire image into a multidimensional array.
473      *
474      * @throws IOException if the underlying stream failed
475      */
476     @Override
477     public Object getCompleteImage() throws IOException {
478 
479         if (this.randomAccessFile == null) {
480             throw new IOException("Attempt to read from null file");
481         }
482         long currentOffset = this.randomAccessFile.getFilePointer();
483         Object o = ArrayFuncs.newInstance(this.base, this.dims);
484         this.randomAccessFile.seek(this.fileOffset);
485         this.randomAccessFile.readImage(o);
486         this.randomAccessFile.seek(currentOffset);
487         return o;
488     }

514     /**
515      * Get a subset of the image. An image tile is returned as a one-dimensional
516      * array although the image will normally be multidimensional.
517      *
518      * @param corners The starting corner (using 0 as the start) for the image.
519      * @param lengths The length requested in each dimension.
520      * @param steps The number of steps to take until the next read in each axis.
521      *            Default should be 1 (next value).
522      * 
523      * @throws IOException if the underlying stream failed
524      */
525     @Override
526     public Object getTile(int[] corners, int[] lengths, int[] steps) throws IOException {
527 
528         if (corners.length != this.dims.length || lengths.length != this.dims.length) {
529             throw new IOException("Inconsistent sub-image request");
530         }
531 
532         int arraySize = 1;
533         for (int i = 0; i < this.dims.length; i += 1) {
534 
535             if (corners[i] < 0 || lengths[i] < 0 || corners[i] + lengths[i] > this.dims[i]) {
536                 throw new IOException("Sub-image not within image");
537             } else if (steps[i] < 1) {
538                 throw new IOException("Step value cannot be less than 1.");
539             }
540 
541             arraySize *= lengths[i];
542         }
543 
544         Object outArray = ArrayFuncs.newInstance(this.base, arraySize);
545 
546         getTile(outArray, corners, lengths, steps);
547         return outArray;
548     }

573     /**
574      * Get a tile, filling in a pre-specified array or writing to a stream. This
575      * version does not check that the user hase entered a valid set of corner
576      * and length arrays. ensure that out matches the length implied by the
577      * lengths array.
578      *
579      * @param output A one-dimensional output array or stream. Data not within
580      *            the valid limits of the image will be left unchanged. For an
581      *            array, the length should be the product of lengths. Optionally
582      *            provide an ArrayDataOutput to stream out data and not fill
583      *            memory; useful for web applications.
584      * @param corners The corners of the tile.
585      * @param lengths The dimensions of the tile.
586      * @param steps The number of steps to take until the next read in each axis.
587      *            Default should be 1 (next value).
588      * 
589      * @throws IOException if the underlying stream failed
590      */
591     @Override
592     public void getTile(Object output, int[] corners, int[] lengths, int[] steps) throws IOException {
593         Object data = getMemoryImage();
594 
595         if (data == null && this.randomAccessFile == null) {
596             throw new IOException("No data source for tile subset");
597         }
598 
599         fillTile(data, output, this.dims, corners, lengths, steps);
