39  /**
40   * This class provides a subset of an N-dimensional image. Modified May 2, 2000
41   * by T. McGlynn to permit tiles that go off the edge of the image.
42   */
43  public abstract class StandardImageTiler implements ImageTiler {
44  
45      private RandomAccess f;
46  
47      private long fileOffset;
48  
49      private int[] dims;
50  
51      private Class base;
52  
53      /**
54       * Create a tiler.
55       * 
56       * @param f
57       *            The random access device from which image data may be read.
58       *            This may be null if the tile information is available from
59       *            memory.
60       * @param fileOffset
61       *            The file offset within the RandomAccess device at which the
62       *            data begins.
63       * @param dims
64       *            The actual dimensions of the image.
65       * @param base
66       *            The base class (should be a primitive type) of the image.
67       */
68      public StandardImageTiler(RandomAccess f, long fileOffset, int[] dims, Class base) {
69          this.f = f;
70          this.fileOffset = fileOffset;
71          this.dims = dims;
72          this.base = base;
73      }
74  
75      /**
76       * See if we can get the image data from memory. This may be overriden by
77       * other classes, notably in nom.tam.fits.ImageData.
78       */
79      protected abstract Object getMemoryImage();
80  
81      /**
82       * Get a subset of the image. An image tile is returned as a one-dimensional
83       * array although the image will normally be multi-dimensional.
84       * 
85       * @param corners
86       *            The starting corner (using 0 as the start) for the image.
87       * @param lengths
88       *            The length requested in each dimension.
89       */
90      public Object getTile(int[] corners, int[] lengths) throws IOException {
91  
92          if (corners.length != dims.length || lengths.length != dims.length) {
93              throw new IOException("Inconsistent sub-image request");
94          }
95  
96          int arraySize = 1;
97          for (int i = 0; i < dims.length; i += 1) {
98  
99              if (corners[i] < 0 || lengths[i] < 0 || corners[i] + lengths[i] > dims[i]) {
100                 throw new IOException("Sub-image not within image");
101             }
102 
103             arraySize *= lengths[i];
104         }
105 
106         Object outArray = ArrayFuncs.newInstance(base, arraySize);
107 
108         getTile(outArray, corners, lengths);
109         return outArray;
110     }
111 
112     /**
113      * Get a tile, filling in a prespecified array. This version does not check
114      * that the user hase entered a valid set of corner and length arrays.
115      * ensure that out matches the length implied by the lengths array.
116      * 
117      * @param outArray
118      *            The output tile array. A one-dimensional array. Data not
119      *            within the valid limits of the image will be left unchanged.
120      *            The length of this array should be the product of lengths.
121      * @param corners
122      *            The corners of the tile.
123      * @param lengths
124      *            The dimensions of the tile.
125      */
126     public void getTile(Object outArray, int[] corners, int[] lengths) throws IOException {
127 
128         Object data = getMemoryImage();
129 
130         if (data == null && f == null) {
131             throw new IOException("No data source for tile subset");
132         }
133         fillTile(data, outArray, dims, corners, lengths);
134     }
135 
136     /**
137      * Fill the subset.
138      * 
139      * @param data
140      *            The memory-resident data image. This may be null if the image
141      *            is to be read from a file. This should be a multi-dimensional
142      *            primitive array.
143      * @param o
144      *            The tile to be filled. This is a simple primitive array.
145      * @param dims
146      *            The dimensions of the full image.
147      * @param corners
148      *            The indices of the corner of the image.
149      * @param lengths
150      *            The dimensions of the subset.
151      */
152     protected void fillTile(Object data, Object o, int[] dims, int[] corners, int[] lengths) throws IOException {
153 
154         int n = dims.length;
155         int[] posits = new int[n];
156         int baseLength = ArrayFuncs.getBaseLength(o);
157         int segment = lengths[n - 1];
158 
159         System.arraycopy(corners, 0, posits, 0, n);
160         long currentOffset = 0;
161         if (data == null) {
162             currentOffset = f.getFilePointer();
163         }
164 
165         int outputOffset = 0;
166 
167         do {
168 
169             // This implies there is some overlap
170             // in the last index (in conjunction
171             // with other tests)
172 
173             int mx = dims.length - 1;
174             boolean validSegment = posits[mx] + lengths[mx] >= 0 && posits[mx] < dims[mx];
175 
176             // Don't do anything for the current
177             // segment if anything but the
178             // last index is out of range.
179 
180             if (validSegment) {
181                 for (int i = 0; i < mx; i += 1) {
182                     if (posits[i] < 0 || posits[i] >= dims[i]) {
183                         validSegment = false;
184                         break;
185                     }
186                 }
187             }
188 
189             if (validSegment) {
190                 if (data != null) {
191                     fillMemData(data, posits, segment, o, outputOffset, 0);
192                 } else {
193                     int offset = getOffset(dims, posits) * baseLength;
194 
195                     // Point to offset at real beginning
196                     // of segment
197                     int actualLen = segment;
198                     int actualOffset = offset;
199                     int actualOutput = outputOffset;
200                     if (posits[mx] < 0) {
201                         actualOffset -= posits[mx] * baseLength;
202                         actualOutput -= posits[mx];
203                         actualLen += posits[mx];
204                     }
205                     if (posits[mx] + segment > dims[mx]) {
206                         actualLen -= posits[mx] + segment - dims[mx];
207                     }
208                     fillFileData(o, actualOffset, actualOutput, actualLen);
209                 }
210             }
211             outputOffset += segment;
212 
213         } while (incrementPosition(corners, posits, lengths));
214         if (data == null) {
215             f.seek(currentOffset);
216         }
217     }

219     /**
220      * Fill a single segment from memory. This routine is called recursively to
221      * handle multi-dimensional arrays. E.g., if data is three-dimensional, this
222      * will recurse two levels until we get a call with a single dimensional
223      * datum. At that point the appropriate data will be copied into the output.
224      * 
225      * @param data
226      *            The in-memory image data.
227      * @param posits
228      *            The current position for which data is requested.
229      * @param length
230      *            The size of the segments.
231      * @param output
232      *            The output tile.
233      * @param outputOffset
234      *            The current offset into the output tile.
235      * @param dim
236      *            The current dimension being
237      */
238     protected void fillMemData(Object data, int[] posits, int length, Object output, int outputOffset, int dim) {
239 
240         if (data instanceof Object[]) {
241 
242             Object[] xo = (Object[]) data;
243             fillMemData(xo[posits[dim]], posits, length, output, outputOffset, dim + 1);
244 
245         } else {
246 
247             // Adjust the spacing for the actual copy.
248             int startFrom = posits[dim];
249             int startTo = outputOffset;
250             int copyLength = length;
251 
252             if (posits[dim] < 0) {
253                 startFrom -= posits[dim];
254                 startTo -= posits[dim];
255                 copyLength += posits[dim];
256             }
257             if (posits[dim] + length > dims[dim]) {
258                 copyLength -= (posits[dim] + length - dims[dim]);
259             }
260 
261             System.arraycopy(data, startFrom, output, startTo, copyLength);
262         }
263     }

265     /**
266      * File a tile segment from a file.
267      * 
268      * @param output
269      *            The output tile.
270      * @param delta
271      *            The offset from the beginning of the image in bytes.
272      * @param outputOffset
273      *            The index into the output array.
274      * @param segment
275      *            The number of elements to be read for this segment.
276      */
277     protected void fillFileData(Object output, int delta, int outputOffset, int segment) throws IOException {
278 
279         f.seek(fileOffset + delta);
280 
281         if (base == float.class) {
282             f.read((float[]) output, outputOffset, segment);
283         } else if (base == int.class) {
284             f.read((int[]) output, outputOffset, segment);
285         } else if (base == short.class) {
286             f.read((short[]) output, outputOffset, segment);
287         } else if (base == double.class) {
288             f.read((double[]) output, outputOffset, segment);
289         } else if (base == byte.class) {
290             f.read((byte[]) output, outputOffset, segment);
291         } else if (base == char.class) {
292             f.read((char[]) output, outputOffset, segment);
293         } else if (base == long.class) {
294             f.read((long[]) output, outputOffset, segment);
295         } else {
296             throw new IOException("Invalid type for tile array");
297         }
298     }

300     /**
301      * Increment the offset within the position array. Note that we never look
302      * at the last index since we copy data a block at a time and not byte by
303      * byte.
304      * 
305      * @param start
306      *            The starting corner values.
307      * @param current
308      *            The current offsets.
309      * @param lengths
310      *            The desired dimensions of the subset.
311      */
312     protected static boolean incrementPosition(int[] start, int[] current, int[] lengths) {
313 
314         for (int i = start.length - 2; i >= 0; i -= 1) {
315             if (current[i] - start[i] < lengths[i] - 1) {
316                 current[i] += 1;
317                 for (int j = i + 1; j < start.length - 1; j += 1) {
318                     current[j] = start[j];
319                 }
320                 return true;
321             }
322         }
323         return false;
324     }
325 
326     /**
327      * Get the offset of a given position.
328      * 
329      * @param dims
330      *            The dimensions of the array.
331      * @param pos
332      *            The index requested.
333      */
334     public static int getOffset(int[] dims, int[] pos) {
335 
336         int offset = 0;
337         for (int i = 0; i < dims.length; i += 1) {
338             if (i > 0) {
339                 offset *= dims[i];
340             }
341             offset += pos[i];
342         }
343         return offset;
344     }

346     /**
347      * Read the entire image into a multidimensional array.
348      */
349     public Object getCompleteImage() throws IOException {
350 
351         if (f == null) {
352             throw new IOException("Attempt to read from null file");
353         }
354         long currentOffset = f.getFilePointer();
355         Object o = ArrayFuncs.newInstance(base, dims);
356         f.seek(fileOffset);
357         f.readLArray(o);
358         f.seek(currentOffset);
359         return o;
360     }
