44  /**
45   * This class tests and illustrates the use of the HashedList class. Tests are
46   * in four parts.
47   * <p>
48   * The first section (in testCollection) tests the methods that are present in
49   * the Collection interface. All of the optional methods of that interface are
50   * supported. This involves tests of the HashedClass interface directly.
51   * <p>
52   * The second set of tests uses the Iterator (in testIterator) returned by the
53   * iterator() method and tests the standard Iterator methods to display and
54   * remove rows from the HashedList.
55   * <p>
56   * The third set of tests (in testCursor) tests the extended capabilities of the
57   * HashedListIterator to add rows to the table, and to work as a cursor to move
58   * in a non-linear fashion through the list.
59   * <p>
60   * The fourth set consists of all other test methods that test special cases of
61   * the implementation of the various methods of HashedList - to obtain 100% code
62   * coverage for HashedList.
63   */
64  public class HashedListTest {
65  
66      static class TestCursorValue implements CursorValue<String> {
67  
68          final String key;
69  
70          final String value;
71  
72          public TestCursorValue(String value) {
73              this.key = value;
74              this.value = value;
75          }
76  
77          public TestCursorValue(String key, String value) {
78              this.key = key;
79              this.value = value;
80          }
81  
82          @Override
83          public boolean equals(Object obj) {
84              if (obj instanceof TestCursorValue) {
85                  return ((TestCursorValue) obj).value.equals(this.value);
86              }
87              return false;
88          }
89  
90          @Override
91          public String getKey() {
92              return this.key;
93          }
94  
95          @Override
96          public int hashCode() {
97              return this.value.hashCode();
98          }
99      }

101     void show(HashedList<?> h, String msg) {
102         Iterator<?> t = h.iterator();
103         System.out.println("\n Looking at list:" + msg);
104         while (t.hasNext()) {
105             System.out.println("Has element:" + t.next());
106         }
107     }

109     @Test
110     public void testCollection() {
111 
112         HashedList<TestCursorValue> h1 = new HashedList<TestCursorValue>();
113         HashedList<TestCursorValue> h2 = new HashedList<TestCursorValue>();
114 
115         // Add a few unkeyed rows.
116 
117         h1.add(new TestCursorValue("Row 1"));
118         h1.add(new TestCursorValue("Row 2"));
119         h1.add(new TestCursorValue("Row 3"));
120 
121         assertEquals("Adding unkeyed rows", 3, h1.size());
122 
123         assertEquals("Has row 1", true, h1.contains(new TestCursorValue("Row 1")));
124         assertEquals("Has row 2", true, h1.contains(new TestCursorValue("Row 2")));
125         h1.remove(new TestCursorValue("Row 2"));
126         assertEquals("Has row 1", true, h1.contains(new TestCursorValue("Row 1")));
127         assertEquals("Has row 2", false, h1.contains(new TestCursorValue("Row 2")));
128 
129         assertEquals("Delete unkeyed rows", 2, h1.size());
130 
131         h1.clear();
132         assertEquals("Cleared unkeyed rows", 0, h1.size());
133 
134         h1.add(new TestCursorValue("key 1", "Row 1"));
135         h1.add(new TestCursorValue("key 2", "Row 2"));
136         h1.add(new TestCursorValue("key 3", "Row 3"));
137 
138         assertEquals("Adding keyed rows", 3, h1.size());
139 
140         assertEquals("Has Row 1", true, h1.contains(new TestCursorValue("Row 1")));
141         assertEquals("Has key 1", true, h1.containsKey("key 1"));
142         assertEquals("Has Row 2", true, h1.contains(new TestCursorValue("Row 2")));
143         assertEquals("Has key 2", true, h1.containsKey("key 2"));
144         assertEquals("Has Row 3", true, h1.contains(new TestCursorValue("Row 3")));
145         assertEquals("Has key 3", true, h1.containsKey("key 3"));
146 
147         h1.removeKey("key 2");
148         assertEquals("Delete keyed row", 2, h1.size());
149         assertEquals("Has Row 1", true, h1.contains(new TestCursorValue("Row 1")));
150         assertEquals("Has key 1", true, h1.containsKey("key 1"));
151         assertEquals("Has Row 2", false, h1.contains(new TestCursorValue("Row 2")));
152         assertEquals("Has key 2", false, h1.containsKey("key 2"));
153         assertEquals("Has Row 3", true, h1.contains(new TestCursorValue("Row 3")));
154         assertEquals("Has key 3", true, h1.containsKey("key 3"));
155 
156         h1.clear();
157         assertEquals("Clear keyed rows", 0, h1.size());
158 
159         h1.add(new TestCursorValue("key 1", "Row 1"));
160         h1.add(new TestCursorValue("key 2", "Row 2"));
161         h1.add(new TestCursorValue("key 3", "Row 3"));
162         assertEquals("Re-Adding keyed rows", 3, h1.size());
163         assertEquals("Has Row 2", true, h1.contains(new TestCursorValue("Row 2")));
164         assertEquals("Has key 2", true, h1.containsKey("key 2"));
165 
166         h2.add(new TestCursorValue("key 4", "Row 4"));
167         h2.add(new TestCursorValue("key 5", "Row 5"));
168 
169         assertEquals("containsAll(beforeAdd)", false, h1.containsAll(h2));
170 
171         h1.addAll(h2);
172 
173         assertEquals("addAll()", 5, h1.size());
174         assertEquals("containsAll(afterAdd)", true, h1.containsAll(h2));
175         assertEquals("has row 4", true, h1.contains(new TestCursorValue("Row 4")));
176         h1.remove(new TestCursorValue("Row 4"));
177         assertEquals("dropped row 4", false, h1.contains(new TestCursorValue("Row 4")));
178         assertEquals("containsAll(afterDrop)", false, h1.containsAll(h2));
179 
180         assertEquals("isEmpty(false)", false, h1.isEmpty());
181         h1.remove(new TestCursorValue("Row 1"));
182         h1.remove(new TestCursorValue("Row 2"));
183         h1.remove(new TestCursorValue("Row 3"));
184         h1.remove(new TestCursorValue("Row 5"));
185         assertEquals("isEmpty(true)", true, h1.isEmpty());
186         h1.add(new TestCursorValue("Row 1"));
187         h1.add(new TestCursorValue("Row 2"));
188         h1.add(new TestCursorValue("Row 3"));
189         h1.addAll(h2);
190         assertEquals("Adding back", 5, h1.size());
191         h1.removeAll(h2);
192 
193         assertEquals("removeAll()", 3, h1.size());
194         h1.addAll(h2);
195 
196         assertEquals("Adding back again", 5, h1.size());
197         h1.retainAll(h2);
198         assertEquals("retainAll()", 2, h1.size());
199 
200         Object[] array1 = h1.toArray();
201         TestCursorValue[] array2 = h1.toArray(new TestCursorValue[h1.size()]);
202         for (int index = 0; index < Math.max(array2.length, array1.length); index++) {
203             Assert.assertSame(array1[index], array2[index]);
204         }
205 
206         Assert.assertTrue(h1.toString().contains(array1[0].toString()));
207     }

209     @Test
210     public void testCursor() {
211         HashedList<TestCursorValue> h1 = new HashedList<TestCursorValue>();
212 
213         h1.add(new TestCursorValue("key 1", "Row 1"));
214         h1.add(new TestCursorValue("Row 3"));
215         h1.add(new TestCursorValue("key 4", "Row 4"));
216         h1.add(new TestCursorValue("Row 5"));
217 
218         Cursor<String, TestCursorValue> j = h1.iterator(0);
219         assertEquals("n1x", true, j.hasNext());
220         assertEquals("n1xv", "Row 1", j.next().value);
221         assertEquals("n1xv", "Row 3", j.next().value);
222 
223         assertEquals("No Row 2", false, h1.containsKey(new TestCursorValue("key 2")));
224         assertEquals("No Row 2", false, h1.contains(new TestCursorValue("Row 2")));
225         j.setKey("key 1");
226         assertEquals("setKey()", "Row 1", j.next().value);
227         j.add(new TestCursorValue("key 2", "Row 2"));
228         assertEquals("has Row 2", true, h1.contains(new TestCursorValue("Row 2")));
229         assertEquals("after add", "Row 3", j.next().value);
230 
231         j.setKey("key 4");
232         assertEquals("setKey(1)", "Row 4", j.next().value);
233         assertEquals("setKey(2)", "Row 5", j.next().value);
234         assertEquals("setKey(3)", false, j.hasNext());
235 
236         j.setKey("key 2");
237         assertEquals("setKey(4)", "Row 2", j.next().value);
238         assertEquals("setKey(5)", "Row 3", j.next().value);
239         j.add(new TestCursorValue("Row 3.5"));
240         j.add(new TestCursorValue("Row 3.6"));
241         assertEquals("After add", 7, h1.size());
242 
243         j = h1.iterator("key 2");
244         j.add(new TestCursorValue("Row 1.5"));
245         j.add(new TestCursorValue("key 1.7", "Row 1.7"));
246         j.add(new TestCursorValue("Row 1.9"));
247         assertEquals("next() after adds", "Row 2", j.next().value);
248         j.setKey("key 1.7");
249         assertEquals("next() after adds", "Row 1.7", j.next().value);
250         assertEquals("prev(1)", "Row 1.7", j.prev().value);
251         assertEquals("prev(2)", "Row 1.5", j.prev().value);
252         assertEquals("prev(3)", true, j.hasPrev());
253         assertEquals("prev(4)", "Row 1", j.prev().value);
254         assertEquals("prev(5)", false, j.hasPrev());
255     }

257     @Test
258     public void testIterator() {
259 
260         HashedList<TestCursorValue> h1 = new HashedList<TestCursorValue>();
261 
262         h1.add(new TestCursorValue("key 4", "Row 4"));
263         h1.add(new TestCursorValue("key 5", "Row 5"));
264 
265         Iterator<TestCursorValue> j = h1.iterator();
266         assertEquals("next1", true, j.hasNext());
267         assertEquals("TestIter1", "Row 4", j.next().value);
268         assertEquals("next2", true, j.hasNext());
269         assertEquals("TestIter2", "Row 5", j.next().value);
270         assertEquals("next3", false, j.hasNext());
271 
272         h1.clear();
273 
274         h1.add(new TestCursorValue("key 1", "Row 1"));
275         h1.add(new TestCursorValue("key 2", "Row 2"));
276         h1.add(new TestCursorValue("Row 3"));
277         h1.add(new TestCursorValue("key 4", "Row 4"));
278         h1.add(new TestCursorValue("Row 5"));
279 
280         assertEquals("Before remove", true, h1.contains(new TestCursorValue("Row 2")));
281         j = h1.iterator();
282         j.next();
283         j.next();
284         j.remove(); // Should get rid of second row
285         assertEquals("After remove", false, h1.contains("Row 2"));
286         assertEquals("n3", true, j.hasNext());
287         assertEquals("n3v", "Row 3", j.next().value);
288         assertEquals("n4", true, j.hasNext());
289         assertEquals("n4v", "Row 4", j.next().value);
290         assertEquals("n5", true, j.hasNext());
291         assertEquals("n5v", "Row 5", j.next().value);
292         assertEquals("n6", false, j.hasNext());
293     }
