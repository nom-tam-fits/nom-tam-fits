40  import java.io.File;

42  import org.junit.After;
43  import org.junit.Test;

45  public class BufferedFileIOTest {
46      
47      @After
48      public void cleanup() {
49          new File("biotest.bin").delete();
50      }
51  
52      @Test(expected = IllegalArgumentException.class)
53      public void testNegativeSeek() throws Exception {
54          BufferedFileIO b = new BufferedFileIO(new File("biotest.bin"), "rw", 256);
55          b.seek(-1);
56      }
57      
58      @Test
59      public void testGetFD() throws Exception {
60          BufferedFileIO b = new BufferedFileIO(new File("biotest.bin"), "rw", 256);
61          assertNotNull(b.getFD());
62      }
63      
64      @Test
65      public void testNotAvailable() throws Exception {
66          BufferedFileIO b = new BufferedFileIO(new File("biotest.bin"), "rw", 256);
67          
68          b.setLength(10);
69          assertEquals("length", 10, b.length());
70          
71          assertTrue(b.hasAvailable(10));
72          assertFalse(b.hasAvailable(11));
73      }
74      
75      @Test
76      public void testTruncate() throws Exception {
77          BufferedFileIO b = new BufferedFileIO(new File("biotest.bin"), "rw", 256);
78   
79          b.setLength(10);
80          assertEquals("length", 10, b.length());
81  
82          b.seek(10);
83          assertEquals("end-pointer", 10, b.getFilePointer());
84          
85          b.setLength(9);
86          assertEquals("truncated-length", 9, b.length());
87          assertEquals("truncated-pointer", 9, b.getFilePointer());
88      }
89      
90      @Test
91      public void testFlushNone() throws Exception {
92          BufferedFileIO b = new BufferedFileIO(new File("biotest.bin"), "rw", 256);
93          b.write(1);
94          b.setLength(0);
95          b.flush();
96      }
97      
98      @Test
99      public void testWriteBeyond() throws Exception {
100         BufferedFileIO b = new BufferedFileIO(new File("biotest.bin"), "rw", 256);
101         b.setLength(10);
102         assertEquals("length0", 10, b.length());
103         
104         b.seek(11);
105         assertEquals("pointer1", 11, b.getFilePointer());
106         b.write(1);
107         assertEquals("pointer1B", 12, b.getFilePointer());
108         assertEquals("length1", 12, b.length());
109         
110         b.seek(20);
111         assertEquals("pointer2", 20, b.getFilePointer());
112         b.write(new byte[40], 0, 40);
113         assertEquals("pointer2B", 60, b.getFilePointer());
114         assertEquals("length2", 60, b.length());
115     }
116     
117     @Test
118     public void testWriteBeyondBuf() throws Exception {
119         BufferedFileIO b = new BufferedFileIO(new File("biotest.bin"), "rw", 16);
120         b.setLength(10);
121         assertEquals("length0", 10, b.length());
122         
123         b.seek(20);
124         assertEquals("pointer1", 20, b.getFilePointer());
125         b.write(1);
126         assertEquals("pointer1B", 21, b.getFilePointer());
127         assertEquals("length1", 21, b.length());
128         
129         b.seek(100);
130         assertEquals("pointer2", 100, b.getFilePointer());
131         b.write(new byte[40], 0, 40);
132         assertEquals("pointer2B", 140, b.getFilePointer());
133         assertEquals("length2", 140, b.length());
134     }
135     
136     @Test
137     public void testReadBeyond() throws Exception {
138         BufferedFileIO b = new BufferedFileIO(new File("biotest.bin"), "rw", 256);
139         b.setLength(10);
140         assertEquals("length0", 10, b.length());
141         
142         b.seek(11);
143         assertEquals("pointer1", 11, b.getFilePointer());
144         assertEquals("read", -1, b.read());
145         
146         assertEquals("pointer2", 11, b.getFilePointer());
147         assertEquals("read", -1, b.read(new byte[40], 0, 40));
148     }
149     
150     @Test(expected = EOFException.class)
151     public void testReadFullyBeyond() throws Exception {
152         BufferedFileIO b = new BufferedFileIO(new File("biotest.bin"), "rw", 256);
153         b.setLength(10);
154         assertEquals("length0", 10, b.length());
155         b.readFully(new byte[40], 0, 40);
156     }
157     
158     @Test
159     public void testSkipBackBuffer() throws Exception {
160         BufferedFileIO b = new BufferedFileIO(new File("biotest.bin"), "rw", 100);
161         b.seek(100);
162         b.write(1);
163         
164         assertEquals("length0", 101, b.length());
165         b.skip(-b.length());
166         assertEquals("beginning", 0, b.getFilePointer());
167     }
168     
169     @Test
170     public void testWriteManySingles() throws Exception {
171         BufferedFileIO b = new BufferedFileIO(new File("biotest.bin"), "rw", 100);
172         for(int i=0; i<300; i++) {
173             b.write(i);
174         }
175     }
176     
177     @Test
178     public void testWriteAhead() throws Exception {
179         BufferedFileIO b = new BufferedFileIO(new File("biotest.bin"), "rw", 100);
180         b.setLength(10);
181         b.seek(200);
182         b.write(1);
183         assertEquals(201, b.length());
184     }
185     
186     @Test
187     public void testWriteAgain() throws Exception {
188         BufferedFileIO b = new BufferedFileIO(new File("biotest.bin"), "rw", 100);
189         b.write(1);
190         b.seek(0);
191         b.write(2);
192         b.seek(0);
193         assertEquals(2, b.read());
194     }
