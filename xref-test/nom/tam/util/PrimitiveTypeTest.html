57      private <T extends Buffer> Buffer bufferAtPosition(ElementType<T> type, int length, int position) {
58          T result = type.newBuffer(length);
59          result.position(position);
60          return type.sliceBuffer(result);
61      }

63      @Test
64      public void testByte() throws Exception {
65          assertSame(PrimitiveTypes.BYTE, PrimitiveTypeHandler.valueOf(8));
66          assertEquals(byte.class, ((byte[]) PrimitiveTypes.BYTE.newArray(5)).getClass().getComponentType());
67          Assert.assertTrue(PrimitiveTypes.BYTE.newBuffer(5) instanceof ByteBuffer);
68          assertEquals(3, bufferAtPosition(PrimitiveTypes.BYTE, 6, 3).capacity());
69  
70          byte expectedValue = 1;
71          ByteBuffer buffer = PrimitiveTypes.BYTE.convertToByteBuffer(new byte[] {1});
72          assertEquals(expectedValue, buffer.get());
73  
74          testGetPutArray(PrimitiveTypes.BYTE, Byte.valueOf((byte) 1), Byte.valueOf((byte) 2));
75  
76          testAppedBuffer(PrimitiveTypes.BYTE, expectedValue);
77      }

79      private <T extends Buffer> void testGetPutArray(ElementType<T> type, Object value, Object other) {
80          Object array = type.newArray(1);
81          Array.set(array, 0, value);
82          T buffer = type.newBuffer(1);
83          type.putArray(buffer, array);
84          Array.set(array, 0, other);
85          buffer.rewind();
86          type.getArray(buffer, array);
87          Assert.assertEquals(value, Array.get(array, 0));
88      }

177     private <T extends Buffer> void testAppedBuffer(ElementType<T> type, Object expectedValue) {
178         Object oneArray = type.newArray(1);
179         Array.set(oneArray, 0, expectedValue);
180         T buffer = type.wrap(oneArray);
181         buffer.rewind();
182         T longerBuffer = type.newBuffer(buffer.remaining() * 10);
183         for (int index = 0; index < 5; index++) {
184             type.appendBuffer(longerBuffer, buffer);
185             buffer.rewind();
186         }
187         longerBuffer.rewind();
188         Object testArray = type.newArray(5);
189         type.getArray(longerBuffer, testArray);
190         for (int index = 0; index < 5; index++) {
191             Assert.assertEquals(expectedValue, Array.get(testArray, index));
192         }
193     }
