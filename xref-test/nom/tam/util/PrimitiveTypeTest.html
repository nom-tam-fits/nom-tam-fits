57      private <T extends Buffer> Buffer bufferAtPosition(ElementType<T> type, int length, int position) {
58          T result = type.newBuffer(length);
59          result.position(position);
60          return type.sliceBuffer(result);
61      }

63      @Test
64      public void testByte() throws Exception {
65          assertSame(PrimitiveTypes.BYTE, PrimitiveTypeHandler.valueOf(8));
66          assertEquals(byte.class, ((byte[]) PrimitiveTypes.BYTE.newArray(5)).getClass().getComponentType());
67          Assert.assertTrue(PrimitiveTypes.BYTE.newBuffer(5) instanceof ByteBuffer);
68          assertEquals(3, bufferAtPosition(PrimitiveTypes.BYTE, 6, 3).capacity());
69  
70          byte expectedValue = 1;
71          ByteBuffer buffer = PrimitiveTypes.BYTE.convertToByteBuffer(new byte[]{
72              1
73          });
74          assertEquals(expectedValue, buffer.get());
75  
76          testGetPutArray(PrimitiveTypes.BYTE, Byte.valueOf((byte) 1), Byte.valueOf((byte) 2));
77  
78          testAppedBuffer(PrimitiveTypes.BYTE, expectedValue);
79      }

81      private <T extends Buffer> void testGetPutArray(ElementType<T> type, Object value, Object other) {
82          Object array = type.newArray(1);
83          Array.set(array, 0, value);
84          T buffer = type.newBuffer(1);
85          type.putArray(buffer, array);
86          Array.set(array, 0, other);
87          buffer.rewind();
88          type.getArray(buffer, array);
89          Assert.assertEquals(value, Array.get(array, 0));
90      }

189     private <T extends Buffer> void testAppedBuffer(ElementType<T> type, Object expectedValue) {
190         Object oneArray = type.newArray(1);
191         Array.set(oneArray, 0, expectedValue);
192         T buffer = type.wrap(oneArray);
193         buffer.rewind();
194         T longerBuffer = type.newBuffer(buffer.remaining() * 10);
195         for (int index = 0; index < 5; index++) {
196             type.appendBuffer(longerBuffer, buffer);
197             buffer.rewind();
198         }
199         longerBuffer.rewind();
200         Object testArray = type.newArray(5);
201         type.getArray(longerBuffer, testArray);
202         for (int index = 0; index < 5; index++) {
203             Assert.assertEquals(expectedValue, Array.get(testArray, index));
204         }
205     }
