38      private TestArrayFuncs(){}

40      /** Compare two double arrays using a given tolerance */
41      public static boolean doubleArrayEquals(double[] x, double[] y, double tol) {
42  
43          for (int i = 0; i < x.length; i += 1) {
44              if (x[i] == 0) {
45                  return y[i] == 0;
46              }
47              if (Math.abs((y[i] - x[i]) / x[i]) > tol) {
48                  return false;
49              }
50          }
51          return true;
52      }

54      /**
55       * Are two objects equal? Arrays have the standard object equals method
56       * which only returns true if the two object are the same. This method
57       * returns true if every element of the arrays match. The inputs may be of
58       * any dimensionality. The dimensionality and dimensions of the arrays must
59       * match as well as any elements. If the elements are non-primitive.
60       * non-tiledImageOperation objects, then the equals method is called for each element. If
61       * both elements are multi-dimensional arrays, then the method recurses.
62       */
63      public static boolean arrayEquals(Object x, Object y) {
64          return arrayEquals(x, y, 0, 0);
65      }

67      /**
68       * Are two objects equal? Arrays have the standard object equals method
69       * which only returns true if the two object are the same. This method
70       * returns true if every element of the arrays match. The inputs may be of
71       * any dimensionality. The dimensionality and dimensions of the arrays must
72       * match as well as any elements. If the elements are non-primitive.
73       * non-tiledImageOperation objects, then the equals method is called for each element. If
74       * both elements are multi-dimensional arrays, then the method recurses.
75       */
76      public static boolean arrayEquals(Object x, Object y, double tolf, double told) {
77  
78          // Handle the special cases first.
79          // We treat null == null so that two object arrays
80          // can match if they have matching null elements.
81          if (x == null && y == null) {
82              return true;
83          }
84  
85          if (x == null || y == null) {
86              return false;
87          }
88  
89          Class<?> xClass = x.getClass();
90          Class<?> yClass = y.getClass();
91  
92          if (xClass != yClass) {
93              return false;
94          }
95  
96          if (!xClass.isArray()) {
97              return x.equals(y);
98  
99          } else {
100             if (xClass.equals(int[].class)) {
101                 return Arrays.equals((int[]) x, (int[]) y);
102 
103             } else if (xClass.equals(double[].class)) {
104                 if (told == 0) {
105                     return Arrays.equals((double[]) x, (double[]) y);
106                 } else {
107                     return doubleArrayEquals((double[]) x, (double[]) y, told);
108                 }
109 
110             } else if (xClass.equals(long[].class)) {
111                 return Arrays.equals((long[]) x, (long[]) y);
112 
113             } else if (xClass.equals(float[].class)) {
114                 if (tolf == 0) {
115                     return Arrays.equals((float[]) x, (float[]) y);
116                 } else {
117                     return floatArrayEquals((float[]) x, (float[]) y, (float) tolf);
118                 }
119 
120             } else if (xClass.equals(byte[].class)) {
121                 return Arrays.equals((byte[]) x, (byte[]) y);
122 
123             } else if (xClass.equals(short[].class)) {
124                 return Arrays.equals((short[]) x, (short[]) y);
125 
126             } else if (xClass.equals(char[].class)) {
127                 return Arrays.equals((char[]) x, (char[]) y);
128 
129             } else if (xClass.equals(boolean[].class)) {
130                 return Arrays.equals((boolean[]) x, (boolean[]) y);
131 
132             } else {
133                 // Non-primitive and multidimensional arrays can be
134                 // cast to Object[]
135                 Object[] xo = (Object[]) x;
136                 Object[] yo = (Object[]) y;
137                 if (xo.length != yo.length) {
138                     return false;
139                 }
140                 for (int i = 0; i < xo.length; i += 1) {
141                     if (!arrayEquals(xo[i], yo[i], tolf, told)) {
142                         return false;
143                     }
144                 }
145 
146                 return true;
147 
148             }
149         }
150     }

166     /**
167      * Just create a simple pattern cycling through valid byte values. We use
168      * bytes because they can be cast to any other numeric type.
169      * 
170      * @param o
171      *            The tiledImageOperation in which the test pattern is to be set.
172      * @param start
173      *            The value for the first element.
174      */
175     public static byte testPattern(Object o, byte start) {
176 
177         int[] dims = ArrayFuncs.getDimensions(o);
178         if (dims.length > 1) {
179             for (int i = 0; i < ((Object[]) o).length; i += 1) {
180                 start = testPattern(((Object[]) o)[i], start);
181             }
182 
183         } else if (dims.length == 1) {
184             for (int i = 0; i < dims[0]; i += 1) {
185                 java.lang.reflect.Array.setByte(o, i, start);
186                 start += 1;
187             }
188         }
189         return start;
190     }

192     /**
193      * Create an tiledImageOperation and populate it with a test pattern.
194      * 
195      * @param baseType
196      *            The base type of the tiledImageOperation. This is expected to be a numeric
197      *            type, but this is not checked.
198      * @param dims
199      *            The desired dimensions.
200      * @return An tiledImageOperation object populated with a simple test pattern.
201      */
202     public static Object generateArray(Class<?> baseType, int[] dims) {
203 
204         // Generate an tiledImageOperation and populate it with a test pattern of
205         // data.
206 
207         Object x = ArrayFuncs.newInstance(baseType, dims);
208         testPattern(x, (byte) 0);
209         return x;
210     }
