41      /** Compare two double arrays using a given tolerance */
42      public static boolean doubleArrayEquals(double[] x, double[] y, double tol) {
43          if (x.length != y.length) {
44              System.err.println("!!! Mismatched sizes: x " + x.length + ", y " + y.length);
45              return false;
46          }
47  
48          for (int i = 0; i < x.length; i++) {
49              if (x[i] == 0) {
50  
51                  if (y[i] != 0) {
52                      System.err.println("!!! Mismatch at " + i + ": x = " + x[i] + ", y = " + y[i]);
53                  }
54                  return y[i] == 0;
55  
56              }
57              if (Math.abs((y[i] - x[i]) / x[i]) > tol) {
58                  System.err.println("!!! Mismatch at " + i + ": x = " + x[i] + ", y = " + y[i]);
59                  return false;
60              }
61          }
62          return true;
63      }

65      /**
66       * Are two objects equal? Arrays have the standard object equals method which only returns true if the two object
67       * are the same. This method returns true if every element of the arrays match. The inputs may be of any
68       * dimensionality. The dimensionality and dimensions of the arrays must match as well as any elements. If the
69       * elements are non-primitive. non-tiledImageOperation objects, then the equals method is called for each element.
70       * If both elements are multi-dimensional arrays, then the method recurses.
71       */
72      public static boolean arrayEquals(Object x, Object y) {
73          return arrayEquals(x, y, 1e-6, 1e-12);
74      }

76      /**
77       * Are two objects equal? Arrays have the standard object equals method which only returns true if the two object
78       * are the same. This method returns true if every element of the arrays match. The inputs may be of any
79       * dimensionality. The dimensionality and dimensions of the arrays must match as well as any elements. If the
80       * elements are non-primitive. non-tiledImageOperation objects, then the equals method is called for each element.
81       * If both elements are multi-dimensional arrays, then the method recurses.
82       */
83      public static boolean arrayEquals(Object x, Object y, double tolf, double told) {
84  
85          // Handle the special cases first.
86          // We treat null == null so that two object arrays
87          // can match if they have matching null elements.
88          if (x == null && y == null) {
89              return true;
90          }
91  
92          if (x == null || y == null) {
93              return false;
94          }
95  
96          Class<?> xClass = x.getClass();
97          Class<?> yClass = y.getClass();
98  
99          if (xClass != yClass) {
100             System.err.println(" Mismatched classes: x " + xClass + ", y " + yClass);
101             return false;
102         }
103 
104         if (!xClass.isArray()) {
105             if (x.equals(y)) {
106                 return true;
107             }
108             System.err.println(" Mismatched Objects: x [" + x + "], y [" + y + "]");
109             return false;
110 
111         }
112         if (xClass.equals(int[].class)) {
113             return Arrays.equals((int[]) x, (int[]) y);
114 
115         }
116         if (xClass.equals(double[].class)) {
117             if (told == 0) {
118                 return Arrays.equals((double[]) x, (double[]) y);
119             }
120             return doubleArrayEquals((double[]) x, (double[]) y, told);
121 
122         }
123         if (xClass.equals(long[].class)) {
124             return Arrays.equals((long[]) x, (long[]) y);
125 
126         }
127         if (xClass.equals(float[].class)) {
128             if (tolf == 0) {
129                 return Arrays.equals((float[]) x, (float[]) y);
130             } else {
131                 return floatArrayEquals((float[]) x, (float[]) y, (float) tolf);
132             }
133 
134         } else if (xClass.equals(byte[].class)) {
135             return Arrays.equals((byte[]) x, (byte[]) y);
136 
137         } else if (xClass.equals(short[].class)) {
138             return Arrays.equals((short[]) x, (short[]) y);
139 
140         } else if (xClass.equals(char[].class)) {
141             return Arrays.equals((char[]) x, (char[]) y);
142 
143         } else if (xClass.equals(boolean[].class)) {
144             return Arrays.equals((boolean[]) x, (boolean[]) y);
145 
146         } else {
147             // Non-primitive and multidimensional arrays can be
148             // cast to Object[]
149             Object[] xo = (Object[]) x;
150             Object[] yo = (Object[]) y;
151 
152             if (xo.length != yo.length) {
153                 System.err.println("!!! Mismatched sizes: x " + xo.length + ", y " + yo.length);
154                 return false;
155             }
156             for (int i = 0; i < xo.length; i++) {
157                 if (!arrayEquals(xo[i], yo[i], tolf, told)) {
158                     System.err.println("   ... differs at index " + i);
159                     return false;
160                 }
161             }
162 
163             return true;
164 
165         }
166     }

168     /** Compare two float arrays using a given tolerance */
169     public static boolean floatArrayEquals(float[] x, float[] y, float tol) {
170         if (x.length != y.length) {
171             System.err.println("!!! Mismatched sizes: x " + x.length + ", y " + y.length);
172             return false;
173         }
174 
175         for (int i = 0; i < x.length; i++) {
176             if (x[i] == 0) {
177                 if (y[i] != 0) {
178                     System.err.println("!!! Mismatch at " + i + ": x = " + x[i] + ", y = " + y[i]);
179                 }
180                 return y[i] == 0;
181             }
182             if (Math.abs((y[i] - x[i]) / x[i]) > tol) {
183                 System.err.println("!!! Mismatch at " + i + ": x = " + x[i] + ", y = " + y[i]);

185             }

190     /**
191      * Just create a simple pattern cycling through valid byte values. We use bytes because they can be cast to any
192      * other numeric type.
193      *
194      * @param o     The tiledImageOperation in which the test pattern is to be set.
195      * @param start The value for the first element.
196      */
197     public static byte testPattern(Object o, byte start) {
198 
199         int[] dims = ArrayFuncs.getDimensions(o);
200         if (dims.length > 1) {
201             for (int i = 0; i < ((Object[]) o).length; i++) {
202                 start = testPattern(((Object[]) o)[i], start);
203             }
204 
205         } else if (dims.length == 1) {
206             for (int i = 0; i < dims[0]; i++) {
207                 java.lang.reflect.Array.setByte(o, i, start);
208                 start++;
209             }
210         }
211         return start;
212     }

214     /**
215      * Create an tiledImageOperation and populate it with a test pattern.
216      *
217      * @param  baseType The base type of the tiledImageOperation. This is expected to be a numeric type, but this is not
218      *                      checked.
219      * @param  dims     The desired dimensions.
220      *
221      * @return          An tiledImageOperation object populated with a simple test pattern.
222      */
223     public static Object generateArray(Class<?> baseType, int[] dims) {
224 
225         // Generate an tiledImageOperation and populate it with a test pattern of
226         // data.
227 
228         Object x = ArrayFuncs.newInstance(baseType, dims);
229         testPattern(x, (byte) 0);
230         return x;
231     }
