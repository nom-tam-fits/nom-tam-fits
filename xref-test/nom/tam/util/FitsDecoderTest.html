116     @Test
117     public void testByteOrder() throws Exception {
118         byte[] data = new byte[100];
119 
120         ByteBuffer b = ByteBuffer.wrap(data);
121         b.putDouble(Math.PI);
122         b.order(ByteOrder.LITTLE_ENDIAN);
123         b.putDouble(Math.PI);
124 
125         double[] d = new double[2];
126         FitsDecoder e = new FitsDecoder(InputReader.from(new ByteArrayInputStream(data)));
127         e.readArray(d);
128 
129         assertEquals("BE", Math.PI, d[0], 1e-12);
130         assertNotEquals("!BE", Math.PI, d[1], 1e-12);
131 
132         e = new FitsDecoder(InputReader.from(new ByteArrayInputStream(data)));
133         e.getInputBuffer().setByteOrder(ByteOrder.LITTLE_ENDIAN);
134         e.read(d, 0, d.length);
135 
136         assertEquals("byteorder", ByteOrder.LITTLE_ENDIAN, e.getInputBuffer().byteOrder());
137         assertNotEquals("!LE", Math.PI, d[0], 1e-12);
138         assertEquals("LE", Math.PI, d[1], 1e-12);
139     }

372     @Test
373     public void testGetMixed() throws Exception {
374         byte[] data = new byte[400];
375         FitsDecoder e = new FitsDecoder(InputReader.from(new ByteArrayInputStream(data)));
376         e.getInputBuffer().loadBytes(400, 1);
377         assertEquals(1, e.getInputBuffer().get(new byte[10], 5, 1));
378 
379         // no view / wrong view (single element)
380         assertEquals(1, e.getInputBuffer().get(new short[10], 5, 1));
381         // creates view (multiple elements)
382         assertEquals(2, e.getInputBuffer().get(new short[10], 5, 2));
383         // uses existing view (multiple elements)
384         assertEquals(2, e.getInputBuffer().get(new short[10], 5, 2));
385         // using view for single element
386         assertEquals(1, e.getInputBuffer().get(new short[10], 5, 1));
387 
388         assertEquals(1, e.getInputBuffer().get(new int[10], 5, 1));
389         assertEquals(2, e.getInputBuffer().get(new int[10], 5, 2));
390         assertEquals(2, e.getInputBuffer().get(new int[10], 5, 2));
391         assertEquals(1, e.getInputBuffer().get(new int[10], 5, 1));
392 
393         assertEquals(1, e.getInputBuffer().get(new long[10], 5, 1));
394         assertEquals(2, e.getInputBuffer().get(new long[10], 5, 2));
395         assertEquals(2, e.getInputBuffer().get(new long[10], 5, 2));
396         assertEquals(1, e.getInputBuffer().get(new long[10], 5, 1));
397 
398         assertEquals(1, e.getInputBuffer().get(new float[10], 5, 1));
399         assertEquals(2, e.getInputBuffer().get(new float[10], 5, 2));
400         assertEquals(2, e.getInputBuffer().get(new float[10], 5, 2));
401         assertEquals(1, e.getInputBuffer().get(new float[10], 5, 1));
402 
403         assertEquals(1, e.getInputBuffer().get(new double[10], 5, 1));
404         assertEquals(2, e.getInputBuffer().get(new double[10], 5, 2));
405         assertEquals(2, e.getInputBuffer().get(new double[10], 5, 2));
406         assertEquals(1, e.getInputBuffer().get(new double[10], 5, 1));
407 
408         assertEquals(1, e.getInputBuffer().get(new short[10], 5, 1));
409         assertEquals(2, e.getInputBuffer().get(new short[10], 5, 2));
410         assertEquals(2, e.getInputBuffer().get(new short[10], 5, 2));
411         assertEquals(1, e.getInputBuffer().get(new short[10], 5, 1));
412     }
