116     @Test
117     public void testByteOrder() throws Exception {   
118         byte[] data = new byte[100];
119        
120         ByteBuffer b = ByteBuffer.wrap(data);
121         b.putDouble(Math.PI);
122         b.order(ByteOrder.LITTLE_ENDIAN);
123         b.putDouble(Math.PI);
124 
125         double[] d = new double[2];
126         FitsDecoder e = new FitsDecoder(InputReader.from(new ByteArrayInputStream(data)));
127         e.readArray(d);
128         
129         assertEquals("BE", Math.PI, d[0], 1e-12);
130         assertNotEquals("!BE", Math.PI, d[1], 1e-12);
131         
132         e = new FitsDecoder(InputReader.from(new ByteArrayInputStream(data)));
133         e.getInputBuffer().setByteOrder(ByteOrder.LITTLE_ENDIAN);
134         e.read(d, 0, d.length);
135         
136         assertEquals("byteorder", ByteOrder.LITTLE_ENDIAN, e.getInputBuffer().byteOrder());
137         assertNotEquals("!LE", Math.PI, d[0], 1e-12);
138         assertEquals("LE", Math.PI, d[1], 1e-12);
139     }

141     @Test
142     public void testBoolean() throws Exception {
143         byte[] data = new byte[] { 'T', 'F', 0, 1, 2};
144         FitsDecoder e = new FitsDecoder(InputReader.from(new ByteArrayInputStream(data)));
145         Boolean[] b = new Boolean[data.length];
146         e.read(b, 0, b.length);
147       
148         assertTrue("T", b[0]);
149         assertFalse("F", b[1]);
150         assertNull("0", b[2]);
151         assertTrue("1", b[3]);      // alternative non-standard 'true' 
152         assertFalse("2", b[4]);     // everything else 'false'
153     }
