100     @Test
101     public void testDouble() throws Exception {
102 
103         for (int i = 0; i < 10; i += 1) {
104             this.buffer[i] = (byte) ' ';
105         }
106         this.bp.setOffset(0);
107         assertEquals("DoubBlank", 0., this.bp.getDouble(10), 0.);
108 
109         double[] dbl = new double[100];
110         for (int i = 6; i < dbl.length; i += 1) {
111             dbl[i] = 2 * (Math.random() - 0.5) * Math.pow(10, 60 * (Math.random() - 0.5));
112         }
113 
114         dbl[0] = Double.MAX_VALUE;
115         dbl[1] = Double.MIN_VALUE;
116         dbl[2] = 0;
117         dbl[3] = Double.NaN;
118         dbl[4] = Double.POSITIVE_INFINITY;
119         dbl[5] = Double.NEGATIVE_INFINITY;
120 
121         this.offset = 0;
122         this.cnt = 0;
123         while (this.cnt < dbl.length) {
124             this.offset = this.bf.format(dbl[this.cnt], this.buffer, this.offset, 25);
125             this.cnt += 1;
126             if (this.cnt % 4 == 0) {
127                 this.offset = this.bf.format("\n", this.buffer, this.offset, 1);
128             } else {
129                 this.offset = this.bf.format(" ", this.buffer, this.offset, 1);
130             }
131         }
132         this.bf.format(99.9, this.bp.getBuffer(), this.offset, 25);
133 
134         this.bp.setOffset(0);
135         for (int i = 0; i < dbl.length; i += 1) {
136 
137             double chk = this.bp.getDouble(25);
138 
139             double dx = Math.abs(chk - dbl[i]);
140             if (dbl[i] != 0) {
141                 dx = dx / Math.abs(dbl[i]);
142             }
143             if (Double.isNaN(dbl[i])) {
144                 assertEquals("Double check:" + i, true, Double.isNaN(chk));
145             } else if (Double.isInfinite(dbl[i])) {
146                 assertEquals("Double check:" + i, dbl[i], chk, 0);
147             } else {
148                 assertEquals("Double check:" + i, 0., dx, 1.e-14);
149             }
150 
151             if ((i + 1) % 4 == 0) {
152                 this.bp.skip(1);
153             }
154         }
155         assertEquals(99.9, this.bp.getDouble(), 1.e-14);
156     }

221     @Test
222     public void testInt() throws Exception {
223 
224         for (int i = 0; i < 10; i += 1) {
225             this.buffer[i] = (byte) ' ';
226         }
227         // this should change nothing
228         this.bp.setBuffer(this.buffer);
229         this.bp.setOffset(0);
230         assertEquals("IntBlank", 0, this.bp.getInt(10));
231 
232         int[] tint = new int[100];
233 
234         tint[0] = Integer.MIN_VALUE;
235         tint[1] = Integer.MAX_VALUE;
236         tint[2] = 0;
237 
238         for (int i = 0; i < tint.length; i += 1) {
239             tint[i] = (int) (Integer.MAX_VALUE * (2 * (Math.random() - .5)));
240         }
241 
242         // Write 100 numbers
243         int colSize = 12;
244         while (this.cnt < tint.length) {
245             this.offset = this.bf.format(tint[this.cnt], this.buffer, this.offset, colSize);
246             this.cnt += 1;
247             if (this.cnt % 8 == 0) {
248                 this.offset = this.bf.format("\n", this.buffer, this.offset, 1);
249             } else {
250                 this.offset = this.bf.format(" ", this.buffer, this.offset, 1);
251             }
252         }
253 
254         this.bf.format(Integer.MIN_VALUE, this.bp.getBuffer(), this.offset, 12);
255         // Now see if we can get them back
256         this.bp.setOffset(0);
257         for (int i = 0; i < tint.length; i += 1) {
258 
259             int chk = this.bp.getInt(colSize);
260 
261             assertEquals("IntegersRA", chk, tint[i]);
262             if ((i + 1) % 8 == 0) {
263                 this.bp.skip(1);
264             }
265         }
266 
267         assertEquals(Integer.MIN_VALUE, this.bp.getInt());
268 
269         this.offset = 0;
270         colSize = 12;
271         this.cnt = 0;
272         this.offset = 0;
273         while (this.cnt < tint.length) {
274             this.offset = this.bf.format(tint[this.cnt], this.buffer, this.offset, colSize);
275             this.offset = this.bf.format(" ", this.buffer, this.offset, 1);
276             this.cnt += 1;
277         }
278         String myStr = new String(this.buffer, 0, this.offset);
279         String[] array = myStr.split(" ");
280 
281         assertEquals("Split size", 100, array.length);
282 
283         for (int i = 0; i < array.length; i += 1) {
284             assertEquals("Parse token", tint[i], Integer.parseInt(array[i]));
285         }
286 
287         int val = 1;
288         Arrays.fill(this.buffer, (byte) ' ');
289 
290         for (int i = 0; i < 10; i += 1) {
291             this.offset = this.bf.format(val, this.buffer, 0, 6);
292             String test = (val + "      ").substring(0, 6);
293             if (i < 6) {
294                 assertEquals("TestTrunc" + i, test, new String(this.buffer, 0, 6));
295             } else {
296                 assertEquals("TestTrunc" + i, "******", new String(this.buffer, 0, 6));
297             }
298             val *= 10;
299         }
300 
301     }

370     @Test
371     public void testBasics() {
372 
373         byte[] data = new byte[20];
374         this.bf.format(true, data);
375         assertArrayEquals(createArray(20, 84), data);
376         Arrays.fill(data, (byte) 0);
377         this.bf.format(99.99d, data);
378         assertArrayEquals(createArray(20, 57, 57, 46, 57, 57, 48, 48, 48, 48, 48, 48, 48, 48, 48, 48, 48, 48, 48, 48), data);
379         Arrays.fill(data, (byte) 0);
380         this.bf.format(99.99f, data);
381         assertArrayEquals(createArray(20, 57, 57, 46, 57, 56, 57, 57, 57, 54, 56), data);
382         Arrays.fill(data, (byte) 0);
383         this.bf.format(99, data);
384         assertArrayEquals(createArray(20, 57, 57), data);
385         Arrays.fill(data, (byte) 0);
386         this.bf.format(99L, data);
387         assertArrayEquals(createArray(20, 57, 57), data);
388 
389         Arrays.fill(data, (byte) 0);
390         this.bf.format(9.9E+3, data);
391         assertArrayEquals(createArray(20, 57, 57, 48, 48, 46, 48, 48, 48, 48, 48, 48, 48, 48, 48, 48, 48, 48, 48, 48), data);
392     }

394     private byte[] createArray(int size, int... values) {
395         byte[] result = new byte[size];
396         for (int index = 0; index < values.length; index++) {
397             result[index] = (byte) values[index];
398         }
399         return result;
400     }
