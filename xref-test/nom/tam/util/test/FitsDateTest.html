98      @Test
99      public void testNow() {
100         String now = FitsDate.getFitsDateString();
101         String now2 = FitsDate.getFitsDateString(new Date());
102         int tryCount = 100;
103         while (tryCount > 0 && !now.regionMatches(0, now2, 0, now.length() - 3)) {
104             now = FitsDate.getFitsDateString();
105             now2 = FitsDate.getFitsDateString(new Date());
106             tryCount--;
107         }
108         Assert.assertTrue(tryCount > 0);
109     }

133     @Test
134     public void equalsAndHashCodeTest() throws FitsException {
135         FitsDate fitsDate = new FitsDate("1997-07-25");
136         FitsDate fitsDate2 = new FitsDate("1997-07-25");
137 
138         assert(fitsDate.equals(fitsDate));
139         assert(fitsDate2.equals(fitsDate) && fitsDate.equals(fitsDate2));
140         assert(fitsDate.hashCode() == fitsDate2.hashCode());
141         assert(!fitsDate.equals(null));
142 
143         fitsDate = new FitsDate("2019-07-12");
144         assert(!fitsDate.equals(fitsDate2));
145         assert(!(fitsDate.hashCode() == fitsDate2.hashCode()));
146 
147         assert(FitsDate.getFitsDateString(new FitsDate("   1997-07-25T10:50:01.999").toDate()).equals(FitsDate.getFitsDateString(new FitsDate("   1997-07-25T10:50:01.999").toDate())));
148     }

150     @Test
151     public void comparableTest() throws FitsException {
152         FitsDate fitsDate1 = new FitsDate("1997-07-25");
153         FitsDate fitsDate2 = new FitsDate("2997-07-25");
154         FitsDate fitsDate3 = new FitsDate("1997-07-25");
155 
156         assert(fitsDate1.compareTo(fitsDate2) < 0);
157         assert(fitsDate2.compareTo(fitsDate1) > 0);
158         assert(fitsDate1.compareTo(fitsDate3) == 0);
159     }
