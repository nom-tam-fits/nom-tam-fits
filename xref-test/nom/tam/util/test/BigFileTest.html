34  import static org.junit.Assert.assertEquals;
35  
36  import java.io.FileInputStream;
37  
38  import nom.tam.util.FitsInputStream;
39  import nom.tam.util.FitsFile;
40  
41  import org.junit.Test;
42  
43  public class BigFileTest {
44  
45      @Test
46      public void test() throws Exception {
47          try {
48              // First create a 3 GB file.
49              String fname = System.getenv("BIGFILETEST");
50              if (fname == null) {
51                  System.out.println("BIGFILETEST environment not set.  Returning without test");
52                  return;
53              }
54              System.out.println("Big file test.  Takes quite a while.");
55              byte[] buf = new byte[100000000]; // 100 MB
56              FitsFile bf = new FitsFile(fname, "rw");
57              byte sample = 13;
58  
59              for (int i = 0; i < 30; i += 1) {
60                  bf.write(buf); // 30 x 100 MB = 3 GB.
61                  if (i == 24) {
62                      bf.write(new byte[]{
63                          sample
64                      });
65                  } // Add a marker.
66              }
67              bf.close();
68  
69              // Now try to skip within the file.
70              bf = new FitsFile(fname, "r");
71              long skip = 2500000000L; // 2.5 G
72  
73              long val1 = bf.getFilePointer();
74              bf.skipAllBytes(skip);
75              val1 = bf.getFilePointer() - val1;
76              long val2 = bf.getFilePointer();
77              int val = bf.read();
78              bf.close();
79  
80              assertEquals("SkipResult", skip, val1);
81              assertEquals("SkipPos", skip, val2);
82              assertEquals("SkipVal", sample, val);
83  
84              FitsInputStream bdis = new FitsInputStream(new FileInputStream(fname));
85              bdis.skipAllBytes(skip);
86              val = bdis.read();
87              bdis.close();
88              assertEquals("SSkipVal", sample, val);
89          } catch (Exception e) {
90              e.printStackTrace(System.err);
91              throw e;
92          }
93      }
94  }
