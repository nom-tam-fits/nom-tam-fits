34  import static org.junit.Assert.assertEquals;
35  
36  import java.io.FileInputStream;
37  
38  import org.junit.Test;
39  
40  import nom.tam.util.FitsFile;
41  import nom.tam.util.FitsInputStream;
42  
43  public class BigFileTest {
44  
45      @Test
46      public void test() throws Exception {
47          try {
48              // First create a 3 GB file.
49              String fname = System.getenv("BIGFILETEST");
50              if (fname == null) {
51                  System.out.println("BIGFILETEST environment not set.  Returning without test");
52                  return;
53              }
54              System.out.println("Big file test.  Takes quite a while.");
55              byte[] buf = new byte[100000000]; // 100 MB
56              FitsFile bf = new FitsFile(fname, "rw");
57              byte sample = 13;
58  
59              for (int i = 0; i < 30; i++) {
60                  bf.write(buf); // 30 x 100 MB = 3 GB.
61                  if (i == 24) {
62                      bf.write(new byte[] {sample});
63                  } // Add a marker.
64              }
65              bf.close();
66  
67              // Now try to skip within the file.
68              bf = new FitsFile(fname, "r");
69              long skip = 2500000000L; // 2.5 G
70  
71              long val1 = bf.getFilePointer();
72              bf.skipAllBytes(skip);
73              val1 = bf.getFilePointer() - val1;
74              long val2 = bf.getFilePointer();
75              int val = bf.read();
76              bf.close();
77  
78              assertEquals("SkipResult", skip, val1);
79              assertEquals("SkipPos", skip, val2);
80              assertEquals("SkipVal", sample, val);
81  
82              FitsInputStream bdis = new FitsInputStream(new FileInputStream(fname));
83              bdis.skipAllBytes(skip);
84              val = bdis.read();
85              bdis.close();
86              assertEquals("SSkipVal", sample, val);
87          } catch (Exception e) {
88              e.printStackTrace(System.err);
89              throw e;
90          }
91      }
92  }
