47      @Test
48      public void testHeapRewriteable() {
49          Assert.assertFalse(new FitsHeap(100).rewriteable());

52      @Test
53      public void testHeapNegativeSize() {
54          IllegalArgumentException actual = null;
55          try {
56              new FitsHeap(-100);
57          } catch (IllegalArgumentException e) {
58              actual = e;
59          }
60          Assert.assertNotNull(actual);
61          Assert.assertTrue(actual.getMessage().toLowerCase().contains("illegal size"));
62      }
63  
64      @Test
65      public void testHeapPositionFailures() {
66          IllegalStateException actual = null;
67          try {
68              new FitsHeap(100).getFileOffset();
69          } catch (IllegalStateException e) {
70              actual = e;
71          }
72          Assert.assertNotNull(actual);
73          Assert.assertTrue(actual.getMessage().toLowerCase().contains("never alone"));
74      }
75  
76      @Test
77      public void testHeapPutManyGigabyte() {
78          FitsException actual = null;
79          try {
80              // fake the gigabytes by filling the fits elements dwith
81              // duplicates;-)
82              long[][][][] data = new long[10][][][];
83              long[][][] data1 = new long[1024][][];
84              long[][] data2 = new long[1024][];
85              long[] data3 = new long[1024];
86              for (int index = 0; index < data.length; index++) {
87                  data[index] = data1;
88                  for (int index2 = 0; index2 < data[0].length; index2++) {
89                      data[index][index2] = data2;
90                      for (int index3 = 0; index3 < data[0][0].length; index3++) {
91                          data[index][index2][index3] = data3;
92                      }
93                  }
94              }
95              new FitsHeap(100).putData(data);
96          } catch (FitsException e) {
97              actual = e;
98          }
99          Assert.assertNotNull(actual);
100         Assert.assertTrue(actual.getMessage().toLowerCase().contains("fits heap > 2 g"));
101     }
102 
103     @Test
104     public void testHeapReadFailures() throws Exception {
105         FitsException actual = null;
106         try {
107             FitsInputStream in = new FitsInputStream(new ByteArrayInputStream(new byte[50]));
108 
109             new FitsHeap(100).read(in);
110         } catch (FitsException e) {
111             actual = e;
112         }
113         Assert.assertNotNull(actual);
114         Assert.assertTrue(actual.getCause() instanceof EOFException);
115 
116         actual = null;
117         try {
118             FitsInputStream in = new FitsInputStream(new ByteArrayInputStream(new byte[50]));
119             in.read(new byte[50]);
120             new FitsHeap(100).read(in);
121         } catch (FitsException e) {
122             actual = e;
123         }
124         Assert.assertNotNull(actual);
125         Assert.assertTrue(actual.getCause() instanceof EOFException);
126     }
127 
128     @Test
129     public void testHeapWriteFailures() throws Exception {
130         FitsException actual = null;
131         try {
132             FitsOutputStream out = new FitsOutputStream(new ByteArrayOutputStream()){
133                 @Override
134                 public synchronized void write(byte[] b, int off, int len) throws IOException {
135                     throw new IOException("testHeapWriteFailures");
136                 }};
137             new FitsHeap(100).write(out);
138         } catch (FitsException e) {
139             actual = e;
140         }
141         Assert.assertNotNull(actual);
142         Assert.assertTrue(actual.getCause().getMessage().equals("testHeapWriteFailures"));
143     }
