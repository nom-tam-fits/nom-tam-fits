59      private static void createCompressedData(int edge, String nr, List<String> types, File fitsFile, File compressedFile,
60              String[] options, String type) throws Exception, IOException {
61          compressedFile.delete();
62          String[] cmdarray = new String[options.length + 3];
63          cmdarray[0] = FPACK;
64          cmdarray[1] = "-w";
65          cmdarray[cmdarray.length - 1] = fitsFile.getAbsolutePath();
66  
67          System.arraycopy(options, 0, cmdarray, 2, options.length);
68          wait(Runtime.getRuntime().exec(cmdarray));
69          File gzip2File = new File("target/compress/test" + edge + "Data" + type + nr + ".fits.fz");
70          gzip2File.delete();
71          compressedFile.renameTo(gzip2File);
72          wait(Runtime.getRuntime().exec(new String[] {FUNPACK, gzip2File.getAbsolutePath()}));
73          new File("target/compress/test" + edge + "Data" + type + nr + ".fits")
74                  .renameTo(new File("target/compress/test" + edge + "Data" + type + nr + ".fits.uncompressed"));
75          types.add(type);
76      }

78      private static void extractCompressedData(int edge, String nr) throws Exception {
79          List<String> types = new ArrayList<>();
80  
81          File fitsFile = new File("target/compress/test" + edge + "Data" + nr + ".fits");
82          File compressedFile = new File("target/compress/test" + edge + "Data" + nr + ".fits.fz");
83  
84          createCompressedData(edge, nr, types, fitsFile, compressedFile, new String[] {"-h", "-s", "0",}, "huf");
85  
86          createCompressedData(edge, nr, types, fitsFile, compressedFile, new String[] {"-h", "-s", "4",}, "4huf");
87  
88          createCompressedData(edge, nr, types, fitsFile, compressedFile, new String[] {"-r"}, "rise");
89  
90          createCompressedData(edge, nr, types, fitsFile, compressedFile, new String[] {"-g2"}, "gzip2");
91  
92          createCompressedData(edge, nr, types, fitsFile, compressedFile, new String[] {"-g1"}, "gzip1");
93  
94          if (nr.equals("8") || nr.equals("16")) {
95              createCompressedData(edge, nr, types, fitsFile, compressedFile, new String[] {"-p",}, "plio");
96          }
97          for (String type : types) {
98              fitsFile = new File("target/compress/test" + edge + "Data" + type + nr + ".fits.fz");
99              if (!fitsFile.exists()) {
100                 System.out.println("ignoring " + fitsFile.getName());
101                 continue;
102             }
103             Fits fits = new Fits(fitsFile);
104             fits.readHDU();
105 
106             BinaryTableHDU hdu2 = (BinaryTableHDU) fits.readHDU();
107             Object element = hdu2.getData().getElement(0, 0);
108 
109             byte[] data;
110             if (element instanceof byte[]) {
111                 data = (byte[]) element;
112             } else {
113                 short[] shorts = (short[]) element;
114                 data = new byte[shorts.length * 2];
115                 ByteBuffer.wrap(data).asShortBuffer().put(shorts);
116             }
117             File compressedDataFile = new File("target/compress/test" + edge + "Data" + nr + "." + type);
118             compressedDataFile.delete();
119             RandomAccessFile file = new RandomAccessFile(compressedDataFile, "rw");
120             file.write(data, 0, data.length);
121             file.close();
122 
123             fits = new Fits(new File("target/compress/test" + edge + "Data" + nr + ".fits"));
124             ImageHDU hdu = (ImageHDU) fits.readHDU();
125             Object dataOrg = hdu.getData().getData();
126 
127             ByteBuffer dataBuffer = getByteData(dataOrg);
128             writeBinData(edge, nr, dataBuffer);
129             {// check uncompressed differes
130                 BasicHDU<?> hdu1;
131                 File uncompressed = new File("target/compress/test" + edge + "Data" + type + nr + ".fits.uncompressed");
132                 if (uncompressed.exists()) {
133                     fits = new Fits(uncompressed);
134                     hdu1 = fits.readHDU();
135                     Object data2Org = hdu1.getData().getData();
136                     ByteBuffer data2Buffer = getByteData(data2Org);
137                     if (notEqual(dataBuffer.array(), data2Buffer.array())) {
138                         file = new RandomAccessFile("target/compress/test" + edge + "Data" + type + nr + ".uncompressed",
139                                 "rw");
140                         file.write(data2Buffer.array(), 0, dataBuffer.position());
141                         file.close();
142                     }
143                     fits.close();
144                     uncompressed.deleteOnExit();
145                 } else {
146                     System.out.println("****************missing uncompress " + uncompressed);
147                 }
148             }
149             fitsFile.deleteOnExit();
150         }
151     }

153     private static ByteBuffer getByteData(Object dataOrg) {
154         ByteBuffer dataBuffer = ByteBuffer.wrap(new byte[1024 * 1024]);
155         if (dataOrg instanceof int[][]) {
156             int[][] intArray = (int[][]) dataOrg;
157             for (int[] element : intArray) {
158                 for (int y = 0; y < intArray[0].length; y++) {
159                     dataBuffer.putInt(element[y]);
160                 }
161             }
162         }
163         if (dataOrg instanceof short[][]) {
164             short[][] intArray = (short[][]) dataOrg;
165             for (short[] element : intArray) {
166                 for (int y = 0; y < intArray[0].length; y++) {
167                     dataBuffer.putShort(element[y]);
168                 }
169             }
170         }
171         if (dataOrg instanceof byte[][]) {
172             byte[][] intArray = (byte[][]) dataOrg;
173             for (byte[] element : intArray) {
174                 for (int y = 0; y < intArray[0].length; y++) {
175                     dataBuffer.put(element[y]);
176                 }
177             }
178         }
179         if (dataOrg instanceof long[][]) {
180             long[][] intArray = (long[][]) dataOrg;
181             for (long[] element : intArray) {
182                 for (int y = 0; y < intArray[0].length; y++) {
183                     dataBuffer.putLong(element[y]);
184                 }
185             }
186         }
187         if (dataOrg instanceof float[][]) {
188             float[][] intArray = (float[][]) dataOrg;
189             for (float[] element : intArray) {
190                 for (int y = 0; y < intArray[0].length; y++) {
191                     dataBuffer.putFloat(element[y]);
192                 }
193             }
194         }
195         if (dataOrg instanceof double[][]) {
196             double[][] intArray = (double[][]) dataOrg;
197             for (double[] element : intArray) {
198                 for (int y = 0; y < intArray[0].length; y++) {
199                     dataBuffer.putDouble(element[y]);
200                 }
201             }
202         }
203         return dataBuffer;
204     }

235     private static boolean notEqual(byte[] data, byte[] data2) {
236         if (data.length != data2.length) {
237             return true;
238         }
239         for (int index = 0; index < data.length; index++) {
240             if (data[index] != data2[index]) {
241                 return true;
242             }
243         }
244         return false;
245     }

247     private static void testDatabyte(int edge) throws FitsException, IOException {
248         int maxValue = Byte.MAX_VALUE / 2;
249         byte[][] image = new byte[edge][edge];
250         for (int x = 0; x < edge; x++) {
251             for (int y = 0; y < edge; y++) {
252                 double distX = (x - edge / 2d) / 50d;
253                 double distY = (y - edge / 2d) / 50d;
254                 double dist = Math.min(Math.sqrt(Math.abs(distX) * Math.abs(distX) + Math.abs(distY) * Math.abs(distY)),
255                         0.999999999999999999999999999999999999);
256                 image[y][x] = (byte) (maxValue - dist * maxValue);
257             }
258         }
259         BasicHDU<?> hdu = FitsFactory.hduFactory(image);
260         Fits fits = new Fits();
261         fits.addHDU(hdu);
262         FitsFile bf = new FitsFile("target/compress/test" + edge + "Data8.fits", "rw");
263         fits.write(bf);
264         bf.flush();
265         bf.close();
266     }

373     private static void wait(Process exec) throws Exception {
374         InputStream in = exec.getErrorStream();
375         BufferedReader reader = new BufferedReader(new InputStreamReader(in));
376         String line;
377         while ((line = reader.readLine()) != null) {
378             System.out.println("ERR:" + line);
379         }
380         in = exec.getInputStream();
381         reader = new BufferedReader(new InputStreamReader(in));
382         while ((line = reader.readLine()) != null) {
383             System.out.println("OUT:" + line);
384         }
385         exec.waitFor();
386     }

388     private static void writeBinData(int edge, String nr, ByteBuffer dataBuffer) throws FileNotFoundException, IOException {
389         File binFileName = new File("target/compress/test" + edge + "Data" + nr + ".bin");

391             RandomAccessFile binfile = new RandomAccessFile(binFileName, "rw");
392             binfile.write(dataBuffer.array(), 0, dataBuffer.position());
393             binfile.close();
394         }
