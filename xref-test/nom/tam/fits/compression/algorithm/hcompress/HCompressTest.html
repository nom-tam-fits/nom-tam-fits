61      @Test
62      public void testHcompressByte() throws Exception {
63          RandomAccessFile file = null;
64          RandomAccessFile expected = null;
65          try {
66              file = new RandomAccessFile("src/test/resources/nom/tam/image/comp/bare/test100Data8.bin", "r");//
67              expected = new RandomAccessFile("src/test/resources/nom/tam/image/comp/hcompress/scale0/test100Data8.huf", "r");//
68  
69              byte[] bytes = new byte[(int) file.length()];
70              file.read(bytes);
71              byte[] expectedBytes = new byte[(int) expected.length()];
72              expected.read(expectedBytes);
73  
74              byte[] byteArray = new byte[bytes.length];
75              ByteBuffer byteBuffer = ByteBuffer.wrap(bytes);
76              byteBuffer.get(byteArray).rewind();
77  
78              ByteBuffer compressed = ByteBuffer.wrap(new byte[byteArray.length]);
79  
80              ByteHCompressor byteHCompress = new ByteHCompressor(
81                      new HCompressorOption().setTileWidth(100).setTileHeight(100).setScale(0));
82              byteHCompress.compress(byteBuffer, compressed);
83  
84              byte[] compressedArray = new byte[compressed.position()];
85              compressed.position(0);
86              compressed.get(compressedArray, 0, compressedArray.length);
87              Assert.assertArrayEquals(expectedBytes, compressedArray);
88  
89              byte[] decompressedArray = new byte[byteArray.length];
90              byteHCompress.decompress(ByteBuffer.wrap(expectedBytes), ByteBuffer.wrap(decompressedArray));
91              Assert.assertArrayEquals(byteArray, decompressedArray);
92          } finally {
93              SafeClose.close(expected);
94              SafeClose.close(file);
95          }
96      }

173     @Test
174     public void testHcompressDouble() throws Exception {
175         RandomAccessFile file = null;
176         try {
177             file = new RandomAccessFile("src/test/resources/nom/tam/image/comp/bare/test100Data-64.bin", "r");
178             HCompressorQuantizeOption quant = new HCompressorQuantizeOption();
179             quant.setDither(true);
180             quant.setSeed(8864L);
181             quant.setQlevel(4);
182             quant.setCheckNull(false);
183             quant.setTileHeight(100);
184             quant.setTileWidth(100);
185             quant.unwrap(HCompressorOption.class).setScale(0);
186             DoubleHCompressor doubleHCompress = new DoubleHCompressor(quant);
187 
188             byte[] bytes = new byte[(int) file.length()];
189             file.read(bytes);
190 
191             double[] doubleArray = new double[bytes.length / 8];
192             DoubleBuffer doubleBuffer = ByteBuffer.wrap(bytes).asDoubleBuffer();
193             doubleBuffer.get(doubleArray).rewind();
194 
195             ByteBuffer compressed = ByteBuffer.wrap(new byte[bytes.length]);
196             doubleHCompress.compress(doubleBuffer, compressed);
197 
198             byte[] compressedArray = new byte[compressed.position()];
199             compressed.rewind();
200             compressed.get(compressedArray, 0, compressedArray.length);
201 
202             double[] decompressedArray = new double[doubleArray.length];
203             doubleHCompress.decompress(ByteBuffer.wrap(compressedArray), DoubleBuffer.wrap(decompressedArray));
204             Assert.assertArrayEquals(doubleArray, decompressedArray, quant.getBScale() * 1.5);
205         } finally {
206             SafeClose.close(file);
207         }
208     }

324     @Test
325     public void testHcompressIntScale4() throws Exception {
326         RandomAccessFile file = null;
327         RandomAccessFile expected = null;
328         RandomAccessFile expectedUncompressed = null;
329         try {
330             file = new RandomAccessFile("src/test/resources/nom/tam/image/comp/bare/test100Data32.bin", "r");//
331             expected = new RandomAccessFile("src/test/resources/nom/tam/image/comp/hcompress/scale4/test100Data32.4huf",
332                     "r");//
333             expectedUncompressed = new RandomAccessFile(
334                     "src/test/resources/nom/tam/image/comp/hcompress/scale4/test100Data4huf32.uncompressed", "r");//
335 
336             byte[] bytes = new byte[(int) file.length()];
337             file.read(bytes);
338             byte[] expectedBytes = new byte[(int) expected.length()];
339             expected.read(expectedBytes);
340 
341             int[] intArray = new int[bytes.length / 4];
342             IntBuffer asIntBuffer = ByteBuffer.wrap(bytes).asIntBuffer();
343             asIntBuffer.get(intArray).rewind();
344 
345             ByteBuffer compressed = ByteBuffer.wrap(new byte[intArray.length * 4]);
346 
347             new IntHCompressor(new HCompressorOption().setTileWidth(100).setTileHeight(100).setScale(0x231a))
348                     .compress(asIntBuffer, compressed);
349 
350             byte[] compressedArray = new byte[expectedBytes.length];
351             compressed.position(0);
352             compressed.get(compressedArray, 0, compressedArray.length);
353             Assert.assertArrayEquals(expectedBytes, compressedArray);
354 
355             expectedUncompressed.read(bytes);
356             asIntBuffer.rewind();
357             asIntBuffer.get(intArray);
358 
359             long[] decompressedLongArray = new long[intArray.length];
360             new HDecompress().decompress(ByteBuffer.wrap(expectedBytes), false, decompressedLongArray);
361             int[] decompressedArray = new int[intArray.length];
362             ArrayFuncs.copyInto(decompressedLongArray, decompressedArray);
363             Assert.assertArrayEquals(intArray, decompressedArray);
364 
365             decompressedLongArray = new long[intArray.length];
366             new HDecompress().decompress(ByteBuffer.wrap(expectedBytes), true, decompressedLongArray);
367             decompressedArray = new int[intArray.length];
368             ArrayFuncs.copyInto(decompressedLongArray, decompressedArray);
369 
370             double sum1 = 0;
371             double sum2 = 0;
372             for (int index = 0; index < intArray.length; index++) {
373                 sum1 += intArray[index];
374                 sum2 += decompressedArray[index];
375             }
376             Assert.assertEquals(sum1 / intArray.length, sum2 / intArray.length, 0.5d);
377         } finally {
378             SafeClose.close(expectedUncompressed);
379             SafeClose.close(expected);
380             SafeClose.close(file);
381         }
382     }

497     @Test
498     public void testOption() throws HeaderCardException {
499         HCompressorOption option = new HCompressorOption() {
500 
501             @Override
502             protected Object clone() throws CloneNotSupportedException {
503                 throw new CloneNotSupportedException("this can not be cloned");
504             }
505         };
506         option.setParameters(new HCompressParameters(option));
507         IllegalStateException expected = null;
508         try {
509             option.copy();
510         } catch (IllegalStateException e) {
511             expected = e;
512         }
513         Assert.assertNotNull(expected);
514 
515         Header header = new Header();
516         header.addValue(Compression.ZNAMEn.n(1).key(), Compression.SCALE, null);
517         header.addValue(Compression.ZVALn.n(1).key(), 1, null);
518         header.addValue(Compression.ZNAMEn.n(2).key(), Compression.SMOOTH, null);
519         header.addValue(Compression.ZVALn.n(2).key(), 1, null);
520         option.getCompressionParameters().getValuesFromHeader(new HeaderAccess(header));
521 
522         Assert.assertTrue(option.isSmooth());
523         Assert.assertEquals(1, option.getScale());
524 
525         Assert.assertNull(option.unwrap(String.class));
526     }
