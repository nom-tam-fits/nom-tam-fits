66      @Test
67      public void testOption() throws HeaderCardException {
68          RiceCompressOption option = new RiceCompressOption() {
69  
70              @Override
71              protected Object clone() throws CloneNotSupportedException {
72                  throw new CloneNotSupportedException("this can not be cloned");
73              }
74          };
75          option.setParameters(new RiceCompressParameters(option));
76          IllegalStateException expected = null;
77          try {
78              option.copy();
79          } catch (IllegalStateException e) {
80              expected = e;
81          }
82          Assert.assertNotNull(expected);
83          Header header = new Header();
84  
85          header.addValue(Compression.ZNAMEn.n(1).key(), Compression.BLOCKSIZE, null);
86          header.addValue(Compression.ZVALn.n(1).key(), 32, null);
87          header.addValue(Compression.ZNAMEn.n(2).key(), Compression.BYTEPIX, null);
88          header.addValue(Compression.ZVALn.n(2).key(), 8, null);
89          option.getCompressionParameters().getValuesFromHeader(new HeaderAccess(header));
90  
91          Assert.assertEquals(32, option.getBlockSize());
92          Assert.assertEquals(8, option.getBytePix());
93  
94          Assert.assertNull(option.unwrap(String.class));
95      }

97      @Test
98      public void testRiceByte() throws Exception {
99          RandomAccessFile file = null;
100         RandomAccessFile expected = null;
101         try {
102             file = new RandomAccessFile("src/test/resources/nom/tam/image/comp/bare/test100Data8.bin", "r");//
103             expected = new RandomAccessFile("src/test/resources/nom/tam/image/comp/rise/test100Data8.rise", "r");//
104 
105             byte[] bytes = new byte[(int) file.length()];
106             file.read(bytes);
107             byte[] expectedBytes = new byte[(int) expected.length()];
108             expected.read(expectedBytes);
109 
110             ByteBuffer compressed = ByteBuffer.wrap(new byte[bytes.length]);
111             ByteRiceCompressor compressor = new ByteRiceCompressor(option.setBytePix(PrimitiveTypes.BYTE.size()));
112             compressor.compress(ByteBuffer.wrap(bytes), compressed);
113 
114             byte[] compressedArray = new byte[compressed.position()];
115             compressed.position(0);
116             compressed.get(compressedArray, 0, compressedArray.length);
117             Assert.assertArrayEquals(expectedBytes, compressedArray);
118 
119             byte[] decompressedArray = new byte[bytes.length];
120             compressed.position(0);
121             compressor.decompress(compressed, ByteBuffer.wrap(decompressedArray));
122             Assert.assertArrayEquals(bytes, decompressedArray);
123         } finally {
124             SafeClose.close(expected);
125             SafeClose.close(file);
126         }
127     }

197     @Test
198     public void testBitBuffer() {
199         byte[] expected = new byte[8];
200         byte[] bytes = new byte[8];
201         BitBuffer bitBuffer = new BitBuffer(ByteBuffer.wrap(bytes));
202         bitBuffer.putInt(99, 0);
203         bitBuffer.putLong(99L, 0);
204         Assert.assertArrayEquals(expected, bytes);
205         bitBuffer.putLong(2L * (Integer.MAX_VALUE), 40);
206         expected = new byte[] {0, -1, -1, -1, -2, 0, 0, 0};
207         Assert.assertArrayEquals(expected, bytes);
208         bytes = new byte[8];
209         bitBuffer = new BitBuffer(ByteBuffer.wrap(bytes));
210         bitBuffer.putLong(3L, 3);
211         expected = new byte[] {96, 0, 0, 0, 0, 0, 0, 0};
212         Assert.assertArrayEquals(expected, bytes);
213

272     @Test
273     public void testRiceCompressFloat() throws Exception {
274         RandomAccessFile file = null;
275         try {
276             file = new RandomAccessFile("src/test/resources/nom/tam/image/comp/bare/test100Data-32.bin", "r");
277             RiceQuantizeCompressOption quant = new RiceQuantizeCompressOption();
278             quant.setDither(true);
279             quant.setSeed(8864L);
280             quant.setQlevel(4);
281             quant.setCheckNull(false);
282             quant.setTileHeight(100);
283             quant.setTileWidth(100);
284             RiceCompressor.FloatRiceCompressor floatCompress = new RiceCompressor.FloatRiceCompressor(quant);
285 
286             byte[] bytes = new byte[(int) file.length()];
287             file.read(bytes);
288 
289             float[] floatArray = new float[bytes.length / 4];
290             FloatBuffer floatBuffer = ByteBuffer.wrap(bytes).asFloatBuffer();
291             floatBuffer.get(floatArray).rewind();
292 
293             ByteBuffer compressed = ByteBuffer.wrap(new byte[bytes.length]);
294             floatCompress.compress(floatBuffer, compressed);
295 
296             byte[] compressedArray = new byte[compressed.position()];
297             compressed.rewind();
298             compressed.get(compressedArray, 0, compressedArray.length);
299 
300             float[] decompressedArray = new float[floatArray.length];
301             floatCompress.decompress(ByteBuffer.wrap(compressedArray), FloatBuffer.wrap(decompressedArray));
302             Assert.assertArrayEquals(floatArray, decompressedArray, (float) (quant.getBScale() * 1.5));
303         } finally {
304             SafeClose.close(file);
305         }
306     }

356     private Object testRoundtrip(Object data) throws Exception {
357         ImageHDU im = (ImageHDU) Fits.makeHDU(data);
358         String fileName = "target/rice-" + ArrayFuncs.getBaseClass(data) + ".fits";
359 
360         CompressedImageHDU c = CompressedImageHDU.fromImageHDU(im);
361 
362         c.setCompressAlgorithm(Compression.ZCMPTYPE_RICE_1);
363         c.compress();
364 
365         Fits cf = new Fits();
366         cf.addHDU(c);
367         cf.write(fileName);
368 
369         cf = new Fits(fileName);
370         c = (CompressedImageHDU) cf.getHDU(1);
371 
372         return c.asImageHDU().getKernel();
373     }
