92      private void checkRequantedValues(QuantizeProcessor quantize, IntBuffer buffer, double[] doubles, QuantizeOption option,
93              boolean check) {
94          double[] output = new double[option.getTileWidth() * option.getTileHeight()];
95          quantize.unquantize(buffer, DoubleBuffer.wrap(output));
96          if (check) {
97              double[] expected = new double[output.length];
98              System.arraycopy(doubles, 0, expected, 0, expected.length);
99              Assert.assertArrayEquals(expected, output, option.getBScale() * 1.5);
100         }
101     }

402     @Test
403     public void testOption() throws HeaderCardException {
404         QuantizeOption option = new QuantizeOption() {
405 
406             @Override
407             protected Object clone() throws CloneNotSupportedException {
408                 throw new CloneNotSupportedException("this can not be cloned");
409             }
410         };
411         option.setParameters(new QuantizeTestParameters(option));
412         IllegalStateException expected = null;
413         try {
414             option.copy();
415         } catch (IllegalStateException e) {
416             expected = e;
417         }
418         Assert.assertNotNull(expected);
419 
420         Header header = new Header();
421         header.addValue(Compression.ZQUANTIZ, Compression.ZQUANTIZ_SUBTRACTIVE_DITHER_2);
422         option.getCompressionParameters().getValuesFromHeader(new HeaderAccess(header));
423         Assert.assertTrue(option.isDither2());
424         Assert.assertTrue(option.isDither());
425         option = new QuantizeOption();
426 
427         Assert.assertFalse(option.isDither2());
428         Assert.assertFalse(option.isDither());
429 
430         option = new QuantizeOption();
431         option.setParameters(new QuantizeTestParameters(option));
432         header = new Header();
433         header.addValue(Compression.ZQUANTIZ, Compression.ZQUANTIZ_SUBTRACTIVE_DITHER_1);
434         option.getCompressionParameters().getValuesFromHeader(new HeaderAccess(header));
435 
436         Assert.assertFalse(option.isDither2());
437         Assert.assertTrue(option.isDither());
438     }
