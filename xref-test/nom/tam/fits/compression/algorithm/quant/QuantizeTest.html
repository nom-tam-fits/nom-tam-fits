90      private void checkRequantedValues(QuantizeProcessor quantize, IntBuffer buffer, double[] doubles, QuantizeOption option, boolean check) {
91          double[] output = new double[option.getTileWidth() * option.getTileHeight()];
92          quantize.unquantize(buffer, DoubleBuffer.wrap(output));
93          if (check) {
94              double[] expected = new double[output.length];
95              System.arraycopy(doubles, 0, expected, 0, expected.length);
96              Assert.assertArrayEquals(expected, output, option.getBScale() * 1.5);
97          }
98      }

104         }

649         header.addValue(Compression.ZQUANTIZ, Compression.ZQUANTIZ_SUBTRACTIVE_DITHER_1);
650         option.getCompressionParameters().getValuesFromHeader(new HeaderAccess(header));
651 
652         Assert.assertFalse(option.isDither2());
