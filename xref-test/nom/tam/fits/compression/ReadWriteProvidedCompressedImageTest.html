120     private void assertArrayEquals(double[] expected, double[] actual, double delta, boolean checkNaN) {
121         Assert.assertEquals(expected.length, actual.length);
122         for (int index = 0; index < actual.length; index++) {
123             double d1 = expected[index];
124             double d2 = actual[index];
125             if (Double.isNaN(d1) || Double.isNaN(d2)) {
126                 if (checkNaN) {
127                     Assert.assertTrue(Double.isNaN(d1) == Double.isNaN(d2)); //
128                 }
129                 Assert.assertTrue(true); // ;-)
130             } else {
131                 Assert.assertEquals(d1, d2, delta);
132             }
133         }
134     }

147     private <T> void assertCompressedToUncompressedImage(String fileName, String unCompfileName, Class<T> clazz, IHDUAsserter<T> reader) throws Exception {
148         Fits f = null;
149         try {
150             f = new Fits(fileName);
151             Fits unFits = new Fits(unCompfileName);
152             ImageHDU hdu = readHDU(unFits, ImageHDU.class);
153             unFits.deleteHDU(0);
154             CompressedImageHDU uncompHdu = readHDU(f, CompressedImageHDU.class);
155             f.deleteHDU(0);
156             while (hdu != null && uncompHdu != null) {
157                 T compressedData = (T) uncompHdu.asImageHDU().getData().getData();
158                 T orgData = (T) hdu.getData().getData();
159                 // show travis something is going on
160                 System.out.println("Asserting image data! " + (assertionCounter++));
161                 reader.assertData(orgData, compressedData);
162                 hdu = readHDU(unFits, ImageHDU.class);
163                 if (hdu != null) {
164                     unFits.deleteHDU(0);
165                 }
166                 uncompHdu = readHDU(f, CompressedImageHDU.class);
167                 if (uncompHdu != null) {
168                     f.deleteHDU(0);
169                 }
170             }
171             Assert.assertFalse(hdu != null || uncompHdu != null);
172         } finally {
173             SafeClose.close(f);
174         }
175     }

194         float[] real = new float[expected[0].length];
195         for (float[] expectedPart : expected) {
196             result.get(real);
197             Assert.assertEquals(expectedPart.length, real.length);
198             for (int subindex = 0; subindex < expectedPart.length; subindex++) {
199                 float expectedFloat = expectedPart[subindex];
200                 float realFloat = real[subindex];
201                 if (!Float.isNaN(expectedFloat) && !Float.isNaN(realFloat)) {
202                     Assert.assertEquals(expectedFloat, realFloat, delta);
203                 }
204             }
205         }

389     private HeaderCard findCompressOption(Header header, String key) {
390         int index = 1;
391         while (true) {
392             HeaderCard card = header.findCard(Compression.ZNAMEn.n(index));
393             if (card.getValue().equals(key)) {
394                 return header.findCard(Compression.ZVALn.n(index));
395             }
396             index++;
397         }
398     }

404     private Object readAll(String fileName, int index) throws Exception {
405         Fits f = null;
406         try {
407             f = new Fits(fileName);
408             BasicHDU<?> hdu = f.getHDU(index);
409             if (hdu instanceof CompressedImageHDU) {
410                 CompressedImageHDU bhdu = (CompressedImageHDU) hdu;
411                 return bhdu.getUncompressedData();
412             }
413             if (hdu instanceof ImageHDU) {
414                 ImageHDU bhdu = (ImageHDU) hdu;
415                 return bhdu.getData().getData();
416             }
417         } finally {
418             SafeClose.close(f);
419         }
420         return null;
421     }

461     private <T> T readHDU(Fits fits, Class<T> clazz) throws FitsException, IOException {
462         BasicHDU<?> result = fits.readHDU();
463         while (result != null && (isEmptyImage(result) || !clazz.isAssignableFrom(result.getClass()))) {
464             result = fits.readHDU();
465         }
466         return (T) result;
467     }

513     private void doTile(String test, Object data, StandardImageTiler t, int x, int y, int nx, int ny) throws Exception {
514         Class<?> baseClass = ArrayFuncs.getBaseClass(data);
515         Object tile = Array.newInstance(baseClass, nx * ny);
516         t.getTile(tile, new int[]{
517             y,
518             x
519         }, new int[]{
520             ny,
521             nx
522         });
523 
524         float sum0 = 0;
525         float sum1 = 0;
526         int length = Array.getLength(tile);
527         for (int i = 0; i < nx; i += 1) {
528             for (int j = 0; j < ny; j += 1) {
529                 int tileOffset = i + j * nx;
530                 if (tileOffset >= length) {
531                     Assert.fail("tile offset wrong");
532                 }
533                 sum0 += ((Number) Array.get(tile, tileOffset)).doubleValue();
534                 sum1 += ((Number) Array.get(Array.get(data, j + y), i + x)).doubleValue();
535             }
536         }
537         assertEquals("Tiler" + test, sum0, sum1, 0);
538     }
