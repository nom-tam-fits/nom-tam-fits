34  import java.nio.Buffer;
35  import java.nio.ByteBuffer;
36  import java.nio.LongBuffer;
37  
38  import nom.tam.fits.compression.algorithm.api.ICompressOption;
39  import nom.tam.fits.compression.algorithm.api.ICompressorControl;
40  import nom.tam.fits.compression.algorithm.gzip2.GZip2Compressor.LongGZip2Compressor;
41  import nom.tam.fits.compression.provider.CompressorProvider;
42  import nom.tam.fits.compression.provider.api.ICompressorProvider;
43  
44  public class TileCompressorAlternativProvider implements ICompressorProvider {
45  
46      public static ICompressorControl createControl(Class<?> clazz) {
47          return new CompressorProvider.TileCompressorControl(clazz);
48      }
49  
50      @Override
51      public ICompressorControl createCompressorControl(String quantAlgorithm, String compressionAlgorithm, Class<?> baseType) {
52          if ("X".equalsIgnoreCase(compressionAlgorithm) && quantAlgorithm == null && baseType.equals(long.class)) {
53              return new ICompressorControl() {
54  
55                  @Override
56                  public boolean compress(Buffer in, ByteBuffer out, ICompressOption options) {
57                      new LongGZip2Compressor().compress((LongBuffer) in, out);
58                      return true;
59                  }
60  
61                  @Override
62                  public void decompress(ByteBuffer in, Buffer out, ICompressOption options) {
63                      new LongGZip2Compressor().decompress(in, (LongBuffer) out);
64                  }
65  
66                  @Override
67                  public ICompressOption option() {
68                      return null;
69                  }
70  
71              };
72          }
73          return null;
74      }
75  }
