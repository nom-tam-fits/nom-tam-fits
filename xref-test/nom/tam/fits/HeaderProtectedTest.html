55      @Test
56      public void testReplaceKey() throws Exception {
57          Header header = new Header();
58          Assert.assertFalse(header.replaceKey("XX", "YY"));
59          header.addValue("XX", "ZZ", null);
60          Assert.assertTrue(header.replaceKey("XX", "YY"));
61  
62          header.addValue("AA", "BB", null);
63          header.addValue("CC", "DD", null);
64          HeaderCardException actual = null;
65          try {
66              header.replaceKey("AA", "CC");
67          } catch (HeaderCardException e) {
68              actual = e;
69          }
70          Assert.assertNotNull(actual);
71      }

101     @Test
102     public void testDataSize() throws Exception {
103         Header header = new Header();
104         // No BITPIX
105         Assert.assertEquals(0L, header.getDataSize());
106         header.addValue(Standard.BITPIX, 32);
107         // No NAXIS
108         Assert.assertEquals(0L, header.getDataSize());
109 
110         header = new Header();
111         header.nullImage();
112         header.write(new FitsOutputStream(new ByteArrayOutputStream(), 80));
113         Assert.assertEquals(0L, header.getDataSize());
114         header.setNaxes(2);
115         header.setNaxis(1, 0);
116         header.setNaxis(2, 2);
117         header.addValue(GROUPS, true);
118         header.write(new FitsOutputStream(new ByteArrayOutputStream(), 80));
119         Assert.assertEquals(FitsUtil.addPadding(2L), header.getDataSize());
120     }
