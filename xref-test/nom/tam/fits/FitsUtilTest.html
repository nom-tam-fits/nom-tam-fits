45      public void singleBooleanToByte() throws Exception {

47          Assert.assertEquals((byte) 'F', (byte) FitsUtil.booleansToBytes(false));
48          Assert.assertEquals((byte) 0, (byte) FitsUtil.booleansToBytes(null));

51      @Test
52      public void singleByteToLogical() throws Exception {
53          Assert.assertTrue((boolean) FitsUtil.bytesToBooleanObjects((byte) 'T'));
54          Assert.assertFalse((boolean) FitsUtil.bytesToBooleanObjects((byte) 'F'));
55          Assert.assertNull(FitsUtil.bytesToBooleanObjects((byte) 0));
56      }

72      @Test
73      public void boolean2DToBytes() throws Exception {
74          boolean[][] bools = new boolean[][] {{true, false}, {false, true}};
75          byte[][] expected = new byte[][] {{'T', 'F'}, {'F', 'T'}};
76          byte[][] got = (byte[][]) FitsUtil.booleansToBytes(bools);
77  
78          Assert.assertEquals(expected.length, got.length);
79  
80          for (int i = 0; i < expected.length; i++) {
81              Assert.assertArrayEquals(expected[i], got[i]);
82          }
83      }

115     @Test
116     public void bitsToBytes() throws Exception {
117         Assert.assertEquals(0x80, FitsIO.BYTE_MASK & FitsUtil.bitsToBytes(new boolean[] {true})[0]);
118         Assert.assertEquals(0x40, FitsIO.BYTE_MASK & FitsUtil.bitsToBytes(new boolean[] {false, true, false})[0]);
119         Assert.assertEquals(0xF0, FitsIO.BYTE_MASK & FitsUtil.bitsToBytes(new boolean[] {true, true, true, true})[0]);
120         Assert.assertEquals(0x90, FitsIO.BYTE_MASK & FitsUtil.bitsToBytes(new boolean[] {true, false, false, true})[0]);
121 
122         boolean[] bits = new boolean[9];
123         bits[8] = true;
124         Assert.assertEquals(0x80, FitsIO.BYTE_MASK & FitsUtil.bitsToBytes(bits)[1]);
125     }
126 
127     @Test
128     public void bytesToBits() throws Exception {
129         Assert.assertArrayEquals(FitsUtil.bytesToBits(new byte[] {(byte) 0x80}, 1), new boolean[] {true});
130         Assert.assertArrayEquals(FitsUtil.bytesToBits(new byte[] {(byte) 0x80}, 2), new boolean[] {true, false});
131         Assert.assertArrayEquals(FitsUtil.bytesToBits(new byte[] {(byte) 0x40}, 3), new boolean[] {false, true, false});
132         Assert.assertArrayEquals(FitsUtil.bytesToBits(new byte[] {(byte) 0xF0}, 4), new boolean[] {true, true, true, true});
133         Assert.assertArrayEquals(FitsUtil.bytesToBits(new byte[] {(byte) 0x90}, 4),
134                 new boolean[] {true, false, false, true});
135 
136         boolean[] bits = new boolean[9];
137         bits[8] = true;
138         Assert.assertArrayEquals(new byte[] {(byte) 0x00, (byte) 0x80}, FitsUtil.bitsToBytes(bits));
139     }
