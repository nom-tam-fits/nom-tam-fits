37  import nom.tam.fits.BinaryTable.ColumnDesc;
38  import nom.tam.fits.header.Standard;
39  
40  @SuppressWarnings("javadoc")
41  public class SubstringTest {
42  
43      @Test
44      public void testParseSimpleSubstring() throws Exception {
45          Header h = new Header();
46          h.addValue(Standard.TFORMn.n(1), "10A2");
47          ColumnDesc c = BinaryTable.getDescriptor(h, 0);
48          Assertions.assertEquals(2, c.getElementWidth());
49          Assertions.assertEquals(5, c.getElementCount());
50      }
51  
52      @Test
53      public void testParseEmptySubstring() throws Exception {
54          Header h = new Header();
55          h.addValue(Standard.TFORMn.n(1), "10A:SSTR");
56          ColumnDesc c = BinaryTable.getDescriptor(h, 0);
57          Assertions.assertEquals(10, c.getElementWidth());
58          Assertions.assertEquals(1, c.getElementCount());
59      }
60  
61      @Test
62      public void testParseSubstringInvalidLength() throws Exception {
63          Header h = new Header();
64          h.addValue(Standard.TFORMn.n(1), "10A:SSTRz");
65          ColumnDesc c = BinaryTable.getDescriptor(h, 0);
66          Assertions.assertEquals(10, c.getElementWidth());
67          Assertions.assertEquals(1, c.getElementCount());
68      }
69  
70      @Test
71      public void testParseSubstringDelimited() throws Exception {
72          Header h = new Header();
73          h.addValue(Standard.TFORMn.n(1), "10A:SSTR2/032");
74          ColumnDesc c = BinaryTable.getDescriptor(h, 0);
75          Assertions.assertEquals(2, c.getElementWidth());
76          Assertions.assertEquals(5, c.getElementCount());
77          Assertions.assertEquals((byte) 32, c.getStringDelimiter());
78      }

85          Assertions.assertEquals(2, c.getElementWidth());
86          Assertions.assertEquals(5, c.getElementCount());

90      @SuppressWarnings("deprecation")
91      @Test
92      public void testParseSubstringSetTFORM() throws Exception {
93          BinaryTable tab = new BinaryTable();
94          tab.addColumn(ColumnDesc.createForDelimitedStringArrays((byte) 32));
95          tab.addRow(new Object[] {new String[] {"abc", "def"}});
96          tab.addRow(new Object[] {new String[] {"1234567890"}});
97  
98          ColumnDesc c = tab.getDescriptor(0);
99          Assertions.assertEquals((byte) 32, c.getStringDelimiter());
100         Assertions.assertEquals(11, c.getStringLength());
101 
102         Header h = new Header();
103         tab.fillHeader(h);
104 
105         Assertions.assertEquals("1PA:SSTR11/032", h.getStringValue(Standard.TFORMn.n(1)));
106     }
107 
108     @Test
109     public void testParseSubstringAndTDIM() throws Exception {
110         Header h = new Header();
111         h.addValue(Standard.TFORMn.n(1), "30A:SSTR5");
112         h.addValue(Standard.TDIMn.n(1), "(10,3)");
113         ColumnDesc c = BinaryTable.getDescriptor(h, 0);
114         Assertions.assertEquals(10, c.getElementWidth());
115         Assertions.assertEquals(3, c.getElementCount());
116         Assertions.assertEquals((byte) 0, c.getStringDelimiter());
117     }
