52  public class BuilderApiTest {
53  
54      public Header createHeader() throws FitsException {
55          byte[][] bimg = new byte[20][20];
56          BasicHDU<?> hdu = Fits.makeHDU(bimg);
57          Header hdr = hdu.getHeader();
58          return hdr;
59      }
60  
61      @Test
62      public void testBasicApi() throws Exception {
63          Header header = createHeader();
64          Date date = new FitsDate("2015-07-11T05:21:25.446").toDate();
65  
66          header.card(DATE_OBS).value(date).comment("observe date")//
67                  .card(INSTRUME).value("The very best")//
68                  .card(ORIGIN).value("private")//
69                  .card(COMMENT).comment("something to comment")//
70                  .card(THEAP).value(2L)//
71                  .card(DATAMIN).value(1)//
72                  .card(DATAMAX).value(2)//
73                  .card(Standard.BSCALE).value(3.0)//
74                  .scale(1)//
75                  .card(TZEROn.n(5)).value(5.55)//
76                  .card(Standard.BZERO).value(6.55)//
77                  .card(Standard.EQUINOX).value(new BigDecimal("5.55"))//
78                  .noScale()//
79                  .card(TZEROn.n(1)).value(1.99999)//
80                  .card(TZEROn.n(2)).value(new BigDecimal("1.99999"))//
81                  .card(AUTHOR).value(true);
82  
83          Assert.assertEquals("2015-07-11T05:21:25.446", header.getStringValue(DATE_OBS));
84          Assert.assertEquals("observe date", header.findCard(DATE_OBS).getComment());
85          Assert.assertEquals("The very best", header.getStringValue(INSTRUME));
86          Assert.assertEquals("private", header.getStringValue(ORIGIN));
87          Assert.assertEquals("something to comment", header.findCard(COMMENT).getComment());
88          Assert.assertEquals(null, header.getStringValue(COMMENT));
89          Assert.assertEquals(2L, header.getLongValue(THEAP));
90          Assert.assertEquals(1, header.getIntValue(DATAMIN));
91          Assert.assertEquals(2.0, header.getFloatValue(DATAMAX), 0.000001);
92          Assert.assertEquals(3.0, header.getDoubleValue(Standard.BSCALE), 0.000001);
93          Assert.assertEquals(6.6, header.getDoubleValue(Standard.BZERO), 0.11);
94          Assert.assertEquals(5.6, header.getDoubleValue(Standard.EQUINOX), 0.11);
95          Assert.assertEquals(5.6, header.getFloatValue(TZEROn.n(5)), 0.11);
96          Assert.assertEquals(1.99999, header.getDoubleValue(TZEROn.n(1)), 0.000001);
97          Assert.assertEquals(1.99999, header.getDoubleValue(TZEROn.n(2)), 0.000001);
98          Assert.assertEquals(true, header.getBooleanValue(AUTHOR));
99  
100         date = new FitsDate("2015-07-12T05:21:25.446").toDate();
101         BasicHDU<?> hdu = Fits.makeHDU(header);
102         hdu.card(DATE_OBS).value(date).comment("observation date")//
103                 .card(INSTRUME).value("The very very best")//
104                 .card(ORIGIN).value("other")//
105                 .card(COMMENT).comment("something else to comment")//
106                 .card(THEAP).value(200L)//
107                 .card(DATAMIN).value(100)//
108                 .card(DATAMAX).value(200)//
109                 .card(Standard.BSCALE).value(300.0)//
110                 .precision(4)//
111                 .card(TZEROn.n(5)).value(50.55)//
112                 .card(Standard.BZERO).value(500.055f)//
113                 .card(Standard.EQUINOX).value(new BigDecimal("500.055"))//
114                 .card(TZEROn.n(3)).value(600.055f)//
115                 .autoPrecision()//
116                 .card(TZEROn.n(1)).value(100.99999d)//
117                 .card(TZEROn.n(2)).value(new BigDecimal("100.99999"))//
118                 .card(TZEROn.n(4)).value(101.999)//
119                 .card(AUTHOR).value(false);
120 
121         Assert.assertEquals("2015-07-12T05:21:25.446", header.getStringValue(DATE_OBS));
122         Assert.assertEquals("observation date", header.findCard(DATE_OBS).getComment());
123         Assert.assertEquals("The very very best", header.getStringValue(INSTRUME));
124         Assert.assertEquals("other", header.getStringValue(ORIGIN));
125         Assert.assertEquals("something else to comment", header.findCard(COMMENT).getComment());
126         Assert.assertEquals(null, header.getStringValue(COMMENT));
127         Assert.assertEquals(200L, header.getLongValue(THEAP));
128         Assert.assertEquals(100, header.getIntValue(DATAMIN));
129         Assert.assertEquals(200.0, header.getFloatValue(DATAMAX), 0.000001);
130         Assert.assertEquals(300.0, header.getDoubleValue(Standard.BSCALE), 0.000001);
131         Assert.assertEquals(500.06f, header.getFloatValue(Standard.BZERO), 0.011f);
132         Assert.assertEquals(500.06, header.getDoubleValue(Standard.EQUINOX), 0.000001);
133         Assert.assertEquals(100.99999, header.getDoubleValue(TZEROn.n(1)), 0.000001);
134         Assert.assertEquals(100.99999, header.getDoubleValue(TZEROn.n(2)), 0.000001);
135         Assert.assertEquals(600.06f, header.getFloatValue(TZEROn.n(3)), 0.011f);
136         Assert.assertEquals(101.999, header.getFloatValue(TZEROn.n(4)), 0.000001);
137         Assert.assertEquals(false, header.getBooleanValue(AUTHOR));
138         Assert.assertEquals(50.55, header.getFloatValue(TZEROn.n(5)), 0.000001);
139 
140     }
