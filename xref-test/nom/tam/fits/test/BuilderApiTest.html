46  import static nom.tam.fits.header.Standard.AUTHOR;
47  import static nom.tam.fits.header.Standard.COMMENT;
48  import static nom.tam.fits.header.Standard.DATAMAX;
49  import static nom.tam.fits.header.Standard.DATAMIN;
50  import static nom.tam.fits.header.Standard.DATE_OBS;
51  import static nom.tam.fits.header.Standard.INSTRUME;
52  import static nom.tam.fits.header.Standard.ORIGIN;
53  import static nom.tam.fits.header.Standard.THEAP;
54  import static nom.tam.fits.header.Standard.TZEROn;
55  
56  public class BuilderApiTest {
57  
58      public Header createHeader() throws FitsException {
59          byte[][] bimg = new byte[20][20];
60          BasicHDU<?> hdu = Fits.makeHDU(bimg);
61          Header hdr = hdu.getHeader();
62          return hdr;
63      }
64  
65      @Test
66      public void testBasicApi() throws Exception {
67          Header header = createHeader();
68          Date date = new FitsDate("2015-07-11T05:21:25.446").toDate();
69  
70          header.card(DATE_OBS).value(date).comment("observe date")//
71                  .card(INSTRUME).value("The very best")//
72                  .card(ORIGIN).value("private")//
73                  .card(COMMENT).comment("something to comment")//
74                  .card(THEAP).value(2L)//
75                  .card(DATAMIN).value(1)//
76                  .card(DATAMAX).value(2)//
77                  .card(Standard.BSCALE).value(3.0)//
78                  .scale(1)//
79                  .card(TZEROn.n(5)).value(5.55)//
80                  .card(Standard.BZERO).value(6.55)//
81                  .card(Standard.EQUINOX).value(new BigDecimal("5.55"))//
82                  .noScale()//
83                  .card(TZEROn.n(1)).value(1.99999)//
84                  .card(TZEROn.n(2)).value(new BigDecimal("1.99999"))//
85                  .card(AUTHOR).value(true);
86  
87          Assert.assertEquals("2015-07-11T05:21:25.446", header.getStringValue(DATE_OBS));
88          Assert.assertEquals("observe date", header.findCard(DATE_OBS).getComment());
89          Assert.assertEquals("The very best", header.getStringValue(INSTRUME));
90          Assert.assertEquals("private", header.getStringValue(ORIGIN));
91          Assert.assertEquals("something to comment", header.findCard(COMMENT).getComment());
92          Assert.assertEquals(null, header.getStringValue(COMMENT));
93          Assert.assertEquals(2L, header.getLongValue(THEAP));
94          Assert.assertEquals(1, header.getIntValue(DATAMIN));
95          Assert.assertEquals(2.0, header.getFloatValue(DATAMAX), 0.000001);
96          Assert.assertEquals(3.0, header.getDoubleValue(Standard.BSCALE), 0.000001);
97          Assert.assertEquals(6.6, header.getDoubleValue(Standard.BZERO), 0.11);
98          Assert.assertEquals(5.6, header.getDoubleValue(Standard.EQUINOX), 0.11);
99          Assert.assertEquals(5.6, header.getFloatValue(TZEROn.n(5)), 0.11);
100         Assert.assertEquals(1.99999, header.getDoubleValue(TZEROn.n(1)), 0.000001);
101         Assert.assertEquals(1.99999, header.getDoubleValue(TZEROn.n(2)), 0.000001);
102         Assert.assertEquals(true, header.getBooleanValue(AUTHOR));
103 
104         date = new FitsDate("2015-07-12T05:21:25.446").toDate();
105         BasicHDU<?> hdu = Fits.makeHDU(header);
106         hdu.card(DATE_OBS).value(date).comment("observation date")//
107                 .card(INSTRUME).value("The very very best")//
108                 .card(ORIGIN).value("other")//
109                 .card(COMMENT).comment("something else to comment")//
110                 .card(THEAP).value(200L)//
111                 .card(DATAMIN).value(100)//
112                 .card(DATAMAX).value(200)//
113                 .card(Standard.BSCALE).value(300.0)//
114                 .precision(4)//
115                 .card(TZEROn.n(5)).value(50.55)//
116                 .card(Standard.BZERO).value(500.055f)//
117                 .card(Standard.EQUINOX).value(new BigDecimal("500.055"))//
118                 .card(TZEROn.n(3)).value(600.055f)//
119                 .autoPrecision()//
120                 .card(TZEROn.n(1)).value(100.99999d)//
121                 .card(TZEROn.n(2)).value(new BigDecimal("100.99999"))//
122                 .card(TZEROn.n(4)).value(101.999)//
123                 .card(AUTHOR).value(false);
124 
125         Assert.assertEquals("2015-07-12T05:21:25.446", header.getStringValue(DATE_OBS));
126         Assert.assertEquals("observation date", header.findCard(DATE_OBS).getComment());
127         Assert.assertEquals("The very very best", header.getStringValue(INSTRUME));
128         Assert.assertEquals("other", header.getStringValue(ORIGIN));
129         Assert.assertEquals("something else to comment", header.findCard(COMMENT).getComment());
130         Assert.assertEquals(null, header.getStringValue(COMMENT));
131         Assert.assertEquals(200L, header.getLongValue(THEAP));
132         Assert.assertEquals(100, header.getIntValue(DATAMIN));
133         Assert.assertEquals(200.0, header.getFloatValue(DATAMAX), 0.000001);
134         Assert.assertEquals(300.0, header.getDoubleValue(Standard.BSCALE), 0.000001);
135         Assert.assertEquals(500.06f, header.getFloatValue(Standard.BZERO), 0.011f);
136         Assert.assertEquals(500.06, header.getDoubleValue(Standard.EQUINOX), 0.000001);
137         Assert.assertEquals(100.99999, header.getDoubleValue(TZEROn.n(1)), 0.000001);
138         Assert.assertEquals(100.99999, header.getDoubleValue(TZEROn.n(2)), 0.000001);
139         Assert.assertEquals(600.06f, header.getFloatValue(TZEROn.n(3)), 0.011f);
140         Assert.assertEquals(101.999, header.getFloatValue(TZEROn.n(4)), 0.000001);
141         Assert.assertEquals(false, header.getBooleanValue(AUTHOR));
142         Assert.assertEquals(50.55, header.getFloatValue(TZEROn.n(5)), 0.000001);
143 
144     }
