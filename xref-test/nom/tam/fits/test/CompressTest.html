51      @Test
52      public void testgz() throws Exception {
53  
54          File fil = new File(".");
55          System.out.println("File is:" + fil.getCanonicalPath());
56          Fits f = new Fits("http://heasarc.gsfc.nasa.gov/FTP/asca/data/rev2/43021000/images/ad43021000gis25670_lo.totsky.gz");
57  
58          BasicHDU h = f.readHDU();
59          int[][] data = (int[][]) h.getKernel();
60          double sum = 0;
61          for (int i = 0; i < data.length; i += 1) {
62              for (int j = 0; j < data[i].length; j += 1) {
63                  sum += data[i][j];
64              }
65          }
66          assertEquals("ZCompress", sum, 296915., 0);
67      }

190     int urlRead(String is, boolean comp, boolean useComp) throws Exception {
191         File fil = new File(is);
192 
193         String path = fil.getCanonicalPath();
194         URL u = new URL("file://" + path);
195 
196         Fits f;
197         if (useComp) {
198             f = new Fits(u, comp);
199         } else {
200             f = new Fits(u);
201         }
202         short[][] data = (short[][]) f.readHDU().getKernel();
203 
204         return total(data);
205     }
206 
207     int streamRead(InputStream is, boolean comp, boolean useComp) throws Exception {
208         Fits f;
209         if (useComp) {
210             f = new Fits(is, comp);
211         } else {
212             f = new Fits(is);
213         }
214         short[][] data = (short[][]) f.readHDU().getKernel();
215         is.close();
216 
217         return total(data);
218     }

232     int stringRead(String is, boolean comp, boolean useComp) throws Exception {
233         Fits f;
234         if (useComp) {
235             f = new Fits(is, comp);
236         } else {
237             f = new Fits(is);
238         }
239         short[][] data = (short[][]) f.readHDU().getKernel();
240 
241         return total(data);
242     }
243 
244     int total(short[][] data) {
245         int total = 0;
246         for (int i = 0; i < data.length; i += 1) {
247             for (int j = 0; j < data[i].length; j += 1) {
248                 total += data[i][j];
249             }
250         }
251         return total;
252     }
