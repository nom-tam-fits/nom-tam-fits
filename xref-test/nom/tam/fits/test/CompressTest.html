44      @Test
45      public void testgz() throws Exception {
46  
47          File fil = new File(".");
48          System.out.println("File is:" + fil.getCanonicalPath());
49          Fits f = new Fits("http://heasarc.gsfc.nasa.gov/FTP/asca/data/rev2/43021000/images/ad43021000gis25670_lo.totsky.gz");
50  
51          BasicHDU h = f.readHDU();
52          int[][] data = (int[][]) h.getKernel();
53          double sum = 0;
54          for (int i = 0; i < data.length; i += 1) {
55              for (int j = 0; j < data[i].length; j += 1) {
56                  sum += data[i][j];
57              }
58          }
59          assertEquals("ZCompress", sum, 296915., 0);
60      }

183     int urlRead(String is, boolean comp, boolean useComp) throws Exception {
184         File fil = new File(is);
185 
186         String path = fil.getCanonicalPath();
187         URL u = new URL("file://" + path);
188 
189         Fits f;
190         if (useComp) {
191             f = new Fits(u, comp);
192         } else {
193             f = new Fits(u);
194         }
195         short[][] data = (short[][]) f.readHDU().getKernel();
196 
197         return total(data);
198     }
199 
200     int streamRead(InputStream is, boolean comp, boolean useComp) throws Exception {
201         Fits f;
202         if (useComp) {
203             f = new Fits(is, comp);
204         } else {
205             f = new Fits(is);
206         }
207         short[][] data = (short[][]) f.readHDU().getKernel();
208         is.close();
209 
210         return total(data);
211     }

237     int total(short[][] data) {
238         int total = 0;
239         for (int i = 0; i < data.length; i += 1) {
240             for (int j = 0; j < data[i].length; j += 1) {
241                 total += data[i][j];
242             }
243         }
244         return total;
245     }
