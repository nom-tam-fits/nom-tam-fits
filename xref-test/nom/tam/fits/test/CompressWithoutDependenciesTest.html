46  import nom.tam.fits.compress.CompressionManager;
47  import nom.tam.util.SafeClose;
48  
49  /**
50   * Test reading .Z and .gz compressed files.
51   */
52  public class CompressWithoutDependenciesTest {
53  
54      @Test
55      public void testWthoutApacheCompression() throws Exception {
56          final List<Object> assertions = new ArrayList<>();
57          Class<?> clazz;
58          Method method;
59          FileInputStream in1 = null;
60          FileInputStream in2 = null;
61          try {
62              try {
63                  clazz = Thread.currentThread().getContextClassLoader().loadClass(CompressionManager.class.getName());
64                  method = clazz.getMethod("decompress", InputStream.class);
65                  in1 = new FileInputStream("src/test/resources/nom/tam/fits/test/test.fits");
66                  in2 = new FileInputStream("src/test/resources/nom/tam/fits/test/test.fits.bz2");
67              } catch (Exception e) {
68                  assertions.add(e);
69                  return;
70              }
71              try {
72                  // first do a normal fits file without compression
73                  method.invoke(clazz, in1);
74                  assertions.add("ok");
75                  // now use the not available compression lib
76                  method.invoke(clazz, in2);
77              } catch (Exception e) {
78                  assertions.add(e);
79              }
80          } finally {
81              SafeClose.close(in1);
82              SafeClose.close(in2);
83          }
84          assertEquals(assertions.get(0), "ok");
85          assertTrue(assertions.get(1) instanceof InvocationTargetException);
86          assertTrue(((InvocationTargetException) assertions.get(1)).getCause() instanceof NoClassDefFoundError);
87      }
