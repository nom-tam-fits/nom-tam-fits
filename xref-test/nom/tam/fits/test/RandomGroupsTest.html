62      @Test
63      public void test() throws Exception {
64  
65          float[][] fa = new float[20][20];
66          float[] pa = new float[3];
67  
68          Object[][] data = new Object[1][2];
69          FitsFile bf = null;
70          try {
71              bf = new FitsFile("target/rg1.fits", "rw");
72  
73              data[0][0] = pa;
74              data[0][1] = fa;
75  
76              // First lets write out the file painfully group by group.
77              BasicHDU<?> hdu = RandomGroupsHDU.createFrom(data);
78              Header hdr = hdu.getHeader();
79              // Change the number of groups
80              hdr.addValue("GCOUNT", 20, "Number of groups");
81              hdr.write(bf);
82  
83              for (int i = 0; i < 20; i++) {
84  
85                  for (int j = 0; j < pa.length; j++) {
86                      pa[j] = i + j;
87                  }
88                  for (int j = 0; j < fa.length; j++) {
89                      fa[j][j] = i * j;
90                  }
91                  // Write a group
92                  bf.writeArray(data);
93              }
94  
95              byte[] padding = new byte[FitsUtil.padding(20 * ArrayFuncs.computeLSize(data))];
96              bf.write(padding);
97  
98              bf.flush();
99          } finally {
100             SafeClose.close(bf);
101         }
102 
103         // Read back the data.
104         Fits f = null;
105         try {
106             f = new Fits("target/rg1.fits");
107             BasicHDU<?>[] hdus = f.read();
108 
109             data = (Object[][]) hdus[0].getKernel();
110 
111             for (int i = 0; i < data.length; i++) {
112 
113                 pa = (float[]) data[i][0];
114                 fa = (float[][]) data[i][1];
115                 for (int j = 0; j < pa.length; j++) {
116                     assertEquals("paramTest:" + i + " " + j, i + j, pa[j], 0);
117                 }
118                 for (int j = 0; j < fa.length; j++) {
119                     assertEquals("dataTest:" + i + " " + j, i * j, fa[j][j], 0);
120                 }
121             }
122         } finally {
123             SafeClose.close(f);
124         }
125 
126         // Now do it in one fell swoop -- but we have to have
127         // all the data in place first.
128         try {
129             f = new Fits();
130             bf = new FitsFile("target/rg2.fits", "rw");
131             // Generate a FITS HDU from the kernel.
132             f.addHDU(RandomGroupsHDU.createFrom(data));
133             f.write(bf);
134 
135             bf.flush();
136         } finally {
137             SafeClose.close(bf);
138             SafeClose.close(f);
139         }
140 
141         try {
142             f = new Fits("target/rg2.fits");
143             BasicHDU<?> groupHDU = f.read()[0];
144             data = (Object[][]) groupHDU.getKernel();
145             for (int i = 0; i < data.length; i++) {
146 
147                 pa = (float[]) data[i][0];
148                 fa = (float[][]) data[i][1];
149                 for (int j = 0; j < pa.length; j++) {
150                     assertEquals("paramTest:" + i + " " + j, i + j, pa[j], 0);
151                 }
152                 for (int j = 0; j < fa.length; j++) {
153                     assertEquals("dataTest:" + i + " " + j, i * j, fa[j][j], 0);
154                 }
155             }
156             ByteArrayOutputStream out = new ByteArrayOutputStream();
157             groupHDU.info(new PrintStream(out));
158             String groupInfo = new String(out.toByteArray());
159             Assert.assertEquals(20, groupHDU.getGroupCount());
160             Assert.assertEquals(3, groupHDU.getParameterCount());
161 
162             Assert.assertTrue(groupInfo.indexOf("Number of groups:20") >= 0);
163             Assert.assertTrue(groupInfo.indexOf("Parameters: float[3]") >= 0);
164             Assert.assertTrue(groupInfo.indexOf("Data:float[20, 20]") >= 0);
165         } finally {
166             SafeClose.close(f);
167         }
168     }

185     @Test
186     public void testReset() throws Exception {
187         float[][] fa = new float[20][20];
188         float[] pa = new float[3];
189         RandomGroupsData groups;
190         FitsFile bf = null;
191         try {
192             bf = new FitsFile("target/testResetData", "rw");
193             Object[][] data = new Object[1][2];
194             data[0][0] = pa;
195             data[0][1] = fa;
196             groups = RandomGroupsHDU.createFrom(data).getData();
197             bf.writeLong(1);
198             groups.write(bf);
199         } finally {
200             SafeClose.close(bf);
201         }
202 
203         // ok now test it
204         try {
205             bf = new FitsFile("target/testResetData", "rw");
206             bf.readLong();
207             groups = new RandomGroupsData();
208             groups.read(bf);
209             Assert.assertEquals(8, groups.getFileOffset());
210             bf.reset();
211             Assert.assertEquals(0, bf.getFilePointer());
212             groups.reset();
213             Assert.assertEquals(8, bf.getFilePointer());
214         } finally {
215             SafeClose.close(bf);
216         }

219     private void testGroupCreationAndRecreationByType(Object fa, Object pa, int bipix, String typeName) throws Exception {
220         Object[][] data = new Object[1][2];
221         data[0][0] = pa;
222         data[0][1] = fa;
223 
224         BasicHDU<?> hdu = RandomGroupsHDU.createFrom(data);
225         Header hdr = hdu.getHeader();
226         Assert.assertEquals(0, hdr.getIntValue("NAXIS1"));
227         Assert.assertEquals(20, hdr.getIntValue("NAXIS2"));
228         Assert.assertEquals(20, hdr.getIntValue("NAXIS3"));
229         Assert.assertEquals(0, hdr.getIntValue("NAXIS4"));
230         Assert.assertEquals(true, hdr.getBooleanValue("GROUPS"));
231         Assert.assertEquals(1, hdr.getIntValue("GCOUNT"));
232         Assert.assertEquals(3, hdr.getIntValue("PCOUNT"));
233         Assert.assertEquals(true, hdr.getBooleanValue("SIMPLE"));
234         Assert.assertEquals(bipix, hdr.getIntValue("BITPIX"));
235 
236         RandomGroupsHDU newHdu = (RandomGroupsHDU) Fits.makeHDU(hdr);
237         RandomGroupsData recreatedData = newHdu.getData();
238         Assert.assertEquals(Bitpix.forValue(bipix).getPrimitiveType(), recreatedData.getElementType());
239         Assert.assertEquals(3, recreatedData.getParameterCount());
240         Assert.assertArrayEquals(new int[] {20, 20}, recreatedData.getDataDims());
241     }

360     @Test
361     public void testRandomGroupsParameterAccess() throws Exception {
362         short[][] img = new short[7][11];
363         short[] parms = new short[] {1, 2, 3, 4, 5, 6};
364         String filename = "target/random-parameters.fits";
365 
366         RandomGroupsData data = new RandomGroupsData(new Object[][] {{parms, img}});
367         RandomGroupsHDU hdu = data.toHDU();
368 
369         hdu.addValue(Standard.PTYPEn.n(1), "Param A"); // --> 1 * 1 + 0 = 1
370         // Default scaling + offset
371 
372         hdu.addValue(Standard.PTYPEn.n(2), "Param A");
373         hdu.addValue(Standard.PSCALn.n(2), 0.5); // --> 0.5 * 2 = 1
374 
375         hdu.addValue(Standard.PTYPEn.n(3), "Param A");
376         hdu.addValue(Standard.PZEROn.n(3), -2); // --> 1 * 3 - 2 = 1
377 
378         hdu.addValue(Standard.PTYPEn.n(4), "Param A");
379         hdu.addValue(Standard.PSCALn.n(4), 0.5);
380         hdu.addValue(Standard.PZEROn.n(4), -1); // --> 0.5 * 4 - 1 = 1
381 
382         hdu.addValue(Standard.PTYPEn.n(5), "Param B"); // --> 1 * 5 + 0 = 5
383 
384         hdu.addValue(Standard.PTYPEn.n(6), "Param B1"); // --> 1 * 16 + 0 = 6
385 
386         try (Fits f = new Fits()) {
387             f.addHDU(hdu);
388             f.write(filename);
389             f.close();
390         }
391 
392         try (Fits f = new Fits(filename)) {
393             hdu = (RandomGroupsHDU) f.readHDU();
394 
395             Set<String> names = hdu.getParameterNames();
396             Assert.assertTrue("contains A", names.contains("Param A"));
397             Assert.assertTrue("contains B", names.contains("Param B"));
398             Assert.assertTrue("contains B1", names.contains("Param B1"));
399             Assert.assertEquals(3, names.size());
400 
401             Assert.assertEquals(4.0, hdu.getParameter("Param A", 0), 1e-12);
402             Assert.assertEquals(5.0, hdu.getParameter("Param B", 0), 1e-12);
403             Assert.assertEquals(6.0, hdu.getParameter("Param B1", 0), 1e-12);
404             Assert.assertTrue("no such parameter", Double.isNaN(hdu.getParameter("blah", 0)));
405 
406             short[][] im = (short[][]) hdu.getData().getImage(0);
407             Assert.assertArrayEquals(img, im);
408             f.close();
409         }
410     }
