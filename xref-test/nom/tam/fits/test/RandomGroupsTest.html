54   * Test random groups formats in FITS data. Write and read random groups data

58      @Test
59      public void test() throws Exception {
60  
61          float[][] fa = new float[20][20];
62          float[] pa = new float[3];
63  
64          Object[][] data = new Object[1][2];
65          BufferedFile bf = null;
66          try {
67              bf = new BufferedFile("target/rg1.fits", "rw");
68  
69              data[0][0] = pa;
70              data[0][1] = fa;
71  
72              // First lets write out the file painfully group by group.
73              BasicHDU<?> hdu = Fits.makeHDU(data);
74              Header hdr = hdu.getHeader();
75              // Change the number of groups
76              hdr.addValue("GCOUNT", 20, "Number of groups");
77              hdr.write(bf);
78  
79              for (int i = 0; i < 20; i += 1) {
80  
81                  for (int j = 0; j < pa.length; j += 1) {
82                      pa[j] = i + j;
83                  }
84                  for (int j = 0; j < fa.length; j += 1) {
85                      fa[j][j] = i * j;
86                  }
87                  // Write a group
88                  bf.writeArray(data);
89              }
90  
91              byte[] padding = new byte[FitsUtil.padding(20 * ArrayFuncs.computeLSize(data))];
92              bf.write(padding);
93  
94              bf.flush();
95          } finally {
96              SafeClose.close(bf);
97          }
98  
99          // Read back the data.
100         Fits f = null;
101         try {
102             f = new Fits("target/rg1.fits");
103             BasicHDU<?>[] hdus = f.read();
104 
105             data = (Object[][]) hdus[0].getKernel();
106 
107             for (int i = 0; i < data.length; i += 1) {
108 
109                 pa = (float[]) data[i][0];
110                 fa = (float[][]) data[i][1];
111                 for (int j = 0; j < pa.length; j += 1) {
112                     assertEquals("paramTest:" + i + " " + j, i + j, pa[j], 0);
113                 }
114                 for (int j = 0; j < fa.length; j += 1) {
115                     assertEquals("dataTest:" + i + " " + j, i * j, fa[j][j], 0);
116                 }
117             }
118         } finally {
119             SafeClose.close(f);
120         }
121 
122         // Now do it in one fell swoop -- but we have to have
123         // all the data in place first.
124         try {
125             f = new Fits();
126             bf = new BufferedFile("target/rg2.fits", "rw");
127             // Generate a FITS HDU from the kernel.
128             f.addHDU(Fits.makeHDU(data));
129             f.write(bf);
130 
131             bf.flush();
132         } finally {
133             SafeClose.close(bf);
134             SafeClose.close(f);
135         }
136 
137         try {
138             f = new Fits("target/rg2.fits");
139             BasicHDU<?> groupHDU = f.read()[0];
140             data = (Object[][]) groupHDU.getKernel();
141             for (int i = 0; i < data.length; i += 1) {
142 
143                 pa = (float[]) data[i][0];
144                 fa = (float[][]) data[i][1];
145                 for (int j = 0; j < pa.length; j += 1) {
146                     assertEquals("paramTest:" + i + " " + j, i + j, pa[j], 0);
147                 }
148                 for (int j = 0; j < fa.length; j += 1) {
149                     assertEquals("dataTest:" + i + " " + j, i * j, fa[j][j], 0);
150                 }
151             }
152             ByteArrayOutputStream out = new ByteArrayOutputStream();
153             groupHDU.info(new PrintStream(out));
154             String groupInfo = new String(out.toByteArray());
155             Assert.assertEquals(20, groupHDU.getGroupCount());
156             Assert.assertEquals(3, groupHDU.getParameterCount());
157 
158             Assert.assertTrue(groupInfo.indexOf("Number of groups:20") >= 0);
159             Assert.assertTrue(groupInfo.indexOf("Parameters: float[3]") >= 0);
160             Assert.assertTrue(groupInfo.indexOf("Data:float[20, 20]") >= 0);
161         } finally {
162             SafeClose.close(f);
163         }
164     }

181     @Test
182     public void testReset() throws Exception {
183         float[][] fa = new float[20][20];
184         float[] pa = new float[3];
185         RandomGroupsData groups;
186         BufferedFile bf = null;
187         try {
188             bf = new BufferedFile("target/testResetData", "rw");
189             Object[][] data = new Object[1][2];
190             data[0][0] = pa;
191             data[0][1] = fa;
192             groups = (RandomGroupsData) Fits.makeHDU(data).getData();
193             bf.writeLong(1);
194             groups.write(bf);
195         } finally {
196             SafeClose.close(bf);
197         }
198 
199         // ok now test it
200         try {
201             bf = new BufferedFile("target/testResetData", "rw");
202             bf.readLong();
203             groups = new RandomGroupsData();
204             groups.read(bf);
205             Assert.assertEquals(8, groups.getFileOffset());
206             bf.reset();
207             Assert.assertEquals(0, bf.getFilePointer());
208             groups.reset();
209             Assert.assertEquals(8, bf.getFilePointer());
210         } finally {
211             SafeClose.close(bf);
212         }
213     }

215     private void testGroupCreationAndRecreationByType(Object fa, Object pa, int bipix, String typeName) throws FitsException {
216         Object[][] data = new Object[1][2];
217         data[0][0] = pa;
218         data[0][1] = fa;
219 
220         BasicHDU<?> hdu = Fits.makeHDU(data);
221         Header hdr = hdu.getHeader();
222         Assert.assertEquals(0, hdr.getIntValue("NAXIS1"));
223         Assert.assertEquals(20, hdr.getIntValue("NAXIS2"));
224         Assert.assertEquals(20, hdr.getIntValue("NAXIS3"));
225         Assert.assertEquals(0, hdr.getIntValue("NAXIS4"));
226         Assert.assertEquals(true, hdr.getBooleanValue("GROUPS"));
227         Assert.assertEquals(1, hdr.getIntValue("GCOUNT"));
228         Assert.assertEquals(3, hdr.getIntValue("PCOUNT"));
229         Assert.assertEquals(true, hdr.getBooleanValue("SIMPLE"));
230         Assert.assertEquals(bipix, hdr.getIntValue("BITPIX"));
231 
232         RandomGroupsHDU newHdu = (RandomGroupsHDU) Fits.makeHDU(hdr);
233         Object[][] recreatedData = (Object[][]) newHdu.getData().getData();
234         Assert.assertEquals(1, recreatedData.length);
235         Assert.assertEquals(2, recreatedData[0].length);
236         Assert.assertEquals(typeName + "[3]", ArrayFuncs.arrayDescription(recreatedData[0][0]));
237         Assert.assertEquals(typeName + "[20, 20]", ArrayFuncs.arrayDescription(recreatedData[0][1]));
238     }
