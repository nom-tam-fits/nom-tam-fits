59      @Test
60      public void test() throws Exception {
61  
62          float[][] fa = new float[20][20];
63          float[] pa = new float[3];
64  
65          Object[][] data = new Object[1][2];
66          FitsFile bf = null;
67          try {
68              bf = new FitsFile("target/rg1.fits", "rw");
69  
70              data[0][0] = pa;
71              data[0][1] = fa;
72  
73              // First lets write out the file painfully group by group.
74              BasicHDU<?> hdu = Fits.makeHDU(data);
75              Header hdr = hdu.getHeader();
76              // Change the number of groups
77              hdr.addValue("GCOUNT", 20, "Number of groups");
78              hdr.write(bf);
79  
80              for (int i = 0; i < 20; i += 1) {
81  
82                  for (int j = 0; j < pa.length; j += 1) {
83                      pa[j] = i + j;
84                  }
85                  for (int j = 0; j < fa.length; j += 1) {
86                      fa[j][j] = i * j;
87                  }
88                  // Write a group
89                  bf.writeArray(data);
90              }
91  
92              byte[] padding = new byte[FitsUtil.padding(20 * ArrayFuncs.computeLSize(data))];
93              bf.write(padding);
94  
95              bf.flush();
96          } finally {
97              SafeClose.close(bf);
98          }
99  
100         // Read back the data.
101         Fits f = null;
102         try {
103             f = new Fits("target/rg1.fits");
104             BasicHDU<?>[] hdus = f.read();
105 
106             data = (Object[][]) hdus[0].getKernel();
107             System.err.println("### [1] " + data.length);
108 
109             for (int i = 0; i < data.length; i += 1) {
110 
111                 pa = (float[]) data[i][0];
112                 fa = (float[][]) data[i][1];
113                 for (int j = 0; j < pa.length; j += 1) {
114                     assertEquals("paramTest:" + i + " " + j, i + j, pa[j], 0);
115                 }
116                 for (int j = 0; j < fa.length; j += 1) {
117                     assertEquals("dataTest:" + i + " " + j, i * j, fa[j][j], 0);
118                 }
119             }
120         } finally {
121             SafeClose.close(f);
122         }
123 
124         // Now do it in one fell swoop -- but we have to have
125         // all the data in place first.
126         try {
127             f = new Fits();
128             bf = new FitsFile("target/rg2.fits", "rw");
129             // Generate a FITS HDU from the kernel.
130             System.err.println("### [2] " + data.length);
131             f.addHDU(Fits.makeHDU(data));
132             f.write(bf);
133 
134             bf.flush();
135         } finally {
136             SafeClose.close(bf);
137             SafeClose.close(f);
138         }
139 
140         try {
141             f = new Fits("target/rg2.fits");
142             BasicHDU<?> groupHDU = f.read()[0];
143             data = (Object[][]) groupHDU.getKernel();
144             for (int i = 0; i < data.length; i += 1) {
145 
146                 pa = (float[]) data[i][0];
147                 fa = (float[][]) data[i][1];
148                 for (int j = 0; j < pa.length; j += 1) {
149                     assertEquals("paramTest:" + i + " " + j, i + j, pa[j], 0);
150                 }
151                 for (int j = 0; j < fa.length; j += 1) {
152                     assertEquals("dataTest:" + i + " " + j, i * j, fa[j][j], 0);
153                 }
154             }
155             ByteArrayOutputStream out = new ByteArrayOutputStream();
156             groupHDU.info(new PrintStream(out));
157             String groupInfo = new String(out.toByteArray());
158             Assert.assertEquals(20, groupHDU.getGroupCount());
159             Assert.assertEquals(3, groupHDU.getParameterCount());
160 
161             Assert.assertTrue(groupInfo.indexOf("Number of groups:20") >= 0);
162             Assert.assertTrue(groupInfo.indexOf("Parameters: float[3]") >= 0);
163             Assert.assertTrue(groupInfo.indexOf("Data:float[20, 20]") >= 0);
164         } finally {
165             SafeClose.close(f);
166         }
167     }

184     @Test
185     public void testReset() throws Exception {
186         float[][] fa = new float[20][20];
187         float[] pa = new float[3];
188         RandomGroupsData groups;
189         FitsFile bf = null;
190         try {
191             bf = new FitsFile("target/testResetData", "rw");
192             Object[][] data = new Object[1][2];
193             data[0][0] = pa;
194             data[0][1] = fa;
195             groups = (RandomGroupsData) Fits.makeHDU(data).getData();
196             bf.writeLong(1);
197             groups.write(bf);
198         } finally {
199             SafeClose.close(bf);
200         }
201 
202         // ok now test it
203         try {
204             bf = new FitsFile("target/testResetData", "rw");
205             bf.readLong();
206             groups = new RandomGroupsData();
207             groups.read(bf);
208             Assert.assertEquals(8, groups.getFileOffset());
209             bf.reset();
210             Assert.assertEquals(0, bf.getFilePointer());
211             groups.reset();
212             Assert.assertEquals(8, bf.getFilePointer());
213         } finally {
214             SafeClose.close(bf);
215         }
216     }

218     private void testGroupCreationAndRecreationByType(Object fa, Object pa, int bipix, String typeName)
219             throws Exception {
220         Object[][] data = new Object[1][2];
221         data[0][0] = pa;
222         data[0][1] = fa;
223 
224         BasicHDU<?> hdu = Fits.makeHDU(data);
225         Header hdr = hdu.getHeader();
226         Assert.assertEquals(0, hdr.getIntValue("NAXIS1"));
227         Assert.assertEquals(20, hdr.getIntValue("NAXIS2"));
228         Assert.assertEquals(20, hdr.getIntValue("NAXIS3"));
229         Assert.assertEquals(0, hdr.getIntValue("NAXIS4"));
230         Assert.assertEquals(true, hdr.getBooleanValue("GROUPS"));
231         Assert.assertEquals(1, hdr.getIntValue("GCOUNT"));
232         Assert.assertEquals(3, hdr.getIntValue("PCOUNT"));
233         Assert.assertEquals(true, hdr.getBooleanValue("SIMPLE"));
234         Assert.assertEquals(bipix, hdr.getIntValue("BITPIX"));
235 
236         RandomGroupsHDU newHdu = (RandomGroupsHDU) Fits.makeHDU(hdr);
237         RandomGroupsData recreatedData = newHdu.getData();
238         Assert.assertEquals(Bitpix.forValue(bipix).getPrimitiveType(), recreatedData.getElementType());
239         Assert.assertEquals(3, recreatedData.getParameterCount());
240         Assert.assertArrayEquals(new int[] {20, 20}, recreatedData.getDataDims());
241     }
