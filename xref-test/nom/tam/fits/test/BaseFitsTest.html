225     private Fits makeAsciiTable() throws Exception {
226         // Create the new ASCII table.
227         Fits f = new Fits();
228         f.addHDU(Fits.makeHDU(getSampleCols(10f)));
229         f.addHDU(Fits.makeHDU(getSampleCols(20f)));
230         f.addHDU(Fits.makeHDU(getSampleCols(30f)));
231         f.addHDU(Fits.makeHDU(getSampleCols(40f)));

234 
235         return new Fits(new File(TARGET_BASIC_FITS_TEST_FITS));

238     private void writeFile(Fits f, String name) throws Exception {
239         FitsFile bf = new FitsFile(name, "rw");
240         f.write(bf);
241         bf.flush();
242         bf.close();
243     }

245     private Object[] getSampleCols(float base) {
246 
247         float[] realCol = new float[50];
248 
249         for (int i = 0; i < realCol.length; i++) {
250             realCol[i] = base * i * i * i + 1;
251         }
252 
253         int[] intCol = (int[]) ArrayFuncs.convertArray(realCol, int.class);
254         long[] longCol = (long[]) ArrayFuncs.convertArray(realCol, long.class);
255         double[] doubleCol = (double[]) ArrayFuncs.convertArray(realCol, double.class);
256 
257         String[] strCol = new String[realCol.length];
258 
259         for (int i = 0; i < realCol.length; i++) {
260             strCol[i] = "ABC" + String.valueOf(realCol[i]) + "CDE";
261         }
262         return new Object[] {realCol, intCol, longCol, doubleCol, strCol};
263     }

804     @Test
805     public void testFitsSettings() throws InterruptedException {
806         boolean longstring = FitsFactory.isLongStringsEnabled();
807         final Boolean[] testFitsSettingsVlaue = new Boolean[6];
808         try {
809             FitsFactory.setLongStringsEnabled(true);
810             new Thread(new Runnable() {
811 
812                 @Override
813                 public void run() {
814                     try {
815                         FitsFactory.useThreadLocalSettings(true);
816                         for (int index = 0; index < testFitsSettingsVlaue.length; index++) {
817                             waitForThread(testFitsSettingsVlaue, 1);
818                             testFitsSettingsVlaue[index++] = FitsFactory.isLongStringsEnabled();
819                             if (testFitsSettingsVlaue[index] != null) {
820                                 FitsFactory.setLongStringsEnabled(testFitsSettingsVlaue[index]);
821                             }
822                         }
823                     } catch (InterruptedException e) {
824                         Arrays.fill(testFitsSettingsVlaue, null);
825                     }
826                 }
827             }).start();
828             testFitsSettingsVlaue[1] = false;
829             waitForThread(testFitsSettingsVlaue, 0);
830             Assert.assertEquals(Boolean.valueOf(true), testFitsSettingsVlaue[0]);
831             Assert.assertEquals(Boolean.valueOf(false), testFitsSettingsVlaue[1]);
832             Assert.assertEquals(true, FitsFactory.isLongStringsEnabled());
833             testFitsSettingsVlaue[3] = true;
834             waitForThread(testFitsSettingsVlaue, 0);
835             Assert.assertEquals(Boolean.valueOf(false), testFitsSettingsVlaue[2]);
836             Assert.assertEquals(Boolean.valueOf(true), testFitsSettingsVlaue[3]);
837             Assert.assertEquals(true, FitsFactory.isLongStringsEnabled());
838             FitsFactory.setLongStringsEnabled(false);
839             testFitsSettingsVlaue[5] = true;
840             waitForThread(testFitsSettingsVlaue, 0);
841             Assert.assertEquals(Boolean.valueOf(false), testFitsSettingsVlaue[2]);
842             Assert.assertEquals(Boolean.valueOf(true), testFitsSettingsVlaue[3]);
843             Assert.assertEquals(false, FitsFactory.isLongStringsEnabled());
844         } finally {
845             FitsFactory.setLongStringsEnabled(longstring);
846         }
847     }

850         int count = 0;
851         while (testFitsSettingsVlaue[index] == null && count < 100) {
852             Thread.sleep(10);
853             count++;
854         }
855     }
