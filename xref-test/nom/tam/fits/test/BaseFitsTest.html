208     private Fits makeAsciiTable() throws Exception {
209         // Create the new ASCII table.
210         try (Fits f = new Fits()) {
211             f.addHDU(Fits.makeHDU(getSampleCols(10f)));
212             f.addHDU(Fits.makeHDU(getSampleCols(20f)));
213             f.addHDU(Fits.makeHDU(getSampleCols(30f)));
214             f.addHDU(Fits.makeHDU(getSampleCols(40f)));
215 
216             writeFile(f, TARGET_BASIC_FITS_TEST_FITS);
217 
218             return new Fits(new File(TARGET_BASIC_FITS_TEST_FITS));
219         }
220     }

222     private void writeFile(Fits f, String name) throws Exception {
223         try (FitsFile bf = new FitsFile(name, "rw")) {
224             f.write(bf);
225             bf.flush();
226             bf.close();
227         }

230     private Object[] getSampleCols(float base) {
231 
232         float[] realCol = new float[50];
233 
234         for (int i = 0; i < realCol.length; i++) {
235             realCol[i] = base * i * i * i + 1;
236         }
237 
238         int[] intCol = (int[]) ArrayFuncs.convertArray(realCol, int.class);
239         long[] longCol = (long[]) ArrayFuncs.convertArray(realCol, long.class);
240         double[] doubleCol = (double[]) ArrayFuncs.convertArray(realCol, double.class);
241 
242         String[] strCol = new String[realCol.length];
243 
244         for (int i = 0; i < realCol.length; i++) {
245             strCol[i] = "ABC" + String.valueOf(realCol[i]) + "CDE";
246         }
247         return new Object[] {realCol, intCol, longCol, doubleCol, strCol};
248     }

721     public void testFitsSettings() throws InterruptedException {
722         boolean longstring = FitsFactory.isLongStringsEnabled();
723         final Boolean[] testFitsSettingsVlaue = new Boolean[6];
724         try {
725             FitsFactory.setLongStringsEnabled(true);
726             new Thread(new Runnable() {
727 
728                 @Override
729                 public void run() {
730                     try {
731                         FitsFactory.useThreadLocalSettings(true);
732                         for (int index = 0; index < testFitsSettingsVlaue.length; index++) {
733                             waitForThread(testFitsSettingsVlaue, 1);
734                             testFitsSettingsVlaue[index++] = FitsFactory.isLongStringsEnabled();
735                             if (testFitsSettingsVlaue[index] != null) {
736                                 FitsFactory.setLongStringsEnabled(testFitsSettingsVlaue[index]);
737                             }
738                         }
739                     } catch (InterruptedException e) {
740                         Arrays.fill(testFitsSettingsVlaue, null);
741                     }
742                 }
743             }).start();
744             testFitsSettingsVlaue[1] = false;
745             waitForThread(testFitsSettingsVlaue, 0);
746             Assertions.assertEquals(Boolean.valueOf(true), testFitsSettingsVlaue[0]);
747             Assertions.assertEquals(Boolean.valueOf(false), testFitsSettingsVlaue[1]);
748             Assertions.assertTrue(FitsFactory.isLongStringsEnabled());
749             testFitsSettingsVlaue[3] = true;
750             waitForThread(testFitsSettingsVlaue, 0);
751             Assertions.assertEquals(Boolean.valueOf(false), testFitsSettingsVlaue[2]);
752             Assertions.assertEquals(Boolean.valueOf(true), testFitsSettingsVlaue[3]);
753             Assertions.assertTrue(FitsFactory.isLongStringsEnabled());
754             FitsFactory.setLongStringsEnabled(false);
755             testFitsSettingsVlaue[5] = true;
756             waitForThread(testFitsSettingsVlaue, 0);
757             Assertions.assertEquals(Boolean.valueOf(false), testFitsSettingsVlaue[2]);
758             Assertions.assertEquals(Boolean.valueOf(true), testFitsSettingsVlaue[3]);
759             Assertions.assertFalse(FitsFactory.isLongStringsEnabled());
760         } finally {
761             FitsFactory.setLongStringsEnabled(longstring);
762         }
763     }

765     private void waitForThread(final Boolean[] testFitsSettingsVlaue, int index) throws InterruptedException {
766         int count = 0;
767         while (testFitsSettingsVlaue[index] == null && count < 100) {
768             Thread.sleep(10);
769             count++;
770         }
771     }
