224     private Fits makeAsciiTable() throws Exception {
225         // Create the new ASCII table.
226         Fits f = new Fits();
227         f.addHDU(Fits.makeHDU(getSampleCols(10f)));
228         f.addHDU(Fits.makeHDU(getSampleCols(20f)));
229         f.addHDU(Fits.makeHDU(getSampleCols(30f)));
230         f.addHDU(Fits.makeHDU(getSampleCols(40f)));
231 
232         writeFile(f, TARGET_BASIC_FITS_TEST_FITS);
233 
234         return new Fits(new File(TARGET_BASIC_FITS_TEST_FITS));
235     }

237     private void writeFile(Fits f, String name) throws Exception {
238         FitsFile bf = new FitsFile(name, "rw");
239         f.write(bf);
240         bf.flush();
241         bf.close();
242     }

244     private Object[] getSampleCols(float base) {
245 
246         float[] realCol = new float[50];
247 
248         for (int i = 0; i < realCol.length; i += 1) {
249             realCol[i] = base * i * i * i + 1;
250         }
251 
252         int[] intCol = (int[]) ArrayFuncs.convertArray(realCol, int.class);
253         long[] longCol = (long[]) ArrayFuncs.convertArray(realCol, long.class);
254         double[] doubleCol = (double[]) ArrayFuncs.convertArray(realCol, double.class);
255 
256         String[] strCol = new String[realCol.length];
257 
258         for (int i = 0; i < realCol.length; i += 1) {
259             strCol[i] = "ABC" + String.valueOf(realCol[i]) + "CDE";
260         }
261         return new Object[]{
262             realCol,
263             intCol,
264             longCol,
265             doubleCol,
266             strCol
267         };
268     }

335     @Test
336     public void testFitsUndefinedHdu3() throws Exception {
337         Fits fits1 = makeAsciiTable();
338         fits1.read();
339         byte[] undefinedData = new byte[1000];
340         for (int index = 0; index < undefinedData.length; index++) {
341             undefinedData[index] = (byte) index;
342         }
343         Data data = UndefinedHDU.encapsulate(undefinedData);
344         Header header = new Header(data);
345 
346         Cursor<String, HeaderCard> iter = header.iterator();
347 
348         String[] headers = new String[header.getNumberOfCards() - 1];
349         int index = 0;
350         while (iter.hasNext()) {
351             HeaderCard headerCard = iter.next();
352             // the EXTEND key will be deleted later on because the header is no
353             // primary header so don't use it
354             if (!headerCard.getKey().equals("EXTEND")) {
355                 headers[index++] = headerCard.toString();
356             }
357         }
358         Header newHeader = new Header(headers);
359         for (index = 0; index < headers.length; index++) {
360             Assert.assertEquals(header.getCard(index), newHeader.getCard(index));
361         }
362 
363         fits1.addHDU(FitsFactory.hduFactory(newHeader, data));
364         FitsOutputStream os = new FitsOutputStream(new FileOutputStream("target/UndefindedHDU3.fits"));
365         fits1.write(os);
366         os.close();
367 
368         Fits fits2 = new Fits("target/UndefindedHDU3.fits");
369         BasicHDU[] hdus = fits2.read();
370 
371         for (index = 0; index < headers.length; index++) {
372             Assert.assertEquals(header.getCard(index), hdus[5].getHeader().getCard(index));
373         }
374 
375     }

827     @Test
828     public void testFitsSettings() throws InterruptedException {
829         boolean longstring = FitsFactory.isLongStringsEnabled();
830         final Boolean[] testFitsSettingsVlaue = new Boolean[6];
831         try {
832             FitsFactory.setLongStringsEnabled(true);
833             new Thread(new Runnable() {
834 
835                 @Override
836                 public void run() {
837                     try {
838                         FitsFactory.useThreadLocalSettings(true);
839                         for (int index = 0; index < testFitsSettingsVlaue.length; index++) {
840                             waitForThread(testFitsSettingsVlaue, 1);
841                             testFitsSettingsVlaue[index++] = FitsFactory.isLongStringsEnabled();
842                             if (testFitsSettingsVlaue[index] != null) {
843                                 FitsFactory.setLongStringsEnabled(testFitsSettingsVlaue[index]);
844                             }
845                         }
846                     } catch (InterruptedException e) {
847                         Arrays.fill(testFitsSettingsVlaue, null);
848                     }
849                 }
850             }).start();
851             testFitsSettingsVlaue[1] = false;
852             waitForThread(testFitsSettingsVlaue, 0);
853             Assert.assertEquals(Boolean.valueOf(true), testFitsSettingsVlaue[0]);
854             Assert.assertEquals(Boolean.valueOf(false), testFitsSettingsVlaue[1]);
855             Assert.assertEquals(true, FitsFactory.isLongStringsEnabled());
856             testFitsSettingsVlaue[3] = true;
857             waitForThread(testFitsSettingsVlaue, 0);
858             Assert.assertEquals(Boolean.valueOf(false), testFitsSettingsVlaue[2]);
859             Assert.assertEquals(Boolean.valueOf(true), testFitsSettingsVlaue[3]);
860             Assert.assertEquals(true, FitsFactory.isLongStringsEnabled());
861             FitsFactory.setLongStringsEnabled(false);
862             testFitsSettingsVlaue[5] = true;
863             waitForThread(testFitsSettingsVlaue, 0);
864             Assert.assertEquals(Boolean.valueOf(false), testFitsSettingsVlaue[2]);
865             Assert.assertEquals(Boolean.valueOf(true), testFitsSettingsVlaue[3]);
866             Assert.assertEquals(false, FitsFactory.isLongStringsEnabled());
867         } finally {
868             FitsFactory.setLongStringsEnabled(longstring);
869         }
870     }
871 
872     private void waitForThread(final Boolean[] testFitsSettingsVlaue, int index) throws InterruptedException {
873         int count = 0;
874         while (testFitsSettingsVlaue[index] == null && count < 100) {
875             Thread.sleep(10);
876             count++;
877         }
878     }
