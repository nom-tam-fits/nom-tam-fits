50  /**
51   * This class tests the ImageTiler. It first creates a FITS file and then reads
52   * it back and allows the user to select tiles. The values of the corner and
53   * center pixels for the selected tile are displayed. Both file and memory tiles
54   * are checked.
55   */

58      private boolean doTile(String test, Object data, StandardImageTiler t, int x, int y, int nx, int ny) throws Exception {
59  
60          Class<?> baseClass = ArrayFuncs.getBaseClass(data);
61          Object tile = Array.newInstance(baseClass, nx * ny);
62          t.getTile(tile, new int[]{
63              y,
64              x
65          }, new int[]{
66              ny,
67              nx
68          });
69  
70          float sum0 = 0;
71          float sum1 = 0;
72          int length = Array.getLength(tile);
73          for (int i = 0; i < nx; i += 1) {
74              for (int j = 0; j < ny; j += 1) {
75                  int tileOffset = i + j * nx;
76                  if (tileOffset >= length) {
77                      return false;
78                  }
79                  sum0 += ((Number) Array.get(tile, tileOffset)).doubleValue();
80                  try {
81                      sum1 += ((Number) Array.get(Array.get(data, j + y), i + x)).doubleValue();
82                  } catch (ArrayIndexOutOfBoundsException e) {
83                      return false;
84                  }
85              }
86          }
87  
88          assertEquals("Tiler" + test, sum0, sum1, 0);
89  
90          return true;
91      }

194     private void doTest(Object data, String suffix) throws IOException, FitsException, Exception {
195         Fits f = null;
196         FitsFile bf = null;
197         try {
198             f = new Fits();
199             bf = new FitsFile("target/tiler" + suffix + ".fits", "rw");
200             f.addHDU(Fits.makeHDU(data));
201             f.write(bf);
202         } finally {
203             SafeClose.close(bf);
204             SafeClose.close(f);
205         }
206 
207         try {
208             f = new Fits("target/tiler" + suffix + ".fits");
209             ImageHDU h = (ImageHDU) f.readHDU();
210 
211             StandardImageTiler t = h.getTiler();
212             doTile("t1", data, t, 200, 200, 50, 50);
213             doTile2("t1", data, t, 200, 200, 50, 50);
214             doTile("t2", data, t, 133, 133, 72, 26);
215             doTile2("t2", data, t, 133, 133, 72, 26);
216 
217             h.getData().getKernel();
218             doTile("t3", data, t, 200, 200, 50, 50);
219             doTile2("t3", data, t, 200, 200, 50, 50);
220             doTile("t4", data, t, 133, 133, 72, 26);
221             doTile2("t4", data, t, 133, 133, 72, 26);
222 
223             Assert.assertFalse(doTile("t5", data, t, 500, 500, 72, 26));
224             IOException expected = null;
225             try {
226                 doTile2("t5", data, t, 500, 500, 72, 26);
227             } catch (IOException e) {
228                 expected = e;
229             }
230             Assert.assertNotNull(expected);
231             Assert.assertTrue(expected.getMessage().contains("within"));
232 
233             expected = null;
234             try {
235                 t.getTile(new int[]{
236                     10,
237                     10
238                 }, new int[]{
239                     20
240                 });
241             } catch (IOException e) {
242                 expected = e;
243             }
244             Assert.assertNotNull(expected);
245             Assert.assertTrue(expected.getMessage().contains("Inconsistent"));
246         } finally {
247             SafeClose.close(f);
248         }
249     }
