57   * This class tests the ImageTiler. It first creates a FITS file and then reads it back and allows the user to select
58   * tiles. The values of the corner and center pixels for the selected tile are displayed. Both file and memory tiles are
59   * checked.

63      private boolean doTile(String test, Object data, StandardImageTiler t, int x, int y, int nx, int ny) throws Exception {
64  
65          Class<?> baseClass = ArrayFuncs.getBaseClass(data);
66          Object tile = Array.newInstance(baseClass, nx * ny);
67          t.getTile(tile, new int[] {y, x}, new int[] {ny, nx});
68  
69          float sum0 = 0;
70          float sum1 = 0;
71          int length = Array.getLength(tile);
72          for (int i = 0; i < nx; i++) {
73              for (int j = 0; j < ny; j++) {
74                  int tileOffset = i + j * nx;
75                  if (tileOffset >= length) {
76                      return false;
77                  }
78                  sum0 += ((Number) Array.get(tile, tileOffset)).doubleValue();
79                  try {
80                      sum1 += ((Number) Array.get(Array.get(data, j + y), i + x)).doubleValue();
81                  } catch (ArrayIndexOutOfBoundsException e) {
82                      return false;
83                  }
84              }
85          }
86  
87          assertEquals("Tiler" + test, sum0, sum1, 0);
88  
89          return true;
90      }

92      private void doTile2(String test, Object data, StandardImageTiler t, int x, int y, int nx, int ny) throws Exception {
93  
94          Object tile = t.getTile(new int[] {y, x}, new int[] {ny, nx});
95  
96          float sum0 = 0;
97          float sum1 = 0;
98  
99          int length = Array.getLength(tile);
100         for (int i = 0; i < nx; i++) {
101             for (int j = 0; j < ny; j++) {
102                 int tileOffset = i + j * nx;
103                 if (tileOffset < length) {
104                     sum0 += ((Number) Array.get(tile, tileOffset)).doubleValue();
105                     sum1 += ((Number) Array.get(Array.get(data, j + y), i + x)).doubleValue();
106                 }
107             }
108         }
109 
110         assertEquals("Tiler" + test, sum0, sum1, 0);
111     }

113     private void doTile3(final String test, final Object data, final ImageTiler t, final int x, final int y, final int nx,
114             final int ny) throws Exception {
115         final ByteArrayOutputStream byteArrayOutputStream = new ByteArrayOutputStream();
116         final ArrayDataOutput output = new FitsOutputStream(byteArrayOutputStream);
117 
118         t.getTile(output, new int[] {y, x}, new int[] {ny, nx});
119 
120         float resultSum = 0;
121         float expectedSum = 0;
122         final ByteArrayInputStream byteArrayInputStream = new ByteArrayInputStream(byteArrayOutputStream.toByteArray());
123         final ArrayDataInput input = new FitsInputStream(byteArrayInputStream);
124         final Class<?> type = ArrayFuncs.getBaseClass(data);
125         final Object testOutput = ArrayFuncs.newInstance(type, ny * nx);
126 
127         input.readLArray(testOutput);
128 
129         int length = Array.getLength(testOutput);
130         for (int i = 0; i < nx; i++) {
131             for (int j = 0; j < ny; j++) {
132                 int tileOffset = i + j * nx;
133                 if (tileOffset < length) {
134                     resultSum += ((Number) Array.get(testOutput, tileOffset)).doubleValue();
135                     expectedSum += ((Number) Array.get(Array.get(data, j + y), i + x)).doubleValue();
136                 }
137             }
138         }
139 
140         assertEquals("StreamTiler_" + test, expectedSum, resultSum, 0);
141     }

215     private void doTest(Object data, String suffix) throws Exception {
216         Fits f = null;
217         FitsFile bf = null;
218         try {
219             f = new Fits();
220             bf = new FitsFile("target/tiler" + suffix + ".fits", "rw");
221             f.addHDU(Fits.makeHDU(data));
222             f.write(bf);
223         } finally {
224             SafeClose.close(bf);
225             SafeClose.close(f);
226         }
227 
228         try {
229             f = new Fits("target/tiler" + suffix + ".fits");
230             ImageHDU h = (ImageHDU) f.readHDU();
231 
232             StandardImageTiler t = h.getTiler();
233             doTile("t1", data, t, 200, 200, 50, 50);
234             doTile2("t1", data, t, 200, 200, 50, 50);
235             doTile3("t1", data, t, 200, 200, 50, 50);
236             doTile("t2", data, t, 133, 133, 72, 26);
237             doTile2("t2", data, t, 133, 133, 72, 26);
238             doTile3("t2", data, t, 133, 133, 72, 26);
239 
240             h.getData().getKernel();
241             doTile("t3", data, t, 200, 200, 50, 50);
242             doTile2("t3", data, t, 200, 200, 50, 50);
243             doTile3("t3", data, t, 200, 200, 50, 50);
244             doTile("t4", data, t, 133, 133, 72, 26);
245             doTile2("t4", data, t, 133, 133, 72, 26);
246             doTile3("t4", data, t, 133, 133, 72, 26);
247 
248             Assert.assertFalse(doTile("t5", data, t, 500, 500, 72, 26));
249             IOException expected = null;
250             try {
251                 doTile2("t5", data, t, 500, 500, 72, 26);
252             } catch (IOException e) {
253                 expected = e;
254             }
255             Assert.assertNotNull(expected);
256             Assert.assertTrue(expected.getMessage().contains("within"));
257 
258             expected = null;
259             try {
260                 doTile3("t5", data, t, 500, 500, 72, 26);
261             } catch (IOException e) {
262                 expected = e;
263             }
264             Assert.assertNotNull(expected);
265             Assert.assertTrue(expected.getMessage().contains("within"));
266 
267             expected = null;
268             try {
269                 t.getTile(new int[] {10, 10}, new int[] {20});
270             } catch (IOException e) {
271                 expected = e;
272             }
273             Assert.assertNotNull(expected);
274             Assert.assertTrue(expected.getMessage().contains("Inconsistent"));
275         } finally {
276             SafeClose.close(f);
277         }
