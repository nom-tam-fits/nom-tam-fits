49      private String extractComment(String string, int end) {
50          int startOfComment = string.indexOf('/', end);
51          int endOfComment = string.length() - 1;
52          if (startOfComment > 0 && endOfComment > startOfComment) {
53              startOfComment++;
54              if (Character.isWhitespace(string.charAt(startOfComment))) {
55                  startOfComment++;
56              }
57              while (Character.isWhitespace(string.charAt(endOfComment))) {
58                  endOfComment--;
59              }
60              if (!Character.isWhitespace(string.charAt(endOfComment))) {
61                  endOfComment++;
62              }
63              if (endOfComment > startOfComment) {
64                  return string.substring(startOfComment, endOfComment);
65              }
66          }
67          return null;
68      }

70      private String replaceAll(Pattern doubleQuotePattern, String quotedString) {
71          Matcher doubleQuoteMatcher = doubleQuotePattern.matcher(quotedString);
72          StringBuffer sb = new StringBuffer();
73          if (doubleQuoteMatcher.find(1)) {
74              do {
75                  doubleQuoteMatcher.appendReplacement(sb, "'");
76              } while (doubleQuoteMatcher.find());
77          }
78          doubleQuoteMatcher.appendTail(sb);
79          sb.deleteCharAt(0);
80          sb.deleteCharAt(sb.length() - 1);
81          return sb.toString();
82      }

97      @Test
98      public void testRegex() {
99          Pattern doubleQuotePattern = Pattern.compile("''");
100         Pattern stringPattern = Pattern.compile("'(?:[^']|'{2})*'");
101         String[] testStrings = {
102             "xx''   ",
103             "xx'test'   / ",
104             "xx''''  ",
105             "xx'test''' /    ",
106             "xx'test''a''aaa' /",
107             "xx'' / yy",
108             "xx'test' / yy",
109             "xx'''' / yy    ",
110             "xx'test''' / yy    ",
111             "xx'test''a''aaa' / yy                    ",
112             "xx'test''a''aaa              ' / yy                    ",
113         };
114         for (String string : testStrings) {
115             try {
116                 System.out.println("---" + string + "---");
117                 Matcher matcher = stringPattern.matcher(string);
118                 while (matcher.find()) {
119                     String result = replaceAll(doubleQuotePattern, matcher.group(0));
120                     String comment = extractComment(string, matcher.end());
121 
122                     System.out.println(result);
123                     System.out.println("comment:" + comment);
124                 }
125             } catch (Exception e) {
126                 e.printStackTrace();
127             }
128         }
