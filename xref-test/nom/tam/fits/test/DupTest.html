48  /**
49   * Test adding a little junk after a valid image. We wish to test three
50   * scenarios: Junk at the beginning (should continue to fail) Short (<80 byte)
51   * junk after valid HDU Long (>80 byte) junk after valid HDU The last two should
52   * succeed after FitsFactory.setAllowTerminalJunk(true).
53   */
54  public class DupTest {
55  
56      @Test
57      public void test() throws Exception {
58  
59          Fits f = new Fits("src/test/resources/nom/tam/fits/test/test_dup.fits");
60          Header hdr = f.readHDU().getHeader();
61          assertEquals("Internal size:", 8640, hdr.getSize());
62          assertEquals("External size:", 8640, hdr.getMinimumSize());
63          assertTrue("Has duplicates:", hdr.hadDuplicates());
64          List<HeaderCard> dups = hdr.getDuplicates();
65          
66          int nDups = dups.size();
67          System.out.println("Number of duplicates:" + nDups);
68          assertTrue("Has dups:", dups != null && dups.size() > 0);
69          // AK: It is rewritable with preallocated blank space that is now supported!
70          assertTrue("Not rewriteable:", hdr.rewriteable());
71             
72          DataOutputStream bf = new DataOutputStream(new FileOutputStream("target/created_dup.fits"));
73          hdr.resetOriginalSize();
74          assertEquals("External size, after reset", 2880, hdr.getMinimumSize());
75          f.write(bf);
76          bf.flush();
77          bf.close();
78          Fits g = new Fits("target/created_dup.fits");
79          hdr = g.readHDU().getHeader();
80          assertEquals("Internal size, after rewrite", 2880, hdr.getSize());
81          assertEquals("External size, after rewrite", 2880, hdr.getMinimumSize());
82          assertTrue("Now rewriteable", hdr.rewriteable());
83          assertFalse("No duplicates", hdr.hadDuplicates());
84          assertTrue("Dups is null", hdr.getDuplicates() == null);
85      }
86  }
