136     @Test
137     public void testAllHeaders() throws Exception {
138         Class<?>[] classes = new Class<?>[] {Checksum.class, CXCExt.class, CXCStclSharedExt.class, DataDescription.class,
139                 HierarchicalGrouping.class, InstrumentDescription.class, MaxImDLExt.class, NOAOExt.class, NonStandard.class,
140                 ObservationDescription.class, ObservationDurationDescription.class, SBFitsExt.class, Standard.class,
141                 STScIExt.class, Compression.class, WCS.class, DateTime.class};
142         for (Class<?> class1 : classes) {
143             for (Object enumConst : class1.getEnumConstants()) {
144                 Assert.assertNotNull(class1.getMethod("valueOf", String.class).invoke(class1,
145                         enumConst.getClass().getMethod("name").invoke(enumConst)));
146                 IFitsHeader iFitsHeader = (IFitsHeader) enumConst;
147                 if (class1 != Standard.class && !FitsKey.isCommentStyleKey(iFitsHeader.key())) {
148                     Assert.assertNotNull(iFitsHeader.comment());
149                 }
150                 String key = iFitsHeader.key();
151                 Assert.assertNotNull(key);
152                 Assert.assertNotNull(iFitsHeader.status());
153                 Assert.assertNotNull(iFitsHeader.valueType());
154                 Assert.assertNotNull(iFitsHeader.hdu());
155                 String url = iFitsHeader.status().url();
156                 if (url != null) {
157                     new URL(url);
158                 }
159                 int nCount = 0;
160                 int offset = 0;
161                 while ((offset = key.indexOf('n', offset)) >= 0) {
162                     nCount++;
163                     offset++;
164                 }
165                 int[] n = new int[nCount];
166                 Arrays.fill(n, 9);
167                 Assert.assertTrue(iFitsHeader.key(), iFitsHeader.n(n).key().indexOf('n') < 0);
168             }
169         }
170 
171     }
