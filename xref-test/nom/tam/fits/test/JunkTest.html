34  import static org.junit.Assert.assertTrue;
35  import nom.tam.fits.Fits;
36  import nom.tam.fits.FitsFactory;
37  import nom.tam.util.FitsFile;
38  import nom.tam.util.SafeClose;
39  
40  import org.junit.Test;
41  
42  /**
43   * Test adding a little junk after a valid image. We wish to test three
44   * scenarios: Junk at the beginning (should continue to fail) Short (<80 byte)
45   * junk after valid HDU Long (>80 byte) junk after valid HDU The last two should
46   * succeed after FitsFactory.setAllowTerminalJunk(true).
47   */
48  public class JunkTest {
49  
50      boolean readSuccess(String file) {
51          Fits f = null;
52          try {
53              f = new Fits(file);
54              f.read();
55              return true;
56          } catch (Exception e) {
57              return false;
58          } finally {
59              SafeClose.close(f);
60          }
61      }
62  
63      @Test
64      public void test() throws Exception {
65          Fits f = null;
66          try {
67              f = new Fits();
68  
69              byte[] bimg = new byte[40];
70              for (int i = 10; i < bimg.length; i += 1) {
71                  bimg[i] = (byte) i;
72              }
73  
74              // Make HDUs of various types.
75              f.addHDU(Fits.makeHDU(bimg));
76  
77              // Write a FITS file.
78  
79              // Valid FITS with one HDU
80              FitsFile bfx = null;
81              try {
82                  bfx = new FitsFile("target/j1.fits", "rw");
83                  f.write(bfx);
84                  bfx.flush();
85              } finally {
86                  SafeClose.close(bfx);
87              }
88  
89              // Invalid junk with no valid FITS.
90              FitsFile bf = new FitsFile("target/j2.fits", "rw");
91              bf.write(new byte[10]);
92              bf.close();
93  
94              // Valid FITS followed by short junk.
95              bf = new FitsFile("target/j3.fits", "rw");
96              f.write(bf);
97              bf.write("JUNKJUNK".getBytes());
98              bf.close();
99  
100             // Valid FITS followed by long junk.
101             bf = new FitsFile("target/j4.fits", "rw");
102             f.write(bf);
103             for (int i = 0; i < 100; i += 1) {
104                 bf.write("A random string".getBytes());
105             }
106             bf.close();
107         } finally {
108             SafeClose.close(f);
109         }
110 
111         int pos = 0;
112         try {
113             f = new Fits("target/j1.fits");
114             f.read();
115         } catch (Exception e) {
116             pos = 1;
117         } finally {
118             SafeClose.close(f);
119         }
120     
121         assertTrue("allow junk", FitsFactory.getAllowTerminalJunk());
122 
123         FitsFactory.setAllowTerminalJunk(true);
124 
125         assertTrue("Junk Test: Valid File OK,with junk", readSuccess("target/j1.fits"));
126         assertTrue("Junk Test: Invalid File Fails, with junk", !readSuccess("target/j2.fits"));
127         assertTrue("Junk Test: Short junk OK, with junk", readSuccess("target/j3.fits"));
128         assertTrue("Junk Test: Long junk OK, with junk", readSuccess("target/j4.fits"));
129 
130         FitsFactory.setAllowTerminalJunk(false);
131 
132         assertTrue("Junk Test: Valid File OK,No junk", readSuccess("target/j1.fits"));
133         assertTrue("Junk Test: Invalid File Fails, No junk", !readSuccess("target/j2.fits"));
134         assertTrue("Junk Test: Short junk fails, No junk", !readSuccess("target/j3.fits"));
135         assertTrue("Junk Test: Long junk fails, No junk", !readSuccess("target/j4.fits"));
136     }
137 }
