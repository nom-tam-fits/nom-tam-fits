52  public class LongCommentCardTest {
53  
54      private static int length = 200;
55      private static boolean enableLongKeywords = true;
56      
57      @Test
58      public void test() throws Exception {
59          length = 200;
60          
61          boolean longEnabled = FitsFactory.isLongStringsEnabled();
62          try {
63              // Enable/disable tthe OGIP 1.0 convention for long entries as
64              // requested...
65              FitsFactory.setLongStringsEnabled(enableLongKeywords);
66  
67              // Create a header only HDU (no data)
68              Header header = new Header();
69              header.setSimple(true);
70              header.setBitpix(32);
71              header.setNaxes(0);
72  
73              // Add a regular keyword of the desired length...
74              header.addLine(new HeaderCard("BLABERY", counts(length), ""));
75  
76              // Add a non-nullable HISTORY entry with the desired length...
77              header.addLine(new HeaderCard("HISTORY", counts(length), false));
78  
79              // Add a non-nullable COMMENT entry with the desired length...
80              header.addLine(HeaderCard.createCommentCard(counts(HeaderCard.MAX_COMMENT_CARD_COMMENT_LENGTH)));
81  
82              boolean thrown = false;
83              try {
84                  header.addLine(HeaderCard.createCommentCard(counts(HeaderCard.MAX_COMMENT_CARD_COMMENT_LENGTH + 1)));
85              } catch (LongValueException e) {
86                  thrown = true;
87              }
88              Assert.assertTrue(thrown);
89                
90              int n = 2;
91              
92              n += header.insertHistory(counts(length));
93              n += header.insertComment(counts(length));
94  
95              // Write the result to 'longcommenttest.fits' in the user's home...
96              Fits fits = new Fits();
97              fits.addHDU(Fits.makeHDU(header));
98              File file = new File("target/longcommenttest.fits");
99              FitsOutputStream stream = new FitsOutputStream(new FileOutputStream(file));
100 
101             try {
102                 fits.write(stream);
103             } catch (FitsException e) {
104                 throw e;
105             } finally {
106                 stream.close();
107             }
108             Assert.assertEquals(2880L, file.length());
109             int commentLike = 0;
110             fits = new Fits(file);
111             Cursor<String, HeaderCard> iterator = fits.getHDU(0).getHeader().iterator();
112             while (iterator.hasNext()) {
113                 HeaderCard headerCard = iterator.next();
114                 if (headerCard.isCommentStyleCard() && !Standard.END.key().equals(headerCard.getKey())) {
115                     commentLike++;
116                     Assert.assertTrue(headerCard.getComment(), headerCard.getComment().length() <= HeaderCard.MAX_COMMENT_CARD_COMMENT_LENGTH);
117                 }
118             }
119             Assert.assertEquals(n, commentLike);
120 
121         } finally {
122             FitsFactory.setLongStringsEnabled(longEnabled);
123         }
124 
125     }
126 
127     public String counts(int n) {
128         StringBuffer buf = new StringBuffer();
129         for (int i = 1; i <= n; i++)
130             buf.append(i % 10);
131         return new String(buf);
132     }
