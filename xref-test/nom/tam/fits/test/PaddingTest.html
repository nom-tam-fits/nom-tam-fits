48  /**
49   * Test that we can read files that fail due to lack of padding in the final
50   * HDU.
51   */
52  public class PaddingTest {
53  
54      @Test
55      public void test1() throws Exception {
56          Fits f = null;
57          FitsFile bf = null;
58          try {
59              f = new Fits();
60              bf = new FitsFile("target/padding1.fits", "rw");
61  
62              byte[][] bimg = new byte[20][20];
63              for (int i = 0; i < 20; i += 1) {
64                  for (int j = 0; j < 20; j += 1) {
65                      bimg[i][j] = (byte) (i + j);
66                  }
67              }
68  
69              BasicHDU<?> hdu = Fits.makeHDU(bimg);
70              Header hdr = hdu.getHeader();
71              hdr.addValue("NEWKEY", "TESTVALUE", "Test keyword");
72              hdr.write(bf);
73              bf.writeArray(bimg); // The data but no following padding.
74              bf.flush();
75          } finally {
76              SafeClose.close(bf);
77              SafeClose.close(f);
78          }
79  
80          // Now try reading this back.
81          try {
82              f = new Fits("target/padding1.fits");
83  
84              try {
85                  f.read();
86              } catch (PaddingException e) {
87                  assertEquals("HDUCount", 0, f.getNumberOfHDUs());
88                  f.addHDU(e.getTruncatedHDU());
89                  assertEquals("HDUCount2", 1, f.getNumberOfHDUs());
90              }
91              ImageHDU hdu0 = (ImageHDU) f.getHDU(0);
92              byte[][] aa = (byte[][]) hdu0.getKernel();
93              int miss = 0;
94              int match = 0;
95              for (int i = 0; i < 20; i += 1) {
96                  for (int j = 0; j < 20; j += 1) {
97                      if (aa[i][j] != (byte) (i + j)) {
98                          miss += 1;
99                      } else {
100                         match += 1;
101                     }
102                 }
103             }
104             assertEquals("PadMiss1:", miss, 0);
105             assertEquals("PadMatch1:", match, 400);
106             // Make sure we got the real header and not the one generated
107             // strictly
108             // from the data.
109             assertEquals("Update header:", hdu0.getHeader().getStringValue("NEWKEY"), "TESTVALUE");
110 
111             nom.tam.image.StandardImageTiler it = hdu0.getTiler();
112 
113             // Remember that the tile is always a flattened
114             // 1-D representation of the data.
115             byte[] data = (byte[]) it.getTile(new int[]{
116                 2,
117                 2
118             }, new int[]{
119                 2,
120                 2
121             });
122 
123             assertEquals("tilet1:", data.length, 4);
124             assertEquals("tilet2:", data[0] + 0, 4);
125             assertEquals("tilet3:", data[1] + 0, 5);
126             assertEquals("tilet4:", data[2] + 0, 5);
127             assertEquals("tilet5:", data[3] + 0, 6);
128         } finally {
129             SafeClose.close(f);
130         }
131     }

133     @Test
134     public void test2() throws Exception {
135         Fits f = null;
136         try {
137             f = new Fits();
138             byte[][] bimg = new byte[20][20];
139             for (int i = 0; i < 20; i += 1) {
140                 for (int j = 0; j < 20; j += 1) {
141                     bimg[i][j] = (byte) (i + j);
142                 }
143             }
144 
145             BasicHDU<?> hdu = Fits.makeHDU(bimg);
146             f.addHDU(hdu);
147 
148             // First create a FITS file with a truncated second HDU.
149             FitsFile bf = new FitsFile("target/padding2.fits", "rw");
150             f.write(bf);
151 
152             hdu.getHeader().setXtension(XTENSION_IMAGE);
153             Cursor<String, HeaderCard> curs = hdu.getHeader().iterator();
154             int cnt = 0;
155             // Write the header
156             while (curs.hasNext()) {
157                 bf.write(curs.next().toString().getBytes());
158                 cnt += 1;
159             }
160 
161             // The padding between header and data
162             byte[] b = new byte[(36 - cnt) * 80]; // Assuming fewer than 36
163                                                   // cards.
164             for (int i = 0; i < b.length; i += 1) {
165                 b[i] = 32; // i.e., a blank
166             }
167             bf.write(b);
168             for (int i = 0; i < 20; i += 1) {
169                 for (int j = 0; j < 20; j += 1) {
170                     bimg[i][j] = (byte) (2 * (i + j));
171                 }
172             }
173             bf.writeArray(bimg); // The data but no following padding.
174             bf.flush();
175             bf.close();
176         } finally {
177             SafeClose.close(f);
178         }
179 
180         // Now try reading this back.
181         try {
182             f = new Fits("target/padding2.fits");
183             try {
184                 f.read();
185             } catch (PaddingException e) {
186                 assertEquals("HDUCount", 1, f.getNumberOfHDUs());
187                 f.addHDU(e.getTruncatedHDU());
188                 assertEquals("HDUCount2", 2, f.getNumberOfHDUs());
189             }
190 
191             ImageHDU hdu0 = (ImageHDU) f.getHDU(0);
192             ImageHDU hdu1 = (ImageHDU) f.getHDU(1);
193             byte[][] aa = (byte[][]) hdu0.getKernel();
194             byte[][] bb = (byte[][]) hdu1.getKernel();
195             int miss = 0;
196             int match = 0;
197             for (int i = 0; i < 20; i += 1) {
198                 for (int j = 0; j < 20; j += 1) {
199                     if (bb[i][j] != (byte) (2 * aa[i][j])) {
200                         miss += 1;
201                     } else {
202                         match += 1;
203                     }
204                 }
205             }
206             assertEquals("PadMiss2:", miss, 0);
207             assertEquals("PadMatch2:", match, 400);
208         } finally {
209             SafeClose.close(f);
210         }
211     }
