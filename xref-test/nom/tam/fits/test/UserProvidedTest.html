101     private static ArrayList<Header> getHeaders(String filename) throws IOException, FitsException {
102         ArrayList<Header> list = new ArrayList<Header>();
103         Fits f = null;
104         try {
105             f = new Fits(filename);
106             BasicHDU<?> readHDU = f.readHDU();
107             while (readHDU != null) {
108                 f.deleteHDU(0);
109                 list.add(readHDU.getHeader());
110                 readHDU = f.readHDU();
111             }
112         } finally {
113             SafeClose.close(f);
114         }
115         return list;
116     }

246     private void printArray(Object array, int indent, StringBuilder builder) {
247         indent++;
248         if (array.getClass().isArray()) {
249             if (builder.charAt(builder.length() - 1) == '[') {
250                 builder.append("\n");
251             }
252             builder.append("[");
253             int length = Array.getLength(array);
254             for (int index = 0; index < length; index++) {
255                 if (index > 0) {
256                     builder.append(", ");
257                 }
258                 printArray(Array.get(array, index), indent, builder);
259             }
260             builder.append("]\n");
261         } else if (array instanceof String) {
262             builder.append("S");
263             builder.append(indent);
264         } else {
265             builder.append(indent);
266         }
267     }
