142     protected void checkByRow(Fits f) throws FitsException, IOException {
143         Object[] output = (Object[]) f.getHDU(1).getKernel();
144         Object[] input = getRowBlock(50);
145 
146         for (int i = 0; i < 50; i += 1) {
147             String[] str = (String[]) output[2];
148             String[] istr = (String[]) input[2];
149             int len1 = str[1].length();
150             str[i] = str[i].trim();
151             // The first row would have set the length for all the
152             // remaining rows...
153             if (istr[i].length() > len1) {
154                 istr[i] = istr[i].substring(0, len1);
155             }
156         }
157 
158         for (int j = 0; j < 3; j += 1) {
159             assertEquals("ByRow:" + j, true, TestArrayFuncs.arrayEquals(input[j], output[j], 1.e-6, 1.e-14));
160         }

257     public void modifyTable() throws Exception {
258 
259         Fits f = new Fits("target/at1.fits");
260         Object[] samp = getSampleCols();
261 
262         AsciiTableHDU hdu = (AsciiTableHDU) f.getHDU(1);
263         AsciiTable data = hdu.getData();
264         float[] f1 = (float[]) data.getColumn(0);
265         float[] f2 = f1.clone();
266         for (int i = 0; i < f2.length; i += 1) {
267             f2[i] = 2 * f2[i];
268         }
269 
270         data.setColumn(0, f2);
271         f1 = new float[]{
272             3.14159f
273         };
274         data.setElement(3, 0, f1);
275 
276         hdu.setNullString(0, "**INVALID**");
277         data.setNull(5, 0, true);
278         hdu.setNull(6, 0, true);
279 
280         Object[] row = new Object[5];
281         row[0] = new float[]{
282             6.28f
283         };
284         row[1] = new int[]{
285             22
286         };
287         row[2] = new long[]{
288             0
289         };
290         row[3] = new double[]{
291             -3
292         };
293         row[4] = new String[]{
294             "A string"
295         };
296 
297         data.setRow(5, row);
298 
299         data.setElement(4, 2, new long[]{
300             54321
301         });
302 
303         BufferedFile bf = new BufferedFile("target/at1x.fits", "rw");
304         f.write(bf);
305 
306         f = new Fits("target/at1x.fits");
307         AsciiTableHDU asciiHdu = (AsciiTableHDU) f.getHDU(1);
308         AsciiTable tab = asciiHdu.getData();
309         Object[] kern = (Object[]) tab.getKernel();
310 
311         float[] fx = (float[]) kern[0];
312         int[] ix = (int[]) kern[1];
313         long[] lx = (long[]) kern[2];
314         double[] dx = (double[]) kern[3];
315         String[] sx = (String[]) kern[4];
316 
317         float[] fy = (float[]) samp[0];
318         int[] iy = (int[]) samp[1];
319         long[] ly = (long[]) samp[2];
320         double[] dy = (double[]) samp[3];
321         String[] sy = (String[]) samp[4];
322 
323         assertEquals("Null", true, tab.isNull(6, 0));
324         assertEquals("Null2", false, tab.isNull(5, 0));
325         assertEquals("Null", true, hdu.isNull(6, 0));
326         assertEquals("Null2", false, hdu.isNull(5, 0));
327 
328         for (int i = 0; i < data.getNRows(); i += 1) {
329             if (i != 5) {
330                 if (i != 6) { // Null
331                     assertEquals("f" + i, 1., f2[i] / fx[i], 1.e-6);
332                 }
333                 assertEquals("i" + i, iy[i], ix[i]);
334                 if (i == 4) {
335                     assertEquals("l4", 54321L, lx[i]);
336                 } else {
337                     assertEquals("l" + i, ly[i], lx[i]);
338                 }
339                 assertEquals("d" + i, 1., dy[i] / dx[i], 1.e-14);
340                 assertEquals("s" + i, sy[i], sx[i].trim());
341             }
342         }
343         Object[] r5 = data.getRow(5);
344         String[] st = (String[]) r5[4];
345         st[0] = st[0].trim();
346         assertEquals("row5", true, TestArrayFuncs.arrayEquals(row, r5, 1.e-6, 1.e-14));
347 
348         addDeleteColumn(asciiHdu);
349     }
