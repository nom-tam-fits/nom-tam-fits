71      @Test
72      public void fileTest() throws Exception {
73          test();
74          byte[][] bimg = new byte[40][40];
75          for (int i = 10; i < 30; i += 1) {
76              for (int j = 10; j < 30; j += 1) {
77                  bimg[i][j] = (byte) (i + j);
78              }
79          }
80  
81          short[][] simg = (short[][]) ArrayFuncs.convertArray(bimg, short.class);
82          int[][] iimg = (int[][]) ArrayFuncs.convertArray(bimg, int.class);
83          long[][] limg = (long[][]) ArrayFuncs.convertArray(bimg, long.class);
84          float[][] fimg = (float[][]) ArrayFuncs.convertArray(bimg, float.class);
85          double[][] dimg = (double[][]) ArrayFuncs.convertArray(bimg, double.class);
86          int[][][] img3 = new int[10][20][30];
87          for (int i = 0; i < 10; i += 1) {
88              for (int j = 0; j < 20; j += 1) {
89                  for (int k = 0; k < 30; k += 1) {
90                      img3[i][j][k] = i + j + k;
91                  }
92              }
93          }
94          double[] img1 = (double[]) ArrayFuncs.flatten(dimg);
95  
96          BasicHDU<?>[] hdus;
97          try (Fits f = new Fits(new File("target/image1.fits"))) {
98              hdus = f.read();
99      
100             assertEquals("fbyte image", true, TestArrayFuncs.arrayEquals(bimg, hdus[0].getData().getKernel()));
101             assertEquals("fshort image", true, TestArrayFuncs.arrayEquals(simg, hdus[1].getData().getKernel()));
102             assertEquals("fint image", true, TestArrayFuncs.arrayEquals(iimg, hdus[2].getData().getKernel()));
103             assertEquals("flong image", true, TestArrayFuncs.arrayEquals(limg, hdus[3].getData().getKernel()));
104             assertEquals("ffloat image", true, TestArrayFuncs.arrayEquals(fimg, hdus[4].getData().getKernel()));
105             assertEquals("fdouble image", true, TestArrayFuncs.arrayEquals(dimg, hdus[5].getData().getKernel()));
106             assertEquals("fint3 image", true, TestArrayFuncs.arrayEquals(img3, hdus[6].getData().getKernel()));
107             assertEquals("fdouble1 image", true, TestArrayFuncs.arrayEquals(img1, hdus[7].getData().getKernel()));
108         }
109     }

111     @Test
112     public void test() throws Exception {
113 
114         byte[][] bimg = new byte[40][40];
115         for (int i = 10; i < 30; i += 1) {
116             for (int j = 10; j < 30; j += 1) {
117                 bimg[i][j] = (byte) (i + j);
118             }
119         }
120 
121         short[][] simg = (short[][]) ArrayFuncs.convertArray(bimg, short.class);
122         int[][] iimg = (int[][]) ArrayFuncs.convertArray(bimg, int.class);
123         long[][] limg = (long[][]) ArrayFuncs.convertArray(bimg, long.class);
124         float[][] fimg = (float[][]) ArrayFuncs.convertArray(bimg, float.class);
125         double[][] dimg = (double[][]) ArrayFuncs.convertArray(bimg, double.class);
126         int[][][] img3 = new int[10][20][30];
127         for (int i = 0; i < 10; i += 1) {
128             for (int j = 0; j < 20; j += 1) {
129                 for (int k = 0; k < 30; k += 1) {
130                     img3[i][j][k] = i + j + k;
131                 }
132             }
133         }
134 
135         double[] img1 = (double[]) ArrayFuncs.flatten(dimg);
136 
137         // Make HDUs of various types.
138         Exception actual = null;
139         try (Fits f = new Fits()) {
140             try {
141                 f.insertHDU(makeHDU(bimg), f.getNumberOfHDUs() + 1);
142             } catch (Exception ex) {
143                 actual = ex;
144             }
145             Assert.assertNotNull(actual);
146             f.insertHDU(makeHDU(bimg), f.getNumberOfHDUs());
147     
148             f.addHDU(Fits.makeHDU(simg));
149             f.addHDU(Fits.makeHDU(iimg));
150             f.addHDU(Fits.makeHDU(limg));
151             f.addHDU(Fits.makeHDU(fimg));
152             f.addHDU(Fits.makeHDU(dimg));
153             f.addHDU(Fits.makeHDU(img3));
154             f.addHDU(Fits.makeHDU(img1));
155     
156             assertEquals("HDU count before", f.getNumberOfHDUs(), 8);
157     
158             // Write a FITS file.
159     
160             try (BufferedFile bf = new BufferedFile("target/image1.fits", "rw")) {
161                 f.write(bf);
162                 bf.flush();
163             }
164         }
165 
166         try (BufferedFile bf = new BufferedFile(new File("target/image1.fits")); Fits f = new Fits("target/image1.fits")) {
167     
168             // Read a FITS file
169             BasicHDU<?>[] hdus = f.read();
170     
171             assertEquals("HDU count after", 8, f.getNumberOfHDUs());
172             assertEquals("byte image", true, TestArrayFuncs.arrayEquals(bimg, hdus[0].getData().getKernel()));
173             assertEquals("[40, 40]", Arrays.toString(hdus[0].getAxes()));
174             assertEquals("he was it", hdus[0].getAuthor());
175             assertEquals(8, hdus[0].getBitPix());
176             assertEquals(1.0, hdus[0].getBScale(), 0.000001);
177             assertEquals(0.0, hdus[0].getBZero(), 0.000001);
178             assertEquals(115, hdus[0].getCreationDate().getYear());
179             assertEquals(2, hdus[0].getCreationDate().getMonth());
180             // Date works in the local time zone which won't cause
181             // issues with the year or month, but may give us an
182             // off by one with the day. So we create a Calendar
183             // object to handle that more uniformly.
184             Calendar cal = Calendar.getInstance();
185             cal.setTimeZone(TimeZone.getTimeZone("GMT+00"));
186             cal.setTime(hdus[0].getCreationDate());
187             assertEquals(22, cal.get(Calendar.DAY_OF_MONTH));
188             assertEquals(2000.0, hdus[0].getEquinox(), 0.000001);
189             assertEquals("the biggest ever", hdus[0].getInstrument());
190             assertEquals(0.0, hdus[0].getMinimumValue(), 0.00001);
191             assertEquals(60.0, hdus[0].getMaximumValue(), 0.00001);
192             assertEquals(115, hdus[0].getObservationDate().getYear());
193             assertEquals(2, hdus[0].getObservationDate().getMonth());
194             cal.setTime(hdus[0].getObservationDate());
195             assertEquals(22, cal.get(Calendar.DAY_OF_MONTH));
196             assertEquals("he was it again", hdus[0].getObserver());
197             assertEquals("thats us", hdus[0].getOrigin());
198             assertEquals("over there", hdus[0].getReference());
199             assertEquals("the biggest ever scope", hdus[0].getTelescope());
200             assertEquals("wow object", hdus[0].getObject());
201             assertEquals(32, hdus[0].getBlankValue());
202             assertEquals("deg", hdus[0].getBUnit());
203             assertEquals(-2000., hdus[0].getEpoch(), 0.0001);
204     
205             assertEquals("short image", true, TestArrayFuncs.arrayEquals(simg, hdus[1].getData().getKernel()));
206             assertEquals("int image", true, TestArrayFuncs.arrayEquals(iimg, hdus[2].getData().getKernel()));
207             assertEquals("long image", true, TestArrayFuncs.arrayEquals(limg, hdus[3].getData().getKernel()));
208             assertEquals("float image", true, TestArrayFuncs.arrayEquals(fimg, hdus[4].getData().getKernel()));
209             assertEquals("double image", true, TestArrayFuncs.arrayEquals(dimg, hdus[5].getData().getKernel()));
210             assertEquals("int3 image", true, TestArrayFuncs.arrayEquals(img3, hdus[6].getData().getKernel()));
211             assertEquals("double1 image", true, TestArrayFuncs.arrayEquals(img1, hdus[7].getData().getKernel()));
212     
213             Assert.assertArrayEquals(new byte[0], (byte[]) new ImageData().getData());
214         }
215     }

217     private BasicHDU<?> makeHDU(Object data) throws FitsException {
218         BasicHDU<?> hdu = Fits.makeHDU(data);
219         hdu.addValue(AUTHOR, "he was it");
220         hdu.addValue(DATE, "2015-03-22");
221         hdu.addValue("EQUINOX", 2000.0, null);
222         hdu.addValue(INSTRUME, "the biggest ever");
223         hdu.addValue(DATAMIN, 0.0);
224         hdu.addValue(DATAMAX, 30 + 30);
225         hdu.addValue(DATE_OBS, "2015-03-22");
226         hdu.addValue(OBSERVER, "he was it again");
227         hdu.addValue(ORIGIN, "thats us");
228         hdu.addValue(REFERENC, "over there");
229         hdu.addValue(TELESCOP, "the biggest ever scope");
230         hdu.addValue(Standard.OBJECT, "wow object");
231         hdu.addValue(Standard.BLANK, 32);
232         hdu.addValue(Standard.BUNIT, "deg");
233         hdu.addValue(Standard.EPOCH, -2000.);
234 
235         return hdu;
236     }
