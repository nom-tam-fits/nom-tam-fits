80          test();
81          byte[][] bimg = new byte[40][40];
82          for (int i = 10; i < 30; i += 1) {
83              for (int j = 10; j < 30; j += 1) {
84                  bimg[i][j] = (byte) (i + j);

88          short[][] simg = (short[][]) ArrayFuncs.convertArray(bimg, short.class);

92          double[][] dimg = (double[][]) ArrayFuncs.convertArray(bimg, double.class);

96                  for (int k = 0; k < 30; k += 1) {
97                      img3[i][j][k] = i + j + k;
98                  }
99              }
100         }
101         double[] img1 = (double[]) ArrayFuncs.flatten(dimg);
102 
103         BasicHDU<?>[] hdus;
104         Fits f = null;
105         try {
106             f = new Fits(new File("target/image1.fits"));
107             hdus = f.read();
108 
109             assertEquals("fbyte image", true, TestArrayFuncs.arrayEquals(bimg, hdus[0].getData().getKernel()));
110             assertEquals("fshort image", true, TestArrayFuncs.arrayEquals(simg, hdus[1].getData().getKernel()));
111             assertEquals("fint image", true, TestArrayFuncs.arrayEquals(iimg, hdus[2].getData().getKernel()));

117         } finally {
118             SafeClose.close(f);

122     @Test
123     public void test() throws Exception {
124 
125         byte[][] bimg = new byte[40][40];
126         for (int i = 10; i < 30; i += 1) {
127             for (int j = 10; j < 30; j += 1) {
128                 bimg[i][j] = (byte) (i + j);
129             }
130         }
131 
132         short[][] simg = (short[][]) ArrayFuncs.convertArray(bimg, short.class);
133         int[][] iimg = (int[][]) ArrayFuncs.convertArray(bimg, int.class);
134         long[][] limg = (long[][]) ArrayFuncs.convertArray(bimg, long.class);
135         float[][] fimg = (float[][]) ArrayFuncs.convertArray(bimg, float.class);
136         double[][] dimg = (double[][]) ArrayFuncs.convertArray(bimg, double.class);
137         int[][][] img3 = new int[10][20][30];
138         for (int i = 0; i < 10; i += 1) {
139             for (int j = 0; j < 20; j += 1) {
140                 for (int k = 0; k < 30; k += 1) {
141                     img3[i][j][k] = i + j + k;
142                 }
143             }
144         }
145 
146         double[] img1 = (double[]) ArrayFuncs.flatten(dimg);
147 
148         // Make HDUs of various types.
149         Exception actual = null;
150         Fits f = null;
151         try {
152             f = new Fits();
153             try {
154                 f.insertHDU(makeHDU(bimg), f.getNumberOfHDUs() + 1);
155             } catch (Exception ex) {
156                 actual = ex;
157             }
158             Assert.assertNotNull(actual);
159             f.insertHDU(makeHDU(bimg), f.getNumberOfHDUs());
160 
161             f.addHDU(Fits.makeHDU(simg));
162             f.addHDU(Fits.makeHDU(iimg));
163             f.addHDU(Fits.makeHDU(limg));
164             f.addHDU(Fits.makeHDU(fimg));
165             f.addHDU(Fits.makeHDU(dimg));
166             f.addHDU(Fits.makeHDU(img3));
167             f.addHDU(Fits.makeHDU(img1));
168 
169             assertEquals("HDU count before", f.getNumberOfHDUs(), 8);
170 
171             // Write a FITS file.
172             FitsFile bf = null;
173             try {
174                 bf = new FitsFile("target/image1.fits", "rw");
175                 f.write(bf);
176                 bf.flush();
177             } finally {
178                 SafeClose.close(bf);
179             }
180         } finally {
181             SafeClose.close(f);
182         }
183 
184         FitsFile bf = null;
185         try {
186             bf = new FitsFile(new File("target/image1.fits"));
187             f = new Fits("target/image1.fits");
188 
189             // Read a FITS file
190             BasicHDU<?>[] hdus = f.read();
191 
192             assertEquals("HDU count after", 8, f.getNumberOfHDUs());
193             assertEquals("byte image", true, TestArrayFuncs.arrayEquals(bimg, hdus[0].getData().getKernel()));
194             assertEquals("[40, 40]", Arrays.toString(hdus[0].getAxes()));
195             assertEquals("he was it", hdus[0].getAuthor());
196             assertEquals(8, hdus[0].getBitPix());
197             assertEquals(1.0, hdus[0].getBScale(), 0.000001);
198             assertEquals(0.0, hdus[0].getBZero(), 0.000001);
199             assertEquals(115, hdus[0].getCreationDate().getYear());
200             assertEquals(2, hdus[0].getCreationDate().getMonth());
201             // Date works in the local time zone which won't cause
202             // issues with the year or month, but may give us an
203             // off by one with the day. So we create a Calendar
204             // object to handle that more uniformly.
205             Calendar cal = Calendar.getInstance();
206             cal.setTimeZone(TimeZone.getTimeZone("GMT+00"));
207             cal.setTime(hdus[0].getCreationDate());
208             assertEquals(22, cal.get(Calendar.DAY_OF_MONTH));
209             assertEquals(2000.0, hdus[0].getEquinox(), 0.000001);
210             assertEquals("the biggest ever", hdus[0].getInstrument());
211             assertEquals(0.0, hdus[0].getMinimumValue(), 0.00001);
212             assertEquals(60.0, hdus[0].getMaximumValue(), 0.00001);
213             assertEquals(115, hdus[0].getObservationDate().getYear());
214             assertEquals(2, hdus[0].getObservationDate().getMonth());
215             cal.setTime(hdus[0].getObservationDate());
216             assertEquals(22, cal.get(Calendar.DAY_OF_MONTH));
217             assertEquals("he was it again", hdus[0].getObserver());
218             assertEquals("thats us", hdus[0].getOrigin());
219             assertEquals("over there", hdus[0].getReference());
220             assertEquals("the biggest ever scope", hdus[0].getTelescope());
221             assertEquals("wow object", hdus[0].getObject());
222             assertEquals(32, hdus[0].getBlankValue());
223             assertEquals("deg", hdus[0].getBUnit());
224             assertEquals(-2000., hdus[0].getEpoch(), 0.0001);
225 
226             assertEquals("short image", true, TestArrayFuncs.arrayEquals(simg, hdus[1].getData().getKernel()));
227             assertEquals("int image", true, TestArrayFuncs.arrayEquals(iimg, hdus[2].getData().getKernel()));
228             assertEquals("long image", true, TestArrayFuncs.arrayEquals(limg, hdus[3].getData().getKernel()));
229             assertEquals("float image", true, TestArrayFuncs.arrayEquals(fimg, hdus[4].getData().getKernel()));
230             assertEquals("double image", true, TestArrayFuncs.arrayEquals(dimg, hdus[5].getData().getKernel()));
231             assertEquals("int3 image", true, TestArrayFuncs.arrayEquals(img3, hdus[6].getData().getKernel()));
232             assertEquals("double1 image", true, TestArrayFuncs.arrayEquals(img1, hdus[7].getData().getKernel()));
233 
234             Assert.assertArrayEquals(new byte[0], (byte[]) new ImageData().getData());
235         } finally {
236             SafeClose.close(f);
237             SafeClose.close(bf);
238         }
239     }

241     private BasicHDU<?> makeHDU(Object data) throws FitsException {
242         BasicHDU<?> hdu = Fits.makeHDU(data);
243         hdu.addValue(AUTHOR, "he was it");
244         hdu.addValue(DATE, "2015-03-22");
245         hdu.addValue("EQUINOX", 2000.0, null);
246         hdu.addValue(INSTRUME, "the biggest ever");
247         hdu.addValue(DATAMIN, 0.0);
248         hdu.addValue(DATAMAX, 30 + 30);
249         hdu.addValue(DATE_OBS, "2015-03-22");
250         hdu.addValue(OBSERVER, "he was it again");
251         hdu.addValue(ORIGIN, "thats us");
252         hdu.addValue(REFERENC, "over there");
253         hdu.addValue(TELESCOP, "the biggest ever scope");
254         hdu.addValue(Standard.OBJECT, "wow object");
255         hdu.addValue(Standard.BLANK, 32);
256         hdu.addValue(Standard.BUNIT, "deg");
257         hdu.addValue(Standard.EPOCH, -2000.);
258 
259         return hdu;
260     }

267     @Test
268     public void testWrongImageInfo() throws FitsException {
269         ImageHDU image = (ImageHDU) FitsFactory.hduFactory(new byte[10][10]);
270         image.getHeader().removeCard(Standard.SIMPLE.key());
271 
272         ByteArrayOutputStream out = new ByteArrayOutputStream();
273         PrintStream print = new PrintStream(out, true);
274         image.info(print);
275         Assert.assertTrue(out.toString().contains("bad header"));
276 
277         image = (ImageHDU) FitsFactory.hduFactory(new byte[10][10]);
278         image = new ImageHDU(image.getHeader(), new ImageData((Object) null));
279 
280         out = new ByteArrayOutputStream();
281         print = new PrintStream(out, true);
282         image.info(print);
283         Assert.assertTrue(out.toString().contains("No Data"));
284 
285         image = (ImageHDU) FitsFactory.hduFactory(new byte[10][10]);
286         image = new ImageHDU(image.getHeader(), new ImageData((Object) null) {
287 
288             @Override
289             public Object getData() {
290                 throw new IllegalStateException();
291             }
292         });
293 
294         out = new ByteArrayOutputStream();
295         print = new PrintStream(out, true);
296         image.info(print);
297         Assert.assertTrue(out.toString().contains("Unable"));
298 
299     }
