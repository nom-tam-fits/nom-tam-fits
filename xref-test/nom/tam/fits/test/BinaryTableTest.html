449     public void columnMetaTest() throws Exception {
450         Object[] data = new Object[]{
451             this.shorts,
452             this.ints,
453             this.floats,
454             this.doubles
455         };
456 
457         Fits f = new Fits();
458 
459         // Add two identical HDUs
460         BinaryTableHDU bhdu = (BinaryTableHDU) Fits.makeHDU(data);
461         f.addHDU(bhdu);
462 
463         // makeHDU creates the TFORM keywords and sometimes
464         // the TDIM keywords. Let's add some additional
465         // column metadata. For each column we'll want a TTYPE, TCOMM,
466         // TUNIT and TX and TY
467         // value and we want the final header to be in this order
468         // TTYPE, TCOMM, TFORM, [TDIM,] TUNIT, TX, TY
469         int oldNCols = bhdu.getNCols();
470 
471         for (int i = 0; i < bhdu.getNCols(); i += 1) {
472             bhdu.setColumnMeta(i, "TTYPE", "NAM" + (i + 1), null, false);
473             bhdu.setColumnMeta(i, "TCOMM", true, "Comment in comment", false);
474             bhdu.setColumnMeta(i, "TUNIT", "UNIT" + (i + 1), null, true);
475             bhdu.setColumnMeta(i, "TX", i + 1, null, true);
476             bhdu.setColumnMeta(i, "TY", 2. * (i + 1), null, true);
477         }
478         bhdu.setCurrentColumn(1);
479         Assert.assertEquals(-1, bhdu.findColumn("XXX"));
480 
481         BufferedFile ff = new BufferedFile("target/bt10.fits", "rw");
482         f.write(ff);
483         ff.close();
484         f = new Fits("target/bt10.fits");
485 
486         bhdu = (BinaryTableHDU) f.getHDU(1);
487         Header hdr = bhdu.getHeader();
488         assertEquals("metaCount", oldNCols, bhdu.getNCols());
489         for (int i = 0; i < bhdu.getNCols(); i += 1) {
490             // If this worked, the first header should be the TTYPE
491             hdr.findCard("TTYPE" + (i + 1));
492             HeaderCard hc = hdr.nextCard();
493             assertEquals("M" + i + "0", "TTYPE" + (i + 1), hc.getKey());
494             hc = hdr.nextCard();
495             assertEquals("M" + i + "A", "TCOMM" + (i + 1), hc.getKey());
496             hc = hdr.nextCard();
497             assertEquals("M" + i + "B", "TFORM" + (i + 1), hc.getKey());
498             hc = hdr.nextCard();
499             // There may have been a TDIM keyword inserted automatically. Let's
500             // skip it if it was. It should only appear immediately after the
501             // TFORM keyword.
502             if (hc.getKey().startsWith("TDIM")) {
503                 hc = hdr.nextCard();
504             }
505             assertEquals("M" + i + "C", "TUNIT" + (i + 1), hc.getKey());
506             hc = hdr.nextCard();
507             assertEquals("M" + i + "D", "TX" + (i + 1), hc.getKey());
508             hc = hdr.nextCard();
509             assertEquals("M" + i + "E", "TY" + (i + 1), hc.getKey());
510         }
