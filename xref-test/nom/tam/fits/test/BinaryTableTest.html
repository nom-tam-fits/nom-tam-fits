423     @Test
424     public void columnMetaTest() throws Exception {
425         Object[] data = new Object[] {this.shorts, this.ints, this.floats, this.doubles};
426 
427         Fits f = new Fits();
428 
429         // Add two identical HDUs
430         BinaryTableHDU bhdu = (BinaryTableHDU) Fits.makeHDU(data);
431         f.addHDU(bhdu);
432 
433         // makeHDU creates the TFORM keywords and sometimes
434         // the TDIM keywords. Let's add some additional
435         // column metadata. For each column we'll want a TTYPE, TCOMM,
436         // TUNIT and TX and TY
437         // value and we want the final header to be in this order
438         // TTYPE, TCOMM, TFORM, [TDIM,] TUNIT, TX, TY
439         int oldNCols = bhdu.getNCols();
440 
441         for (int i = 0; i < bhdu.getNCols(); i += 1) {
442             bhdu.setColumnMeta(i, "TTYPE", "NAM" + (i + 1), null, false);
443             bhdu.setColumnMeta(i, "TCOMM", true, "Comment in comment", false);
444             bhdu.setColumnMeta(i, "TUNIT", "UNIT" + (i + 1), null, true);
445             bhdu.setColumnMeta(i, "TX", i + 1, null, true);
446             bhdu.setColumnMeta(i, "TY", 2. * (i + 1), null, true);
447             bhdu.setColumnMeta(i, "TZ", 3. * (i + 1), i + 1, null, true);
448             bhdu.setColumnMeta(i, "TSCI", 3. * (i + 1), i + 1, null, true);
449         }
450         bhdu.setCurrentColumn(1);
451         Assert.assertEquals(-1, bhdu.findColumn("XXX"));
452 
453         FitsFile ff = new FitsFile("target/bt10.fits", "rw");
454         f.write(ff);
455         ff.close();
456         f = new Fits("target/bt10.fits");
457 
458         bhdu = (BinaryTableHDU) f.getHDU(1);
459         Header hdr = bhdu.getHeader();
460         assertEquals("metaCount", oldNCols, bhdu.getNCols());
461         for (int i = 0; i < bhdu.getNCols(); i += 1) {
462             // If this worked, the first header should be the TTYPE
463             hdr.findCard("TTYPE" + (i + 1));
464             HeaderCard hc = hdr.nextCard();
465             assertEquals("M" + i + "0", "TTYPE" + (i + 1), hc.getKey());
466             assertEquals(String.format("NAM%d", i + 1), hc.getValue());
467             hc = hdr.nextCard();
468             assertEquals("M" + i + "A", "TCOMM" + (i + 1), hc.getKey());
469             assertEquals("T", hc.getValue());
470             hc = hdr.nextCard();
471             assertEquals("M" + i + "B", "TFORM" + (i + 1), hc.getKey());
472 
473             hc = hdr.nextCard();
474             // There may have been a TDIM keyword inserted automatically. Let's
475             // skip it if it was. It should only appear immediately after the
476             // TFORM keyword.
477             if (hc.getKey().startsWith("TDIM")) {
478                 hc = hdr.nextCard();
479             }
480 
481             assertEquals("M" + i + "C", "TUNIT" + (i + 1), hc.getKey());
482             assertEquals(hc.getValue(), String.format("UNIT%d", i + 1));
483             hc = hdr.nextCard();
484             assertEquals("M" + i + "D", "TX" + (i + 1), hc.getKey());
485             assertEquals(i + 1, (int) hc.getValue(Integer.class, null));
486             hc = hdr.nextCard();
487             assertEquals("M" + i + "E", "TY" + (i + 1), hc.getKey());
488             assertEquals(2. * (i + 1), hc.getValue(Double.class, null), 1e-12);
489             hc = hdr.nextCard();
490             assertEquals("M" + i + "F", "TZ" + (i + 1), hc.getKey());
491             assertEquals(3. * (i + 1), hc.getValue(Double.class, null), 1e-12);
492             hc = hdr.nextCard();
493             assertEquals("M" + i + "F", "TSCI" + (i + 1), hc.getKey());
494 
495             assertEquals("3x" + (i + 1), 3.0 * (i + 1), hc.getValue(Double.class, 0.0).doubleValue(), 1e-12);
496         }
