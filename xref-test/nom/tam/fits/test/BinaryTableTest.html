448         int oldNCols = bhdu.getNCols();

451             bhdu.setColumnMeta(i, "TTYPE", "NAM" + (i + 1), null, false);
452             bhdu.setColumnMeta(i, "TCOMM", true, "Comment in comment", false);
453             bhdu.setColumnMeta(i, "TUNIT", "UNIT" + (i + 1), null, true);
454             bhdu.setColumnMeta(i, "TX", i + 1, null, true);
455             bhdu.setColumnMeta(i, "TY", 2. * (i + 1), null, true);
456             bhdu.setColumnMeta(i, "TZ", 3. * (i + 1), i + 1, null, true);
457             bhdu.setColumnMeta(i, "TSCI", 3. * (i + 1), i + 1, null, true);
458         }
459         bhdu.setCurrentColumn(1);
460         Assert.assertEquals(-1, bhdu.findColumn("XXX"));
461 
462         FitsFile ff = new FitsFile("target/bt10.fits", "rw");
463         f.write(ff);
464         ff.close();
465         f = new Fits("target/bt10.fits");
466 
467         bhdu = (BinaryTableHDU) f.getHDU(1);
468         Header hdr = bhdu.getHeader();
469         assertEquals("metaCount", oldNCols, bhdu.getNCols());
470         for (int i = 0; i < bhdu.getNCols(); i++) {
471             // If this worked, the first header should be the TTYPE
472             hdr.findCard("TTYPE" + (i + 1));
473             HeaderCard hc = hdr.nextCard();
474             assertEquals("M" + i + "0", "TTYPE" + (i + 1), hc.getKey());
475             assertEquals(String.format("NAM%d", i + 1), hc.getValue());
476             hc = hdr.nextCard();
477             assertEquals("M" + i + "A", "TCOMM" + (i + 1), hc.getKey());
478             assertEquals("T", hc.getValue());
479             hc = hdr.nextCard();
480             assertEquals("M" + i + "B", "TFORM" + (i + 1), hc.getKey());
481 
482             hc = hdr.nextCard();
483             // There may have been a TDIM keyword inserted automatically. Let's
484             // skip it if it was. It should only appear immediately after the
485             // TFORM keyword.
486             if (hc.getKey().startsWith("TDIM")) {
487                 hc = hdr.nextCard();
488             }
489 
490             assertEquals("M" + i + "C", "TUNIT" + (i + 1), hc.getKey());
491             assertEquals(hc.getValue(), String.format("UNIT%d", i + 1));
492             hc = hdr.nextCard();
493             assertEquals("M" + i + "D", "TX" + (i + 1), hc.getKey());
494             assertEquals(i + 1, (int) hc.getValue(Integer.class, null));
495             hc = hdr.nextCard();
496             assertEquals("M" + i + "E", "TY" + (i + 1), hc.getKey());
