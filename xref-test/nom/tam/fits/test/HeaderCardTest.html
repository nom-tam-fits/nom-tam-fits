54      @Test
55      public void test1() throws Exception {
56  
57          HeaderCard p;
58          p = HeaderCard.create("SIMPLE  =                     T");
59  
60          assertEquals("t1", "SIMPLE", p.getKey());
61          assertEquals("t2", "T", p.getValue());
62          assertNull("t3", p.getComment());
63  
64          p = HeaderCard.create("VALUE   =                   123");
65          assertEquals("t4", "VALUE", p.getKey());
66          assertEquals("t5", "123", p.getValue());
67          assertNull("t3", p.getComment());
68  
69          p = HeaderCard.create("VALUE   =    1.23698789798798E23 / Comment ");
70          assertEquals("t6", "VALUE", p.getKey());
71          assertEquals("t7", "1.23698789798798E23", p.getValue());
72          assertEquals("t8", "Comment", p.getComment());
73  
74          String lng = "111111111111111111111111111111111111111111111111111111111111111111111111";
75          p = HeaderCard.create("COMMENT " + lng);
76          assertEquals("t9", "COMMENT", p.getKey());
77          assertNull("t10", p.getValue());
78          assertEquals("t11", lng, p.getComment());
79  
80          boolean thrown = false;
81          try {
82              //
83              p = HeaderCard.create("VALUE   = '   ");
84          } catch (Exception e) {
85              thrown = true;
86          }
87          assertEquals("t12", true, thrown);
88  
89          p = HeaderCard.create("COMMENT " + lng + lng);
90          assertEquals("t13", lng, p.getComment());
91  
92          HeaderCard z = new HeaderCard("TTTT", 1.234567891234567891234567e101, "a comment");
93          assertTrue("t14", z.toString().indexOf("E") > 0);
94      }

96      @Test
97      public void test3() throws Exception {
98  
99          HeaderCard p = new HeaderCard("KEY", "VALUE", "COMMENT");
100         assertEquals("x1", "KEY     = 'VALUE   '           / COMMENT                                        ", p.toString());
101 
102         p = new HeaderCard("KEY", 123, "COMMENT");
103         assertEquals("x2", "KEY     =                  123 / COMMENT                                        ", p.toString());
104         p = new HeaderCard("KEY", 1.23, "COMMENT");
105         assertEquals("x3", "KEY     =                 1.23 / COMMENT                                        ", p.toString());
106         p = new HeaderCard("KEY", true, "COMMENT");
107         assertEquals("x4", "KEY     =                    T / COMMENT                                        ", p.toString());
108 
109         boolean thrown = false;
110         try {
111             p = new HeaderCard("LONGKEYWORD", 123, "COMMENT");
112         } catch (Exception e) {
113             thrown = true;
114         }
115         assertEquals("x5", true, thrown);
116 
117         thrown = false;
118         String lng = "00000000001111111111222222222233333333334444444444555555555566666666667777777777";
119         try {
120             Header.setLongStringsEnabled(false);
121             p = new HeaderCard("KEY", lng, "COMMENT");
122         } catch (Exception e) {
123             thrown = true;
124         }
125         assertEquals("x6", true, thrown);
126 
127         // Only trailing spaces are stripped.
128         p = new HeaderCard("STRING", "VALUE", null);
129         assertEquals("x6", "VALUE", p.getValue());
130 
131         p = new HeaderCard("STRING", "VALUE ", null);
132         assertEquals("x7", "VALUE", p.getValue());
133 
134         p = new HeaderCard("STRING", " VALUE", null);
135         assertEquals("x8", " VALUE", p.getValue());
136 
137         p = new HeaderCard("STRING", " VALUE ", null);
138         assertEquals("x9", " VALUE", p.getValue());
139 
140         p = new HeaderCard("QUOTES", "ABC'DEF", null);
141         assertEquals("x10", "ABC'DEF", p.getValue());
142         assertEquals("x10b", p.toString().indexOf("''") > 0, true);
143 
144         p = new HeaderCard("QUOTES", "ABC''DEF", null);
145         assertEquals("x11", "ABC''DEF", p.getValue());
146         assertEquals("x10b", p.toString().indexOf("''''") > 0, true);
147     }

241     @Test
242     public void testHierarch() throws Exception {
243 
244         HeaderCard hc;
245         String key = "HIERARCH.TEST1.TEST2.INT";
246         boolean thrown = false;
247         try {
248             hc = new HeaderCard(key, 123, "Comment");
249         } catch (Exception e) {
250             thrown = true;
251         }
252         assertEquals("h1", true, thrown);
253 
254         String card = "HIERARCH TEST1 TEST2 INT=           123 / Comment                               ";
255         hc = HeaderCard.create(card);
256         assertEquals("h2", "HIERARCH", hc.getKey());
257         assertNull("h3", hc.getValue());
258         // its wrong because setUseHierarch -> false
259         assertEquals("h4", "TEST1 TEST2 INT=           123 / Comment", hc.getComment());
260 
261         FitsFactory.setUseHierarch(true);
262 
263         hc = new HeaderCard(key, 123, "Comment");
264 
265         assertEquals("HIERARCH TEST1 TEST2 INT=           123 / Comment", hc.toString().trim());
266 
267         assertEquals("h5", key, hc.getKey());
268         assertEquals("h6", "123", hc.getValue());
269         assertEquals("h7", "Comment", hc.getComment());
270 
271         hc = HeaderCard.create(card);
272         assertEquals("h8", key, hc.getKey());
273         assertEquals("h9", "123", hc.getValue());
274         assertEquals("h10", "Comment", hc.getComment());
275 
276         // now test a longString
277 
278         FitsFactory.setLongStringsEnabled(true);
279 
280         hc = new HeaderCard(key, "a verly long value that must be splitted over multiple lines to fit the card", "the comment is also not the smallest");
281 
282         assertEquals("HIERARCH TEST1 TEST2 INT= 'a verly long value that must be splitted over multi&'" + //
283                 "CONTINUE  'ple lines to fit the card' / the comment is also not the smallest    ", hc.toString());
284 
285         FitsFactory.setLongStringsEnabled(false);
286     }
