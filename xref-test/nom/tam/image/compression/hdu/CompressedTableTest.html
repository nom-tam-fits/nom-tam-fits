75      @Test
76      public void testBasicBinaryTable() throws Exception {
77  
78          Fits fitsComp = new Fits("src/test/resources/nom/tam/table/comp/testBinaryTable.fits.fz");
79          CompressedTableHDU cfitsioTable = (CompressedTableHDU) fitsComp.getHDU(1);
80  
81          cfitsioTable.info(System.out);
82  
83          BinaryTableHDU binaryTable2HDU = cfitsioTable.asBinaryTableHDU();
84  
85          Fits fits = new Fits();
86          fits.addHDU(binaryTable2HDU);
87          fits.write(new File("target/testBinaryTable_uncompressed.fits"));
88          fits.close();
89  
90          CompressedTableHDU compressed = CompressedTableHDU.fromBinaryTableHDU(binaryTable2HDU, 10).compress();
91          compressed.compress();
92          fits = new Fits();
93          fits.addHDU(compressed);
94          fits.write(new File("target/testBinaryTable_recompressed.fits"));
95          fits.close();
96  
97          fits = new Fits("target/testBinaryTable_uncompressed.fits");
98          Header header = fits.getHDU(1).getHeader();
99          Cursor<String, HeaderCard> iter = header.iterator();
100         assertStringCard(Standard.XTENSION, XTENSION_BINTABLE, iter.next());
101         assertIntCard(Standard.BITPIX, 8, iter.next());
102         assertIntCard(Standard.NAXIS, 2, iter.next());
103         assertIntCard(Standard.NAXISn.n(1), 200, iter.next());
104         assertIntCard(Standard.NAXISn.n(2), 50, iter.next());
105         assertIntCard(Standard.PCOUNT, 0, iter.next());
106         assertIntCard(Standard.GCOUNT, 1, iter.next());
107         assertIntCard(Standard.TFIELDS, 1, iter.next());
108         // the order of the next two fields is not fix
109         assertStringCard(Standard.TFORMn.n(1), "25D", header.card(Standard.TFORMn.n(1)).card());
110         assertStringCard(Standard.TDIMn.n(1), "(5,5)", header.card(Standard.TDIMn.n(1)).card());
111         fits.close();
112 
113         fits = new Fits("target/testBinaryTable_recompressed.fits");
114         header = fits.getHDU(1).getHeader();
115         iter = header.iterator();
116         assertStringCard(Standard.XTENSION, XTENSION_BINTABLE, iter.next());
117         assertIntCard(Standard.BITPIX, 8, iter.next());
118         assertIntCard(Standard.NAXIS, 2, iter.next());
119         assertIntCard(Standard.NAXISn.n(1), 8, iter.next());
120         assertIntCard(Standard.NAXISn.n(2), 5, iter.next());
121 
122         HeaderCard hc = iter.next();
123         Assert.assertEquals(Standard.PCOUNT.key(), hc.getKey());
124         Assert.assertNotEquals((long) hc.getValue(Long.class, 0L), 0L);
125 
126         assertIntCard(Standard.GCOUNT, 1, iter.next());
127         assertIntCard(Standard.TFIELDS, 1, iter.next());
128         // the order of the next two fields is not fix
129         assertStringCard(Standard.TFORMn.n(1), "1PB", header.card(Standard.TFORMn.n(1)).card());
130         assertStringCard(Standard.TDIMn.n(1), "(5,5)", header.card(Standard.TDIMn.n(1)).card());
131         fits.close();
132     }

136         String compressedVarTable = BlackBoxImages.getBlackBoxImage("map_one_source_a_level_1_cal.fits.fz");
137         Fits fitsComp = new Fits(compressedVarTable);
138         CompressedTableHDU cfitsioTable = (CompressedTableHDU) fitsComp.getHDU(1);
139 
140         cfitsioTable.info(System.out);
141 
142         BinaryTableHDU binaryTable2HDU = cfitsioTable.asBinaryTableHDU();
143         Fits fits = new Fits();
144         fits.addHDU(binaryTable2HDU);
145         fits.write(new File("target/map_one_source_a_level_1_cal_uncompressed.fits"));
146         fits.close();
147         fitsComp.close();
148 
149         fits = new Fits("target/map_one_source_a_level_1_cal_uncompressed.fits");
150         BinaryTableHDU table = (BinaryTableHDU) fits.getHDU(1);
151 
152         Assert.assertNotNull(table); // table could be read.
153 
154         Assert.assertEquals(21, table.getNCols());
155         Assert.assertEquals(1, table.getNRows());
156 
157         String originalVarTable = BlackBoxImages.getBlackBoxImage("map_one_source_a_level_1_cal.fits");
158         Fits fitsOrg = new Fits(originalVarTable);
159         Header orgHeader = fitsOrg.getHDU(1).getHeader();
160 
161         Assert.assertEquals(orgHeader.getSize(), table.getHeader().getSize());
162         Cursor<String, HeaderCard> iterator = orgHeader.iterator();
163         while (iterator.hasNext()) {
164             HeaderCard headerCard = iterator.next();
165             String uncompressed = table.getHeader().getStringValue(headerCard.getKey());
166             String original = orgHeader.getStringValue(headerCard.getKey());
167             if (uncompressed != null || original != null) {
168                 Assert.assertEquals(original, uncompressed);
169             }
170         }
171         fitsOrg.close();
172     }

208                 decompressedElement = decompressedTable.getElement(row, 4);
209                 orgElement = orgTable.getElement(row, 4);
210                 Assert.assertArrayEquals((String[]) orgElement, (String[]) decompressedElement);
211             }
212 
213         } finally {
214             SafeClose.close(fitsComp);

242                 Assert.assertArrayEquals("column " + column, org, decompressed);
243             }
244 
245         } finally {
246             SafeClose.close(fitsComp);
247         }

250     private int[] calculateOffsets(byte[] byteArray, int primitiveSize) {
251         int[] offset = new int[primitiveSize];
252         offset[0] = 0;
253         for (int primitivIndex = 1; primitivIndex < primitiveSize; primitivIndex++) {
254             offset[primitivIndex] = offset[primitivIndex - 1] + byteArray.length / primitiveSize;
255         }
256         return offset;
257     }
258 
259     public byte[] unshuffle(byte[] byteArray, int primitiveSize) {
260         byte[] result = new byte[byteArray.length];
261         int resultIndex = 0;
262         int[] offset = calculateOffsets(byteArray, primitiveSize);
263         for (int index = 0; index < byteArray.length; index += primitiveSize) {
264             for (int primitiveIndex = 0; primitiveIndex < primitiveSize; primitiveIndex++) {
265                 result[index + primitiveIndex] = byteArray[resultIndex + offset[primitiveIndex]];
266             }
267             resultIndex++;
268         }
269         return result;
270     }
271 
272     private byte[] decompress(Object decompressedElement) throws IOException {
273         if (decompressedElement instanceof byte[]) {
274             if (((byte[]) decompressedElement)[0] != 31 && ((byte[]) decompressedElement)[1] != -117) {
275                 return (byte[]) decompressedElement;
276             }
277         }
278         ByteArrayOutputStream decompressed = new ByteArrayOutputStream();
279         byte[] buffer = new byte[1024];
280         GZIPInputStream gzipInputStream = new GZIPInputStream(new ByteArrayInputStream((byte[]) decompressedElement));
281         int count;
282         while ((count = gzipInputStream.read(buffer)) >= 0) {
283             decompressed.write(buffer, 0, count);
284         }
285         return decompressed.toByteArray();
286     }

288     @Test
289     public void testB12TableTileDecompress() throws Exception {
290         Fits fitsUncompressed = new Fits("src/test/resources/nom/tam/table/comp/bt12.fits");
291         int tileSize = 5;
292 
293         CompressedTableHDU compressedTable = CompressedTableHDU
294                 .fromBinaryTableHDU((BinaryTableHDU) fitsUncompressed.getHDU(1), tileSize).compress();
295         compressedTable.compress();
296 
297         BinaryTableHDU hdu = compressedTable.asBinaryTableHDU(1);
298 
299         Assert.assertEquals(tileSize, hdu.getNRows());
300 
301         BinaryTableHDU hdu0 = (BinaryTableHDU) fitsUncompressed.getHDU(1);
302 
303         BinaryTable tab0 = hdu0.getData();
304         BinaryTable tab = hdu.getData();
305 
306         for (int row = 0; row < hdu.getNRows(); row++) {
307             int row0 = tileSize + row;
308             Assert.assertEquals("row " + row, tab0.getElement(row0, 0), tab.getElement(row, 0));
309             Assert.assertArrayEquals("row " + row, (short[]) tab0.getElement(row0, 1), (short[]) tab.getElement(row, 1));
310 
311             float[][] f0 = (float[][]) tab0.getElement(row0, 2);
312             float[][] f = (float[][]) tab.getElement(row, 2);
313             Assert.assertEquals(f0.length, f.length);
314             for (int j = 0; j < f0.length; j++) {
315                 Assert.assertArrayEquals(f0[j], f[j], 1e-4f);
316             }
317 
318             Assert.assertArrayEquals("row " + row, (double[]) tab0.getElement(row0, 3), (double[]) tab.getElement(row, 3),
319                     1e-8);
320 
321             String[] s0 = (String[]) tab0.getElement(row0, 4);
322             String[] s = (String[]) tab.getElement(row, 4);
323             Assert.assertEquals(s0.length, s.length);
324             for (int j = 0; j < s0.length; j++) {
325                 Assert.assertEquals(s0[j], s[j]);
326             }
327         }
328 
329         fitsUncompressed.close();
330     }
