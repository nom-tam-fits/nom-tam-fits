398     private void testTileSizes(int tileWidth, int tileHeight) throws HeaderCardException, FitsException {
399         int imageSize = 100;
400         TiledImageCompressionOperation operationsOfImage = new TiledImageCompressionOperation(null);
401         Buffer buffer = IntBuffer.allocate(imageSize * imageSize);
402         Header header = new Header();
403         header.addValue(ZBITPIX, 32);
404         header.addValue(ZNAXIS, 2);
405         header.addValue(ZNAXISn.n(1), imageSize);
406         header.addValue(ZNAXISn.n(2), imageSize);
407         header.addValue(ZTILEn.n(1), tileWidth);
408         header.addValue(ZTILEn.n(2), tileHeight);
409 
410         operationsOfImage.readPrimaryHeaders(header);
411         operationsOfImage.prepareUncompressedData(buffer);
412         List<TileCompressionOperation> tiles = getTileCompressionOperations(operationsOfImage);
413         int heigth = 0;
414         int width = 0;
415         int pixels = 0;
416         for (TileCompressionOperation tileOperation : tiles) {
417             if (tileWidth == imageSize) {
418                 heigth += Access.getTileBuffer(tileOperation).getHeight();
419             } else if (tileHeight == imageSize) {
420                 width += Access.getTileBuffer(tileOperation).getWidth();
421             }
422             pixels += Access.getTileBuffer(tileOperation).getHeight() * Access.getTileBuffer(tileOperation).getWidth();
423         }
424         Assert.assertEquals(imageSize * imageSize, pixels);
425         if (heigth != 0) {
426             Assert.assertEquals(imageSize, heigth);
427         }
428         if (width != 0) {
429             Assert.assertEquals(imageSize, width);
430         }
431     }

433     private static List<TileCompressionOperation> getTileCompressionOperations(TiledImageCompressionOperation operationsOfImage) {
434         List<TileCompressionOperation> tiles = new ArrayList<TileCompressionOperation>();
435         try {
436             for (int index = 0; index < 10000; index++) {
437                 tiles.add((TileCompressionOperation) Access.getTile(operationsOfImage, index));
438             }
439         } catch (ArrayIndexOutOfBoundsException e) {
440             return tiles;
441         }
442         return tiles;
443     }
