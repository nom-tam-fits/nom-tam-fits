51  public class TileCompressionTest {
52  
53      public int[][] getRectangularImage(int nx, int ny) {
54          int[][] im = new int[ny][nx];
55  
56          for (int y = 0; y < im.length; y++) for (int x = 0; x < im[y].length; x++) {
57              im[y][x] = x + y;
58          }
59  
60          return im;
61      }
62      
63      @Test
64      public void rectangularRiceCompressTest() throws Exception {
65          int[][] im = getRectangularImage(32, 80);
66          String fileName = "target/rect_comp.fits.fz";
67  
68          ImageHDU hdu = (ImageHDU) FitsFactory.hduFactory(im);
69          CompressedImageHDU cHDU = CompressedImageHDU.fromImageHDU(hdu, -1, 1);
70  
71          cHDU.setCompressAlgorithm(Compression.ZCMPTYPE_RICE_1)
72          .setQuantAlgorithm(null)
73          .getCompressOption(RiceCompressOption.class)
74          .setBlockSize(32);
75  
76          cHDU.compress();
77  
78          Fits f = new Fits();
79          f.addHDU(cHDU);
80          f.write(fileName);
81  
82          f = new Fits(fileName);
83          cHDU = (CompressedImageHDU) f.read()[1];
84  
85          hdu = cHDU.asImageHDU();
86          int[][] im2 = (int[][]) hdu.getKernel();
87          
88          Assert.assertArrayEquals(im, im2);
89      }
