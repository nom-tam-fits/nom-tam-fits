50      private final class TestImageTiler extends StandardImageTiler {
51  
52          private TestImageTiler(RandomAccess f, long fileOffset, int[] dims, Class<?> base) {
53              super(f, fileOffset, dims, base);
54          }
55  
56          @Override
57          protected Object getMemoryImage() {
58              return StandardImageTilerTest.this.dataArray;
59          }
60  
61          public void setFile(Object file) throws Exception {
62              Field declaredField = StandardImageTiler.class.getDeclaredField("randomAccessFile");
63              declaredField.setAccessible(true);
64              declaredField.set(this, file);
65          }
66  
67          public void setBase(Object base) throws Exception {
68              Field declaredField = StandardImageTiler.class.getDeclaredField("base");
69              declaredField.setAccessible(true);
70              declaredField.set(this, base);
71          }
72  
73      }
74  
75      private TestImageTiler tiler;
76  
77      private FitsFile file;
78  
79      private int[][] dataArray;
80  
81      @Before
82      public void setup() throws Exception {
83          dataArray = new int[10][10];
84          for (int index = 0; index < dataArray.length; index++) {
85              for (int index2 = 0; index2 < dataArray[index].length; index2++) {
86                  dataArray[index][index2] = 1;
87              }
88          }
89          FitsFile file = new FitsFile("target/StandardImageTilerTest", "rw");
90          file.writeArray(dataArray);
91          file.seek(0);
92          tiler = new TestImageTiler(file, 0, ArrayFuncs.getDimensions(dataArray), ArrayFuncs.getBaseClass(dataArray));
93  
94      }
95  
96      @After
97      public void close() {
98          SafeClose.close(file);
99      }

101     @Test
102     public void testFailedGetTile() throws Exception {
103         dataArray = null;
104         tiler.setFile(null);
105         IOException actual = null;
106         try {

109             actual = e;

130     @Test
131     public void testFillTileNegativeBounds() throws Exception {
132 
133         int[] lengths = new int[]{
134             2,
135             2
136         };
137         int[] corners = new int[]{
138             -1,
139             -1
140         };
141         int[] newDims = new int[]{
142             2,
143             2
144         };
145         int[] tile = new int[25];
146         tiler.fillTile(null, tile, newDims, corners, lengths);
147         Assert.assertEquals(1, tile[3]);
148         tile[3] = 0;
149         // check the rest should be 0
150         Assert.assertArrayEquals(new int[25], tile);
151     }
