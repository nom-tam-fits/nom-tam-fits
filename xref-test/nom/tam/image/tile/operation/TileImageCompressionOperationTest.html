54      @Test
55      public void tileImageSubtileTest() throws Exception {
56          Buffer buf = ByteBuffer.wrap(new byte[10000]);
57  
58          TiledImageCompressionOperation op = new TiledImageCompressionOperation(new BinaryTable());
59          op.setTileAxes(new int[] {10});
60          op.setAxes(new int[] {100, 100});
61          op.setBaseType(ElementType.forNearestBitpix(8));
62          op.prepareUncompressedData(buf);
63          Assert.assertArrayEquals(new int[] {1, 10}, op.getTileAxes());
64      }

78      @Test
79      public void testZquantizValues() throws Exception {
80          TiledImageCompressionOperation op = new TiledImageCompressionOperation(null);
81  
82          op.setQuantAlgorithm(HeaderCard.create(Compression.ZQUANTIZ, Compression.ZQUANTIZ_NO_DITHER));
83          Assert.assertEquals(op.getQuantAlgorithm(), Compression.ZQUANTIZ_NO_DITHER);
84  
85          op.setQuantAlgorithm(HeaderCard.create(Compression.ZQUANTIZ, Compression.ZQUANTIZ_SUBTRACTIVE_DITHER_1));
86          Assert.assertEquals(op.getQuantAlgorithm(), Compression.ZQUANTIZ_SUBTRACTIVE_DITHER_1);
87  
88          op.setQuantAlgorithm(HeaderCard.create(Compression.ZQUANTIZ, Compression.ZQUANTIZ_SUBTRACTIVE_DITHER_2));
89          Assert.assertEquals(op.getQuantAlgorithm(), Compression.ZQUANTIZ_SUBTRACTIVE_DITHER_2);
90  
91          op.setQuantAlgorithm(HeaderCard.create(Compression.ZQUANTIZ, "invalid value"));
92          Assert.assertNull(op.getQuantAlgorithm());
93  
94          op.setQuantAlgorithm(null);
95          Assert.assertNull(op.getQuantAlgorithm());
96      }
97  
98      @Test
99      public void testZcmptypeValues() throws Exception {
100         TiledImageCompressionOperation op = new TiledImageCompressionOperation(null);
101 
102         op.setCompressAlgorithm(HeaderCard.create(Compression.ZCMPTYPE, Compression.ZCMPTYPE_NOCOMPRESS));
103         Assert.assertEquals(op.getCompressAlgorithm(), Compression.ZCMPTYPE_NOCOMPRESS);
104 
105         op.setCompressAlgorithm(HeaderCard.create(Compression.ZCMPTYPE, Compression.ZCMPTYPE_GZIP_1));
106         Assert.assertEquals(op.getCompressAlgorithm(), Compression.ZCMPTYPE_GZIP_1);
107 
108         op.setCompressAlgorithm(HeaderCard.create(Compression.ZCMPTYPE, Compression.ZCMPTYPE_GZIP_2));
109         Assert.assertEquals(op.getCompressAlgorithm(), Compression.ZCMPTYPE_GZIP_2);
110 
111         op.setCompressAlgorithm(HeaderCard.create(Compression.ZCMPTYPE, Compression.ZCMPTYPE_RICE_1));
112         Assert.assertEquals(op.getCompressAlgorithm(), Compression.ZCMPTYPE_RICE_1);
113 
114         op.setCompressAlgorithm(HeaderCard.create(Compression.ZCMPTYPE, Compression.ZCMPTYPE_PLIO_1));
115         Assert.assertEquals(op.getCompressAlgorithm(), Compression.ZCMPTYPE_PLIO_1);
116 
117         op.setCompressAlgorithm(HeaderCard.create(Compression.ZCMPTYPE, Compression.ZCMPTYPE_HCOMPRESS_1));
118         Assert.assertEquals(op.getCompressAlgorithm(), Compression.ZCMPTYPE_HCOMPRESS_1);
119 
120         op.setCompressAlgorithm(HeaderCard.create(Compression.ZCMPTYPE, "invalid value"));
121         Assert.assertNull(op.getCompressAlgorithm());
122 
123         op.setCompressAlgorithm(null);
124         Assert.assertNull(op.getCompressAlgorithm());
125     }
